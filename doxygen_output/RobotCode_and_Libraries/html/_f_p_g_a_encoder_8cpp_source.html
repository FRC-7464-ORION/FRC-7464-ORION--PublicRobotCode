44 HAL_FPGAEncoderHandle HAL_InitializeFPGAEncoder(
45  HAL_Handle digitalSourceHandleA, HAL_AnalogTriggerType analogTriggerTypeA,
46  HAL_Handle digitalSourceHandleB, HAL_AnalogTriggerType analogTriggerTypeB,
47  HAL_Bool reverseDirection, int32_t* index, int32_t* status) {
48  hal::init::CheckInit();
49  bool routingAnalogTriggerA = false;
50  uint8_t routingChannelA = 0;
51  uint8_t routingModuleA = 0;
52  bool successA = remapDigitalSource(digitalSourceHandleA, analogTriggerTypeA,
53  routingChannelA, routingModuleA,
54  routingAnalogTriggerA);
55  bool routingAnalogTriggerB = false;
56  uint8_t routingChannelB = 0;
57  uint8_t routingModuleB = 0;
58  bool successB = remapDigitalSource(digitalSourceHandleB, analogTriggerTypeB,
59  routingChannelB, routingModuleB,
60  routingAnalogTriggerB);
61 
62  if (!successA || !successB) {
63  *status = HAL_HANDLE_ERROR;
64  return HAL_kInvalidHandle;
65  }
66 
67  auto handle = fpgaEncoderHandles->Allocate();
68  if (handle == HAL_kInvalidHandle) { // out of resources
69  *status = NO_AVAILABLE_RESOURCES;
70  return HAL_kInvalidHandle;
71  }
72 
73  auto encoder = fpgaEncoderHandles->Get(handle);
74  if (encoder == nullptr) { // will only error on thread issue
75  *status = HAL_HANDLE_ERROR;
76  return HAL_kInvalidHandle;
77  }
78 
79  encoder->index = static_cast<uint8_t>(getHandleIndex(handle));
80  *index = encoder->index;
81  // TODO: if (index == ~0ul) { CloneError(quadEncoders); return; }
82  encoder->encoder.reset(tEncoder::create(encoder->index, status));
83  encoder->encoder->writeConfig_ASource_Module(routingModuleA, status);
84  encoder->encoder->writeConfig_ASource_Channel(routingChannelA, status);
85  encoder->encoder->writeConfig_ASource_AnalogTrigger(routingAnalogTriggerA,
86  status);
87  encoder->encoder->writeConfig_BSource_Module(routingModuleB, status);
88  encoder->encoder->writeConfig_BSource_Channel(routingChannelB, status);
89  encoder->encoder->writeConfig_BSource_AnalogTrigger(routingAnalogTriggerB,
90  status);
91  encoder->encoder->strobeReset(status);
92  encoder->encoder->writeConfig_Reverse(reverseDirection, status);
93  encoder->encoder->writeTimerConfig_AverageSize(4, status);
94 
95  return handle;
96 }

123 double HAL_GetFPGAEncoderPeriod(HAL_FPGAEncoderHandle fpgaEncoderHandle,
124  int32_t* status) {
125  auto encoder = fpgaEncoderHandles->Get(fpgaEncoderHandle);
126  if (encoder == nullptr) {
127  *status = HAL_HANDLE_ERROR;
128  return 0.0;
129  }
130  tEncoder::tTimerOutput output = encoder->encoder->readTimerOutput(status);
131  double value;
132  if (output.Stalled) {
133  // Return infinity
134  double zero = 0.0;
135  value = 1.0 / zero;
136  } else {
137  // output.Period is a fixed point number that counts by 2 (24 bits, 25
138  // integer bits)
139  value = static_cast<double>(output.Period << 1) /
140  static_cast<double>(output.Count);
141  }
142  double measuredPeriod = value * 2.5e-8;
143  return measuredPeriod / DECODING_SCALING_FACTOR;
144 }
