49 bool gui::PlatformInitRenderer() {
50  glfwMakeContextCurrent(gContext->window);
51 
52  glfwSwapInterval(1); // Enable vsync
53 
54  // Initialize OpenGL loader
55  if (gl3wInit() != 0) {
56  std::fprintf(stderr, "Failed to initialize OpenGL loader!\n");
57  return false;
58  }
59 
60  // Turn on multisampling
61  glEnable(GL_MULTISAMPLE);
62 
63  // Setup Platform/Renderer bindings
64  ImGui_ImplGlfw_InitForOpenGL(gContext->window, true);
65 
66  // Decide GLSL versions
67 #if __APPLE__
68  const char* glsl_version = "#version 150";
69 #else
70  const char* glsl_version = "#version 130";
71 #endif
72  ImGui_ImplOpenGL3_Init(glsl_version);
73 
74  gPlatformValid = true;
75  return true;
76 }

112 ImTextureID gui::CreateTexture(PixelFormat format, int width, int height,
113  const unsigned char* data) {
114  if (!gPlatformValid) {
115  return nullptr;
116  }
117 
118  // Create a OpenGL texture identifier
119  GLuint texture;
120  glGenTextures(1, &texture);
121  glBindTexture(GL_TEXTURE_2D, texture);
122 
123  // Setup filtering parameters for display
124  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
125  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
126 
127  // Upload pixels into texture
128  glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
129  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0,
130  GLPixelFormat(format), GL_UNSIGNED_BYTE, data);
131 
132  return reinterpret_cast<ImTextureID>(static_cast<uintptr_t>(texture));
133 }

135 void gui::UpdateTexture(ImTextureID texture, PixelFormat format, int width,
136  int height, const unsigned char* data) {
137  GLuint glTexture = static_cast<GLuint>(reinterpret_cast<uintptr_t>(texture));
138  if (glTexture == 0) {
139  return;
140  }
141  glBindTexture(GL_TEXTURE_2D, glTexture);
142  glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, width, height, GLPixelFormat(format),
143  GL_UNSIGNED_BYTE, data);
144 }
