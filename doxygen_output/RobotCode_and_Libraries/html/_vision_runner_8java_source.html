80  if (id != null && Thread.currentThread().getId() == id) {

87  private void runOnceInternal() {
88  long frameTime = m_cvSink.grabFrame(m_image);
89  if (frameTime == 0) {
90  // There was an error, report it
91  String error = m_cvSink.getError();
92  CameraServerSharedStore.getCameraServerShared().reportDriverStationError(error);
93  } else {
94  // No errors, process the image
95  m_pipeline.process(m_image);
96  m_listener.copyPipelineOutputs(m_pipeline);
97  }
