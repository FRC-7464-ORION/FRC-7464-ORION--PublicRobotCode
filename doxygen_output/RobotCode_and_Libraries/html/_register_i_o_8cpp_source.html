73 void RegisterIO::Run() {
74  io_provider->Init();
75 
76  /* Initial Device Configuration */
77  SetUpdateRateHz(this->update_rate_hz);
78  GetConfiguration();
79 
80  double update_rate_ms = 1.0/(double)this->update_rate_hz;
81  if ( update_rate_ms > DELAY_OVERHEAD_MILLISECONDS) {
82  update_rate_ms -= DELAY_OVERHEAD_MILLISECONDS;
83  }
84 
85  /* IO Loop */
86  while (!stop) {
87  if ( board_state.update_rate_hz != this->update_rate_hz ) {
88  SetUpdateRateHz(this->update_rate_hz);
89  }
90  GetCurrentData();
91  delayMillis(update_rate_ms);
92  }
93 }

103 bool RegisterIO::GetConfiguration() {
104  bool success = false;
105  int retry_count = 0;
106  while ( retry_count < 3 && !success ) {
107  char config[NAVX_REG_SENSOR_STATUS_H+1] = {0};
108  if ( io_provider->Read(NAVX_REG_WHOAMI, (uint8_t *)config, sizeof(config)) ) {
109  board_id.hw_rev = config[NAVX_REG_HW_REV];
110  board_id.fw_ver_major = config[NAVX_REG_FW_VER_MAJOR];
111  board_id.fw_ver_minor = config[NAVX_REG_FW_VER_MINOR];
112  board_id.type = config[NAVX_REG_WHOAMI];
113  notify_sink->SetBoardID(board_id);
114 
115  board_state.cal_status = config[NAVX_REG_CAL_STATUS];
116  board_state.op_status = config[NAVX_REG_OP_STATUS];
117  board_state.selftest_status = config[NAVX_REG_SELFTEST_STATUS];
118  board_state.sensor_status = IMURegisters::decodeProtocolUint16(config + NAVX_REG_SENSOR_STATUS_L);
119  board_state.gyro_fsr_dps = IMURegisters::decodeProtocolUint16(config + NAVX_REG_GYRO_FSR_DPS_L);
120  board_state.accel_fsr_g = (int16_t)config[NAVX_REG_ACCEL_FSR_G];
121  board_state.update_rate_hz = config[NAVX_REG_UPDATE_RATE_HZ];
122  board_state.capability_flags = IMURegisters::decodeProtocolUint16(config + NAVX_REG_CAPABILITY_FLAGS_L);
123  bool update_board_status = true;
124  notify_sink->SetBoardState(board_state, update_board_status);
125  success = true;
126  } else {
127  success = false;
128  delayMillis(50);
129  }
130  retry_count++;
131  }
132  return success;
133 }

135 void RegisterIO::GetCurrentData() {
136  uint8_t first_address = NAVX_REG_UPDATE_RATE_HZ;
137  bool displacement_registers = board_capabilities->IsDisplacementSupported();
138  uint8_t buffer_len;
139  char curr_data[NAVX_REG_LAST + 1];
140  /* If firmware supports displacement data, acquire it - otherwise implement */
141  /* similar (but potentially less accurate) calculations on this processor. */
142  if ( displacement_registers ) {
143  buffer_len = NAVX_REG_LAST + 1 - first_address;
144  } else {
145  buffer_len = NAVX_REG_QUAT_OFFSET_Z_H + 1 - first_address;
146  }
147  if ( io_provider->Read(first_address,(uint8_t *)curr_data, buffer_len) ) {
148  long sensor_timestamp = IMURegisters::decodeProtocolUint32(curr_data + NAVX_REG_TIMESTAMP_L_L-first_address);
149  if (!connect_reported) {
150  notify_sink->ConnectDetected();
151  connect_reported = true;
152  disconnect_reported = false;
153  }
154  if ( sensor_timestamp == last_sensor_timestamp ) {
155  return;
156  }
157  last_sensor_timestamp = sensor_timestamp;
158  ahrspos_update.op_status = curr_data[NAVX_REG_OP_STATUS - first_address];
159  ahrspos_update.selftest_status = curr_data[NAVX_REG_SELFTEST_STATUS - first_address];
160  ahrspos_update.cal_status = curr_data[NAVX_REG_CAL_STATUS - first_address];
161  ahrspos_update.sensor_status = curr_data[NAVX_REG_SENSOR_STATUS_L - first_address];
162  ahrspos_update.yaw = IMURegisters::decodeProtocolSignedHundredthsFloat(curr_data + NAVX_REG_YAW_L-first_address);
163  ahrspos_update.pitch = IMURegisters::decodeProtocolSignedHundredthsFloat(curr_data + NAVX_REG_PITCH_L-first_address);
164  ahrspos_update.roll = IMURegisters::decodeProtocolSignedHundredthsFloat(curr_data + NAVX_REG_ROLL_L-first_address);
165  ahrspos_update.compass_heading = IMURegisters::decodeProtocolUnsignedHundredthsFloat(curr_data + NAVX_REG_HEADING_L-first_address);
166  ahrspos_update.mpu_temp = IMURegisters::decodeProtocolSignedHundredthsFloat(curr_data + NAVX_REG_MPU_TEMP_C_L - first_address);
167  ahrspos_update.linear_accel_x = IMURegisters::decodeProtocolSignedThousandthsFloat(curr_data + NAVX_REG_LINEAR_ACC_X_L-first_address);
168  ahrspos_update.linear_accel_y = IMURegisters::decodeProtocolSignedThousandthsFloat(curr_data + NAVX_REG_LINEAR_ACC_Y_L-first_address);
169  ahrspos_update.linear_accel_z = IMURegisters::decodeProtocolSignedThousandthsFloat(curr_data + NAVX_REG_LINEAR_ACC_Z_L-first_address);
170  ahrspos_update.altitude = IMURegisters::decodeProtocol1616Float(curr_data + NAVX_REG_ALTITUDE_D_L - first_address);
171  ahrspos_update.barometric_pressure = IMURegisters::decodeProtocol1616Float(curr_data + NAVX_REG_PRESSURE_DL - first_address);
172  ahrspos_update.fused_heading = IMURegisters::decodeProtocolUnsignedHundredthsFloat(curr_data + NAVX_REG_FUSED_HEADING_L-first_address);
173  ahrspos_update.quat_w = ((float)IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_QUAT_W_L-first_address)) / 32768.0f;
174  ahrspos_update.quat_x = ((float)IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_QUAT_X_L-first_address)) / 32768.0f;
175  ahrspos_update.quat_y = ((float)IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_QUAT_Y_L-first_address)) / 32768.0f;
176  ahrspos_update.quat_z = ((float)IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_QUAT_Z_L-first_address)) / 32768.0f;
177  if ( displacement_registers ) {
178  ahrspos_update.vel_x = IMURegisters::decodeProtocol1616Float(curr_data + NAVX_REG_VEL_X_I_L-first_address);
179  ahrspos_update.vel_y = IMURegisters::decodeProtocol1616Float(curr_data + NAVX_REG_VEL_Y_I_L-first_address);
180  ahrspos_update.vel_z = IMURegisters::decodeProtocol1616Float(curr_data + NAVX_REG_VEL_Z_I_L-first_address);
181  ahrspos_update.disp_x = IMURegisters::decodeProtocol1616Float(curr_data + NAVX_REG_DISP_X_I_L-first_address);
182  ahrspos_update.disp_y = IMURegisters::decodeProtocol1616Float(curr_data + NAVX_REG_DISP_Y_I_L-first_address);
183  ahrspos_update.disp_z = IMURegisters::decodeProtocol1616Float(curr_data + NAVX_REG_DISP_Z_I_L-first_address);
184  notify_sink->SetAHRSPosData(ahrspos_update, sensor_timestamp);
185  } else {
186  ahrs_update.op_status = ahrspos_update.op_status;
187  ahrs_update.selftest_status = ahrspos_update.selftest_status;
188  ahrs_update.cal_status = ahrspos_update.cal_status;
189  ahrs_update.sensor_status = ahrspos_update.sensor_status;
190  ahrs_update.yaw = ahrspos_update.yaw;
191  ahrs_update.pitch = ahrspos_update.pitch;
192  ahrs_update.roll = ahrspos_update.roll;
193  ahrs_update.compass_heading = ahrspos_update.compass_heading;
194  ahrs_update.mpu_temp = ahrspos_update.mpu_temp;
195  ahrs_update.linear_accel_x = ahrspos_update.linear_accel_x;
196  ahrs_update.linear_accel_y = ahrspos_update.linear_accel_y;
197  ahrs_update.linear_accel_z = ahrspos_update.linear_accel_z;
198  ahrs_update.altitude = ahrspos_update.altitude;
199  ahrs_update.barometric_pressure = ahrspos_update.barometric_pressure;
200  ahrs_update.fused_heading = ahrspos_update.fused_heading;
201  notify_sink->SetAHRSData( ahrs_update, sensor_timestamp );
202  }
203 
204  board_state.update_rate_hz = curr_data[NAVX_REG_UPDATE_RATE_HZ-first_address];
205  board_state.gyro_fsr_dps = IMURegisters::decodeProtocolUint16(curr_data + NAVX_REG_GYRO_FSR_DPS_L-first_address);
206  board_state.accel_fsr_g = (int16_t)curr_data[NAVX_REG_ACCEL_FSR_G-first_address];
207  board_state.capability_flags= IMURegisters::decodeProtocolUint16(curr_data + NAVX_REG_CAPABILITY_FLAGS_L-first_address);
208  bool update_board_status = false;
209  notify_sink->SetBoardState(board_state, update_board_status); // Board status already updated previously above
210 
211  raw_data_update.gyro_x = IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_GYRO_X_L-first_address);
212  raw_data_update.gyro_y = IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_GYRO_Y_L-first_address);
213  raw_data_update.gyro_z = IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_GYRO_Z_L-first_address);
214  raw_data_update.accel_x = IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_ACC_X_L-first_address);
215  raw_data_update.accel_y = IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_ACC_Y_L-first_address);
216  raw_data_update.accel_z = IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_ACC_Z_L-first_address);
217  raw_data_update.mag_x = IMURegisters::decodeProtocolInt16(curr_data + NAVX_REG_MAG_X_L-first_address);
218  raw_data_update.temp_c = ahrspos_update.mpu_temp;
219  notify_sink->SetRawData(raw_data_update, sensor_timestamp);
220 
221  this->last_update_time = Timer::GetFPGATimestamp();
222  byte_count += buffer_len;
223  update_count++;
224  }
225  if (connect_reported && !disconnect_reported && !this->IsConnected()) {
226  notify_sink->DisconnectDetected();
227  disconnect_reported = true;
228  connect_reported = false;
229  }
230 }
