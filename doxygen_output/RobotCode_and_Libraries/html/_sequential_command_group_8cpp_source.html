58 void SequentialCommandGroup::AddCommands(
59  std::vector<std::unique_ptr<Command>>&& commands) {
60  if (!RequireUngrouped(commands)) {
61  return;
62  }
63 
64  if (m_currentCommandIndex != invalid_index) {
65  throw FRC_MakeError(frc::err::CommandIllegalUse, "{}",
66  "Commands cannot be added to a CommandGroup "
67  "while the group is running");
68  }
69 
70  for (auto&& command : commands) {
71  command->SetGrouped(true);
72  AddRequirements(command->GetRequirements());
73  m_runWhenDisabled &= command->RunsWhenDisabled();
74  m_commands.emplace_back(std::move(command));
75  }
76 }
