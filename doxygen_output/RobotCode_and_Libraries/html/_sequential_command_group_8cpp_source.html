57 void SequentialCommandGroup::AddCommands(
58  std::vector<std::unique_ptr<Command>>&& commands) {

60  return;

62

64  wpi_setWPIErrorWithContext(CommandIllegalUse,
65  "Commands cannot be added to a CommandGroup "
66  "while the group is running");
67  }
68 
69  for (auto&& command : commands) {
