65 static void EntryGroupCallback(AvahiEntryGroup* group,
66  AvahiEntryGroupState state, void* userdata) {

79 static void RegisterService(AvahiClient* client,
80  MulticastServiceAnnouncer::Impl* impl) {
81  if (impl->group == nullptr) {
82  impl->group = impl->table.entry_group_new(client, EntryGroupCallback, impl);
83  }
84 
85  while (true) {
86  if (impl->table.entry_group_is_empty(impl->group)) {
87  auto ret = impl->table.entry_group_add_service_strlst(
88  impl->group, AVAHI_IF_UNSPEC, AVAHI_PROTO_UNSPEC,
89  AVAHI_PUBLISH_USE_MULTICAST, impl->serviceName.c_str(),
90  impl->serviceType.c_str(), "local", nullptr, impl->port,
91  impl->stringList);
92  if (ret == AVAHI_ERR_COLLISION) {
93  // Local collision
94  char* newName =
95  impl->table.alternative_service_name(impl->serviceName.c_str());
96  impl->serviceName = newName;
97  impl->table.free(newName);
98  continue;
99  } else if (ret != AVAHI_OK) {
100  break;
101  }
102  impl->table.entry_group_commit(impl->group);
103  break;
104  }
105  }
106 }

108 static void ClientCallback(AvahiClient* client, AvahiClientState state,
109  void* userdata) {
110  MulticastServiceAnnouncer::Impl* impl =
111  reinterpret_cast<MulticastServiceAnnouncer::Impl*>(userdata);
112 
113  if (state == AVAHI_CLIENT_S_RUNNING) {
114  RegisterService(client, impl);
115  } else if (state == AVAHI_CLIENT_S_COLLISION ||
116  state == AVAHI_CLIENT_S_REGISTERING) {
117  if (impl->group) {
118  impl->table.entry_group_reset(impl->group);
119  }
120  }
121 }
