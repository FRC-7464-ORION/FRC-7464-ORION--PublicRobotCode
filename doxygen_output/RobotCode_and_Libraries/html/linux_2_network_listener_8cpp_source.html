69 void NetworkListener::Impl::Thread::Main() {
70  // Create event socket so we can be shut down
71  m_command_fd = ::eventfd(0, 0);
72  if (m_command_fd < 0) {
73  ERROR("NetworkListener: could not create eventfd: {}",
74  std::strerror(errno));
75  return;
76  }
77 
78  // Create netlink socket
79  int sd = ::socket(AF_NETLINK, SOCK_RAW, NETLINK_ROUTE);
80  if (sd < 0) {
81  ERROR("NetworkListener: could not create socket: {}", std::strerror(errno));
82  ::close(m_command_fd);
83  m_command_fd = -1;
84  return;
85  }
86 
87  // Bind to netlink socket
88  struct sockaddr_nl addr;
89  std::memset(&addr, 0, sizeof(addr));
90  addr.nl_family = AF_NETLINK;
91  addr.nl_groups = RTMGRP_LINK | RTMGRP_IPV4_IFADDR;
92  if (bind(sd, reinterpret_cast<struct sockaddr*>(&addr), sizeof(addr)) < 0) {
93  ERROR("NetworkListener: could not create socket: {}", std::strerror(errno));
94  ::close(sd);
95  ::close(m_command_fd);
96  m_command_fd = -1;
97  return;
98  }
99 
100  char buf[4096];
101 
102  while (m_active) {
103  // select timeout
104  struct timeval tv;
105  tv.tv_sec = 10;
106  tv.tv_usec = 0;
107 
108  // select on applicable read descriptors
109  fd_set readfds;
110  FD_ZERO(&readfds);
111  FD_SET(m_command_fd, &readfds);
112  FD_SET(sd, &readfds);
113  int nfds = std::max(m_command_fd, sd) + 1;
114 
115  if (::select(nfds, &readfds, nullptr, nullptr, &tv) < 0) {
116  ERROR("NetworkListener: select(): {}", std::strerror(errno));
117  break; // XXX: is this the right thing to do here?
118  }
119 
120  // Double-check to see if we're shutting down
121  if (!m_active) {
122  break;
123  }
124 
125  if (!FD_ISSET(sd, &readfds)) {
126  continue;
127  }
128 
129  std::memset(&addr, 0, sizeof(addr));
130  struct iovec iov = {buf, sizeof(buf)};
131  struct msghdr msg = {&addr, sizeof(addr), &iov, 1, nullptr, 0, 0};
132  int len = ::recvmsg(sd, &msg, 0);
133  if (len < 0) {
134  if (errno == EWOULDBLOCK || errno == EAGAIN) {
135  continue;
136  }
137  ERROR("NetworkListener: could not read netlink: {}",
138  std::strerror(errno));
139  break; // XXX: is this the right thing to do here?
140  }
141  if (len == 0) {
142  continue; // EOF?
143  }
144  unsigned int ulen = static_cast<unsigned int>(len);
145  for (struct nlmsghdr* nh = reinterpret_cast<struct nlmsghdr*>(buf);
146  NLMSG_OK(nh, ulen); nh = NLMSG_NEXT(nh, ulen)) {
147  if (nh->nlmsg_type == NLMSG_DONE) {
148  break;
149  }
150  if (nh->nlmsg_type == RTM_NEWLINK || nh->nlmsg_type == RTM_DELLINK ||
151  nh->nlmsg_type == RTM_NEWADDR || nh->nlmsg_type == RTM_DELADDR) {
152  m_notifier.NotifyNetworkInterfacesChanged();
153  }
154  }
155  }
156  ::close(sd);
157  ::close(m_command_fd);
158  m_command_fd = -1;
159 }
