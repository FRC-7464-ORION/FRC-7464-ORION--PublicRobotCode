52 template <typename ReturnT, typename... ParamTs>
53 class unique_function<ReturnT(ParamTs...)> {
54  static constexpr size_t InlineStorageSize = sizeof(void *) * 4;
55 
56  // MSVC has a bug and ICEs if we give it a particular dependent value
57  // expression as part of the `std::conditional` below. To work around this,
58  // we build that into a template struct's constexpr bool.
59  template <typename T> struct IsSizeLessThanThresholdT {
60  static constexpr bool value = sizeof(T) <= (2 * sizeof(void *));
61  };
62 
63  // Provide a type function to map parameters that won't observe extra copies
64  // or moves and which are small enough to likely pass in register to values
65  // and all other types to l-value reference types. We use this to compute the
66  // types used in our erased call utility to minimize copies and moves unless
67  // doing so would force things unnecessarily into memory.
68  //
69  // The heuristic used is related to common ABI register passing conventions.
70  // It doesn't have to be exact though, and in one way it is more strict
71  // because we want to still be able to observe either moves *or* copies.
72  template <typename T>
73  using AdjustedParamT = typename std::conditional<
74  !std::is_reference<T>::value &&
75  std::is_trivially_copy_constructible<T>::value &&
76  std::is_trivially_move_constructible<T>::value &&
77  IsSizeLessThanThresholdT<T>::value,
78  T, T &>::type;
79 
80  // The type of the erased function pointer we use as a callback to dispatch to
81  // the stored callable when it is trivial to move and destroy.
82  using CallPtrT = ReturnT (*)(void *CallableAddr,
83  AdjustedParamT<ParamTs>... Params);
84  using MovePtrT = void (*)(void *LHSCallableAddr, void *RHSCallableAddr);
85  using DestroyPtrT = void (*)(void *CallableAddr);
86 
89  struct alignas(8) TrivialCallback {
90  CallPtrT CallPtr;
91  };
92 
95  struct alignas(8) NonTrivialCallbacks {
96  CallPtrT CallPtr;
97  MovePtrT MovePtr;
98  DestroyPtrT DestroyPtr;
99  };
100 
101  // Create a pointer union between either a pointer to a static trivial call
102  // pointer in a struct or a pointer to a static struct of the call, move, and
103  // destroy pointers.
104  using CallbackPointerUnionT =
105  PointerUnion<TrivialCallback *, NonTrivialCallbacks *>;
106 
107  // The main storage buffer. This will either have a pointer to out-of-line
108  // storage or an inline buffer storing the callable.
109  union StorageUnionT {
110  // For out-of-line storage we keep a pointer to the underlying storage and
111  // the size. This is enough to deallocate the memory.
112  struct OutOfLineStorageT {
113  void *StoragePtr;
114  size_t Size;
115  size_t Alignment;
116  } OutOfLineStorage;
117  static_assert(
118  sizeof(OutOfLineStorageT) <= InlineStorageSize,
119  "Should always use all of the out-of-line storage for inline storage!");
120 
121  // For in-line storage, we just provide an aligned character buffer. We
122  // provide four pointers worth of storage here.
123  typename std::aligned_storage<InlineStorageSize, alignof(void *)>::type
124  InlineStorage;
125  } StorageUnion;
126 
127  // A compressed pointer to either our dispatching callback or our table of
128  // dispatching callbacks and the flag for whether the callable itself is
129  // stored inline or not.
130  PointerIntPair<CallbackPointerUnionT, 1, bool> CallbackAndInlineFlag;
131 
132  bool isInlineStorage() const { return CallbackAndInlineFlag.getInt(); }
133 
134  bool isTrivialCallback() const {
135  return CallbackAndInlineFlag.getPointer().template is<TrivialCallback *>();
136  }
137 
138  CallPtrT getTrivialCallback() const {
139  return CallbackAndInlineFlag.getPointer().template get<TrivialCallback *>()->CallPtr;
140  }

143  return CallbackAndInlineFlag.getPointer()

145  }

163  template <typename CallableT>
164  static ReturnT CallImpl(void *CallableAddr, AdjustedParamT<ParamTs>... Params) {
165  return (*reinterpret_cast<CallableT *>(CallableAddr))(
166  std::forward<ParamTs>(Params)...);
167  }

169  template <typename CallableT>
170  static void MoveImpl(void *LHSCallableAddr, void *RHSCallableAddr) noexcept {
171  new (LHSCallableAddr)
172  CallableT(std::move(*reinterpret_cast<CallableT *>(RHSCallableAddr)));
173  }
174 
175  template <typename CallableT>
176  static void DestroyImpl(void *CallableAddr) noexcept {
177  reinterpret_cast<CallableT *>(CallableAddr)->~CallableT();
178  }

180 public:
181  unique_function() = default;
182  unique_function(std::nullptr_t /*null_callable*/) {}
183 
184  ~unique_function() {
185  if (!CallbackAndInlineFlag.getPointer())
186  return;
187 
188  // Cache this value so we don't re-check it after type-erased operations.
189  bool IsInlineStorage = isInlineStorage();
190 
191  if (!isTrivialCallback())
192  getNonTrivialCallbacks()->DestroyPtr(
193  IsInlineStorage ? getInlineStorage() : getOutOfLineStorage());
194 
195  if (!IsInlineStorage)
196  deallocate_buffer(getOutOfLineStorage(), getOutOfLineStorageSize(),
197  getOutOfLineStorageAlignment());
198  }

200  unique_function(unique_function &&RHS) noexcept {
201  // Copy the callback and inline flag.
202  CallbackAndInlineFlag = RHS.CallbackAndInlineFlag;
203 
204  // If the RHS is empty, just copying the above is sufficient.
205  if (!RHS)
206  return;
207 
208  if (!isInlineStorage()) {
209  // The out-of-line case is easiest to move.
210  StorageUnion.OutOfLineStorage = RHS.StorageUnion.OutOfLineStorage;
211  } else if (isTrivialCallback()) {
212  // Move is trivial, just memcpy the bytes across.
213  memcpy(getInlineStorage(), RHS.getInlineStorage(), InlineStorageSize);
214  } else {
215  // Non-trivial move, so dispatch to a type-erased implementation.
216  getNonTrivialCallbacks()->MovePtr(getInlineStorage(),
217  RHS.getInlineStorage());
218  }
219 
220  // Clear the old callback and inline flag to get back to as-if-null.
221  RHS.CallbackAndInlineFlag = {};
222 
223 #ifndef NDEBUG
224  // In debug builds, we also scribble across the rest of the storage.
225  memset(RHS.getInlineStorage(), 0xAD, InlineStorageSize);
226 #endif
227  }
228 
229  unique_function &operator=(unique_function &&RHS) noexcept {
230  if (this == &RHS)
231  return *this;
232 
233  // Because we don't try to provide any exception safety guarantees we can
234  // implement move assignment very simply by first destroying the current
235  // object and then move-constructing over top of it.
236  this->~unique_function();
237  new (this) unique_function(std::move(RHS));
238  return *this;
239  }

241  template <typename CallableT>
242  unique_function(CallableT Callable,
243  std::enable_if_t<
244  std::is_invocable_r_v<
245  ReturnT, CallableT, ParamTs...>>* = nullptr) {
246  bool IsInlineStorage = true;
247  void *CallableAddr = getInlineStorage();
248  if (sizeof(CallableT) > InlineStorageSize ||
249  alignof(CallableT) > alignof(decltype(StorageUnion.InlineStorage))) {
250  IsInlineStorage = false;
251  // Allocate out-of-line storage. FIXME: Use an explicit alignment
252  // parameter in C++17 mode.
253  auto Size = sizeof(CallableT);
254  auto Alignment = alignof(CallableT);
255  CallableAddr = allocate_buffer(Size, Alignment);
256  setOutOfLineStorage(CallableAddr, Size, Alignment);
257  }
258 
259  // Now move into the storage.
260  new (CallableAddr) CallableT(std::move(Callable));
261 
262  // See if we can create a trivial callback. We need the callable to be
263  // trivially moved and trivially destroyed so that we don't have to store
264  // type erased callbacks for those operations.
265  //
266  // FIXME: We should use constexpr if here and below to avoid instantiating
267  // the non-trivial static objects when unnecessary. While the linker should
268  // remove them, it is still wasteful.
269  if (std::is_trivially_move_constructible<CallableT>::value &&
270  std::is_trivially_destructible<CallableT>::value) {
271  // We need to create a nicely aligned object. We use a static variable
272  // for this because it is a trivial struct.
273  static TrivialCallback Callback = { &CallImpl<CallableT> };
274 
275  CallbackAndInlineFlag = {&Callback, IsInlineStorage};
276  return;
277  }
278 
279  // Otherwise, we need to point at an object that contains all the different
280  // type erased behaviors needed. Create a static instance of the struct type
281  // here and then use a pointer to that.
282  static NonTrivialCallbacks Callbacks = {
283  &CallImpl<CallableT>, &MoveImpl<CallableT>, &DestroyImpl<CallableT>};
284 
285  CallbackAndInlineFlag = {&Callbacks, IsInlineStorage};
286  }
287 
288  ReturnT operator()(ParamTs... Params) {
289  void *CallableAddr =
290  isInlineStorage() ? getInlineStorage() : getOutOfLineStorage();
291 
292  return (isTrivialCallback()
293  ? getTrivialCallback()
294  : getNonTrivialCallbacks()->CallPtr)(CallableAddr, Params...);
295  }
296 
297  explicit operator bool() const {
298  return (bool)CallbackAndInlineFlag.getPointer();
299  }
300 };
