41 ShuffleboardLayout& ShuffleboardContainer::GetLayout(std::string_view title,
42  std::string_view type) {
43  if (m_layouts.count(title) == 0) {
44  auto layout = std::make_unique<ShuffleboardLayout>(*this, title, type);
45  auto ptr = layout.get();
46  m_components.emplace_back(std::move(layout));
47  m_layouts.insert(std::make_pair(title, ptr));
48  }
49  return *m_layouts[title];
50 }
