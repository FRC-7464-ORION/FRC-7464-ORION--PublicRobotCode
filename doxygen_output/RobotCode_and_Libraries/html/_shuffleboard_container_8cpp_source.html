45 ShuffleboardLayout& ShuffleboardContainer::GetLayout(const wpi::Twine& title,
46  const wpi::Twine& type) {
47  wpi::SmallVector<char, 16> storage;
48  auto titleRef = title.toStringRef(storage);
49  if (m_layouts.count(titleRef) == 0) {
50  auto layout = std::make_unique<ShuffleboardLayout>(*this, type, titleRef);
51  auto ptr = layout.get();
52  m_components.emplace_back(std::move(layout));
53  m_layouts.insert(std::make_pair(titleRef, ptr));
54  }
55  return *m_layouts[titleRef];
56 }
