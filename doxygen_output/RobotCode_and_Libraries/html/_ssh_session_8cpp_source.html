69 void SshSession::Execute(std::string_view cmd) {
70  // Allocate a new channel.
71  ssh_channel channel = ssh_channel_new(m_session);
72  if (!channel) {
73  throw SshException(ssh_get_error(m_session));
74  }
75 
76  // Open the channel.
77  int rc = ssh_channel_open_session(channel);
78  if (rc != SSH_OK) {
79  throw SshException(ssh_get_error(m_session));
80  }
81 
82  // Execute the command.
83  std::string command{cmd};
84  rc = ssh_channel_request_exec(channel, command.c_str());
85  if (rc != SSH_OK) {
86  ssh_channel_close(channel);
87  ssh_channel_free(channel);
88  throw SshException(ssh_get_error(m_session));
89  }
90  INFO("{}", cmd);
91 
92  // Log output.
93  char buf[512];
94  int read = ssh_channel_read(channel, buf, sizeof(buf), 0);
95  if (read != 0) {
96  INFO("{}", cmd);
97  }
98 
99  // Close and free channel.
100  ssh_channel_close(channel);
101  ssh_channel_free(channel);
102 }

104 void SshSession::Put(std::string_view path, std::string_view contents) {
105  // Allocate the SFTP session.
106  sftp_session sftp = sftp_new(m_session);
107  if (!sftp) {
108  throw SshException(ssh_get_error(m_session));
109  }
110 
111  // Initialize.
112  int rc = sftp_init(sftp);
113  if (rc != SSH_OK) {
114  sftp_free(sftp);
115  throw SshException(ssh_get_error(m_session));
116  }
117 
118  // Copy.
119  sftp_file file =
120  sftp_open(sftp, path.data(), O_WRONLY | O_CREAT | O_TRUNC, S_IFMT);
121  if (!file) {
122  sftp_free(sftp);
123  throw SshException(ssh_get_error(m_session));
124  }
125 
126  // Send 150K at a time.
127  static constexpr size_t kChunkSize = 150000;
128  for (size_t i = 0; i < contents.size(); i += kChunkSize) {
129  size_t len = (std::min)(kChunkSize, contents.size() - i);
130  size_t written = sftp_write(file, contents.data() + i, len);
131  if (written != len) {
132  sftp_close(file);
133  sftp_free(sftp);
134  throw SshException(ssh_get_error(m_session));
135  }
136  }
137 
138  INFO("[SFTP] Deployed {}!", path);
139 
140  // Close file, free memory.
141  sftp_close(file);
142  sftp_free(sftp);
143 }
