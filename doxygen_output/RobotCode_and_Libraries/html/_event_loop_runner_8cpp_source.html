54 void EventLoopRunner::Stop() {
55  ExecAsync([](uv::Loop& loop) {
56  // close all handles; this will (eventually) stop the loop
57  loop.Walk([](uv::Handle& h) {
58  h.SetLoopClosing(true);
59  h.Close();
60  });
61  });
62  m_owner.Join();
63 }
