72 void HALSimWS::Start() {
73  m_tcp_client->SetNoDelay(true);
74 
75  // Hook up TCP client events
76  m_tcp_client->error.connect(
77  [this, socket = m_tcp_client.get()](wpi::uv::Error err) {
78  if (m_tcp_connected) {
79  m_tcp_connected = false;
80  m_connect_attempts = 0;
81  return;
82  }
83 
84  // If we weren't previously connected, attempt a reconnection
85  m_connect_timer->Start(uv::Timer::Time(kTcpConnectAttemptTimeout));
86  });
87 
88  m_tcp_client->closed.connect([]() { std::puts("TCP connection closed"); });
89 
90  // Set up the connection timer
91  std::puts("HALSimWS Initialized");
92  fmt::print("Will attempt to connect to ws://{}:{}{}\n", m_host, m_port,
93  m_uri);
94 
95  // Set up the timer to attempt connection
96  m_connect_timer->timeout.connect([this] { AttemptConnect(); });
97 
98  // Run the initial connect immediately
99  m_connect_timer->Start(uv::Timer::Time(0));
100  m_connect_timer->Unreference();
101 }
