64 static bool CalcParity(int v) {
65  bool parity = false;
66  while (v != 0) {
67  parity = !parity;
68  v = v & (v - 1);
69  }
70  return parity;
71 }

73 static inline int BytesToIntBE(uint8_t* buf) {
74  int result = static_cast<int>(buf[0]) << 24;
75  result |= static_cast<int>(buf[1]) << 16;
76  result |= static_cast<int>(buf[2]) << 8;
77  result |= static_cast<int>(buf[3]);

81 uint16_t ADXRS450_Gyro::ReadRegister(int reg) {
82  int cmd = 0x80000000 | static_cast<int>(reg) << 17;
83  if (!CalcParity(cmd)) {
84  cmd |= 1u;
85  }
86 
87  // big endian
88  uint8_t buf[4] = {static_cast<uint8_t>((cmd >> 24) & 0xff),
89  static_cast<uint8_t>((cmd >> 16) & 0xff),
90  static_cast<uint8_t>((cmd >> 8) & 0xff),
91  static_cast<uint8_t>(cmd & 0xff)};
92 
93  m_spi.Write(buf, 4);
94  m_spi.Read(false, buf, 4);
95  if ((buf[0] & 0xe0) == 0) {
96  return 0; // error, return 0
97  }
98  return static_cast<uint16_t>((BytesToIntBE(buf) >> 5) & 0xffff);
99 }
