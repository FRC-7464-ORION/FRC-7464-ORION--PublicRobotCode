65 static bool CalcParity(int v) {
66  bool parity = false;
67  while (v != 0) {
68  parity = !parity;
69  v = v & (v - 1);
70  }
71  return parity;
72 }

74 static inline int BytesToIntBE(uint8_t* buf) {
75  int result = static_cast<int>(buf[0]) << 24;
76  result |= static_cast<int>(buf[1]) << 16;
77  result |= static_cast<int>(buf[2]) << 8;
78  result |= static_cast<int>(buf[3]);

82 uint16_t ADXRS450_Gyro::ReadRegister(int reg) {
83  int cmd = 0x80000000 | static_cast<int>(reg) << 17;
84  if (!CalcParity(cmd)) cmd |= 1u;
85 
86  // big endian
87  uint8_t buf[4] = {static_cast<uint8_t>((cmd >> 24) & 0xff),
88  static_cast<uint8_t>((cmd >> 16) & 0xff),
89  static_cast<uint8_t>((cmd >> 8) & 0xff),
90  static_cast<uint8_t>(cmd & 0xff)};
91 
92  m_spi.Write(buf, 4);
93  m_spi.Read(false, buf, 4);
94  if ((buf[0] & 0xe0) == 0) return 0; // error, return 0
95  return static_cast<uint16_t>((BytesToIntBE(buf) >> 5) & 0xffff);
96 }
