75 int TCPAcceptor::start() {
76  if (m_listening) {
77  return 0;
78  }
79 
80  m_lsd = socket(PF_INET, SOCK_STREAM, 0);
81  if (m_lsd < 0) {
82  WPI_ERROR(m_logger, "{}", "could not create socket");
83  return -1;
84  }
85  struct sockaddr_in address;
86 
87  std::memset(&address, 0, sizeof(address));
88  address.sin_family = PF_INET;
89  if (m_address.size() > 0) {
90 #ifdef _WIN32
91  SmallString<128> addr_copy(m_address);
92  addr_copy.push_back('\0');
93  int res = InetPton(PF_INET, addr_copy.data(), &(address.sin_addr));
94 #else
95  int res = inet_pton(PF_INET, m_address.c_str(), &(address.sin_addr));
96 #endif
97  if (res != 1) {
98  WPI_ERROR(m_logger, "could not resolve {} address", m_address);
99  return -1;
100  }
101  } else {
102  address.sin_addr.s_addr = INADDR_ANY;
103  }
104  address.sin_port = htons(m_port);
105 
106 #ifdef _WIN32
107  int optval = 1;
108  setsockopt(m_lsd, SOL_SOCKET, SO_EXCLUSIVEADDRUSE,
109  reinterpret_cast<char*>(&optval), sizeof optval);
110 #else
111  int optval = 1;
112  setsockopt(m_lsd, SOL_SOCKET, SO_REUSEADDR, reinterpret_cast<char*>(&optval),
113  sizeof optval);
114 #endif
115 
116  int result = bind(m_lsd, reinterpret_cast<struct sockaddr*>(&address),
117  sizeof(address));
118  if (result != 0) {
119  WPI_ERROR(m_logger, "bind() to port {} failed: {}", m_port,
120  SocketStrerror());
121  return result;
122  }
123 
124  result = listen(m_lsd, 5);
125  if (result != 0) {
126  WPI_ERROR(m_logger, "listen() on port {} failed: {}", m_port,
127  SocketStrerror());
128  return result;
129  }
130  m_listening = true;
131  return result;
132 }

134 void TCPAcceptor::shutdown() {
135  m_shutdown = true;
136 #ifdef _WIN32
137  ::shutdown(m_lsd, SD_BOTH);
138 
139  // this is ugly, but the easiest way to do this
140  // force wakeup of accept() with a non-blocking connect to ourselves
141  struct sockaddr_in address;
142 
143  std::memset(&address, 0, sizeof(address));
144  address.sin_family = PF_INET;
145  SmallString<128> addr_copy;
146  if (m_address.size() > 0)
147  addr_copy = m_address;
148  else
149  addr_copy = "127.0.0.1";
150  addr_copy.push_back('\0');
151  int size = sizeof(address);
152  if (WSAStringToAddress(addr_copy.data(), PF_INET, nullptr,
153  (struct sockaddr*)&address, &size) != 0)
154  return;
155  address.sin_port = htons(m_port);
156 
157  int result = -1, sd = socket(AF_INET, SOCK_STREAM, 0);
158  if (sd < 0)
159  return;
160 
161  // Set socket to non-blocking
162  u_long mode = 1;
163  ioctlsocket(sd, FIONBIO, &mode);
164 
165  // Try to connect
166  ::connect(sd, (struct sockaddr*)&address, sizeof(address));
167 
168  // Close
169  ::closesocket(sd);
170 
171 #else
172  ::shutdown(m_lsd, SHUT_RDWR);
173  int nullfd = ::open("/dev/null", O_RDONLY);
174  if (nullfd >= 0) {
175  ::dup2(nullfd, m_lsd);
176  ::close(nullfd);
177  }
178 #endif
179 }

181 std::unique_ptr<NetworkStream> TCPAcceptor::accept() {
182  if (!m_listening || m_shutdown) {
183  return nullptr;
184  }
185 
186  struct sockaddr_in address;
187 #ifdef _WIN32
188  int len = sizeof(address);
189 #else
190  socklen_t len = sizeof(address);
191 #endif
192  std::memset(&address, 0, sizeof(address));
193  int sd = ::accept(m_lsd, reinterpret_cast<struct sockaddr*>(&address), &len);
194  if (sd < 0) {
195  if (!m_shutdown) {
196  WPI_ERROR(m_logger, "accept() on port {} failed: {}", m_port,
197  SocketStrerror());
198  }
199  return nullptr;
200  }
201  if (m_shutdown) {
202 #ifdef _WIN32
203  closesocket(sd);
204 #else
205  close(sd);
206 #endif
207  return nullptr;
208  }
209  return std::unique_ptr<NetworkStream>(new TCPStream(sd, &address));
210 }
