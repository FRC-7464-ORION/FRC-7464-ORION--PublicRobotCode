9 QuinticHermiteSpline::QuinticHermiteSpline(
10  wpi::array<double, 3> xInitialControlVector,
11  wpi::array<double, 3> xFinalControlVector,
12  wpi::array<double, 3> yInitialControlVector,
13  wpi::array<double, 3> yFinalControlVector) {
14  const auto hermite = MakeHermiteBasis();
15  const auto x =
16  ControlVectorFromArrays(xInitialControlVector, xFinalControlVector);
17  const auto y =
18  ControlVectorFromArrays(yInitialControlVector, yFinalControlVector);
19 
20  // Populate first two rows with coefficients.
21  m_coefficients.template block<1, 6>(0, 0) = (hermite * x).transpose();
22  m_coefficients.template block<1, 6>(1, 0) = (hermite * y).transpose();
23 
24  // Populate Row 2 and Row 3 with the derivatives of the equations above.
25  // Then populate row 4 and 5 with the second derivatives.
26  for (int i = 0; i < 6; i++) {
27  // Here, we are multiplying by (5 - i) to manually take the derivative. The
28  // power of the term in index 0 is 5, index 1 is 4 and so on. To find the
29  // coefficient of the derivative, we can use the power rule and multiply
30  // the existing coefficient by its power.
31  m_coefficients.template block<2, 1>(2, i) =
32  m_coefficients.template block<2, 1>(0, i) * (5 - i);
33  }
34  for (int i = 0; i < 5; i++) {
35  // Here, we are multiplying by (4 - i) to manually take the derivative. The
36  // power of the term in index 0 is 4, index 1 is 3 and so on. To find the
37  // coefficient of the derivative, we can use the power rule and multiply
38  // the existing coefficient by its power.
39  m_coefficients.template block<2, 1>(4, i) =
40  m_coefficients.template block<2, 1>(2, i) * (4 - i);
41  }
42 }
