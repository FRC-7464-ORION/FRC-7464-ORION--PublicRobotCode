62 ColorSensorV3::ColorSensorV3(frc::I2C::Port port)
63  : m_i2c(port, kAddress),
64  m_simDevice("REV Color Sensor V3", static_cast<int>(port), kAddress) {
65  if (m_simDevice) {
66  m_simR = m_simDevice.CreateDouble("Red", false, 0.0);
67  m_simG = m_simDevice.CreateDouble("Green", false, 0.0);
68  m_simB = m_simDevice.CreateDouble("Blue", false, 0.0);
69  m_simIR = m_simDevice.CreateDouble("IR", false, 0.0);
70  m_simProx = m_simDevice.CreateDouble("Proximity", false, 0.0);
71  return;
72  }
73 
74  // Only report errors if this is not a simulation
75  if (!CheckDeviceID(!m_simDevice)) return;
76 
77  InitializeDevice();
78 
79  // Clear the reset flag
80  HasReset();
81 }

90 frc::Color ColorSensorV3::GetColor() {
91  RawColor color = GetRawColor();

93  double g = static_cast<double>(color.green);
94  double b = static_cast<double>(color.blue);
95  double mag = r + g + b;
96  return frc::Color(r / mag, g / mag, b / mag);

102 ColorSensorV3::RawColor ColorSensorV3::GetRawColor() {
103  uint8_t raw[12];
104 
105  if (m_simDevice) {
106  return ColorSensorV3::RawColor(static_cast<uint32_t>(m_simR.Get()),
107  static_cast<uint32_t>(m_simG.Get()),
108  static_cast<uint32_t>(m_simB.Get()),
109  static_cast<uint32_t>(m_simIR.Get()));
110  }
111 
112  if (!Read(Register::kDataInfrared, 12, raw)) {
113  return ColorSensorV3::RawColor(To20Bit(&raw[9]), To20Bit(&raw[3]),
114  To20Bit(&raw[6]), To20Bit(&raw[0]));
115  }
116 
117  return ColorSensorV3::RawColor(0, 0, 0, 0);
118 }

120 rev::CIEColor ColorSensorV3::GetCIEColor() {
121  RawColor raw = GetRawColor();
122  return rev::CIEColor(
123  Cmatrix[0] * raw.red + Cmatrix[1] * raw.green + Cmatrix[2] * raw.blue,
124  Cmatrix[3] * raw.red + Cmatrix[4] * raw.green + Cmatrix[5] * raw.blue,
125  Cmatrix[6] * raw.red + Cmatrix[7] * raw.green + Cmatrix[8] * raw.blue);
126 }

139 bool ColorSensorV3::CheckDeviceID(bool reportErrors) {
140  uint8_t partID = 0;
141  if (Read(Register::kPartID, 1, &partID)) {
142  if (reportErrors) {
143  FRC_ReportError(frc::err::Error, "{}",
144  "Could not find REV color sensor");
145  }
146  return false;
147  }
148 
149  if (partID != kExpectedPartID) {
150  if (reportErrors) {
151  FRC_ReportError(
152  frc::err::Error, "{}",
153  "Unknown device found with same I2C address as REV color "
154  "sensor");
155  }
156  return false;
157  }
158 
159  return true;
160 }

168 void ColorSensorV3::InitializeDevice() {
169  Write(Register::kMainCtrl,
170  static_cast<uint8_t>(MainCtrlFields::kRGBMode) |
171  static_cast<uint8_t>(MainCtrlFields::kLightSensorEnable) |
172  static_cast<uint8_t>(MainCtrlFields::kProximitySensorEnable));
173 
174  ConfigureProximitySensorLED(kDefaultPulseFreq, kDefaultLEDCurrent,
175  kDefaultPulses);
176  ConfigureProximitySensor(kDefaultProxRes, kDefaultProxRate);
177  ConfigureColorSensor(kDefaultColorRes, kDefaultColorRate);
178  SetGain(kDefaultGain);
179 }
