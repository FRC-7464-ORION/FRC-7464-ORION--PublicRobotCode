52 template <class Robot>
53 int StartRobot() {
54  int halInit = RunHALInitialization();
55  if (halInit != 0) {
56  return halInit;
57  }
58 
59  static wpi::mutex m;
60  static wpi::condition_variable cv;
61  static Robot* robot = nullptr;
62  static bool exited = false;
63 
64  if (HAL_HasMain()) {
65  std::thread thr([] {
66  try {
67  impl::RunRobot<Robot>(m, &robot);
68  } catch (...) {
69  HAL_ExitMain();
70  {
71  std::scoped_lock lock{m};
72  robot = nullptr;
73  exited = true;
74  }
75  cv.notify_all();
76  throw;
77  }
78 
79  HAL_ExitMain();
80  {
81  std::scoped_lock lock{m};
82  robot = nullptr;
83  exited = true;
84  }
85  cv.notify_all();
86  });
87 
88  HAL_RunMain();
89 
90  // signal loop to exit
91  if (robot) {
92  robot->EndCompetition();
93  }
94 
95  // prefer to join, but detach to exit if it doesn't exit in a timely manner
96  using namespace std::chrono_literals;
97  std::unique_lock lock{m};
98  if (cv.wait_for(lock, 1s, [] { return exited; })) {
99  thr.join();
100  } else {
101  thr.detach();
102  }
103  } else {
104  impl::RunRobot<Robot>(m, &robot);
105  }
106 
107  HAL_Shutdown();
108 
109  return 0;
110 }
