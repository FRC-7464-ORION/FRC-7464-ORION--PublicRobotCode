66 units::ampere_t SingleJointedArmSim::GetCurrentDraw() const {

78 Eigen::Vector<double, 2> SingleJointedArmSim::UpdateX(
79  const Eigen::Vector<double, 2>& currentXhat,
80  const Eigen::Vector<double, 1>& u, units::second_t dt) {
81  // Horizontal case:
82  // Torque = F * r = I * alpha
83  // alpha = F * r / I
84  // Since F = mg,
85  // alpha = m * g * r / I
86  // Finally, multiply RHS by cos(theta) to account for the arm angle
87  // This acceleration is added to the linear system dynamics x-dot = Ax + Bu
88  // We therefore find that f(x, u) = Ax + Bu + [[0] [m * g * r / I *
89  // std::cos(theta)]]
90 
91  Eigen::Vector<double, 2> updatedXhat = RKDP(
92  [&](const auto& x, const auto& u) -> Eigen::Vector<double, 2> {
93  Eigen::Vector<double, 2> xdot = m_plant.A() * x + m_plant.B() * u;
94 
95  if (m_simulateGravity) {
96  xdot += Eigen::Vector<double, 2>{
97  0.0, (m_mass * m_r * -9.8 * 3.0 / (m_mass * m_r * m_r) *
98  std::cos(x(0)))
99  .value()};
100  }
101  return xdot;
102  },
103  currentXhat, u, dt);
104 
105  // Check for collisions.
106  if (WouldHitLowerLimit(units::radian_t(updatedXhat(0)))) {
107  return Eigen::Vector<double, 2>{m_minAngle.value(), 0.0};
108  } else if (WouldHitUpperLimit(units::radian_t(updatedXhat(0)))) {
109  return Eigen::Vector<double, 2>{m_maxAngle.value(), 0.0};
110  }
111  return updatedXhat;
112 }
