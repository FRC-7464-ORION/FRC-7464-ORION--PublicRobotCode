62 void ErrorBase::SetErrnoError(const wpi::Twine& contextMessage,
63  wpi::StringRef filename, wpi::StringRef function,
64  int lineNumber) const {
65  wpi::SmallString<128> buf;
66  wpi::raw_svector_ostream err(buf);
67  int errNo = errno;
68  if (errNo == 0) {
69  err << "OK: ";
70  } else {
71  err << std::strerror(errNo) << " (" << wpi::format_hex(errNo, 10, true)
72  << "): ";
73  }
74 
75  // Set the current error information for this object.
76  m_error.Set(-1, err.str() + contextMessage, filename, function, lineNumber,
77  this);
78 
79  // Update the global error if there is not one already set.
80  GlobalErrors::Insert(m_error);
81 }
