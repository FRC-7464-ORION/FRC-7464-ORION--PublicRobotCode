57 static int inet_ntop4(const unsigned char *src, char *dst, size_t size) {
58  static const char fmt[] = "%u.%u.%u.%u";
59  char tmp[UV__INET_ADDRSTRLEN];
60  int l;
61 
62  l = snprintf(tmp, sizeof(tmp), fmt, src[0], src[1], src[2], src[3]);
63  if (l <= 0 || (size_t) l >= size) {
64  return UV_ENOSPC;
65  }
66  uv__strscpy(dst, tmp, size);
67  return 0;
68 }

71 static int inet_ntop6(const unsigned char *src, char *dst, size_t size) {
72  /*
73  * Note that int32_t and int16_t need only be "at least" large enough
74  * to contain a value of the specified size. On some systems, like
75  * Crays, there is no such thing as an integer variable with 16 bits.
76  * Keep this in mind if you think this function should have been coded
77  * to use pointer overlays. All the world's not a VAX.
78  */
79  char tmp[UV__INET6_ADDRSTRLEN], *tp;
80  struct { int base, len; } best, cur;
81  unsigned int words[sizeof(struct in6_addr) / sizeof(uint16_t)];
82  int i;
83 
84  /*
85  * Preprocess:
86  * Copy the input (bytewise) array into a wordwise array.
87  * Find the longest run of 0x00's in src[] for :: shorthanding.
88  */
89  memset(words, '\0', sizeof words);
90  for (i = 0; i < (int) sizeof(struct in6_addr); i++)
91  words[i / 2] |= (src[i] << ((1 - (i % 2)) << 3));
92  best.base = -1;
93  best.len = 0;
94  cur.base = -1;
95  cur.len = 0;
96  for (i = 0; i < (int) ARRAY_SIZE(words); i++) {
97  if (words[i] == 0) {
98  if (cur.base == -1)
99  cur.base = i, cur.len = 1;
100  else
101  cur.len++;
102  } else {
103  if (cur.base != -1) {
104  if (best.base == -1 || cur.len > best.len)
105  best = cur;
106  cur.base = -1;
107  }
108  }
109  }
110  if (cur.base != -1) {
111  if (best.base == -1 || cur.len > best.len)
112  best = cur;
113  }
114  if (best.base != -1 && best.len < 2)
115  best.base = -1;
116 
117  /*
118  * Format the result.
119  */
120  tp = tmp;
121  for (i = 0; i < (int) ARRAY_SIZE(words); i++) {
122  /* Are we inside the best run of 0x00's? */
123  if (best.base != -1 && i >= best.base &&
124  i < (best.base + best.len)) {
125  if (i == best.base)
126  *tp++ = ':';
127  continue;
128  }
129  /* Are we following an initial run of 0x00s or any real hex? */
130  if (i != 0)
131  *tp++ = ':';
132  /* Is this address an encapsulated IPv4? */
133  if (i == 6 && best.base == 0 && (best.len == 6 ||
134  (best.len == 7 && words[7] != 0x0001) ||
135  (best.len == 5 && words[5] == 0xffff))) {
136  int err = inet_ntop4(src+12, tp, sizeof tmp - (tp - tmp));
137  if (err)
138  return err;
139  tp += strlen(tp);
140  break;
141  }
142  tp += sprintf(tp, "%x", words[i]);
143  }
144  /* Was it a trailing run of 0x00's? */
145  if (best.base != -1 && (best.base + best.len) == ARRAY_SIZE(words))
146  *tp++ = ':';
147  *tp++ = '\0';
148  if (UV_E2BIG == uv__strscpy(dst, tmp, size))
149  return UV_ENOSPC;
150  return 0;
151 }

154 int uv_inet_pton(int af, const char* src, void* dst) {
155  if (src == NULL || dst == NULL)
156  return UV_EINVAL;
157 
158  switch (af) {
159  case AF_INET:
160  return (inet_pton4(src, (unsigned char*)dst));
161  case AF_INET6: {
162  int len;
163  char tmp[UV__INET6_ADDRSTRLEN], *s;
164  const char *p;
165  s = (char*) src;
166  p = strchr(src, '%');
167  if (p != NULL) {
168  s = tmp;
169  len = p - src;
170  if (len > UV__INET6_ADDRSTRLEN-1)
171  return UV_EINVAL;
172  memcpy(s, src, len);
173  s[len] = '\0';
174  }
175  return inet_pton6(s, (unsigned char*)dst);
176  }
177  default:
178  return UV_EAFNOSUPPORT;
179  }
180  /* NOTREACHED */
181 }

184 static int inet_pton4(const char *src, unsigned char *dst) {
185  static const char digits[] = "0123456789";
186  int saw_digit, octets, ch;
187  unsigned char tmp[sizeof(struct in_addr)], *tp;
188 
189  saw_digit = 0;
190  octets = 0;
191  *(tp = tmp) = 0;
192  while ((ch = *src++) != '\0') {
193  const char *pch;
194 
195  if ((pch = strchr(digits, ch)) != NULL) {
196  unsigned int nw = *tp * 10 + (pch - digits);
197 
198  if (saw_digit && *tp == 0)
199  return UV_EINVAL;
200  if (nw > 255)
201  return UV_EINVAL;
202  *tp = nw;
203  if (!saw_digit) {
204  if (++octets > 4)
205  return UV_EINVAL;
206  saw_digit = 1;
207  }
208  } else if (ch == '.' && saw_digit) {
209  if (octets == 4)
210  return UV_EINVAL;
211  *++tp = 0;
212  saw_digit = 0;
213  } else
214  return UV_EINVAL;
215  }
216  if (octets < 4)
217  return UV_EINVAL;
218  memcpy(dst, tmp, sizeof(struct in_addr));
219  return 0;
220 }

223 static int inet_pton6(const char *src, unsigned char *dst) {
224  static const char xdigits_l[] = "0123456789abcdef",
225  xdigits_u[] = "0123456789ABCDEF";
226  unsigned char tmp[sizeof(struct in6_addr)], *tp, *endp, *colonp;
227  const char *xdigits, *curtok;
228  int ch, seen_xdigits;
229  unsigned int val;
230 
231  memset((tp = tmp), '\0', sizeof tmp);
232  endp = tp + sizeof tmp;
233  colonp = NULL;
234  /* Leading :: requires some special handling. */
235  if (*src == ':')
236  if (*++src != ':')
237  return UV_EINVAL;
238  curtok = src;
239  seen_xdigits = 0;
240  val = 0;
241  while ((ch = *src++) != '\0') {
242  const char *pch;
243 
244  if ((pch = strchr((xdigits = xdigits_l), ch)) == NULL)
245  pch = strchr((xdigits = xdigits_u), ch);
246  if (pch != NULL) {
247  val <<= 4;
248  val |= (pch - xdigits);
249  if (++seen_xdigits > 4)
250  return UV_EINVAL;
251  continue;
252  }
253  if (ch == ':') {
254  curtok = src;
255  if (!seen_xdigits) {
256  if (colonp)
257  return UV_EINVAL;
258  colonp = tp;
259  continue;
260  } else if (*src == '\0') {
261  return UV_EINVAL;
262  }
263  if (tp + sizeof(uint16_t) > endp)
264  return UV_EINVAL;
265  *tp++ = (unsigned char) (val >> 8) & 0xff;
266  *tp++ = (unsigned char) val & 0xff;
267  seen_xdigits = 0;
268  val = 0;
269  continue;
270  }
271  if (ch == '.' && ((tp + sizeof(struct in_addr)) <= endp)) {
272  int err = inet_pton4(curtok, tp);
273  if (err == 0) {
274  tp += sizeof(struct in_addr);
275  seen_xdigits = 0;
276  break; /*%< '\\0' was seen by inet_pton4(). */
277  }
278  }
279  return UV_EINVAL;
280  }
281  if (seen_xdigits) {
282  if (tp + sizeof(uint16_t) > endp)
283  return UV_EINVAL;
284  *tp++ = (unsigned char) (val >> 8) & 0xff;
285  *tp++ = (unsigned char) val & 0xff;
286  }
287  if (colonp != NULL) {
288  /*
289  * Since some memmove()'s erroneously fail to handle
290  * overlapping regions, we'll do the shift by hand.
291  */
292  const int n = tp - colonp;
293  int i;
294 
295  if (tp == endp)
296  return UV_EINVAL;
297  for (i = 1; i <= n; i++) {
298  endp[- i] = colonp[n - i];
299  colonp[n - i] = 0;
300  }
301  tp = endp;
302  }
303  if (tp != endp)
304  return UV_EINVAL;
305  memcpy(dst, tmp, sizeof tmp);
306  return 0;
307 }
