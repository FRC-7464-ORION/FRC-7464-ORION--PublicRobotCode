110 void ReportError(JNIEnv* env, int32_t status, bool doThrow) {
111  if (status == 0) {
112  return;
113  }
114  const char* message = HAL_GetLastError(&status);
115  if (status == HAL_HANDLE_ERROR) {
116  ThrowHalHandleException(env, status);
117  return;
118  }
119  if (doThrow && status < 0) {
120  ThrowUncleanStatusException(
121  env, fmt::format(" Code: {}. {}", status, message).c_str(), status);
122  } else {
123  std::string func;
124  auto stack = GetJavaStackTrace(env, &func, "edu.wpi.first");
125  // Make a copy of message for safety, calling back into the HAL might
126  // invalidate the string.
127  std::string lastMessage{message};
128  HAL_SendError(1, status, 0, lastMessage.c_str(), func.c_str(),
129  stack.c_str(), 1);
130  }
131 }

133 void ThrowError(JNIEnv* env, int32_t status, int32_t minRange, int32_t maxRange,
134  int32_t requestedValue) {
135  if (status == 0) {
136  return;
137  }
138  const char* lastError = HAL_GetLastError(&status);
139  if (status == NO_AVAILABLE_RESOURCES || status == RESOURCE_IS_ALLOCATED ||
140  status == RESOURCE_OUT_OF_RANGE) {
141  ThrowAllocationException(env, lastError, status);
142  return;
143  }
144  if (status == HAL_HANDLE_ERROR) {
145  ThrowHalHandleException(env, status);
146  return;
147  }
148  ThrowUncleanStatusException(
149  env, fmt::format(" Code: {}. {}", status, lastError).c_str(), status);
150 }

152 void ReportCANError(JNIEnv* env, int32_t status, int message_id) {
153  if (status >= 0) {
154  return;
155  }
156  switch (status) {
157  case kRioStatusSuccess:
158  // Everything is ok... don't throw.
159  break;
160  case HAL_ERR_CANSessionMux_InvalidBuffer:
161  case kRIOStatusBufferInvalidSize: {
162  static jmethodID invalidBufConstruct = nullptr;
163  if (!invalidBufConstruct) {
164  invalidBufConstruct =
165  env->GetMethodID(canInvalidBufferExCls, "<init>", "()V");
166  }
167  jobject exception =
168  env->NewObject(canInvalidBufferExCls, invalidBufConstruct);
169  env->Throw(static_cast<jthrowable>(exception));
170  break;
171  }
172  case HAL_ERR_CANSessionMux_MessageNotFound:
173  case kRIOStatusOperationTimedOut: {
174  static jmethodID messageNotFoundConstruct = nullptr;
175  if (!messageNotFoundConstruct) {
176  messageNotFoundConstruct =
177  env->GetMethodID(canMessageNotFoundExCls, "<init>", "()V");
178  }
179  jobject exception =
180  env->NewObject(canMessageNotFoundExCls, messageNotFoundConstruct);
181  env->Throw(static_cast<jthrowable>(exception));
182  break;
183  }
184  case HAL_ERR_CANSessionMux_NotAllowed:
185  case kRIOStatusFeatureNotSupported: {
186  canMessageNotAllowedExCls.Throw(
187  env, fmt::format("MessageID = {}", message_id).c_str());
188  break;
189  }
190  case HAL_ERR_CANSessionMux_NotInitialized:
191  case kRIOStatusResourceNotInitialized: {
192  static jmethodID notInitConstruct = nullptr;
193  if (!notInitConstruct) {
194  notInitConstruct =
195  env->GetMethodID(canNotInitializedExCls, "<init>", "()V");
196  }
197  jobject exception =
198  env->NewObject(canNotInitializedExCls, notInitConstruct);
199  env->Throw(static_cast<jthrowable>(exception));
200  break;
201  }
202  default: {
203  uncleanStatusExCls.Throw(
204  env, fmt::format("Fatal status code detected: {}", status).c_str());
205  break;
206  }
207  }
208 }

215  double upper) {
216  static jmethodID getMessage = nullptr;
217  if (!getMessage) {
218  getMessage = env->GetStaticMethodID(boundaryExCls, "getMessage",

221

228  jobject msg = env->CallStaticObjectMethod(
229  boundaryExCls, getMessage, static_cast<jdouble>(value),

369 //
370 // indicate JNI version support desired and load classes
371 //
372 JNIEXPORT jint JNICALL JNI_OnLoad(JavaVM* vm, void* reserved) {
373  jvm = vm;
374 
375  JNIEnv* env;
376  if (vm->GetEnv(reinterpret_cast<void**>(&env), JNI_VERSION_1_6) != JNI_OK) {
377  return JNI_ERR;
378  }
379 
380  for (auto& c : classes) {
381  *c.cls = JClass(env, c.name);
382  if (!*c.cls) {
383  return JNI_ERR;
384  }
385  }
386 
387  for (auto& c : exceptions) {
388  *c.cls = JException(env, c.name);
389  if (!*c.cls) {
390  return JNI_ERR;
391  }
392  }
393 
394  return sim::SimOnLoad(vm, reserved);
395 }
396 
397 JNIEXPORT void JNICALL JNI_OnUnload(JavaVM* vm, void* reserved) {
398  sim::SimOnUnload(vm, reserved);
399 
400  JNIEnv* env;
401  if (vm->GetEnv(reinterpret_cast<void**>(&env), JNI_VERSION_1_6) != JNI_OK) {
402  return;
403  }
404  // Delete global references
405 
406  for (auto& c : classes) {
407  c.cls->free(env);
408  }
409  for (auto& c : exceptions) {
410  c.cls->free(env);
411  }
412  jvm = nullptr;
413 }
