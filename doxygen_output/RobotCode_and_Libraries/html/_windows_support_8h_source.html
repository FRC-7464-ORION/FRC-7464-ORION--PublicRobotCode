58 inline wpi::VersionTuple GetWindowsOSVersion() {
59  typedef NTSTATUS(WINAPI* RtlGetVersionPtr)(PRTL_OSVERSIONINFOW);
60  HMODULE hMod = ::GetModuleHandleW(L"ntdll.dll");
61  if (hMod) {
62  auto getVer = (RtlGetVersionPtr)::GetProcAddress(hMod, "RtlGetVersion");
63  if (getVer) {
64  RTL_OSVERSIONINFOEXW info{};
65  info.dwOSVersionInfoSize = sizeof(info);
66  if (getVer((PRTL_OSVERSIONINFOW)&info) == ((NTSTATUS)0x00000000L)) {
67  return wpi::VersionTuple(info.dwMajorVersion, info.dwMinorVersion, 0,
68  info.dwBuildNumber);
69  }
70  }
71  }
72  return wpi::VersionTuple(0, 0, 0, 0);
73 }

84 inline bool MakeErrMsg(std::string *ErrMsg, const std::string &prefix) {
85  if (!ErrMsg)
86  return true;
87  char *buffer = NULL;
88  DWORD LastError = GetLastError();
89  DWORD R = FormatMessageA(FORMAT_MESSAGE_ALLOCATE_BUFFER |
90  FORMAT_MESSAGE_FROM_SYSTEM |
91  FORMAT_MESSAGE_MAX_WIDTH_MASK,
92  NULL, LastError, 0, (LPSTR)&buffer, 1, NULL);
93  if (R)
94  *ErrMsg = prefix + ": " + buffer;
95  else
96  *ErrMsg = prefix + ": Unknown error";
97  *ErrMsg += " (0x" + wpi::utohexstr(LastError) + ")";
98 
99  LocalFree(buffer);
100  return R != 0;
101 }

103 template <typename HandleTraits>
104 class ScopedHandle {
105  typedef typename HandleTraits::handle_type handle_type;
106  handle_type Handle;
107 
108  ScopedHandle(const ScopedHandle &other) = delete;
109  void operator=(const ScopedHandle &other) = delete;
110 public:
111  ScopedHandle()
112  : Handle(HandleTraits::GetInvalid()) {}
113 
114  explicit ScopedHandle(handle_type h)
115  : Handle(h) {}
116 
117  ~ScopedHandle() {
118  if (HandleTraits::IsValid(Handle))
119  HandleTraits::Close(Handle);
120  }
121 
122  handle_type take() {
123  handle_type t = Handle;
124  Handle = HandleTraits::GetInvalid();
125  return t;
126  }
127 
128  ScopedHandle &operator=(handle_type h) {
129  if (HandleTraits::IsValid(Handle))
130  HandleTraits::Close(Handle);
131  Handle = h;
132  return *this;
133  }
134 
135  // True if Handle is valid.
136  explicit operator bool() const {
137  return HandleTraits::IsValid(Handle) ? true : false;
138  }
139 
140  operator handle_type() const {
141  return Handle;
142  }
143 };
