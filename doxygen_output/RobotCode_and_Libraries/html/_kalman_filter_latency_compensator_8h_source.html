50  void AddObserverState(const KalmanFilterType& observer,
51  Eigen::Vector<double, Inputs> u,
52  Eigen::Vector<double, Outputs> localY,
53  units::second_t timestamp) {
54  // Add the new state into the vector.
55  m_pastObserverSnapshots.emplace_back(timestamp,
56  ObserverSnapshot{observer, u, localY});
57 
58  // Remove the oldest snapshot if the vector exceeds our maximum size.
59  if (m_pastObserverSnapshots.size() > kMaxPastObserverStates) {
60  m_pastObserverSnapshots.erase(m_pastObserverSnapshots.begin());
61  }
62  }

75  template <int Rows>
76  void ApplyPastGlobalMeasurement(
77  KalmanFilterType* observer, units::second_t nominalDt,
78  Eigen::Vector<double, Rows> y,
79  std::function<void(const Eigen::Vector<double, Inputs>& u,
80  const Eigen::Vector<double, Rows>& y)>
81  globalMeasurementCorrect,
82  units::second_t timestamp) {
83  if (m_pastObserverSnapshots.size() == 0) {
84  // State map was empty, which means that we got a measurement right at
85  // startup. The only thing we can do is ignore the measurement.
86  return;
87  }
88 
89  // Perform a binary search to find the index of first snapshot whose
90  // timestamp is greater than or equal to the global measurement timestamp
91  auto it = std::lower_bound(
92  m_pastObserverSnapshots.cbegin(), m_pastObserverSnapshots.cend(),
93  timestamp,
94  [](const auto& entry, const auto& ts) { return entry.first < ts; });
95 
96  size_t indexOfClosestEntry;
97 
98  if (it == m_pastObserverSnapshots.cbegin()) {
99  // If the global measurement is older than any snapshot, throw out the
100  // measurement because there's no state estimate into which to incorporate
101  // the measurement
102  if (timestamp < it->first) {
103  return;
104  }
105 
106  // If the first snapshot has same timestamp as the global measurement, use
107  // that snapshot
108  indexOfClosestEntry = 0;
109  } else if (it == m_pastObserverSnapshots.cend()) {
110  // If all snapshots are older than the global measurement, use the newest
111  // snapshot
112  indexOfClosestEntry = m_pastObserverSnapshots.size() - 1;
113  } else {
114  // Index of snapshot taken after the global measurement
115  int nextIdx = std::distance(m_pastObserverSnapshots.cbegin(), it);
116 
117  // Index of snapshot taken before the global measurement. Since we already
118  // handled the case where the index points to the first snapshot, this
119  // computation is guaranteed to be nonnegative.
120  int prevIdx = nextIdx - 1;
121 
122  // Find the snapshot closest in time to global measurement
123  units::second_t prevTimeDiff =
124  units::math::abs(timestamp - m_pastObserverSnapshots[prevIdx].first);
125  units::second_t nextTimeDiff =
126  units::math::abs(timestamp - m_pastObserverSnapshots[nextIdx].first);
127  indexOfClosestEntry = prevTimeDiff < nextTimeDiff ? prevIdx : nextIdx;
128  }
129 
130  units::second_t lastTimestamp =
131  m_pastObserverSnapshots[indexOfClosestEntry].first - nominalDt;
132 
133  // We will now go back in time to the state of the system at the time when
134  // the measurement was captured. We will reset the observer to that state,
135  // and apply correction based on the measurement. Then, we will go back
136  // through all observer states until the present and apply past inputs to
137  // get the present estimated state.
138  for (size_t i = indexOfClosestEntry; i < m_pastObserverSnapshots.size();
139  ++i) {
140  auto& [key, snapshot] = m_pastObserverSnapshots[i];
141 
142  if (i == indexOfClosestEntry) {
143  observer->SetP(snapshot.errorCovariances);
144  observer->SetXhat(snapshot.xHat);
145  }
146 
147  observer->Predict(snapshot.inputs, key - lastTimestamp);
148  observer->Correct(snapshot.inputs, snapshot.localMeasurements);
149 
150  if (i == indexOfClosestEntry) {
151  // Note that the measurement is at a timestep close but probably not
152  // exactly equal to the timestep for which we called predict. This makes
153  // the assumption that the dt is small enough that the difference
154  // between the measurement time and the time that the inputs were
155  // captured at is very small.
156  globalMeasurementCorrect(snapshot.inputs, y);
157  }
158 
159  lastTimestamp = key;
160  snapshot = ObserverSnapshot{*observer, snapshot.inputs,
161  snapshot.localMeasurements};
162  }
163  }
164 
165  private:
166  static constexpr size_t kMaxPastObserverStates = 300;
167  std::vector<std::pair<units::second_t, ObserverSnapshot>>
168  m_pastObserverSnapshots;
169 };
