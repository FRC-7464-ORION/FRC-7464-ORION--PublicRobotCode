193  /*
194  @brief read a number from the input
195 
196  @tparam NumberType the type of the number
197 
198  @return number of type @a NumberType
199 
200  @note This function needs to respect the system's endianess, because
201  bytes in CBOR and MessagePack are stored in network order (big
202  endian) and therefore need reordering on little endian systems.
203 
204  @throw parse_error.110 if input has less than `sizeof(NumberType)` bytes
205  */
206  template<typename NumberType> NumberType get_number()
207  {
208  // step 1: read input into array with system's byte order
209  std::array<uint8_t, sizeof(NumberType)> vec;
210  for (std::size_t i = 0; i < sizeof(NumberType); ++i)
211  {
212  get();
213  unexpect_eof();
214 
215  // reverse byte order prior to conversion if necessary
216  if (is_little_endian)
217  {
218  vec[sizeof(NumberType) - i - 1] = static_cast<uint8_t>(current);
219  }
220  else
221  {
222  vec[i] = static_cast<uint8_t>(current); // LCOV_EXCL_LINE
223  }
224  }
225 
226  // step 2: convert array into number of type T and return
227  NumberType result;
228  std::memcpy(&result, vec.data(), sizeof(NumberType));
229  return result;
230  }

406 {
407  switch (get_char ? get() : current)
408  {
409  // EOF
410  case std::char_traits<char>::eof():
411  JSON_THROW(parse_error::create(110, chars_read, "unexpected end of input"));
412 
413  // Integer 0x00..0x17 (0..23)

415  case 0x01:
416  case 0x02:

436  case 0x16:
437  case 0x17:
438  return static_cast<uint64_t>(current);
439

441  return get_number<uint8_t>();
442 
443  case 0x19: // Unsigned integer (two-byte uint16_t follows)

445

448

474  case 0x35:
475  case 0x36:
476  case 0x37:

625  }

627  case 0xB9: // map (two-byte uint16_t for n follow)

630  }
631 
632  case 0xBA: // map (four-byte uint32_t for n follow)

634  return get_cbor_object(get_number<uint32_t>());
635  }
636 
637  case 0xBB: // map (eight-byte uint64_t for n follow)
638  {
639  return get_cbor_object(get_number<uint64_t>());
640  }
641 
642  case 0xBF: // map (indefinite length)
643  {
644  json result = value_t::object;
645  while (get() != 0xFF)
646  {
647  auto key = get_cbor_string();
648  result[key] = parse_cbor_internal();
649  }
650  return result;
651  }
652 
653  case 0xF4: // false
654  {
655  return false;
656  }
657 
658  case 0xF5: // true
659  {
660  return true;
661  }
662 
663  case 0xF6: // null
664  {
665  return value_t::null;
666  }
667 
668  case 0xF9: // Half-Precision Float (two-byte IEEE 754)
669  {
670  const int byte1 = get();
671  unexpect_eof();
672  const int byte2 = get();
673  unexpect_eof();
674 
675  // code from RFC 7049, Appendix D, Figure 3:
676  // As half-precision floating-point numbers were only added
677  // to IEEE 754 in 2008, today's programming platforms often
678  // still only have limited support for them. It is very
679  // easy to include at least decoding support for them even
680  // without such support. An example of a small decoder for
681  // half-precision floating-point numbers in the C language
682  // is shown in Fig. 3.
683  const int half = (byte1 << 8) + byte2;
684  const int exp = (half >> 10) & 0x1F;
685  const int mant = half & 0x3FF;
686  double val;
687  if (exp == 0)
688  {
689  val = std::ldexp(mant, -24);
690  }
691  else if (exp != 31)
692  {
693  val = std::ldexp(mant + 1024, exp - 25);
694  }
695  else
696  {
697  val = (mant == 0) ? std::numeric_limits<double>::infinity()
698  : std::numeric_limits<double>::quiet_NaN();
699  }
700  return (half & 0x8000) != 0 ? -val : val;
701  }
702 
703  case 0xFA: // Single-Precision Float (four-byte IEEE 754)
704  {
705  return get_number<float>();
706  }
707 
708  case 0xFB: // Double-Precision Float (eight-byte IEEE 754)
709  {
710  return get_number<double>();
711  }
712 
713  default: // anything else (0xFF is handled inside the other types)
714  {
715  JSON_THROW(parse_error::create(112, chars_read, fmt::format("error reading CBOR; last byte: {:#02x}", current)));
716  }
717  }
