14 int PropertyContainer::GetPropertyIndex(std::string_view name) const {
15  // We can't fail, so instead we create a new index if caching fails.
16  CS_Status status = 0;
17  if (!m_properties_cached) {
18  CacheProperties(&status);
19  }
20  std::scoped_lock lock(m_mutex);
21  int& ndx = m_properties[name];
22  if (ndx == 0) {
23  // create a new index
24  ndx = m_propertyData.size() + 1;
25  m_propertyData.emplace_back(CreateEmptyProperty(name));
26  }
27  return ndx;
28 }

237 bool PropertyContainer::SetPropertiesJson(const wpi::json& config,
238  wpi::Logger& logger,
239  std::string_view logName,
240  CS_Status* status) {
241  for (auto&& prop : config) {
242  std::string name;
243  try {
244  name = prop.at("name").get<std::string>();
245  } catch (const wpi::json::exception& e) {
246  WPI_WARNING(logger, "{}: SetConfigJson: could not read property name: {}",
247  logName, e.what());
248  continue;
249  }
250  int n = GetPropertyIndex(name);
251  try {
252  auto& v = prop.at("value");
253  if (v.is_string()) {
254  std::string val = v.get<std::string>();
255  WPI_INFO(logger, "{}: SetConfigJson: setting property '{}' to '{}'",
256  logName, name, val);
257  SetStringProperty(n, val, status);
258  } else if (v.is_boolean()) {
259  bool val = v.get<bool>();
260  WPI_INFO(logger, "{}: SetConfigJson: setting property '{}' to {}",
261  logName, name, val);
262  SetProperty(n, val, status);
263  } else {
264  int val = v.get<int>();
265  WPI_INFO(logger, "{}: SetConfigJson: setting property '{}' to {}",
266  logName, name, val);
267  SetProperty(n, val, status);
268  }
269  } catch (const wpi::json::exception& e) {
270  WPI_WARNING(logger,
271  "{}: SetConfigJson: could not read property value: {}",
272  logName, e.what());
273  continue;
274  }
275  }
276 
277  return true;
278 }

280 wpi::json PropertyContainer::GetPropertiesJsonObject(CS_Status* status) {
281  wpi::json j;
282  wpi::SmallVector<int, 32> propVec;
283  for (int p : EnumerateProperties(propVec, status)) {
284  wpi::json prop;
285  wpi::SmallString<128> strBuf;
286  prop.emplace("name", GetPropertyName(p, strBuf, status));
287  switch (GetPropertyKind(p)) {
288  case CS_PROP_BOOLEAN:
289  prop.emplace("value", static_cast<bool>(GetProperty(p, status)));
290  break;
291  case CS_PROP_INTEGER:
292  case CS_PROP_ENUM:
293  prop.emplace("value", GetProperty(p, status));
294  break;
295  case CS_PROP_STRING:
296  prop.emplace("value", GetStringProperty(p, strBuf, status));
297  break;
298  default:
299  continue;
300  }
301  j.emplace_back(prop);
302  }
303 
304  return j;
305 }
