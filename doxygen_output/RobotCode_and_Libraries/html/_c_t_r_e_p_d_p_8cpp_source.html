127 HAL_PDPHandle HAL_InitializePDP(int32_t module, const char* allocationLocation,
128  int32_t* status) {
129  hal::init::CheckInit();
130  if (!HAL_CheckPDPModule(module)) {
131  *status = PARAMETER_OUT_OF_RANGE;
132  hal::SetLastError(status, fmt::format("Invalid pdp module {}", module));
133  return HAL_kInvalidHandle;
134  }
135 
136  HAL_PDPHandle handle;
137  auto pdp = pdpHandles->Allocate(module, &handle, status);
138 
139  if (*status != 0) {
140  if (pdp) {
141  hal::SetLastErrorPreviouslyAllocated(status, "CTRE PDP", module,
142  pdp->previousAllocation);
143  } else {
144  hal::SetLastErrorIndexOutOfRange(status, "Invalid Index for CTRE PDP", 0,
145  kNumCTREPDPModules, module);
146  }
147  return HAL_kInvalidHandle; // failed to allocate. Pass error back.
148  }
149 
150  pdp->canHandle = HAL_InitializeCAN(manufacturer, module, deviceType, status);
151  if (*status != 0) {
152  pdpHandles->Free(handle);
153  return HAL_kInvalidHandle;
154  }
155 
156  pdp->previousAllocation = allocationLocation ? allocationLocation : "";
157 
158  return handle;
159 }

223 double HAL_GetPDPChannelCurrent(HAL_PDPHandle handle, int32_t channel,
224  int32_t* status) {
225  if (!HAL_CheckPDPChannel(channel)) {
226  *status = PARAMETER_OUT_OF_RANGE;
227  hal::SetLastError(status, fmt::format("Invalid pdp channel {}", channel));
228  return 0;
229  }
230 
231  auto pdp = pdpHandles->Get(handle);
232  if (pdp == nullptr) {

234  return 0;

236 
237  int32_t length = 0;
238  uint64_t receivedTimestamp = 0;
239 
240  double raw = 0;
241 
242  if (channel <= 5) {
243  PdpStatus1 pdpStatus;
244  HAL_ReadCANPacketTimeout(pdp->canHandle, Status1, pdpStatus.data, &length,
245  &receivedTimestamp, TimeoutMs, status);
246  if (*status != 0) {
247  return 0;
248  }
249  switch (channel) {
250  case 0:
251  raw = (static_cast<uint32_t>(pdpStatus.bits.chan1_h8) << 2) |
252  pdpStatus.bits.chan1_l2;
253  break;
254  case 1:
255  raw = (static_cast<uint32_t>(pdpStatus.bits.chan2_h6) << 4) |

261  break;
262  case 3:
263  raw = (static_cast<uint32_t>(pdpStatus.bits.chan4_h2) << 8) |
264  pdpStatus.bits.chan4_l8;
265  break;
266  case 4:
267  raw = (static_cast<uint32_t>(pdpStatus.bits.chan5_h8) << 2) |
268  pdpStatus.bits.chan5_l2;
269  break;
270  case 5:
271  raw = (static_cast<uint32_t>(pdpStatus.bits.chan6_h6) << 4) |
272  pdpStatus.bits.chan6_l4;
273  break;
274  }
275  } else if (channel <= 11) {
276  PdpStatus2 pdpStatus;
277  HAL_ReadCANPacketTimeout(pdp->canHandle, Status2, pdpStatus.data, &length,
278  &receivedTimestamp, TimeoutMs, status);
279  if (*status != 0) {

283  case 6:
284  raw = (static_cast<uint32_t>(pdpStatus.bits.chan7_h8) << 2) |
285  pdpStatus.bits.chan7_l2;
286  break;
287  case 7:
288  raw = (static_cast<uint32_t>(pdpStatus.bits.chan8_h6) << 4) |
289  pdpStatus.bits.chan8_l4;
290  break;
291  case 8:
292  raw = (static_cast<uint32_t>(pdpStatus.bits.chan9_h4) << 6) |
293  pdpStatus.bits.chan9_l6;
294  break;
295  case 9:
296  raw = (static_cast<uint32_t>(pdpStatus.bits.chan10_h2) << 8) |
297  pdpStatus.bits.chan10_l8;
298  break;
299  case 10:
300  raw = (static_cast<uint32_t>(pdpStatus.bits.chan11_h8) << 2) |
301  pdpStatus.bits.chan11_l2;
302  break;
303  case 11:
304  raw = (static_cast<uint32_t>(pdpStatus.bits.chan12_h6) << 4) |
305  pdpStatus.bits.chan12_l4;
306  break;
307  }
308  } else {
309  PdpStatus3 pdpStatus;
310  HAL_ReadCANPacketTimeout(pdp->canHandle, Status3, pdpStatus.data, &length,
311  &receivedTimestamp, TimeoutMs, status);
312  if (*status != 0) {
313  return 0;
314  }
315  switch (channel) {
316  case 12:
317  raw = (static_cast<uint32_t>(pdpStatus.bits.chan13_h8) << 2) |
318  pdpStatus.bits.chan13_l2;
319  break;
320  case 13:
321  raw = (static_cast<uint32_t>(pdpStatus.bits.chan14_h6) << 4) |
322  pdpStatus.bits.chan14_l4;
323  break;
324  case 14:
325  raw = (static_cast<uint32_t>(pdpStatus.bits.chan15_h4) << 6) |
326  pdpStatus.bits.chan15_l6;
327  break;
328  case 15:
329  raw = (static_cast<uint32_t>(pdpStatus.bits.chan16_h2) << 8) |
330  pdpStatus.bits.chan16_l8;
331  break;
332  }
333  }
334 
335  /* convert to amps */
336  return raw * 0.125; /* 7.3 fixed pt value in Amps */
