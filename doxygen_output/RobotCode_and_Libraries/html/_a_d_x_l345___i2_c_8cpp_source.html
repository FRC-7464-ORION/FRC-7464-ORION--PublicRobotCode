13 ADXL345_I2C::ADXL345_I2C(I2C::Port port, Range range, int deviceAddress)

60 double ADXL345_I2C::GetAcceleration(ADXL345_I2C::Axes axis) {
61  if (axis == kAxis_X && m_simX) {
62  return m_simX.Get();
63  }
64  if (axis == kAxis_Y && m_simY) {
65  return m_simY.Get();
66  }
67  if (axis == kAxis_Z && m_simZ) {
68  return m_simZ.Get();
69  }
70  int16_t rawAccel = 0;
71  m_i2c.Read(kDataRegister + static_cast<int>(axis), sizeof(rawAccel),
72  reinterpret_cast<uint8_t*>(&rawAccel));
73  return rawAccel * kGsPerLSB;
74 }

76 ADXL345_I2C::AllAxes ADXL345_I2C::GetAccelerations() {
77  AllAxes data;
78  if (m_simX && m_simY && m_simZ) {
79  data.XAxis = m_simX.Get();
80  data.YAxis = m_simY.Get();
81  data.ZAxis = m_simZ.Get();
82  return data;
83  }
84  int16_t rawData[3];
85  m_i2c.Read(kDataRegister, sizeof(rawData),
86  reinterpret_cast<uint8_t*>(rawData));
87 
88  data.XAxis = rawData[0] * kGsPerLSB;
89  data.YAxis = rawData[1] * kGsPerLSB;
90  data.ZAxis = rawData[2] * kGsPerLSB;
91  return data;
92 }
