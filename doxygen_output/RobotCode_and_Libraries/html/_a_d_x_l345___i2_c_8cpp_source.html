49 double ADXL345_I2C::GetAcceleration(ADXL345_I2C::Axes axis) {
50  if (axis == kAxis_X && m_simX) return m_simX.Get();
51  if (axis == kAxis_Y && m_simY) return m_simY.Get();
52  if (axis == kAxis_Z && m_simZ) return m_simZ.Get();
53  int16_t rawAccel = 0;
54  m_i2c.Read(kDataRegister + static_cast<int>(axis), sizeof(rawAccel),
55  reinterpret_cast<uint8_t*>(&rawAccel));
56  return rawAccel * kGsPerLSB;
57 }

59 ADXL345_I2C::AllAxes ADXL345_I2C::GetAccelerations() {
60  AllAxes data;
61  if (m_simX && m_simY && m_simZ) {
62  data.XAxis = m_simX.Get();
63  data.YAxis = m_simY.Get();
64  data.ZAxis = m_simZ.Get();
65  return data;
66  }
67  int16_t rawData[3];
68  m_i2c.Read(kDataRegister, sizeof(rawData),
69  reinterpret_cast<uint8_t*>(rawData));
70 
71  data.XAxis = rawData[0] * kGsPerLSB;
72  data.YAxis = rawData[1] * kGsPerLSB;
73  data.ZAxis = rawData[2] * kGsPerLSB;
74  return data;
75 }
