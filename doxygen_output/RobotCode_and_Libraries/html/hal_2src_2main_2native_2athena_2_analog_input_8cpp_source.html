25 HAL_AnalogInputHandle HAL_InitializeAnalogInputPort(
26  HAL_PortHandle portHandle, const char* allocationLocation,
27  int32_t* status) {
28  hal::init::CheckInit();
29  initializeAnalog(status);
30 
31  if (*status != 0) {
32  return HAL_kInvalidHandle;
33  }
34 
35  int16_t channel = getPortHandleChannel(portHandle);
36  if (channel == InvalidHandleIndex || channel >= kNumAnalogInputs) {
37  *status = RESOURCE_OUT_OF_RANGE;
38  hal::SetLastErrorIndexOutOfRange(status, "Invalid Index for Analog Input",
39  0, kNumAnalogInputs, channel);
40  return HAL_kInvalidHandle;
41  }
42 
43  HAL_AnalogInputHandle handle;
44  auto analog_port = analogInputHandles->Allocate(channel, &handle, status);
45 
46  if (*status != 0) {
47  if (analog_port) {
48  hal::SetLastErrorPreviouslyAllocated(status, "Analog Input", channel,
49  analog_port->previousAllocation);
50  } else {
51  hal::SetLastErrorIndexOutOfRange(status, "Invalid Index for Analog Input",
52  0, kNumAnalogInputs, channel);
53  }
54  return HAL_kInvalidHandle; // failed to allocate. Pass error back.
55  }
56 
57  // Initialize port structure
58  analog_port->channel = static_cast<uint8_t>(channel);
59  if (HAL_IsAccumulatorChannel(handle, status)) {
60  analog_port->accumulator.reset(tAccumulator::create(channel, status));
61  } else {
62  analog_port->accumulator = nullptr;
63  }
64 
65  // Set default configuration
66  analogInputSystem->writeScanList(channel, channel, status);
67  HAL_SetAnalogAverageBits(handle, kDefaultAverageBits, status);
68  HAL_SetAnalogOversampleBits(handle, kDefaultOversampleBits, status);
69  analog_port->previousAllocation =
70  allocationLocation ? allocationLocation : "";
71  return handle;
72 }

191 int32_t HAL_GetAnalogVoltsToValue(HAL_AnalogInputHandle analogPortHandle,
192  double voltage, int32_t* status) {
193  if (voltage > 5.0) {
194  voltage = 5.0;
195  *status = VOLTAGE_OUT_OF_RANGE;
196  }
197  if (voltage < 0.0) {
198  voltage = 0.0;
199  *status = VOLTAGE_OUT_OF_RANGE;
200  }
201  int32_t LSBWeight = HAL_GetAnalogLSBWeight(analogPortHandle, status);
202  int32_t offset = HAL_GetAnalogOffset(analogPortHandle, status);
203  int32_t value =
204  static_cast<int32_t>((voltage + offset * 1.0e-9) / (LSBWeight * 1.0e-9));
205  return value;
206 }

225 double HAL_GetAnalogAverageVoltage(HAL_AnalogInputHandle analogPortHandle,
226  int32_t* status) {
227  int32_t value = HAL_GetAnalogAverageValue(analogPortHandle, status);
228  int32_t LSBWeight = HAL_GetAnalogLSBWeight(analogPortHandle, status);
229  int32_t offset = HAL_GetAnalogOffset(analogPortHandle, status);
230  int32_t oversampleBits =
231  HAL_GetAnalogOversampleBits(analogPortHandle, status);
232  double voltage =
233  LSBWeight * 1.0e-9 * value / static_cast<double>(1 << oversampleBits) -
234  offset * 1.0e-9;
235  return voltage;
236 }
