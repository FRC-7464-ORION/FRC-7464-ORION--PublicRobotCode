37 void WireEncoder::WriteType(NT_Type type) {
38  char ch;
39  // Convert from enum to actual byte value.
40  switch (type) {
41  case NT_BOOLEAN:
42  ch = 0x00;
43  break;
44  case NT_DOUBLE:
45  ch = 0x01;
46  break;
47  case NT_STRING:
48  ch = 0x02;
49  break;
50  case NT_RAW:
51  if (m_proto_rev < 0x0300u) {
52  m_error = "raw type not supported in protocol < 3.0";
53  return;
54  }
55  ch = 0x03;
56  break;
57  case NT_BOOLEAN_ARRAY:
58  ch = 0x10;
59  break;
60  case NT_DOUBLE_ARRAY:
61  ch = 0x11;
62  break;
63  case NT_STRING_ARRAY:
64  ch = 0x12;
65  break;
66  case NT_RPC:
67  if (m_proto_rev < 0x0300u) {
68  m_error = "RPC type not supported in protocol < 3.0";
69  return;
70  }
71  ch = 0x20;
72  break;
73  default:
74  m_error = "unrecognized type";
75  return;
76  }
77  m_data.push_back(ch);
78 }

80 size_t WireEncoder::GetValueSize(const Value& value) const {
81  switch (value.type()) {
82  case NT_BOOLEAN:
83  return 1;
84  case NT_DOUBLE:
85  return 8;
86  case NT_STRING:
87  return GetStringSize(value.GetString());
88  case NT_RAW:
89  if (m_proto_rev < 0x0300u) {
90  return 0;
91  }
92  return GetStringSize(value.GetRaw());
93  case NT_RPC:
94  if (m_proto_rev < 0x0300u) {
95  return 0;
96  }
97  return GetStringSize(value.GetRpc());
98  case NT_BOOLEAN_ARRAY: {
99  // 1-byte size, 1 byte per element
100  size_t size = value.GetBooleanArray().size();
101  if (size > 0xff) {
102  size = 0xff; // size is only 1 byte, truncate
103  }
104  return 1 + size;
105  }
106  case NT_DOUBLE_ARRAY: {
107  // 1-byte size, 8 bytes per element
108  size_t size = value.GetDoubleArray().size();
109  if (size > 0xff) {
110  size = 0xff; // size is only 1 byte, truncate
111  }
112  return 1 + size * 8;
113  }
114  case NT_STRING_ARRAY: {
115  auto v = value.GetStringArray();
116  size_t size = v.size();
117  if (size > 0xff) {
118  size = 0xff; // size is only 1 byte, truncate
119  }
120  size_t len = 1; // 1-byte size
121  for (size_t i = 0; i < size; ++i) {
122  len += GetStringSize(v[i]);
123  }
124  return len;
125  }
126  default:
127  return 0;
128  }
129 }

131 void WireEncoder::WriteValue(const Value& value) {
132  switch (value.type()) {
133  case NT_BOOLEAN:
134  Write8(value.GetBoolean() ? 1 : 0);
135  break;
136  case NT_DOUBLE:
137  WriteDouble(value.GetDouble());
138  break;
139  case NT_STRING:
140  WriteString(value.GetString());
141  break;
142  case NT_RAW:
143  if (m_proto_rev < 0x0300u) {
144  m_error = "raw values not supported in protocol < 3.0";
145  return;
146  }
147  WriteString(value.GetRaw());
148  break;
149  case NT_RPC:
150  if (m_proto_rev < 0x0300u) {
151  m_error = "RPC values not supported in protocol < 3.0";
152  return;
153  }
154  WriteString(value.GetRpc());
155  break;
156  case NT_BOOLEAN_ARRAY: {
157  auto v = value.GetBooleanArray();
158  size_t size = v.size();
159  if (size > 0xff) {
160  size = 0xff; // size is only 1 byte, truncate
161  }
162  Write8(size);
163 
164  for (size_t i = 0; i < size; ++i) {
165  Write8(v[i] ? 1 : 0);
166  }
167  break;
168  }
169  case NT_DOUBLE_ARRAY: {
170  auto v = value.GetDoubleArray();
171  size_t size = v.size();
172  if (size > 0xff) {
173  size = 0xff; // size is only 1 byte, truncate
174  }
175  Write8(size);
176 
177  for (size_t i = 0; i < size; ++i) {
178  WriteDouble(v[i]);
179  }
180  break;
181  }
182  case NT_STRING_ARRAY: {
183  auto v = value.GetStringArray();
184  size_t size = v.size();
185  if (size > 0xff) {
186  size = 0xff; // size is only 1 byte, truncate
187  }
188  Write8(size);
189 
190  for (size_t i = 0; i < size; ++i) {
191  WriteString(v[i]);
192  }
193  break;
194  }
195  default:
196  m_error = "unrecognized type when writing value";
197  return;
198  }
199 }

201 size_t WireEncoder::GetStringSize(std::string_view str) const {
202  if (m_proto_rev < 0x0300u) {
203  size_t len = str.size();
204  if (len > 0xffff) {
205  len = 0xffff; // Limited to 64K length; truncate

212 void WireEncoder::WriteString(std::string_view str) {
213  // length
214  size_t len = str.size();
215  if (m_proto_rev < 0x0300u) {
216  if (len > 0xffff) {
217  len = 0xffff; // Limited to 64K length; truncate
218  }
219  Write16(len);
220  } else {
221  WriteUleb128(len);
222  }
223 
224  // contents
225  m_data.append(str.data(), str.data() + len);
226 }
