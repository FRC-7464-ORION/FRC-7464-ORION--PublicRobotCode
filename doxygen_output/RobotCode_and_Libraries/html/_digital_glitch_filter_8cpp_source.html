27 DigitalGlitchFilter::DigitalGlitchFilter() {

52 void DigitalGlitchFilter::DoAdd(DigitalSource* input, int requestedIndex) {
53  // Some sources from Counters and Encoders are null. By pushing the check
54  // here, we catch the issue more generally.
55  if (input) {
56  // We don't support GlitchFilters on AnalogTriggers.
57  if (input->IsAnalogTrigger()) {
58  throw FRC_MakeError(
59  -1, "{}", "Analog Triggers not supported for DigitalGlitchFilters");
60  }
61  int32_t status = 0;
62  HAL_SetFilterSelect(input->GetPortHandleForRouting(), requestedIndex,
63  &status);
64  FRC_CheckErrorStatus(status, "requested index {}", requestedIndex);
65 
66  // Validate that we set it correctly.
67  int actualIndex =
68  HAL_GetFilterSelect(input->GetPortHandleForRouting(), &status);
69  FRC_CheckErrorStatus(status, "requested index {}", requestedIndex);
70  FRC_Assert(actualIndex == requestedIndex);
71  }
72 }
