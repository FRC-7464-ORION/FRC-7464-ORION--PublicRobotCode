71 void DigitalGlitchFilter::DoAdd(DigitalSource* input, int requestedIndex) {
72  // Some sources from Counters and Encoders are null. By pushing the check
73  // here, we catch the issue more generally.
74  if (input) {
75  // We don't support GlitchFilters on AnalogTriggers.
76  if (input->IsAnalogTrigger()) {
77  wpi_setErrorWithContext(
78  -1, "Analog Triggers not supported for DigitalGlitchFilters");
79  return;
80  }
81  int32_t status = 0;
82  HAL_SetFilterSelect(input->GetPortHandleForRouting(), requestedIndex,
83  &status);
84  wpi_setHALError(status);
85 
86  // Validate that we set it correctly.
87  int actualIndex =
88  HAL_GetFilterSelect(input->GetPortHandleForRouting(), &status);
89  wpi_assertEqual(actualIndex, requestedIndex);
90  }
91 }
