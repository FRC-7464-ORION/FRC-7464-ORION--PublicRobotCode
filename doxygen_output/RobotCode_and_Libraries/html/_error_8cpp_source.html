61 void Error::Set(Code code, const wpi::Twine& contextMessage,
62  wpi::StringRef filename, wpi::StringRef function,
63  int lineNumber, const ErrorBase* originatingObject) {
64  bool report = true;
65 
66  if (code == m_code && GetTime() - m_timestamp < 1) {
67  report = false;
68  }
69 
70  m_code = code;
71  m_message = contextMessage.str();
72  m_filename = filename;
73  m_function = function;
74  m_lineNumber = lineNumber;
75  m_originatingObject = originatingObject;
76 
77  if (report) {
78  m_timestamp = GetTime();
79  Report();
80  }
81 }
