61 template <int States, int Inputs>
62 bool IsStabilizableImpl(const Eigen::Matrix<double, States, States>& A,
63  const Eigen::Matrix<double, States, Inputs>& B) {
64  Eigen::EigenSolver<Eigen::Matrix<double, States, States>> es{A};
65 
66  for (int i = 0; i < States; ++i) {
67  if (es.eigenvalues()[i].real() * es.eigenvalues()[i].real() +
68  es.eigenvalues()[i].imag() * es.eigenvalues()[i].imag() <
69  1) {
70  continue;
71  }
72 
73  Eigen::Matrix<std::complex<double>, States, States + Inputs> E;
74  E << es.eigenvalues()[i] * Eigen::Matrix<std::complex<double>, States,
75  States>::Identity() -
76  A,
77  B;
78 
79  Eigen::ColPivHouseholderQR<
80  Eigen::Matrix<std::complex<double>, States, States + Inputs>>
81  qr{E};
82  if (qr.rank() < States) {
83  return false;
84  }
85  }
86  return true;
87 }

218 template <int N>
219 Eigen::Vector<double, N> MakeWhiteNoiseVector(
220  const std::array<double, N>& stdDevs) {
221  std::random_device rd;
222  std::mt19937 gen{rd()};
223 
224  Eigen::Vector<double, N> result;
225  for (int i = 0; i < N; ++i) {
226  // Passing a standard deviation of 0.0 to std::normal_distribution is
227  // undefined behavior
228  if (stdDevs[i] == 0.0) {
229  result(i) = 0.0;
230  } else {
231  std::normal_distribution distr{0.0, stdDevs[i]};
232  result(i) = distr(gen);
233  }
234  }
235  return result;
236 }
