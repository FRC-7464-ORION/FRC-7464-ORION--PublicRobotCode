9 CubicHermiteSpline::CubicHermiteSpline(
10  wpi::array<double, 2> xInitialControlVector,
11  wpi::array<double, 2> xFinalControlVector,
12  wpi::array<double, 2> yInitialControlVector,
13  wpi::array<double, 2> yFinalControlVector) {
14  const auto hermite = MakeHermiteBasis();
15  const auto x =
16  ControlVectorFromArrays(xInitialControlVector, xFinalControlVector);
17  const auto y =
18  ControlVectorFromArrays(yInitialControlVector, yFinalControlVector);
19 
20  // Populate first two rows with coefficients.
21  m_coefficients.template block<1, 4>(0, 0) = hermite * x;
22  m_coefficients.template block<1, 4>(1, 0) = hermite * y;
23 
24  // Populate Row 2 and Row 3 with the derivatives of the equations above.
25  // Then populate row 4 and 5 with the second derivatives.
26  for (int i = 0; i < 4; i++) {
27  // Here, we are multiplying by (3 - i) to manually take the derivative. The
28  // power of the term in index 0 is 3, index 1 is 2 and so on. To find the
29  // coefficient of the derivative, we can use the power rule and multiply
30  // the existing coefficient by its power.
31  m_coefficients.template block<2, 1>(2, i) =
32  m_coefficients.template block<2, 1>(0, i) * (3 - i);
33  }
34 
35  for (int i = 0; i < 3; i++) {
36  // Here, we are multiplying by (2 - i) to manually take the derivative. The
37  // power of the term in index 0 is 2, index 1 is 1 and so on. To find the
38  // coefficient of the derivative, we can use the power rule and multiply
39  // the existing coefficient by its power.
40  m_coefficients.template block<2, 1>(4, i) =
41  m_coefficients.template block<2, 1>(2, i) * (2 - i);
42  }
43 }
