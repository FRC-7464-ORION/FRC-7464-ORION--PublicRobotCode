77  @SuppressWarnings("JavadocMethod")
78  private static String makeSourceValue(int source) {
79  switch (VideoSource.getKindFromInt(CameraServerJNI.getSourceKind(source))) {
80  case kUsb:
81  return "usb:" + CameraServerJNI.getUsbCameraPath(source);
82  case kHttp: {
83  String[] urls = CameraServerJNI.getHttpCameraUrls(source);
84  if (urls.length > 0) {
85  return "ip:" + urls[0];
86  } else {
87  return "ip:";
88  }
89  }
90  case kCv:
91  return "cv:";
92  default:
93  return "unknown:";
94  }
95  }

102  @SuppressWarnings({"JavadocMethod", "PMD.AvoidUsingHardCodedIP"})
103  private synchronized String[] getSinkStreamValues(int sink) {
104  // Ignore all but MjpegServer
105  if (VideoSink.getKindFromInt(CameraServerJNI.getSinkKind(sink)) != VideoSink.Kind.kMjpeg) {
106  return new String[0];
107  }
108 
109  // Get port
110  int port = CameraServerJNI.getMjpegServerPort(sink);
111 
112  // Generate values
113  ArrayList<String> values = new ArrayList<>(m_addresses.length + 1);
114  String listenAddress = CameraServerJNI.getMjpegServerListenAddress(sink);
115  if (!listenAddress.isEmpty()) {
116  // If a listen address is specified, only use that
117  values.add(makeStreamValue(listenAddress, port));
118  } else {
119  // Otherwise generate for hostname and all interface addresses
120  values.add(makeStreamValue(CameraServerJNI.getHostname() + ".local", port));
121  for (String addr : m_addresses) {
122  if ("127.0.0.1".equals(addr)) {
123  continue; // ignore localhost
124  }
125  values.add(makeStreamValue(addr, port));
126  }
127  }
128 
129  return values.toArray(new String[0]);
130  }

132  @SuppressWarnings({"JavadocMethod", "PMD.AvoidUsingHardCodedIP"})
133  private synchronized String[] getSourceStreamValues(int source) {
134  // Ignore all but HttpCamera
135  if (VideoSource.getKindFromInt(CameraServerJNI.getSourceKind(source))
136  != VideoSource.Kind.kHttp) {
137  return new String[0];
138  }
139 
140  // Generate values
141  String[] values = CameraServerJNI.getHttpCameraUrls(source);
142  for (int j = 0; j < values.length; j++) {
143  values[j] = "mjpg:" + values[j];
144  }
145 
146  if (CameraServerSharedStore.getCameraServerShared().isRoboRIO()) {
147  // Look to see if we have a passthrough server for this source
148  // Only do this on the roboRIO
149  for (VideoSink i : m_sinks.values()) {
150  int sink = i.getHandle();
151  int sinkSource = CameraServerJNI.getSinkSource(sink);
152  if (source == sinkSource
153  && VideoSink.getKindFromInt(CameraServerJNI.getSinkKind(sink))
154  == VideoSink.Kind.kMjpeg) {
155  // Add USB-only passthrough
156  String[] finalValues = Arrays.copyOf(values, values.length + 1);
157  int port = CameraServerJNI.getMjpegServerPort(sink);
158  finalValues[values.length] = makeStreamValue("172.22.11.2", port);
159  return finalValues;
160  }
161  }
162  }
163 
164  return values;
165  }

167  @SuppressWarnings({"JavadocMethod", "PMD.AvoidUsingHardCodedIP", "PMD.CyclomaticComplexity"})
168  private synchronized void updateStreamValues() {
169  // Over all the sinks...
170  for (VideoSink i : m_sinks.values()) {
171  int sink = i.getHandle();
172 
173  // Get the source's subtable (if none exists, we're done)
174  int source = Objects.requireNonNullElseGet(m_fixedSources.get(sink),
175  () -> CameraServerJNI.getSinkSource(sink));
176 
177  if (source == 0) {
178  continue;
179  }
180  NetworkTable table = m_tables.get(source);
181  if (table != null) {
182  // Don't set stream values if this is a HttpCamera passthrough
183  if (VideoSource.getKindFromInt(CameraServerJNI.getSourceKind(source))
184  == VideoSource.Kind.kHttp) {
185  continue;
186  }
187 
188  // Set table value
189  String[] values = getSinkStreamValues(sink);
190  if (values.length > 0) {
191  table.getEntry("streams").setStringArray(values);
192  }
193  }
194  }
195 
196  // Over all the sources...
197  for (VideoSource i : m_sources.values()) {
198  int source = i.getHandle();
199 
200  // Get the source's subtable (if none exists, we're done)
201  NetworkTable table = m_tables.get(source);
202  if (table != null) {
203  // Set table value
204  String[] values = getSourceStreamValues(source);
205  if (values.length > 0) {
206  table.getEntry("streams").setStringArray(values);
207  }
208  }
209  }

248  @SuppressWarnings({"JavadocMethod", "PMD.CyclomaticComplexity"})
249  private static void putSourcePropertyValue(NetworkTable table, VideoEvent event, boolean isNew) {
250  String name;
251  String infoName;
252  if (event.name.startsWith("raw_")) {
253  name = "RawProperty/" + event.name;
254  infoName = "RawPropertyInfo/" + event.name;
255  } else {
256  name = "Property/" + event.name;
257  infoName = "PropertyInfo/" + event.name;
258  }
259 
260  NetworkTableEntry entry = table.getEntry(name);
261  try {
262  switch (event.propertyKind) {
263  case kBoolean:
264  if (isNew) {
265  entry.setDefaultBoolean(event.value != 0);
266  } else {
267  entry.setBoolean(event.value != 0);
268  }
269  break;
270  case kInteger:
271  case kEnum:
272  if (isNew) {
273  entry.setDefaultDouble(event.value);
274  table.getEntry(infoName + "/min").setDouble(
275  CameraServerJNI.getPropertyMin(event.propertyHandle));
276  table.getEntry(infoName + "/max").setDouble(
277  CameraServerJNI.getPropertyMax(event.propertyHandle));
278  table.getEntry(infoName + "/step").setDouble(
279  CameraServerJNI.getPropertyStep(event.propertyHandle));
280  table.getEntry(infoName + "/default").setDouble(
281  CameraServerJNI.getPropertyDefault(event.propertyHandle));
282  } else {
283  entry.setDouble(event.value);
284  }
285  break;
286  case kString:
287  if (isNew) {
288  entry.setDefaultString(event.valueStr);
289  } else {
290  entry.setString(event.valueStr);
291  }
292  break;
293  default:
294  break;
295  }
296  } catch (VideoException ignored) {
297  // ignore
298  }
299  }

302  "PMD.NPathComplexity"})
303  private CameraServer() {
304  m_defaultUsbDevice = new AtomicInteger();
305  m_sources = new HashMap<>();
306  m_sinks = new HashMap<>();
307  m_fixedSources = new HashMap<>();
308  m_tables = new HashMap<>();
309  m_publishTable = NetworkTableInstance.getDefault().getTable(kPublishName);
310  m_nextPort = kBasePort;
311  m_addresses = new String[0];
312 
313  // We publish sources to NetworkTables using the following structure:
314  // "/CameraPublisher/{Source.Name}/" - root
315  // - "source" (string): Descriptive, prefixed with type (e.g. "usb:0")
316  // - "streams" (string array): URLs that can be used to stream data
317  // - "description" (string): Description of the source
318  // - "connected" (boolean): Whether source is connected
319  // - "mode" (string): Current video mode
320  // - "modes" (string array): Available video modes
321  // - "Property/{Property}" - Property values
322  // - "PropertyInfo/{Property}" - Property supporting information
323 
324  // Listener for video events
325  m_videoListener = new VideoListener(event -> {
326  switch (event.kind) {
327  case kSourceCreated: {
328  // Create subtable for the camera
329  NetworkTable table = m_publishTable.getSubTable(event.name);
330  m_tables.put(event.sourceHandle, table);
331  table.getEntry("source").setString(makeSourceValue(event.sourceHandle));
332  table.getEntry("description").setString(
333  CameraServerJNI.getSourceDescription(event.sourceHandle));
334  table.getEntry("connected").setBoolean(
335  CameraServerJNI.isSourceConnected(event.sourceHandle));
336  table.getEntry("streams").setStringArray(getSourceStreamValues(event.sourceHandle));
337  try {
338  VideoMode mode = CameraServerJNI.getSourceVideoMode(event.sourceHandle);
339  table.getEntry("mode").setDefaultString(videoModeToString(mode));
340  table.getEntry("modes").setStringArray(getSourceModeValues(event.sourceHandle));
341  } catch (VideoException ignored) {
342  // Do nothing. Let the other event handlers update this if there is an error.
343  }
344  break;
345  }
346  case kSourceDestroyed: {
347  NetworkTable table = m_tables.get(event.sourceHandle);
348  if (table != null) {
349  table.getEntry("source").setString("");
350  table.getEntry("streams").setStringArray(new String[0]);
351  table.getEntry("modes").setStringArray(new String[0]);
352  }
353  break;
354  }
355  case kSourceConnected: {
356  NetworkTable table = m_tables.get(event.sourceHandle);
357  if (table != null) {
358  // update the description too (as it may have changed)
359  table.getEntry("description").setString(
360  CameraServerJNI.getSourceDescription(event.sourceHandle));
361  table.getEntry("connected").setBoolean(true);
362  }
363  break;
364  }
365  case kSourceDisconnected: {
366  NetworkTable table = m_tables.get(event.sourceHandle);
367  if (table != null) {
368  table.getEntry("connected").setBoolean(false);
369  }
370  break;
371  }
372  case kSourceVideoModesUpdated: {
373  NetworkTable table = m_tables.get(event.sourceHandle);
374  if (table != null) {
375  table.getEntry("modes").setStringArray(getSourceModeValues(event.sourceHandle));
376  }
377  break;
378  }
379  case kSourceVideoModeChanged: {
380  NetworkTable table = m_tables.get(event.sourceHandle);
381  if (table != null) {
382  table.getEntry("mode").setString(videoModeToString(event.mode));
383  }
384  break;
385  }
386  case kSourcePropertyCreated: {
387  NetworkTable table = m_tables.get(event.sourceHandle);
388  if (table != null) {
389  putSourcePropertyValue(table, event, true);
390  }
391  break;
392  }
393  case kSourcePropertyValueUpdated: {
394  NetworkTable table = m_tables.get(event.sourceHandle);
395  if (table != null) {
396  putSourcePropertyValue(table, event, false);
397  }
398  break;
399  }
400  case kSourcePropertyChoicesUpdated: {
401  NetworkTable table = m_tables.get(event.sourceHandle);
402  if (table != null) {
403  try {
404  String[] choices = CameraServerJNI.getEnumPropertyChoices(event.propertyHandle);
405  table.getEntry("PropertyInfo/" + event.name + "/choices").setStringArray(choices);
406  } catch (VideoException ignored) {
407  // ignore
408  }
409  }
410  break;
411  }
412  case kSinkSourceChanged:
413  case kSinkCreated:
414  case kSinkDestroyed:
415  case kNetworkInterfacesChanged: {
416  m_addresses = CameraServerJNI.getNetworkInterfaces();
417  updateStreamValues();
418  break;
419  }
420  default:
421  break;
422  }
423  }, 0x4fff, true);
424 
425  // Listener for NetworkTable events
426  // We don't currently support changing settings via NT due to
427  // synchronization issues, so just update to current setting if someone
428  // else tries to change it.
429  m_tableListener = NetworkTableInstance.getDefault().addEntryListener(kPublishName + "/",
430  event -> {
431  String relativeKey = event.name.substring(kPublishName.length() + 1);
432 
433  // get source (sourceName/...)
434  int subKeyIndex = relativeKey.indexOf('/');
435  if (subKeyIndex == -1) {
436  return;
437  }
438  String sourceName = relativeKey.substring(0, subKeyIndex);
439  VideoSource source = m_sources.get(sourceName);
440  if (source == null) {
441  return;
442  }
443 
444  // get subkey
445  relativeKey = relativeKey.substring(subKeyIndex + 1);
446 
447  // handle standard names
448  String propName;
449  if ("mode".equals(relativeKey)) {
450  // reset to current mode
451  event.getEntry().setString(videoModeToString(source.getVideoMode()));
452  return;
453  } else if (relativeKey.startsWith("Property/")) {
454  propName = relativeKey.substring(9);
455  } else if (relativeKey.startsWith("RawProperty/")) {
456  propName = relativeKey.substring(12);
457  } else {
458  return; // ignore
459  }
460 
461  // everything else is a property
462  VideoProperty property = source.getProperty(propName);
463  switch (property.getKind()) {
464  case kNone:
465  return;
466  case kBoolean:
467  // reset to current setting
468  event.getEntry().setBoolean(property.get() != 0);
469  return;
470  case kInteger:
471  case kEnum:
472  // reset to current setting
473  event.getEntry().setDouble(property.get());
474  return;
475  case kString:
476  // reset to current setting
477  event.getEntry().setString(property.getString());
478  return;
479  default:
480  return;
481  }
482  }, EntryListenerFlags.kImmediate | EntryListenerFlags.kUpdate);
483  }
