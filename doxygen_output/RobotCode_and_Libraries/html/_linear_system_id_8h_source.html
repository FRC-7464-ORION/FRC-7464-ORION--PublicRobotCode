46  static LinearSystem<2, 1, 1> ElevatorSystem(DCMotor motor,
47  units::kilogram_t m,
48  units::meter_t r, double G) {
49  if (m <= 0_kg) {
50  throw std::domain_error("m must be greater than zero.");
51  }
52  if (r <= 0_m) {
53  throw std::domain_error("r must be greater than zero.");
54  }
55  if (G <= 0.0) {
56  throw std::domain_error("G must be greater than zero.");
57  }
58 
59  Eigen::Matrix<double, 2, 2> A{
60  {0.0, 1.0},
61  {0.0, (-std::pow(G, 2) * motor.Kt /
62  (motor.R * units::math::pow<2>(r) * m * motor.Kv))
63  .value()}};
64  Eigen::Matrix<double, 2, 1> B{0.0,
65  (G * motor.Kt / (motor.R * r * m)).value()};
66  Eigen::Matrix<double, 1, 2> C{1.0, 0.0};
67  Eigen::Matrix<double, 1, 1> D{0.0};
68 
69  return LinearSystem<2, 1, 1>(A, B, C, D);
70  }

204  static LinearSystem<2, 2, 2> IdentifyDrivetrainSystem(
205  decltype(1_V / 1_mps) kVlinear, decltype(1_V / 1_mps_sq) kAlinear,
206  decltype(1_V / 1_mps) kVangular, decltype(1_V / 1_mps_sq) kAangular) {
207  if (kVlinear <= decltype(kVlinear){0}) {
208  throw std::domain_error("Kv,linear must be greater than zero.");
209  }
210  if (kAlinear <= decltype(kAlinear){0}) {
211  throw std::domain_error("Ka,linear must be greater than zero.");
212  }
213  if (kVangular <= decltype(kVangular){0}) {
214  throw std::domain_error("Kv,angular must be greater than zero.");
215  }
216  if (kAangular <= decltype(kAangular){0}) {
217  throw std::domain_error("Ka,angular must be greater than zero.");
218  }
219 
220  double A1 = -(kVlinear.value() / kAlinear.value() +
221  kVangular.value() / kAangular.value());
222  double A2 = -(kVlinear.value() / kAlinear.value() -
223  kVangular.value() / kAangular.value());
224  double B1 = 1.0 / kAlinear.value() + 1.0 / kAangular.value();
225  double B2 = 1.0 / kAlinear.value() - 1.0 / kAangular.value();
226 
227  Eigen::Matrix<double, 2, 2> A =
228  0.5 * Eigen::Matrix<double, 2, 2>{{A1, A2}, {A2, A1}};
229  Eigen::Matrix<double, 2, 2> B =
230  0.5 * Eigen::Matrix<double, 2, 2>{{B1, B2}, {B2, B1}};
231  Eigen::Matrix<double, 2, 2> C{{1.0, 0.0}, {0.0, 1.0}};
232  Eigen::Matrix<double, 2, 2> D{{0.0, 0.0}, {0.0, 0.0}};
233 
234  return LinearSystem<2, 2, 2>(A, B, C, D);
235  }

256  static LinearSystem<2, 2, 2> IdentifyDrivetrainSystem(
257  decltype(1_V / 1_mps) kVlinear, decltype(1_V / 1_mps_sq) kAlinear,
258  decltype(1_V / 1_rad_per_s) kVangular,
259  decltype(1_V / 1_rad_per_s_sq) kAangular, units::meter_t trackwidth) {
260  if (kVlinear <= decltype(kVlinear){0}) {
261  throw std::domain_error("Kv,linear must be greater than zero.");
262  }
263  if (kAlinear <= decltype(kAlinear){0}) {
264  throw std::domain_error("Ka,linear must be greater than zero.");
265  }
266  if (kVangular <= decltype(kVangular){0}) {
267  throw std::domain_error("Kv,angular must be greater than zero.");
268  }
269  if (kAangular <= decltype(kAangular){0}) {
270  throw std::domain_error("Ka,angular must be greater than zero.");
271  }
272  if (trackwidth <= 0_m) {
273  throw std::domain_error("r must be greater than zero.");
274  }
275 
276  // We want to find a factor to include in Kv,angular that will convert
277  // `u = Kv,angular omega` to `u = Kv,angular v`.
278  //
279  // v = omega r
280  // omega = v/r
281  // omega = 1/r v
282  // omega = 1/(trackwidth/2) v
283  // omega = 2/trackwidth v
284  //
285  // So multiplying by 2/trackwidth converts the angular gains from V/(rad/s)
286  // to V/(m/s).
287  return IdentifyDrivetrainSystem(kVlinear, kAlinear,
288  kVangular * 2.0 / trackwidth * 1_rad,
289  kAangular * 2.0 / trackwidth * 1_rad);
290  }

371  static LinearSystem<2, 2, 2> DrivetrainVelocitySystem(
372  const DCMotor& motor, units::kilogram_t m, units::meter_t r,
373  units::meter_t rb, units::kilogram_square_meter_t J, double G) {
374  if (m <= 0_kg) {
375  throw std::domain_error("m must be greater than zero.");
376  }
377  if (r <= 0_m) {
378  throw std::domain_error("r must be greater than zero.");
379  }
380  if (rb <= 0_m) {
381  throw std::domain_error("rb must be greater than zero.");
382  }
383  if (J <= 0_kg_sq_m) {
384  throw std::domain_error("J must be greater than zero.");
385  }
386  if (G <= 0.0) {
387  throw std::domain_error("G must be greater than zero.");
388  }
389 
390  auto C1 = -std::pow(G, 2) * motor.Kt /
391  (motor.Kv * motor.R * units::math::pow<2>(r));
392  auto C2 = G * motor.Kt / (motor.R * r);
393 
394  Eigen::Matrix<double, 2, 2> A{
395  {((1 / m + units::math::pow<2>(rb) / J) * C1).value(),
396  ((1 / m - units::math::pow<2>(rb) / J) * C1).value()},
397  {((1 / m - units::math::pow<2>(rb) / J) * C1).value(),
398  ((1 / m + units::math::pow<2>(rb) / J) * C1).value()}};
399  Eigen::Matrix<double, 2, 2> B{
400  {((1 / m + units::math::pow<2>(rb) / J) * C2).value(),
401  ((1 / m - units::math::pow<2>(rb) / J) * C2).value()},
402  {((1 / m - units::math::pow<2>(rb) / J) * C2).value(),
403  ((1 / m + units::math::pow<2>(rb) / J) * C2).value()}};
404  Eigen::Matrix<double, 2, 2> C{{1.0, 0.0}, {0.0, 1.0}};
405  Eigen::Matrix<double, 2, 2> D{{0.0, 0.0}, {0.0, 0.0}};
406 
407  return LinearSystem<2, 2, 2>(A, B, C, D);
408  }
