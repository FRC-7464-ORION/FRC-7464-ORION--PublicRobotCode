14 std::vector<CubicHermiteSpline> SplineHelper::CubicSplinesFromControlVectors(
15  const Spline<3>::ControlVector& start, std::vector<Translation2d> waypoints,
16  const Spline<3>::ControlVector& end) {
17  std::vector<CubicHermiteSpline> splines;
18 
19  std::array<double, 2> xInitial = start.x;
20  std::array<double, 2> yInitial = start.y;
21  std::array<double, 2> xFinal = end.x;
22  std::array<double, 2> yFinal = end.y;
23 
24  if (waypoints.size() > 1) {
25  waypoints.emplace(waypoints.begin(),
26  Translation2d{units::meter_t(xInitial[0]),
27  units::meter_t(yInitial[0])});
28  waypoints.emplace_back(
29  Translation2d{units::meter_t(xFinal[0]), units::meter_t(yFinal[0])});
30 
31  // Populate tridiagonal system for clamped cubic
32  /* See:
33  https://www.uio.no/studier/emner/matnat/ifi/nedlagte-emner/INF-MAT4350/h08
34  /undervisningsmateriale/chap7alecture.pdf
35  */
36 
37  // Above-diagonal of tridiagonal matrix, zero-padded
38  std::vector<double> a;
39  // Diagonal of tridiagonal matrix
40  std::vector<double> b(waypoints.size() - 2, 4.0);
41  // Below-diagonal of tridiagonal matrix, zero-padded
42  std::vector<double> c;
43  // rhs vectors
44  std::vector<double> dx, dy;
45  // solution vectors
46  std::vector<double> fx(waypoints.size() - 2, 0.0),
47  fy(waypoints.size() - 2, 0.0);
48 
49  // populate above-diagonal and below-diagonal vectors
50  a.emplace_back(0);
51  for (size_t i = 0; i < waypoints.size() - 3; ++i) {
52  a.emplace_back(1);
53  c.emplace_back(1);
54  }
55  c.emplace_back(0);
56 
57  // populate rhs vectors
58  dx.emplace_back(
59  3 * (waypoints[2].X().to<double>() - waypoints[0].X().to<double>()) -
60  xInitial[1]);
61  dy.emplace_back(
62  3 * (waypoints[2].Y().to<double>() - waypoints[0].Y().to<double>()) -
63  yInitial[1]);
64  if (waypoints.size() > 4) {
65  for (size_t i = 1; i <= waypoints.size() - 4; ++i) {
66  dx.emplace_back(3 * (waypoints[i + 1].X().to<double>() -
67  waypoints[i - 1].X().to<double>()));
68  dy.emplace_back(3 * (waypoints[i + 1].Y().to<double>() -
69  waypoints[i - 1].Y().to<double>()));
70  }
71  }
72  dx.emplace_back(3 * (waypoints[waypoints.size() - 1].X().to<double>() -
73  waypoints[waypoints.size() - 3].X().to<double>()) -
74  xFinal[1]);
75  dy.emplace_back(3 * (waypoints[waypoints.size() - 1].Y().to<double>() -
76  waypoints[waypoints.size() - 3].Y().to<double>()) -
77  yFinal[1]);
78 
79  // Compute solution to tridiagonal system
80  ThomasAlgorithm(a, b, c, dx, &fx);
81  ThomasAlgorithm(a, b, c, dy, &fy);
82 
83  fx.emplace(fx.begin(), xInitial[1]);
84  fx.emplace_back(xFinal[1]);
85  fy.emplace(fy.begin(), yInitial[1]);
86  fy.emplace_back(yFinal[1]);
87 
88  for (size_t i = 0; i < fx.size() - 1; ++i) {
89  // Create the spline.
90  const CubicHermiteSpline spline{
91  {waypoints[i].X().to<double>(), fx[i]},
92  {waypoints[i + 1].X().to<double>(), fx[i + 1]},
93  {waypoints[i].Y().to<double>(), fy[i]},
94  {waypoints[i + 1].Y().to<double>(), fy[i + 1]}};
95 
96  splines.push_back(spline);
97  }
98  } else if (waypoints.size() == 1) {
99  const double xDeriv =
100  (3 * (xFinal[0] - xInitial[0]) - xFinal[1] - xInitial[1]) / 4.0;
101  const double yDeriv =
102  (3 * (yFinal[0] - yInitial[0]) - yFinal[1] - yInitial[1]) / 4.0;
103 
104  std::array<double, 2> midXControlVector{waypoints[0].X().to<double>(),
105  xDeriv};
106  std::array<double, 2> midYControlVector{waypoints[0].Y().to<double>(),
107  yDeriv};
108 
109  splines.emplace_back(xInitial, midXControlVector, yInitial,
110  midYControlVector);
111  splines.emplace_back(midXControlVector, xFinal, midYControlVector, yFinal);
112 
113  } else {
114  // Create the spline.
115  const CubicHermiteSpline spline{xInitial, xFinal, yInitial, yFinal};
116  splines.push_back(spline);
117  }
118 
119  return splines;
120 }

122 std::vector<QuinticHermiteSpline>
123 SplineHelper::QuinticSplinesFromControlVectors(
124  const std::vector<Spline<5>::ControlVector>& controlVectors) {
125  std::vector<QuinticHermiteSpline> splines;
126  for (size_t i = 0; i < controlVectors.size() - 1; ++i) {
127  auto& xInitial = controlVectors[i].x;
128  auto& yInitial = controlVectors[i].y;
129  auto& xFinal = controlVectors[i + 1].x;
130  auto& yFinal = controlVectors[i + 1].y;
131  splines.emplace_back(xInitial, xFinal, yInitial, yFinal);
132  }
133  return splines;
134 }

136 std::array<Spline<3>::ControlVector, 2>
137 SplineHelper::CubicControlVectorsFromWaypoints(
138  const Pose2d& start, const std::vector<Translation2d>& interiorWaypoints,
139  const Pose2d& end) {
140  double scalar;
141  if (interiorWaypoints.empty()) {
142  scalar = 1.2 * start.Translation().Distance(end.Translation()).to<double>();
143  } else {
144  scalar =
145  1.2 *
146  start.Translation().Distance(interiorWaypoints.front()).to<double>();
147  }
148  const auto initialCV = CubicControlVector(scalar, start);
149  if (!interiorWaypoints.empty()) {
150  scalar =
151  1.2 * end.Translation().Distance(interiorWaypoints.back()).to<double>();
152  }
153  const auto finalCV = CubicControlVector(scalar, end);
154  return {initialCV, finalCV};
155 }

157 std::vector<Spline<5>::ControlVector>
158 SplineHelper::QuinticControlVectorsFromWaypoints(
159  const std::vector<Pose2d>& waypoints) {
160  std::vector<Spline<5>::ControlVector> vectors;
161  for (size_t i = 0; i < waypoints.size() - 1; ++i) {
162  auto& p0 = waypoints[i];
163  auto& p1 = waypoints[i + 1];
164 
165  // This just makes the splines look better.
166  const auto scalar =
167  1.2 * p0.Translation().Distance(p1.Translation()).to<double>();
168 
169  // Only add the first control vector if this is the first iteration. The
170  // last control vector of this iteration is the first control vector of
171  // the next iteration.
172  if (i == 0) {
173  vectors.push_back(QuinticControlVector(scalar, p0));
174  }
175  vectors.push_back(QuinticControlVector(scalar, p1));
176  }
177  return vectors;
178 }

180 void SplineHelper::ThomasAlgorithm(const std::vector<double>& a,
181  const std::vector<double>& b,
182  const std::vector<double>& c,
183  const std::vector<double>& d,
184  std::vector<double>* solutionVector) {
185  auto& f = *solutionVector;
186  size_t N = d.size();
187 
188  // Create the temporary vectors
189  // Note that this is inefficient as it is possible to call
190  // this function many times. A better implementation would
191  // pass these temporary matrices by non-const reference to
192  // save excess allocation and deallocation
193  std::vector<double> c_star(N, 0.0);
194  std::vector<double> d_star(N, 0.0);
195 
196  // This updates the coefficients in the first row
197  // Note that we should be checking for division by zero here
198  c_star[0] = c[0] / b[0];
199  d_star[0] = d[0] / b[0];
200 
201  // Create the c_star and d_star coefficients in the forward sweep
202  for (size_t i = 1; i < N; ++i) {
203  double m = 1.0 / (b[i] - a[i] * c_star[i - 1]);
204  c_star[i] = c[i] * m;
205  d_star[i] = (d[i] - a[i] * d_star[i - 1]) * m;
206  }
207 
208  f[N - 1] = d_star[N - 1];
209  // This is the reverse sweep, used to update the solution vector f
210  for (int i = N - 2; i >= 0; i--) {
211  f[i] = d_star[i] - c_star[i] * f[i + 1];
212  }
213 }
