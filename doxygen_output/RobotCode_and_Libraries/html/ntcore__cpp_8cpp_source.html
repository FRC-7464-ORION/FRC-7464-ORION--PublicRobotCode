709 std::string PackRpcDefinition(const RpcDefinition& def) {
710  WireEncoder enc(0x0300);
711  enc.Write8(def.version);
712  enc.WriteString(def.name);
713 
714  // parameters
715  unsigned int params_size = def.params.size();
716  if (params_size > 0xff) {
717  params_size = 0xff;
718  }
719  enc.Write8(params_size);
720  for (size_t i = 0; i < params_size; ++i) {
721  enc.WriteType(def.params[i].def_value->type());
722  enc.WriteString(def.params[i].name);
723  enc.WriteValue(*def.params[i].def_value);
724  }
725 
726  // results
727  unsigned int results_size = def.results.size();
728  if (results_size > 0xff) {

740 bool UnpackRpcDefinition(std::string_view packed, RpcDefinition* def) {
741  wpi::raw_mem_istream is(packed.data(), packed.size());
742  wpi::Logger logger;
743  WireDecoder dec(is, 0x0300, logger);
744  if (!dec.Read8(&def->version)) {
745  return false;
746  }
747  if (!dec.ReadString(&def->name)) {
748  return false;
749  }
750 
751  // parameters
752  unsigned int params_size;
753  if (!dec.Read8(&params_size)) {
754  return false;
755  }
756  def->params.resize(0);
757  def->params.reserve(params_size);
758  for (size_t i = 0; i < params_size; ++i) {
759  RpcParamDef pdef;
760  NT_Type type;
761  if (!dec.ReadType(&type)) {
762  return false;
763  }
764  if (!dec.ReadString(&pdef.name)) {
765  return false;
766  }
767  pdef.def_value = dec.ReadValue(type);
768  if (!pdef.def_value) {
769  return false;
770  }
771  def->params.emplace_back(std::move(pdef));
772  }
773 
774  // results
775  unsigned int results_size;
776  if (!dec.Read8(&results_size)) {
777  return false;
778  }
779  def->results.resize(0);
780  def->results.reserve(results_size);
781  for (size_t i = 0; i < results_size; ++i) {
782  RpcResultDef rdef;
783  if (!dec.ReadType(&rdef.type)) {
784  return false;
785  }
786  if (!dec.ReadString(&rdef.name)) {
787  return false;
788  }
789  def->results.emplace_back(std::move(rdef));
790  }
791 
792  return true;
793 }
