132 Eigen::Vector<double, 7> DifferentialDrivetrainSim::Dynamics(
133  const Eigen::Vector<double, 7>& x, const Eigen::Vector<double, 2>& u) {
134  // Because G^2 can be factored out of A, we can divide by the old ratio
135  // squared and multiply by the new ratio squared to get a new drivetrain
136  // model.
137  Eigen::Matrix<double, 4, 2> B;
138  B.block<2, 2>(0, 0) = m_plant.B() * m_currentGearing * m_currentGearing /
139  m_originalGearing / m_originalGearing;
140  B.block<2, 2>(2, 0).setZero();
141 
142  // Because G can be factored out of B, we can divide by the old ratio and
143  // multiply by the new ratio to get a new drivetrain model.
144  Eigen::Matrix<double, 4, 4> A;
145  A.block<2, 2>(0, 0) = m_plant.A() * m_currentGearing / m_originalGearing;
146 
147  A.block<2, 2>(2, 0).setIdentity();
148  A.block<4, 2>(0, 2).setZero();
149 
150  double v = (x(State::kLeftVelocity) + x(State::kRightVelocity)) / 2.0;
151 
152  Eigen::Vector<double, 7> xdot;
153  xdot(0) = v * std::cos(x(State::kHeading));
154  xdot(1) = v * std::sin(x(State::kHeading));
155  xdot(2) =
156  ((x(State::kRightVelocity) - x(State::kLeftVelocity)) / (2.0 * m_rb))
157  .value();
158  xdot.block<4, 1>(3, 0) = A * x.block<4, 1>(3, 0) + B * u;
159  return xdot;
160 }
