295  * Signature: (ID[I[I)J
296  */
297 JNIEXPORT jlong JNICALL
298 Java_edu_wpi_first_hal_DMAJNI_readDMA
299  (JNIEnv* env, jclass, jint handle, jdouble timeoutSeconds, jintArray buf,
300  jintArray store)
301 {
302  int32_t status = 0;
303  HAL_DMASample dmaSample;
304  std::memset(&dmaSample, 0, sizeof(dmaSample));
305  int32_t remaining = 0;
306  HAL_DMAReadStatus readStatus =
307  HAL_ReadDMA(handle, &dmaSample, timeoutSeconds, &remaining, &status);
308  CheckStatus(env, status);
309 
310  static_assert(sizeof(uint32_t) == sizeof(jint), "Java ints must be 32 bits");
311 
312  env->SetIntArrayRegion(buf, 0, dmaSample.captureSize,
313  reinterpret_cast<jint*>(dmaSample.readBuffer));
314 
315  int32_t* nativeArr =
316  static_cast<int32_t*>(env->GetPrimitiveArrayCritical(store, nullptr));
317 
318  std::copy_n(
319  dmaSample.channelOffsets,
320  sizeof(dmaSample.channelOffsets) / sizeof(dmaSample.channelOffsets[0]),
321  nativeArr);
322  nativeArr[22] = static_cast<int32_t>(dmaSample.captureSize);
323  nativeArr[23] = static_cast<int32_t>(dmaSample.triggerChannels);
324  nativeArr[24] = remaining;
325  nativeArr[25] = readStatus;
326 
327  env->ReleasePrimitiveArrayCritical(store, nativeArr, JNI_ABORT);
328 
329  return dmaSample.timeStamp;
330 }

360  * Method: getSensorReadData
361  * Signature: (I)Ljava/lang/Object;
362  */
363 JNIEXPORT jobject JNICALL
364 Java_edu_wpi_first_hal_DMAJNI_getSensorReadData
365  (JNIEnv* env, jclass, jint handle)
366 {
367  HAL_Handle halHandle = static_cast<HAL_Handle>(handle);
368 
369  // Check for encoder/counter handle
370  HAL_FPGAEncoderHandle fpgaEncoderHandle = 0;
371  HAL_CounterHandle counterHandle = 0;
372  bool validEncoderHandle =
373  hal::GetEncoderBaseHandle(halHandle, &fpgaEncoderHandle, &counterHandle);
374 
375  if (validEncoderHandle) {
376  if (counterHandle != HAL_kInvalidHandle) {
377  int32_t cindex = getHandleIndex(counterHandle);
378  if (cindex < 4) {
379  return CreateDMABaseStore(env, kEnable_Counters_Low, cindex);
380  } else {
381  return CreateDMABaseStore(env, kEnable_Counters_High, cindex - 4);
382  }
383  } else {
384  int32_t cindex = getHandleIndex(fpgaEncoderHandle);
385  if (cindex < 4) {
386  return CreateDMABaseStore(env, kEnable_Encoders_Low, cindex);
387  } else {
388  return CreateDMABaseStore(env, kEnable_Encoders_High, cindex - 4);
389  }
390  }
391  }
392 
393  HAL_HandleEnum handleType = getHandleType(halHandle);
394  int32_t index = getHandleIndex(halHandle);
395  if (handleType == HAL_HandleEnum::DIO) {
396  return CreateDMABaseStore(env, kEnable_DI, index);
397  } else if (handleType == HAL_HandleEnum::AnalogTrigger) {
398  return CreateDMABaseStore(env, kEnable_AnalogTriggers, index);
399  } else if (handleType == HAL_HandleEnum::AnalogInput) {
400  if (index < 4) {
401  return CreateDMABaseStore(env, kEnable_AI0_Low, index);
402  } else {
403  return CreateDMABaseStore(env, kEnable_AI0_High, index - 4);
404  }
405  } else if (handleType == HAL_HandleEnum::DutyCycle) {
406  if (index < 4) {
407  return CreateDMABaseStore(env, kEnable_DutyCycle_Low, index);
408  } else {
409  return CreateDMABaseStore(env, kEnable_DutyCycle_High, index - 4);
410  }
411  } else {
412  return nullptr;
413  }
414 }
