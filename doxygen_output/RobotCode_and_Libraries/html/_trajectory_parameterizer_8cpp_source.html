36 Trajectory TrajectoryParameterizer::TimeParameterizeTrajectory(
37  const std::vector<PoseWithCurvature>& points,
38  const std::vector<std::unique_ptr<TrajectoryConstraint>>& constraints,
39  units::meters_per_second_t startVelocity,
40  units::meters_per_second_t endVelocity,
41  units::meters_per_second_t maxVelocity,
42  units::meters_per_second_squared_t maxAcceleration, bool reversed) {
43  std::vector<ConstrainedState> constrainedStates(points.size());
44 
45  ConstrainedState predecessor{points.front(), 0_m, startVelocity,
46  -maxAcceleration, maxAcceleration};
47 
48  constrainedStates[0] = predecessor;
49 
50  // Forward pass
51  for (unsigned int i = 0; i < points.size(); i++) {
52  auto& constrainedState = constrainedStates[i];
53  constrainedState.pose = points[i];
54 
55  // Begin constraining based on predecessor
56  units::meter_t ds = constrainedState.pose.first.Translation().Distance(
57  predecessor.pose.first.Translation());
58  constrainedState.distance = ds + predecessor.distance;
59 
60  // We may need to iterate to find the maximum end velocity and common
61  // acceleration, since acceleration limits may be a function of velocity.
62  while (true) {
63  // Enforce global max velocity and max reachable velocity by global
64  // acceleration limit. vf = std::sqrt(vi^2 + 2*a*d).
65 
66  constrainedState.maxVelocity = units::math::min(
67  maxVelocity,
68  units::math::sqrt(predecessor.maxVelocity * predecessor.maxVelocity +
69  predecessor.maxAcceleration * ds * 2.0));
70 
71  constrainedState.minAcceleration = -maxAcceleration;
72  constrainedState.maxAcceleration = maxAcceleration;
73 
74  // At this point, the constrained state is fully constructed apart from
75  // all the custom-defined user constraints.
76  for (const auto& constraint : constraints) {
77  constrainedState.maxVelocity = units::math::min(
78  constrainedState.maxVelocity,
79  constraint->MaxVelocity(constrainedState.pose.first,
80  constrainedState.pose.second,
81  constrainedState.maxVelocity));
82  }
83 
84  // Now enforce all acceleration limits.
85  EnforceAccelerationLimits(reversed, constraints, &constrainedState);
86 
87  if (ds.to<double>() < kEpsilon) break;
88 
89  // If the actual acceleration for this state is higher than the max
90  // acceleration that we applied, then we need to reduce the max
91  // acceleration of the predecessor and try again.
92  units::meters_per_second_squared_t actualAcceleration =
93  (constrainedState.maxVelocity * constrainedState.maxVelocity -
94  predecessor.maxVelocity * predecessor.maxVelocity) /
95  (ds * 2.0);
96 
97  // If we violate the max acceleration constraint, let's modify the
98  // predecessor.
99  if (constrainedState.maxAcceleration < actualAcceleration - 1E-6_mps_sq) {
100  predecessor.maxAcceleration = constrainedState.maxAcceleration;
101  } else {
102  // Constrain the predecessor's max acceleration to the current
103  // acceleration.
104  if (actualAcceleration > predecessor.minAcceleration + 1E-6_mps_sq) {
105  predecessor.maxAcceleration = actualAcceleration;
106  }
107  // If the actual acceleration is less than the predecessor's min
108  // acceleration, it will be repaired in the backward pass.
109  break;
110  }
111  }
112  predecessor = constrainedState;
113  }
114 
115  ConstrainedState successor{points.back(), constrainedStates.back().distance,
116  endVelocity, -maxAcceleration, maxAcceleration};
117 
118  // Backward pass
119  for (int i = points.size() - 1; i >= 0; i--) {
120  auto& constrainedState = constrainedStates[i];
121  units::meter_t ds =
122  constrainedState.distance - successor.distance; // negative
123 
124  while (true) {
125  // Enforce max velocity limit (reverse)
126  // vf = std::sqrt(vi^2 + 2*a*d), where vi = successor.
127  units::meters_per_second_t newMaxVelocity =
128  units::math::sqrt(successor.maxVelocity * successor.maxVelocity +
129  successor.minAcceleration * ds * 2.0);
130 
131  // No more limits to impose! This state can be finalized.
132  if (newMaxVelocity >= constrainedState.maxVelocity) break;
133 
134  constrainedState.maxVelocity = newMaxVelocity;
135 
136  // Check all acceleration constraints with the new max velocity.
137  EnforceAccelerationLimits(reversed, constraints, &constrainedState);
138 
139  if (ds.to<double>() > -kEpsilon) break;
140 
141  // If the actual acceleration for this state is lower than the min
142  // acceleration, then we need to lower the min acceleration of the
143  // successor and try again.
144  units::meters_per_second_squared_t actualAcceleration =
145  (constrainedState.maxVelocity * constrainedState.maxVelocity -
146  successor.maxVelocity * successor.maxVelocity) /
147  (ds * 2.0);
148  if (constrainedState.minAcceleration > actualAcceleration + 1E-6_mps_sq) {
149  successor.minAcceleration = constrainedState.minAcceleration;
150  } else {
151  successor.minAcceleration = actualAcceleration;
152  break;
153  }
154  }
155  successor = constrainedState;
156  }
157 
158  // Now we can integrate the constrained states forward in time to obtain our
159  // trajectory states.
160 
161  std::vector<Trajectory::State> states(points.size());
162  units::second_t t = 0_s;
163  units::meter_t s = 0_m;
164  units::meters_per_second_t v = 0_mps;
165 
166  for (unsigned int i = 0; i < constrainedStates.size(); i++) {
167  auto state = constrainedStates[i];
168 
169  // Calculate the change in position between the current state and the
170  // previous state.
171  units::meter_t ds = state.distance - s;
172 
173  // Calculate the acceleration between the current state and the previous
174  // state.
175  units::meters_per_second_squared_t accel =
176  (state.maxVelocity * state.maxVelocity - v * v) / (ds * 2);
177 
178  // Calculate dt.
179  units::second_t dt = 0_s;
180  if (i > 0) {
181  states.at(i - 1).acceleration = reversed ? -accel : accel;
182  if (units::math::abs(accel) > 1E-6_mps_sq) {
183  // v_f = v_0 + a * t
184  dt = (state.maxVelocity - v) / accel;
185  } else if (units::math::abs(v) > 1E-6_mps) {
186  // delta_x = v * t
187  dt = ds / v;
188  } else {
189  throw std::runtime_error(
190  "Something went wrong during trajectory generation.");
191  }
192  }
193 
194  v = state.maxVelocity;
195  s = state.distance;
196 
197  t += dt;
198 
199  states[i] = {t, reversed ? -v : v, reversed ? -accel : accel,
200  state.pose.first, state.pose.second};
201  }
202 
203  return Trajectory(states);
204 }

206 void TrajectoryParameterizer::EnforceAccelerationLimits(
207  bool reverse,
208  const std::vector<std::unique_ptr<TrajectoryConstraint>>& constraints,
209  ConstrainedState* state) {
210  for (auto&& constraint : constraints) {
211  double factor = reverse ? -1.0 : 1.0;
212 
213  auto minMaxAccel = constraint->MinMaxAcceleration(
214  state->pose.first, state->pose.second, state->maxVelocity * factor);
215 
216  state->minAcceleration = units::math::max(
217  state->minAcceleration,
218  reverse ? -minMaxAccel.maxAcceleration : minMaxAccel.minAcceleration);
219 
220  state->maxAcceleration = units::math::min(
221  state->maxAcceleration,
222  reverse ? -minMaxAccel.minAcceleration : minMaxAccel.maxAcceleration);
223  }
224 }
