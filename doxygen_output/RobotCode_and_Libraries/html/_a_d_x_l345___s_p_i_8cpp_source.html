63 double ADXL345_SPI::GetAcceleration(ADXL345_SPI::Axes axis) {
64  if (axis == kAxis_X && m_simX) return m_simX.Get();
65  if (axis == kAxis_Y && m_simY) return m_simY.Get();
66  if (axis == kAxis_Z && m_simZ) return m_simZ.Get();
67  uint8_t buffer[3];
68  uint8_t command[3] = {0, 0, 0};
69  command[0] = (kAddress_Read | kAddress_MultiByte | kDataRegister) +
70  static_cast<uint8_t>(axis);
71  m_spi.Transaction(command, buffer, 3);
72 
73  // Sensor is little endian... swap bytes
74  int16_t rawAccel = buffer[2] << 8 | buffer[1];
75  return rawAccel * kGsPerLSB;
76 }

78 ADXL345_SPI::AllAxes ADXL345_SPI::GetAccelerations() {
79  AllAxes data;
80  if (m_simX && m_simY && m_simZ) {
81  data.XAxis = m_simX.Get();
82  data.YAxis = m_simY.Get();
83  data.ZAxis = m_simZ.Get();
84  return data;
85  }
86 
87  uint8_t dataBuffer[7] = {0, 0, 0, 0, 0, 0, 0};
88  int16_t rawData[3];
89 
90  // Select the data address.
91  dataBuffer[0] = (kAddress_Read | kAddress_MultiByte | kDataRegister);
92  m_spi.Transaction(dataBuffer, dataBuffer, 7);
93 
94  for (int i = 0; i < 3; i++) {
95  // Sensor is little endian... swap bytes
96  rawData[i] = dataBuffer[i * 2 + 2] << 8 | dataBuffer[i * 2 + 1];
97  }
98 
99  data.XAxis = rawData[0] * kGsPerLSB;
100  data.YAxis = rawData[1] * kGsPerLSB;
101  data.ZAxis = rawData[2] * kGsPerLSB;
102 
103  return data;
104 }
