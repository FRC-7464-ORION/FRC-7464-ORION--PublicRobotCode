40 SparkMaxLimitSwitch::SparkMaxLimitSwitch(CANSparkMax& device,
41  Direction direction, Type switchType)
42  : m_device(&device), m_direction(direction) {
43  if (c_SparkMax_AttemptToSetDataPortConfig(
44  static_cast<c_SparkMax_handle>(m_device->m_sparkMaxHandle),
45  c_SparkMax_kDataPortConfigLimitSwitches) ==
46  c_REVLibError_SparkMaxDataPortAlreadyConfiguredDifferently) {
47  throw std::runtime_error(fmt::format(
48  "Limit switches cannot be used on SPARK MAX #{}, because it "
49  "has an alternate encoder configured",
50  m_device->m_deviceID));
51  }
52 
53  c_SparkMax_LimitDirection dir;
54 
55  if (direction == Direction::kForward) {
56  dir = c_SparkMax_kForward;
57  } else {
58  dir = c_SparkMax_kReverse;
59  }
60  c_SparkMax_SetLimitPolarity(
61  static_cast<c_SparkMax_handle>(m_device->m_sparkMaxHandle), dir,
62  static_cast<c_SparkMax_LimitPolarity>(switchType));
63 }

65 bool SparkMaxLimitSwitch::Get() const {
66  if (m_direction == Direction::kForward) {
67  return m_device->GetFault(CANSparkMax::FaultID::kHardLimitFwd);
68  } else {
69  return m_device->GetFault(CANSparkMax::FaultID::kHardLimitRev);

73 REVLibError SparkMaxLimitSwitch::EnableLimitSwitch(bool enable) {
74  c_SparkMax_LimitDirection dir;
75 
76  if (m_direction == Direction::kForward) {
77  dir = c_SparkMax_kForward;
78  } else {
79  dir = c_SparkMax_kReverse;
80  }
81  auto status = c_SparkMax_EnableLimitSwitch(
82  static_cast<c_SparkMax_handle>(m_device->m_sparkMaxHandle), dir,
83  enable ? 1 : 0);
84  return static_cast<REVLibError>(status);
85 }
