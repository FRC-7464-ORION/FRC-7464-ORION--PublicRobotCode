92 double PIDController::Calculate(double measurement) {
93  m_prevError = m_positionError;
94  m_positionError = GetContinuousError(m_setpoint - measurement);
95  m_velocityError = (m_positionError - m_prevError) / m_period.to<double>();
96 
97  if (m_Ki != 0) {
98  m_totalError =
99  std::clamp(m_totalError + m_positionError * m_period.to<double>(),
100  m_minimumIntegral / m_Ki, m_maximumIntegral / m_Ki);
101  }
102 
103  return m_Kp * m_positionError + m_Ki * m_totalError + m_Kd * m_velocityError;
104 }

129 double PIDController::GetContinuousError(double error) const {
130  if (m_continuous && m_inputRange > 0) {
131  error = std::fmod(error, m_inputRange);
132  if (std::abs(error) > m_inputRange / 2) {
133  if (error > 0) {
134  return error - m_inputRange;
135  } else {
136  return error + m_inputRange;
137  }
138  }
139  }
140 
141  return error;
142 }
