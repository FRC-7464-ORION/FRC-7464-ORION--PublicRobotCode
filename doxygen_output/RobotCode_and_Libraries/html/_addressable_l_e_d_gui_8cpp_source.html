58 void AddressableLEDsModel::Update() {
59  for (int i = 0; i < static_cast<int>(m_models.size()); ++i) {
60  auto& model = m_models[i];
61  if (HALSIM_GetAddressableLEDInitialized(i)) {
62  if (!model) {
63  model = std::make_unique<AddressableLEDModel>(i);
64  }
65  if (model) {
66  model->Update();
67  }
68  } else {
69  model.reset();
70  }
71  }
72 }
