97 DifferentialDrive::WheelSpeeds DifferentialDrive::ArcadeDriveIK(
98  double xSpeed, double zRotation, bool squareInputs) {
99  xSpeed = std::clamp(xSpeed, -1.0, 1.0);
100  zRotation = std::clamp(zRotation, -1.0, 1.0);
101 
102  // Square the inputs (while preserving the sign) to increase fine control
103  // while permitting full power.
104  if (squareInputs) {
105  xSpeed = std::copysign(xSpeed * xSpeed, xSpeed);
106  zRotation = std::copysign(zRotation * zRotation, zRotation);
107  }
108 
109  double leftSpeed;
110  double rightSpeed;
111 
112  double maxInput =
113  std::copysign(std::max(std::abs(xSpeed), std::abs(zRotation)), xSpeed);
114 
115  if (xSpeed >= 0.0) {
116  // First quadrant, else second quadrant
117  if (zRotation >= 0.0) {
118  leftSpeed = maxInput;
119  rightSpeed = xSpeed - zRotation;
120  } else {
121  leftSpeed = xSpeed + zRotation;
122  rightSpeed = maxInput;
123  }
124  } else {
125  // Third quadrant, else fourth quadrant
126  if (zRotation >= 0.0) {
127  leftSpeed = xSpeed + zRotation;
128  rightSpeed = maxInput;
129  } else {
130  leftSpeed = maxInput;
131  rightSpeed = xSpeed - zRotation;
132  }
133  }
134 
135  // Normalize the wheel speeds
136  double maxMagnitude = std::max(std::abs(leftSpeed), std::abs(rightSpeed));
137  if (maxMagnitude > 1.0) {
138  leftSpeed /= maxMagnitude;
139  rightSpeed /= maxMagnitude;
140  }
141 
142  return {leftSpeed, rightSpeed};
143 }

145 DifferentialDrive::WheelSpeeds DifferentialDrive::CurvatureDriveIK(
146  double xSpeed, double zRotation, bool allowTurnInPlace) {
147  xSpeed = std::clamp(xSpeed, -1.0, 1.0);
148  zRotation = std::clamp(zRotation, -1.0, 1.0);
149 
150  double leftSpeed = 0.0;
151  double rightSpeed = 0.0;
152 
153  if (allowTurnInPlace) {
154  leftSpeed = xSpeed + zRotation;
155  rightSpeed = xSpeed - zRotation;
156  } else {
157  leftSpeed = xSpeed + std::abs(xSpeed) * zRotation;
158  rightSpeed = xSpeed - std::abs(xSpeed) * zRotation;
159  }
160 
161  // Normalize wheel speeds
162  double maxMagnitude = std::max(std::abs(leftSpeed), std::abs(rightSpeed));
163  if (maxMagnitude > 1.0) {
164  leftSpeed /= maxMagnitude;
165  rightSpeed /= maxMagnitude;
166  }
167 
168  return {leftSpeed, rightSpeed};
