13 template <class Distance>
14 TrapezoidProfile<Distance>::TrapezoidProfile(Constraints constraints,
15  State goal, State initial)
16  : m_direction{ShouldFlipAcceleration(initial, goal) ? -1 : 1},
17  m_constraints(constraints),
18  m_initial(Direct(initial)),
19  m_goal(Direct(goal)) {
20  if (m_initial.velocity > m_constraints.maxVelocity) {
21  m_initial.velocity = m_constraints.maxVelocity;
22  }
23 
24  // Deal with a possibly truncated motion profile (with nonzero initial or
25  // final velocity) by calculating the parameters as if the profile began and
26  // ended at zero velocity
27  units::second_t cutoffBegin =
28  m_initial.velocity / m_constraints.maxAcceleration;
29  Distance_t cutoffDistBegin =
30  cutoffBegin * cutoffBegin * m_constraints.maxAcceleration / 2.0;
31 
32  units::second_t cutoffEnd = m_goal.velocity / m_constraints.maxAcceleration;
33  Distance_t cutoffDistEnd =
34  cutoffEnd * cutoffEnd * m_constraints.maxAcceleration / 2.0;
35 
36  // Now we can calculate the parameters as if it was a full trapezoid instead
37  // of a truncated one
38 
39  Distance_t fullTrapezoidDist =
40  cutoffDistBegin + (m_goal.position - m_initial.position) + cutoffDistEnd;
41  units::second_t accelerationTime =
42  m_constraints.maxVelocity / m_constraints.maxAcceleration;
43 
44  Distance_t fullSpeedDist =
45  fullTrapezoidDist -
46  accelerationTime * accelerationTime * m_constraints.maxAcceleration;
47 
48  // Handle the case where the profile never reaches full speed
49  if (fullSpeedDist < Distance_t(0)) {
50  accelerationTime =
51  units::math::sqrt(fullTrapezoidDist / m_constraints.maxAcceleration);
52  fullSpeedDist = Distance_t(0);
53  }
54 
55  m_endAccel = accelerationTime - cutoffBegin;
56  m_endFullSpeed = m_endAccel + fullSpeedDist / m_constraints.maxVelocity;
57  m_endDeccel = m_endFullSpeed + accelerationTime - cutoffEnd;
58 }

60 template <class Distance>
61 typename TrapezoidProfile<Distance>::State
62 TrapezoidProfile<Distance>::Calculate(units::second_t t) const {
63  State result = m_initial;
64 
65  if (t < m_endAccel) {
66  result.velocity += t * m_constraints.maxAcceleration;
67  result.position +=
68  (m_initial.velocity + t * m_constraints.maxAcceleration / 2.0) * t;
69  } else if (t < m_endFullSpeed) {
70  result.velocity = m_constraints.maxVelocity;
71  result.position += (m_initial.velocity +
72  m_endAccel * m_constraints.maxAcceleration / 2.0) *
73  m_endAccel +
74  m_constraints.maxVelocity * (t - m_endAccel);
75  } else if (t <= m_endDeccel) {
76  result.velocity =
77  m_goal.velocity + (m_endDeccel - t) * m_constraints.maxAcceleration;
78  units::second_t timeLeft = m_endDeccel - t;
79  result.position =
80  m_goal.position -
81  (m_goal.velocity + timeLeft * m_constraints.maxAcceleration / 2.0) *
82  timeLeft;
83  } else {
84  result = m_goal;
85  }
86 
87  return Direct(result);
88 }

90 template <class Distance>
91 units::second_t TrapezoidProfile<Distance>::TimeLeftUntil(
92  Distance_t target) const {
93  Distance_t position = m_initial.position * m_direction;
94  Velocity_t velocity = m_initial.velocity * m_direction;
95 
96  units::second_t endAccel = m_endAccel * m_direction;
97  units::second_t endFullSpeed = m_endFullSpeed * m_direction - endAccel;
98 
99  if (target < position) {
100  endAccel *= -1.0;
101  endFullSpeed *= -1.0;
102  velocity *= -1.0;
103  }
104 
105  endAccel = units::math::max(endAccel, 0_s);
106  endFullSpeed = units::math::max(endFullSpeed, 0_s);
107  units::second_t endDeccel = m_endDeccel - endAccel - endFullSpeed;
108  endDeccel = units::math::max(endDeccel, 0_s);
109 
110  const Acceleration_t acceleration = m_constraints.maxAcceleration;
111  const Acceleration_t decceleration = -m_constraints.maxAcceleration;
112 
113  Distance_t distToTarget = units::math::abs(target - position);
114 
115  if (distToTarget < Distance_t(1e-6)) {
116  return 0_s;
117  }
118 
119  Distance_t accelDist =
120  velocity * endAccel + 0.5 * acceleration * endAccel * endAccel;
121 
122  Velocity_t deccelVelocity;
123  if (endAccel > 0_s) {
124  deccelVelocity = units::math::sqrt(
125  units::math::abs(velocity * velocity + 2 * acceleration * accelDist));
126  } else {
127  deccelVelocity = velocity;
128  }
129 
130  Distance_t deccelDist =
131  deccelVelocity * endDeccel + 0.5 * decceleration * endDeccel * endDeccel;
132 
133  deccelDist = units::math::max(deccelDist, Distance_t(0));
134 
135  Distance_t fullSpeedDist = m_constraints.maxVelocity * endFullSpeed;
136 
137  if (accelDist > distToTarget) {
138  accelDist = distToTarget;
139  fullSpeedDist = Distance_t(0);
140  deccelDist = Distance_t(0);
141  } else if (accelDist + fullSpeedDist > distToTarget) {
142  fullSpeedDist = distToTarget - accelDist;
143  deccelDist = Distance_t(0);
144  } else {
145  deccelDist = distToTarget - fullSpeedDist - accelDist;
146  }
147 
148  units::second_t accelTime =
149  (-velocity + units::math::sqrt(units::math::abs(
150  velocity * velocity + 2 * acceleration * accelDist))) /
151  acceleration;
152 
153  units::second_t deccelTime =
154  (-deccelVelocity +
155  units::math::sqrt(units::math::abs(deccelVelocity * deccelVelocity +
156  2 * decceleration * deccelDist))) /
157  decceleration;
158 
159  units::second_t fullSpeedTime = fullSpeedDist / m_constraints.maxVelocity;
160 
161  return accelTime + fullSpeedTime + deccelTime;
162 }
