24 Trajectory::State Trajectory::State::Interpolate(State endValue,
25  double i) const {
26  // Find the new [t] value.
27  const auto newT = Lerp(t, endValue.t, i);
28 
29  // Find the delta time between the current state and the interpolated state.
30  const auto deltaT = newT - t;
31 
32  // If delta time is negative, flip the order of interpolation.
33  if (deltaT < 0_s) return endValue.Interpolate(*this, 1.0 - i);
34 
35  // Check whether the robot is reversing at this stage.
36  const auto reversing =
37  velocity < 0_mps ||
38  (units::math::abs(velocity) < 1E-9_mps && acceleration < 0_mps_sq);
39 
40  // Calculate the new velocity.
41  // v = v_0 + at
42  const units::meters_per_second_t newV = velocity + (acceleration * deltaT);
43 
44  // Calculate the change in position.
45  // delta_s = v_0 t + 0.5 at^2
46  const units::meter_t newS =
47  (velocity * deltaT + 0.5 * acceleration * deltaT * deltaT) *
48  (reversing ? -1.0 : 1.0);
49 
50  // Return the new state. To find the new position for the new state, we need
51  // to interpolate between the two endpoint poses. The fraction for
52  // interpolation is the change in position (delta s) divided by the total
53  // distance between the two endpoints.
54  const double interpolationFrac =
55  newS / endValue.pose.Translation().Distance(pose.Translation());
56 
57  return {newT, newV, acceleration,
58  Lerp(pose, endValue.pose, interpolationFrac),
59  Lerp(curvature, endValue.curvature, interpolationFrac)};
60 }

66 Trajectory::State Trajectory::Sample(units::second_t t) const {
67  if (t <= m_states.front().t) return m_states.front();
68  if (t >= m_totalTime) return m_states.back();
69 
70  // To get the element that we want, we will use a binary search algorithm
71  // instead of iterating over a for-loop. A binary search is O(std::log(n))
72  // whereas searching using a loop is O(n).
73 
74  // This starts at 1 because we use the previous state later on for
75  // interpolation.
76  int low = 1;
77  int high = m_states.size() - 1;
78 
79  while (low != high) {
80  int mid = (low + high) / 2;
81  if (m_states[mid].t < t) {
82  // This index and everything under it are less than the requested
83  // timestamp. Therefore, we can discard them.
84  low = mid + 1;
85  } else {
86  // t is at least as large as the element at this index. This means that
87  // anything after it cannot be what we are looking for.
88  high = mid;
89  }
90  }
91 
92  // High and Low should be the same.
93 
94  // The sample's timestamp is now greater than or equal to the requested
95  // timestamp. If it is greater, we need to interpolate between the
96  // previous state and the current state to get the exact state that we
97  // want.
98  const auto sample = m_states[low];
99  const auto prevSample = m_states[low - 1];
100 
101  // If the difference in states is negligible, then we are spot on!
102  if (units::math::abs(sample.t - prevSample.t) < 1E-9_s) {
103  return sample;
104  }
105  // Interpolate between the two states for the state that we want.
106  return prevSample.Interpolate(sample,
107  (t - prevSample.t) / (sample.t - prevSample.t));
108 }

110 Trajectory Trajectory::TransformBy(const Transform2d& transform) {
111  auto& firstState = m_states[0];
112  auto& firstPose = firstState.pose;
113 
114  // Calculate the transformed first pose.
115  auto newFirstPose = firstPose + transform;
116  auto newStates = m_states;
117  newStates[0].pose = newFirstPose;
118 
119  for (unsigned int i = 1; i < newStates.size(); i++) {
120  auto& state = newStates[i];
121  // We are transforming relative to the coordinate frame of the new initial
122  // pose.
123  state.pose = newFirstPose + (state.pose - firstPose);
124  }
125 
126  return Trajectory(newStates);
127 }
