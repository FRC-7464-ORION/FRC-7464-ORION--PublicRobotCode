26 Trajectory::State Trajectory::State::Interpolate(State endValue,
27  double i) const {
28  // Find the new [t] value.
29  const auto newT = wpi::Lerp(t, endValue.t, i);
30 
31  // Find the delta time between the current state and the interpolated state.
32  const auto deltaT = newT - t;
33 
34  // If delta time is negative, flip the order of interpolation.
35  if (deltaT < 0_s) {
36  return endValue.Interpolate(*this, 1.0 - i);
37  }
38 
39  // Check whether the robot is reversing at this stage.
40  const auto reversing =
41  velocity < 0_mps ||
42  (units::math::abs(velocity) < 1E-9_mps && acceleration < 0_mps_sq);
43 
44  // Calculate the new velocity.
45  // v = v_0 + at
46  const units::meters_per_second_t newV = velocity + (acceleration * deltaT);
47 
48  // Calculate the change in position.
49  // delta_s = v_0 t + 0.5 at^2
50  const units::meter_t newS =
51  (velocity * deltaT + 0.5 * acceleration * deltaT * deltaT) *
52  (reversing ? -1.0 : 1.0);
53 
54  // Return the new state. To find the new position for the new state, we need
55  // to interpolate between the two endpoint poses. The fraction for
56  // interpolation is the change in position (delta s) divided by the total
57  // distance between the two endpoints.
58  const double interpolationFrac =
59  newS / endValue.pose.Translation().Distance(pose.Translation());
60 
61  return {newT, newV, acceleration,
62  wpi::Lerp(pose, endValue.pose, interpolationFrac),
63  wpi::Lerp(curvature, endValue.curvature, interpolationFrac)};
64 }

70 Trajectory::State Trajectory::Sample(units::second_t t) const {
71  if (t <= m_states.front().t) {
72  return m_states.front();
73  }
74  if (t >= m_totalTime) {
75  return m_states.back();
76  }
77 
78  // Use binary search to get the element with a timestamp no less than the
79  // requested timestamp. This starts at 1 because we use the previous state
80  // later on for interpolation.
81  auto sample =
82  std::lower_bound(m_states.cbegin() + 1, m_states.cend(), t,
83  [](const auto& a, const auto& b) { return a.t < b; });
84 
85  auto prevSample = sample - 1;
86 
87  // The sample's timestamp is now greater than or equal to the requested
88  // timestamp. If it is greater, we need to interpolate between the
89  // previous state and the current state to get the exact state that we
90  // want.
91 
92  // If the difference in states is negligible, then we are spot on!
93  if (units::math::abs(sample->t - prevSample->t) < 1E-9_s) {
94  return *sample;
95  }
96  // Interpolate between the two states for the state that we want.
97  return prevSample->Interpolate(
98  *sample, (t - prevSample->t) / (sample->t - prevSample->t));
99 }

101 Trajectory Trajectory::TransformBy(const Transform2d& transform) {
102  auto& firstState = m_states[0];
103  auto& firstPose = firstState.pose;
104 
105  // Calculate the transformed first pose.
106  auto newFirstPose = firstPose + transform;
107  auto newStates = m_states;
108  newStates[0].pose = newFirstPose;
109 
110  for (unsigned int i = 1; i < newStates.size(); i++) {
111  auto& state = newStates[i];
112  // We are transforming relative to the coordinate frame of the new initial
113  // pose.
114  state.pose = newFirstPose + (state.pose - firstPose);
115  }
116 
117  return Trajectory(newStates);
118 }

128 Trajectory Trajectory::operator+(const Trajectory& other) const {
129  // If this is a default constructed trajectory with no states, then we can
130  // simply return the rhs trajectory.
131  if (m_states.empty()) {
132  return other;
133  }
134 
135  auto states = m_states;
136  auto otherStates = other.States();
137  for (auto& otherState : otherStates) {
138  otherState.t += m_totalTime;
139  }
140 
141  // Here we omit the first state of the other trajectory because we don't want
142  // two time points with different states. Sample() will automatically
143  // interpolate between the end of this trajectory and the second state of the
144  // other trajectory.
145  states.insert(states.end(), otherStates.begin() + 1, otherStates.end());
146  return Trajectory(states);
147 }
