92 void CANBusScanner::run() {
93  HAL_CANStreamMessage* msgbuf = new HAL_CANStreamMessage[m_streamBufferSize];
94  while (!m_stopThread) {
95  if (m_running) {
96  uint32_t numRead = 0;
97  int32_t status = 0;
98  HAL_CAN_ReadStreamSession(m_streamHandle, msgbuf,
99  m_streamBufferSize, &numRead, &status);
100 
101  if (status != 0) {
102  Stop();
103  m_lastError = "Failed to read stream session, status:" +
104  std::to_string(status);
105  } else {
106  for (int i = 0; i < static_cast<int>(numRead); i++) {
107  // Read entire buffer
108  auto element = m_registeredDevices.find(
109  msgbuf[i].messageID & ~CAN_DEVICEID_FIELD_MASK);
110  if (element != m_registeredDevices.end()) {
111  // Message is registered
112  int canID =
113  msgbuf[i].messageID & CAN_DEVICEID_FIELD_MASK;
114 
115  element->second->AddOrUpdateDevice(canID);
116  }
117  }
118  }
119  }
120 
121  if (!m_stopThread) {
122  std::this_thread::sleep_for(
123  std::chrono::milliseconds(m_threadInterval));
124  }
125  }
126  delete[] msgbuf;
127 }

129 void CANBusScanner::RegisterDevice(std::string name,
130  std::vector<uint32_t> validIds,
131  int32_t maxFramePeriodMs) {
132  if (validIds.empty()) {
133  return;

136  std::shared_ptr<CANScanCollection> ptr =
137  std::make_shared<CANScanCollection>(name, validIds[0],
138  maxFramePeriodMs);
139  m_registeredList.push_back(ptr);
140  for (auto const& id : validIds) {
141  m_registeredDevices[id] = ptr;
142  }

145 std::vector<CANScanIdentifier> CANBusScanner::CANBusScan() {
146  std::vector<CANScanIdentifier> result;
147  for (auto const& devElements : m_registeredList) {
148  uint32_t arbid = devElements->ArbId();
149  std::vector<int> deviceIds = devElements->ActiveDevices();
150 
151  for (auto const& devId : deviceIds) {
152  result.push_back(
153  CANScanIdentifier(arbid | devId, devElements->Name()));
154  }
155  }
156  return result;
157 }
