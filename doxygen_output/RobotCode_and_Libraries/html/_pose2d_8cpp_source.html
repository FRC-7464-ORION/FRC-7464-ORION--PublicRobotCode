55 Pose2d Pose2d::Exp(const Twist2d& twist) const {
56  const auto dx = twist.dx;
57  const auto dy = twist.dy;
58  const auto dtheta = twist.dtheta.to<double>();
59 
60  const auto sinTheta = std::sin(dtheta);
61  const auto cosTheta = std::cos(dtheta);
62 
63  double s, c;
64  if (std::abs(dtheta) < 1E-9) {
65  s = 1.0 - 1.0 / 6.0 * dtheta * dtheta;
66  c = 0.5 * dtheta;
67  } else {
68  s = sinTheta / dtheta;
69  c = (1 - cosTheta) / dtheta;
70  }
71 
72  const Transform2d transform{Translation2d{dx * s - dy * c, dx * c + dy * s},
73  Rotation2d{cosTheta, sinTheta}};
74 
75  return *this + transform;
76 }

78 Twist2d Pose2d::Log(const Pose2d& end) const {
79  const auto transform = end.RelativeTo(*this);
80  const auto dtheta = transform.Rotation().Radians().to<double>();
81  const auto halfDtheta = dtheta / 2.0;
82 
83  const auto cosMinusOne = transform.Rotation().Cos() - 1;
84 
85  double halfThetaByTanOfHalfDtheta;
86 
87  if (std::abs(cosMinusOne) < 1E-9) {
88  halfThetaByTanOfHalfDtheta = 1.0 - 1.0 / 12.0 * dtheta * dtheta;
89  } else {
90  halfThetaByTanOfHalfDtheta =
91  -(halfDtheta * transform.Rotation().Sin()) / cosMinusOne;
92  }
93 
94  const Translation2d translationPart =
95  transform.Translation().RotateBy(
96  {halfThetaByTanOfHalfDtheta, -halfDtheta}) *
97  std::hypot(halfThetaByTanOfHalfDtheta, halfDtheta);
98 
99  return {translationPart.X(), translationPart.Y(), units::radian_t(dtheta)};
100 }
