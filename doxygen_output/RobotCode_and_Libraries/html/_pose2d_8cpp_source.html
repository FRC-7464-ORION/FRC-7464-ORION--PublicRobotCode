46 Pose2d Pose2d::Exp(const Twist2d& twist) const {
47  const auto dx = twist.dx;
48  const auto dy = twist.dy;
49  const auto dtheta = twist.dtheta.value();
50 
51  const auto sinTheta = std::sin(dtheta);
52  const auto cosTheta = std::cos(dtheta);
53 
54  double s, c;
55  if (std::abs(dtheta) < 1E-9) {
56  s = 1.0 - 1.0 / 6.0 * dtheta * dtheta;
57  c = 0.5 * dtheta;
58  } else {
59  s = sinTheta / dtheta;
60  c = (1 - cosTheta) / dtheta;
61  }
62 
63  const Transform2d transform{Translation2d{dx * s - dy * c, dx * c + dy * s},
64  Rotation2d{cosTheta, sinTheta}};
65 
66  return *this + transform;
67 }

69 Twist2d Pose2d::Log(const Pose2d& end) const {
70  const auto transform = end.RelativeTo(*this);
71  const auto dtheta = transform.Rotation().Radians().value();
72  const auto halfDtheta = dtheta / 2.0;
73 
74  const auto cosMinusOne = transform.Rotation().Cos() - 1;
75 
76  double halfThetaByTanOfHalfDtheta;
77 
78  if (std::abs(cosMinusOne) < 1E-9) {
79  halfThetaByTanOfHalfDtheta = 1.0 - 1.0 / 12.0 * dtheta * dtheta;
80  } else {
81  halfThetaByTanOfHalfDtheta =
82  -(halfDtheta * transform.Rotation().Sin()) / cosMinusOne;
83  }
84 
85  const Translation2d translationPart =
86  transform.Translation().RotateBy(
87  {halfThetaByTanOfHalfDtheta, -halfDtheta}) *
88  std::hypot(halfThetaByTanOfHalfDtheta, halfDtheta);
89 
90  return {translationPart.X(), translationPart.Y(), units::radian_t(dtheta)};
91 }
