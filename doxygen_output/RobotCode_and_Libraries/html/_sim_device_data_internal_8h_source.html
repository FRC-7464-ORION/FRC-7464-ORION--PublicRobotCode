108  // Must return -1 on a null callback for error handling
109  if (callback == nullptr) {
110  return -1;
111  }
112  if (!m_callbacks) {
113  m_callbacks = std::make_unique<CallbackVector>();
114  }
115  return m_callbacks->emplace_back(prefix, param, callback) + 1;
116  }
