28 /*****************************************************************************/
29 /* This protocol, introduced first with the navX MXP, expands upon the IMU */
30 /* protocol by adding the following new functionality: */
31 /* */
32 /* AHRS Update: Includes Fused Heading and Altitude Info */
33 /* Magnetometer Calibration: Enables configuration of coefficients from PC */
34 /* Board Identity: Enables retrieval of Board Identification Info */
35 /* Fusion Tuning: Enables configuration of key thresholds/coefficients used */
36 /* in data fusion algorithms from a remote client */
37 /* */
38 /* In addition, the navX enable stream command has been extended with a new */
39 /* Stream type, in order to enable AHRS Updates. */
40 /*****************************************************************************/

50 /* AHRS Protocol encodes certain data in binary format, unlike the IMU */
51 /* protocol, which encodes all data in ASCII characters. Thus, the */
52 /* packet start and message termination sequences may occur within the */
53 /* message content itself. To support the binary format, the binary */
54 /* message has this format: */
55 /* */
56 /* [start][binary indicator][len][msgid]<MESSAGE>[checksum][terminator] */
57 /* */
58 /* (The binary indicator and len are not present in the ASCII protocol) */
59 /* */
60 /* The [len] does not include the length of the start and binary */
61 /* indicator characters, but does include all other message items, */
62 /* including the checksum and terminator sequence. */
63

351  static int encodeIntegrationControlCmd( char *protocol_buffer, struct IntegrationControl& cmd )
352  {
353  // Header
354  protocol_buffer[0] = PACKET_START_CHAR;
355  protocol_buffer[1] = BINARY_PACKET_INDICATOR_CHAR;
356  protocol_buffer[2] = INTEGRATION_CONTROL_CMD_MESSAGE_LENGTH - 2;
357  protocol_buffer[3] = MSGID_INTEGRATION_CONTROL_CMD;
358  // Data
359  protocol_buffer[INTEGRATION_CONTROL_CMD_ACTION_INDEX] = cmd.action;
360  IMURegisters::encodeProtocolInt32(cmd.parameter,&protocol_buffer[INTEGRATION_CONTROL_CMD_PARAMETER_INDEX]);
361  // Footer
362  encodeTermination( protocol_buffer, INTEGRATION_CONTROL_CMD_MESSAGE_LENGTH, INTEGRATION_CONTROL_CMD_MESSAGE_LENGTH - 4 );
363  return INTEGRATION_CONTROL_CMD_MESSAGE_LENGTH;
364  }
365 
366  static int decodeIntegrationControlCmd( char *buffer, int length, uint8_t& action, int32_t& parameter)
367  {
368  if ( length < INTEGRATION_CONTROL_CMD_MESSAGE_LENGTH ) return 0;
369  if ( ( buffer[0] == PACKET_START_CHAR ) &&
370  ( buffer[1] == BINARY_PACKET_INDICATOR_CHAR ) &&
371  ( buffer[2] == INTEGRATION_CONTROL_CMD_MESSAGE_LENGTH - 2) &&
372  ( buffer[3] == MSGID_INTEGRATION_CONTROL_CMD ) )
373  {
374  if ( !verifyChecksum( buffer, INTEGRATION_CONTROL_CMD_MESSAGE_CHECKSUM_INDEX ) ) return 0;
375 
376  // Data
377  action = (uint8_t)buffer[INTEGRATION_CONTROL_CMD_ACTION_INDEX];
378  parameter = IMURegisters::decodeProtocolInt32(&buffer[INTEGRATION_CONTROL_CMD_PARAMETER_INDEX]);
379  return INTEGRATION_CONTROL_CMD_MESSAGE_LENGTH;
380  }
381  return 0;
382  }
