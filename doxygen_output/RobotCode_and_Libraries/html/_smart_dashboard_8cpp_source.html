80 void SmartDashboard::PutData(std::string_view key, wpi::Sendable* data) {
81  if (!data) {
82  throw FRC_MakeError(err::NullParameter, "{}", "value");
83  }
84  auto& inst = GetInstance();
85  std::scoped_lock lock(inst.tablesToDataMutex);
86  auto& uid = inst.tablesToData[key];
87  wpi::Sendable* sddata = wpi::SendableRegistry::GetSendable(uid);
88  if (sddata != data) {
89  uid = wpi::SendableRegistry::GetUniqueId(data);
90  auto dataTable = inst.table->GetSubTable(key);
91  auto builder = std::make_unique<SendableBuilderImpl>();
92  auto builderPtr = builder.get();
93  builderPtr->SetTable(dataTable);
94  wpi::SendableRegistry::Publish(uid, std::move(builder));
95  builderPtr->StartListeners();
96  dataTable->GetEntry(".name").SetString(key);
97  }
98 }

110 wpi::Sendable* SmartDashboard::GetData(std::string_view key) {
111  auto& inst = GetInstance();
112  std::scoped_lock lock(inst.tablesToDataMutex);
113  auto it = inst.tablesToData.find(key);
114  if (it == inst.tablesToData.end()) {
115  throw FRC_MakeError(err::SmartDashboardMissingKey, "{}", key);
116  }
117  return wpi::SendableRegistry::GetSendable(it->getValue());
118 }
