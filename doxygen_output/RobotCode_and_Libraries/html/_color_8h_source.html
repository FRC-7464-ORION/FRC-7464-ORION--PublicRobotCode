744  constexpr Color() = default;
745 
753  constexpr Color(double r, double g, double b)
754  : red(roundAndClamp(r)),
755  green(roundAndClamp(g)),
756  blue(roundAndClamp(b)) {}
757 
758  double red = 0.0;
759  double green = 0.0;
760  double blue = 0.0;
761 
762  private:
763  static constexpr double kPrecision = 1.0 / (1 << 12);
764 
765  static constexpr double roundAndClamp(double value) {
766  const auto rounded =
767  (static_cast<int>(value / kPrecision) + 0.5) * kPrecision;
768  return std::clamp(rounded, 0.0, 1.0);
769  }
770 };
771 
772 inline bool operator==(const Color& c1, const Color& c2) {
773  return c1.red == c2.red && c1.green == c2.green && c1.blue == c2.blue;
774 }
