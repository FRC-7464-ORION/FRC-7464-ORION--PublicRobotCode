62 JNIEXPORT jint JNICALL JNI_OnLoad(JavaVM* vm, void* reserved) {
63  jvm = vm;
64 
65  JNIEnv* env;
66  if (vm->GetEnv(reinterpret_cast<void**>(&env), JNI_VERSION_1_6) != JNI_OK) {
67  return JNI_ERR;
68  }
69 
70  // Cache references to classes
71  for (auto& c : classes) {
72  *c.cls = JClass(env, c.name);
73  if (!*c.cls) {
74  return JNI_ERR;
75  }
76  }
77 
78  for (auto& c : exceptions) {
79  *c.cls = JException(env, c.name);
80  if (!*c.cls) {
81  return JNI_ERR;
82  }
83  }
84 
85  return JNI_VERSION_1_6;
86 }

88 JNIEXPORT void JNICALL JNI_OnUnload(JavaVM* vm, void* reserved) {
89  JNIEnv* env;
90  if (vm->GetEnv(reinterpret_cast<void**>(&env), JNI_VERSION_1_6) != JNI_OK) {
91  return;
92  }
93  // Delete global references
94  for (auto& c : classes) {
95  c.cls->free(env);
96  }
97  for (auto& c : exceptions) {
98  c.cls->free(env);
99  }
100  jvm = nullptr;
101 }

181 static jobject MakeJObject(JNIEnv* env, const nt::Value& value) {
182  static jmethodID booleanConstructor = nullptr;
183  static jmethodID doubleConstructor = nullptr;
184  if (!booleanConstructor) {
185  booleanConstructor = env->GetMethodID(booleanCls, "<init>", "(Z)V");
186  }
187  if (!doubleConstructor) {
188  doubleConstructor = env->GetMethodID(doubleCls, "<init>", "(D)V");
189  }
190 
191  switch (value.type()) {
192  case NT_BOOLEAN:
193  return env->NewObject(booleanCls, booleanConstructor,
194  static_cast<jboolean>(value.GetBoolean() ? 1 : 0));
195  case NT_DOUBLE:
196  return env->NewObject(doubleCls, doubleConstructor,
197  static_cast<jdouble>(value.GetDouble()));
198  case NT_STRING:
199  return MakeJString(env, value.GetString());
200  case NT_RAW:
201  return MakeJByteArray(env, value.GetRaw());
202  case NT_BOOLEAN_ARRAY:
203  return MakeJBooleanArray(env, value.GetBooleanArray());
204  case NT_DOUBLE_ARRAY:
205  return MakeJDoubleArray(env, value.GetDoubleArray());
206  case NT_STRING_ARRAY:
207  return MakeJStringArray(env, value.GetStringArray());
208  case NT_RPC:
209  return MakeJByteArray(env, value.GetRpc());
210  default:
211  return nullptr;
212  }
213 }
