62  -> decltype(_unwrap_smart_ptr(m_choices[""])) {
63  std::string selected = m_defaultChoice;

75 template <class T>
76 void SendableChooser<T>::InitSendable(SendableBuilder& builder) {
77  builder.SetSmartDashboardType("String Chooser");
78  builder.GetEntry(kInstance).SetDouble(m_instance);
79  builder.AddStringArrayProperty(kOptions,
80  [=]() {
81  std::vector<std::string> keys;
82  for (const auto& choice : m_choices) {
83  keys.push_back(choice.first());
84  }
85 
86  // Unlike std::map, wpi::StringMap elements
87  // are not sorted
88  std::sort(keys.begin(), keys.end());
89 
90  return keys;
91  },
92  nullptr);
93  builder.AddSmallStringProperty(
94  kDefault,
95  [=](wpi::SmallVectorImpl<char>&) -> wpi::StringRef {
96  return m_defaultChoice;
97  },
98  nullptr);
99  builder.AddSmallStringProperty(
100  kActive,
101  [=](wpi::SmallVectorImpl<char>& buf) -> wpi::StringRef {
102  std::lock_guard<wpi::mutex> lock(m_mutex);
103  if (m_haveSelected) {
104  buf.assign(m_selected.begin(), m_selected.end());
105  return wpi::StringRef(buf.data(), buf.size());
106  } else {
107  return m_defaultChoice;
108  }
109  },
110  nullptr);
111  {
112  std::lock_guard<wpi::mutex> lock(m_mutex);
113  m_activeEntries.emplace_back(builder.GetEntry(kActive));
114  }
115  builder.AddStringProperty(kSelected, nullptr, [=](wpi::StringRef val) {
116  std::lock_guard<wpi::mutex> lock(m_mutex);
117  m_haveSelected = true;
118  m_selected = val;
119  for (auto& entry : m_activeEntries) entry.SetString(val);
120  });
121 }
