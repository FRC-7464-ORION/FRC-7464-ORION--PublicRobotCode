36 static bool IsRaspbian() {
37  bool is_raspbian = false;
38 
39  std::string raspbian_suffix("aspbian");
40  std::ifstream os_release_file;
41  os_release_file.open("/etc/os-release", std::ios::in);
42  if (os_release_file.fail()) return false;
43 
44  std::string line;
45  while (getline(os_release_file, line)) {
46  if (line.find(raspbian_suffix) != std::string::npos) {
47  is_raspbian = true;
48  break;
49  }
50  }
51 
52  os_release_file.close();
53  return is_raspbian;
54 }

278  void UpdateBoardStatus( uint8_t op_status, int16_t sensor_status, uint8_t cal_status, uint8_t selftest_status) {
279  /* Detect/Report Board operational status transitions */
280  bool poweron_init_completed = false;
281  if (ahrs->op_status == NAVX_OP_STATUS_NORMAL) {
282  if (op_status != NAVX_OP_STATUS_NORMAL) {
283  /* Board reset detected */
284  printf("navX-Sensor Reset Detected.\n");
285  }
286  } else {
287  if (op_status == NAVX_OP_STATUS_NORMAL) {
288  poweron_init_completed = true;
289  if ((cal_status & NAVX_CAL_STATUS_IMU_CAL_STATE_MASK) != NAVX_CAL_STATUS_IMU_CAL_COMPLETE) {
290  printf("navX-Sensor startup initialization underway; startup calibration in progress.\n");
291  } else {
292  printf("navX-Sensor startup initialization and startup calibration complete.\n");
293  }
294  }
295  }
296 
297  /* Detect/report reset of yaw angle tracker upon transition to startup calibration complete state. */
298  if (((ahrs->cal_status & NAVX_CAL_STATUS_IMU_CAL_STATE_MASK) != NAVX_CAL_STATUS_IMU_CAL_COMPLETE) &&
299  ((cal_status & NAVX_CAL_STATUS_IMU_CAL_STATE_MASK) == NAVX_CAL_STATUS_IMU_CAL_COMPLETE)) {
300  printf("navX-Sensor onboard startup calibration complete.\n");
301  /* Carefully, only reset the software yaw reset offset if calibration completion upon */
302  /* initial poweron or after board reset occurs. */
303  if (poweron_init_completed || ahrs->disconnect_startupcalibration_recovery_pending) {
304  ahrs->disconnect_startupcalibration_recovery_pending = false;
305  ahrs->yaw_angle_tracker->Init();
306  printf("navX-Sensor Yaw angle auto-reset to 0.0 due to startup calibration.\n");
307  }
308  }
309 
310  ahrs->op_status = op_status;
311  ahrs->sensor_status = sensor_status;
312  ahrs->cal_status = cal_status;
313  ahrs->selftest_status = selftest_status;
314  }

551 #define NUM_SUPPRESSED_SUCCESSIVE_YAWRESET_MESSAGES 5
552 #define SUPPRESSED_SUCESSIVE_YAWRESET_PERIOD_SECONDS 0.2
553 
566 void AHRS::ZeroYaw() {
567  double curr_timestamp = Timer::GetFPGATimestamp();
568  double delta_time_since_last_yawreset_request = curr_timestamp - last_yawreset_request_timestamp;
569  if (delta_time_since_last_yawreset_request < SUPPRESSED_SUCESSIVE_YAWRESET_PERIOD_SECONDS) {
570  successive_suppressed_yawreset_request_count++;
571  if ((successive_suppressed_yawreset_request_count % NUM_SUPPRESSED_SUCCESSIVE_YAWRESET_MESSAGES) == 1) {
572  if (logging_enabled) printf("navX-Sensor rapidly-repeated Yaw Reset ignored%s\n",
573  ((successive_suppressed_yawreset_request_count < NUM_SUPPRESSED_SUCCESSIVE_YAWRESET_MESSAGES)
574  ? "." : (" (repeated messages suppressed).")));
575  }
576  return;
577  }
578 
579  if (IsCalibrating()) {
580  double delta_time_since_last_yawreset_while_calibrating_request =
581  curr_timestamp - last_yawreset_while_calibrating_request_timestamp;
582  if ((delta_time_since_last_yawreset_while_calibrating_request > SUPPRESSED_SUCESSIVE_YAWRESET_PERIOD_SECONDS)) {
583  printf("navX-Sensor Yaw Reset request ignored - startup calibration is currently in progress.\n");
584  }
585  last_yawreset_while_calibrating_request_timestamp = curr_timestamp;
586  return;
587  }
588 
589  successive_suppressed_yawreset_request_count = 0;
590 
591  last_yawreset_request_timestamp = curr_timestamp;
592  if ( enable_boardlevel_yawreset && ahrs_internal->IsBoardYawResetSupported() ) {
593  io->ZeroYaw();
594  printf("navX-Sensor Board-level Yaw Reset requested.\n");
595  /* Note: Notification is deferred until action is complete. */
596  } else {
597  yaw_offset_tracker->SetOffset();
598  /* Notification occurs immediately. */
599  ahrs_internal->YawResetComplete();
600  }
601 }
