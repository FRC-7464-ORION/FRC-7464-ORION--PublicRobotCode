31 template <class T>
32 auto SendableChooser<T>::GetSelected()
33  -> decltype(_unwrap_smart_ptr(m_choices[""])) {
34  std::string selected = m_defaultChoice;
35  {
36  std::scoped_lock lock(m_mutex);
37  if (m_haveSelected) {
38  selected = m_selected;
39  }
40  }
41  if (selected.empty()) {
42  return decltype(_unwrap_smart_ptr(m_choices[""])){};
43  } else {
44  return _unwrap_smart_ptr(m_choices[selected]);
45  }
46 }

48 template <class T>
49 void SendableChooser<T>::InitSendable(nt::NTSendableBuilder& builder) {
50  builder.SetSmartDashboardType("String Chooser");
51  builder.GetEntry(kInstance).SetDouble(m_instance);
52  builder.AddStringArrayProperty(
53  kOptions,
54  [=] {
55  std::vector<std::string> keys;
56  for (const auto& choice : m_choices) {
57  keys.emplace_back(choice.first());
58  }
59 
60  // Unlike std::map, wpi::StringMap elements
61  // are not sorted
62  std::sort(keys.begin(), keys.end());
63 
64  return keys;
65  },
66  nullptr);
67  builder.AddSmallStringProperty(
68  kDefault,
69  [=](wpi::SmallVectorImpl<char>&) -> std::string_view {
70  return m_defaultChoice;
71  },
72  nullptr);
73  builder.AddSmallStringProperty(
74  kActive,
75  [=](wpi::SmallVectorImpl<char>& buf) -> std::string_view {
76  std::scoped_lock lock(m_mutex);
77  if (m_haveSelected) {
78  buf.assign(m_selected.begin(), m_selected.end());
79  return {buf.data(), buf.size()};
80  } else {
81  return m_defaultChoice;
82  }
83  },
84  nullptr);
85  {
86  std::scoped_lock lock(m_mutex);
87  m_activeEntries.emplace_back(builder.GetEntry(kActive));
88  }
89  builder.AddStringProperty(kSelected, nullptr, [=](std::string_view val) {
90  std::scoped_lock lock(m_mutex);
91  m_haveSelected = true;
92  m_selected = val;
93  for (auto& entry : m_activeEntries) {
94  entry.SetString(val);
95  }
96  });
97 }
