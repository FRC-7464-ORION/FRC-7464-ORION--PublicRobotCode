48 void DSCommPacket::ReadMatchtimeTag(wpi::span<const uint8_t> tagData) {
49  if (tagData.size() < 6) {
50  return;
51  }
52 
53  uint32_t store = tagData[2] << 24;
54  store |= tagData[3] << 16;
55  store |= tagData[4] << 8;
56  store |= tagData[5];
57 
58  static_assert(sizeof(uint32_t) == sizeof(float), "float must be 32 bits");
59 
60  float matchTime = 0;
61 
62  std::memcpy(&matchTime, &store, sizeof(float));
63  m_match_time = matchTime;
64 }

66 void DSCommPacket::ReadJoystickTag(wpi::span<const uint8_t> dataInput,
67  int index) {
68  DSCommJoystickPacket& stick = m_joystick_packets[index];
69  stick.ResetUdp();
70 
71  if (dataInput.size() == 2) {
72  return;
73  }
74 
75  dataInput = dataInput.subspan(2);
76 
77  // Read axes
78  int axesLength = dataInput[0];
79  for (int i = 0; i < axesLength; i++) {
80  int8_t value = dataInput[1 + i];
81  if (value < 0) {
82  stick.axes.axes[i] = value / 128.0;
83  } else {
84  stick.axes.axes[i] = value / 127.0;
85  }
86  }
87  stick.axes.count = axesLength;
88 
89  dataInput = dataInput.subspan(1 + axesLength);
90 
91  // Read Buttons
92  int buttonCount = dataInput[0];
93  int numBytes = (buttonCount + 7) / 8;
94  stick.buttons.buttons = 0;
95  for (int i = 0; i < numBytes; i++) {
96  stick.buttons.buttons |= dataInput[numBytes - i] << (8 * (i));
97  }
98  stick.buttons.count = buttonCount;
99 
100  dataInput = dataInput.subspan(1 + numBytes);
101 
102  int povsLength = dataInput[0];
103  for (int i = 0; i < povsLength * 2; i += 2) {
104  stick.povs.povs[i] = (dataInput[1 + i] << 8) | dataInput[2 + i];
105  }
106 
107  stick.povs.count = povsLength;
108 
109  return;
110 }

115 void DSCommPacket::DecodeTCP(wpi::span<const uint8_t> packet) {
116  // No header
117  while (!packet.empty()) {
118  int tagLength = packet[0] << 8 | packet[1];
119  auto tagPacket = packet.subspan(0, tagLength + 2);
120 
121  if (tagLength == 0) {
122  return;
123  }
124 
125  switch (packet[2]) {
126  case kJoystickNameTag:
127  ReadJoystickDescriptionTag(tagPacket);
128  break;
129  case kGameDataTag:
130  ReadGameSpecificMessageTag(tagPacket);
131  break;
132  case kMatchInfoTag:
133  ReadNewMatchInfoTag(tagPacket);
134  break;
135  }
136  packet = packet.subspan(tagLength + 2);
137  }
138 }

140 void DSCommPacket::DecodeUDP(wpi::span<const uint8_t> packet) {
141  if (packet.size() < 6) {
142  return;
143  }
144  // Decode fixed header
145  m_hi = packet[0];
146  m_lo = packet[1];
147  // Comm Version is packet 2, ignore
148  SetControl(packet[3], packet[4]);
149  SetAlliance(packet[5]);
150 
151  // Return if packet finished
152  if (packet.size() == 6) {
153  return;
154  }
155 
156  // Else, handle tagged data
157  packet = packet.subspan(6);
158 
159  int joystickNum = 0;
160 
161  // Loop to handle multiple tags
162  while (!packet.empty()) {
163  auto tagLength = packet[0];
164  auto tagPacket = packet.subspan(0, tagLength + 1);
165 
166  switch (packet[1]) {
167  case kJoystickDataTag:
168  ReadJoystickTag(tagPacket, joystickNum);
169  joystickNum++;
170  break;
171  case kMatchTimeTag:
172  ReadMatchtimeTag(tagPacket);
173  break;
174  }
175  packet = packet.subspan(tagLength + 1);
176  }
177 }

179 void DSCommPacket::ReadNewMatchInfoTag(wpi::span<const uint8_t> data) {
180  // Size 2 bytes, tag 1 byte
181  if (data.size() <= 3) {
182  return;
183  }
184 
185  int nameLength = std::min<size_t>(data[3], sizeof(matchInfo.eventName) - 1);
186 
187  for (int i = 0; i < nameLength; i++) {
188  matchInfo.eventName[i] = data[4 + i];
189  }
190 
191  matchInfo.eventName[nameLength] = '\0';
192 
193  data = data.subspan(4 + nameLength);
194 
195  if (data.size() < 4) {
196  return;
197  }
198 
199  matchInfo.matchType = static_cast<HAL_MatchType>(
200  data[0]); // None, Practice, Qualification, Elimination, Test
201  matchInfo.matchNumber = (data[1] << 8) | data[2];
202  matchInfo.replayNumber = data[3];
203 
204  HALSIM_SetMatchInfo(&matchInfo);
205 }

223 void DSCommPacket::ReadJoystickDescriptionTag(wpi::span<const uint8_t> data) {
224  if (data.size() < 3) {
225  return;
226  }
227  data = data.subspan(3);
228  int joystickNum = data[0];
229  DSCommJoystickPacket& packet = m_joystick_packets[joystickNum];
230  packet.ResetTcp();
231  packet.descriptor.isXbox = data[1] != 0 ? 1 : 0;
232  packet.descriptor.type = data[2];
233  int nameLength =
234  std::min<size_t>(data[3], (sizeof(packet.descriptor.name) - 1));
235  for (int i = 0; i < nameLength; i++) {
236  packet.descriptor.name[i] = data[4 + i];
237  }
238  data = data.subspan(4 + nameLength);
239  packet.descriptor.name[nameLength] = '\0';
240  int axesCount = data[0];
241  packet.descriptor.axisCount = axesCount;
242  for (int i = 0,
243  len = std::min<int>(axesCount, sizeof(packet.descriptor.axisTypes));
244  i < len; i++) {
245  packet.descriptor.axisTypes[i] = data[1 + i];
246  }
247  data = data.subspan(1 + axesCount);
248 
249  packet.descriptor.buttonCount = data[0];
250  packet.descriptor.povCount = data[1];
251 }

268 void DSCommPacket::SetupSendHeader(wpi::raw_uv_ostream& buf) {
269  static constexpr uint8_t kCommVersion = 0x01;
270 
271  // High low packet index, comm version
272  buf << m_hi << m_lo << kCommVersion;
273 
274  // Control word and status check
275  buf << m_control_sent
276  << static_cast<uint8_t>(HALSIM_GetProgramStarted() ? kRobotHasCode : 0);
277 
278  // Battery voltage high and low

280 
281  // Request (Always 0)
282  buf << static_cast<uint8_t>(0);

285 void DSCommPacket::SetupJoystickTag(wpi::raw_uv_ostream& buf) {
286  static constexpr uint8_t kHIDTag = 0x01;
287 
288  // HID tags are sent 1 per device
289  int64_t outputs;
290  int32_t rightRumble;
291  int32_t leftRumble;
292  for (size_t i = 0; i < m_joystick_packets.size(); i++) {
293  // Length is 9, 1 tag and 8 data.
294  buf << static_cast<uint8_t>(9) << kHIDTag;
295  HALSIM_GetJoystickOutputs(i, &outputs, &leftRumble, &rightRumble);
296  auto op = static_cast<uint32_t>(outputs);
297  auto rr = static_cast<uint16_t>(rightRumble);
298  auto lr = static_cast<uint16_t>(leftRumble);
299  buf.write((op >> 24 & 0xFF));
300  buf.write((op >> 16 & 0xFF));
301  buf.write((op >> 8 & 0xFF));
302  buf.write((op & 0xFF));
303  buf.write((rr >> 8 & 0xFF));
304  buf.write((rr & 0xFF));
305  buf.write((lr >> 8 & 0xFF));
306  buf.write((lr & 0xFF));
307  }
308 }
