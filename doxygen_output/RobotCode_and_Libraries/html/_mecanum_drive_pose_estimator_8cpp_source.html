14 frc::MecanumDrivePoseEstimator::MecanumDrivePoseEstimator(
15  const Rotation2d& gyroAngle, const Pose2d& initialPose,
16  MecanumDriveKinematics kinematics,
17  const wpi::array<double, 3>& stateStdDevs,
18  const wpi::array<double, 1>& localMeasurementStdDevs,
19  const wpi::array<double, 3>& visionMeasurementStdDevs,
20  units::second_t nominalDt)
21  : m_observer(
22  [](const Eigen::Vector<double, 3>& x,
23  const Eigen::Vector<double, 3>& u) { return u; },
24  [](const Eigen::Vector<double, 3>& x,
25  const Eigen::Vector<double, 3>& u) { return x.block<1, 1>(2, 0); },
26  stateStdDevs, localMeasurementStdDevs, frc::AngleMean<3, 3>(2),
27  frc::AngleMean<1, 3>(0), frc::AngleResidual<3>(2),
28  frc::AngleResidual<1>(0), frc::AngleAdd<3>(2), nominalDt),
29  m_kinematics(kinematics),
30  m_nominalDt(nominalDt) {
31  SetVisionMeasurementStdDevs(visionMeasurementStdDevs);
32 
33  // Create vision correction mechanism.
34  m_visionCorrect = [&](const Eigen::Vector<double, 3>& u,
35  const Eigen::Vector<double, 3>& y) {
36  m_observer.Correct<3>(
37  u, y,
38  [](const Eigen::Vector<double, 3>& x, const Eigen::Vector<double, 3>&) {
39  return x;
40  },
41  m_visionContR, frc::AngleMean<3, 3>(2), frc::AngleResidual<3>(2),
42  frc::AngleResidual<3>(2), frc::AngleAdd<3>(2));
43  };
44 
45  // Set initial state.
46  m_observer.SetXhat(PoseTo3dVector(initialPose));
47 
48  // Calculate offsets.
49  m_gyroOffset = initialPose.Rotation() - gyroAngle;
50  m_previousAngle = initialPose.Rotation();
51 }

89 Pose2d frc::MecanumDrivePoseEstimator::UpdateWithTime(
90  units::second_t currentTime, const Rotation2d& gyroAngle,
91  const MecanumDriveWheelSpeeds& wheelSpeeds) {
92  auto dt = m_prevTime >= 0_s ? currentTime - m_prevTime : m_nominalDt;
93  m_prevTime = currentTime;
94 
95  auto angle = gyroAngle + m_gyroOffset;
96  auto omega = (angle - m_previousAngle).Radians() / dt;
97 
98  auto chassisSpeeds = m_kinematics.ToChassisSpeeds(wheelSpeeds);
99  auto fieldRelativeVelocities =
100  Translation2d(chassisSpeeds.vx * 1_s, chassisSpeeds.vy * 1_s)
101  .RotateBy(angle);
102 
103  Eigen::Vector<double, 3> u{fieldRelativeVelocities.X().value(),
104  fieldRelativeVelocities.Y().value(),
105  omega.value()};
106 
107  Eigen::Vector<double, 1> localY{angle.Radians().value()};
108  m_previousAngle = angle;
109 
110  m_latencyCompensator.AddObserverState(m_observer, u, localY, currentTime);
111 
112  m_observer.Predict(u, dt);
113  m_observer.Correct(u, localY);
114 
115  return GetEstimatedPosition();
116 }
