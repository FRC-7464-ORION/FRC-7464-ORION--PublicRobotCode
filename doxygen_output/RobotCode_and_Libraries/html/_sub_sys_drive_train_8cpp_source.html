116 // The method for driving arcade style
117 void SubSysDriveTrain::DriveArcadeStyle(double y, double x) {
118 
119  // The multiplier for putting turbo mode on
120  double const TurboMultiplier =
121  k_DriveTrainAbsValMaxSpeedTurboOn/k_DriveTrainAbsValMaxSpeedNoTurbo;
122 
123  // The y axis value after being nulled, desensitized, and speed limited
124  double y_Nulled_Desensed_Limited;
125  // The x axis value after being nulled, desensitized, and speed limited
126  double x_Nulled_Desensed_Limited;
127  // The y axis value after being nulled, desensitized, speed limited,
128  // and turboed (on or off)
129  double y_Turboed;
130  // The x axis value after being nulled, desensitized, speed limited,
131  // and turboed (on or off)
132  double x_Turboed;
133  // The y axis value after being nulled, desensitized, speed limited,
134  // turboed (on or off), and smoothed (on or off)
135  double y_Smoothed;
136  // The x axis value after being nulled, desensitized, speed limited,
137  // turboed (on or off), and smoothed (on or off)
138  double x_Smoothed;
139 
140  // If the drive direction is switched...
141  if(m_drive_direction_switched)
142  {
143  // Invert the y axis
144  y = k_ReverseSpeedMultiplier * y;
145  }
146 
147  // We were passed in a nulled/desensed/limited value for y,
148  // so we'll use that
149  y_Nulled_Desensed_Limited = y;
150  // We were passed in a nulled/desensed/limited value for x,
151  // so we'll use that
152  x_Nulled_Desensed_Limited = x;
153 
154  // If turbo mode is not enabled...
155  if(!m_TurboModeEnabled) {
156 
157  // We'll just used the passed in values...
158  y_Turboed = y_Nulled_Desensed_Limited;
159  x_Turboed = x_Nulled_Desensed_Limited;
160 
161  } // end if(!m_TurboModeEnabled)
162 
163  else { // Turbo mode is enabled...
164 
165  // We'll use the passed in values, multiplied by our
166  // turbo factor
167  y_Turboed = y_Nulled_Desensed_Limited*TurboMultiplier;
168  x_Turboed = x_Nulled_Desensed_Limited*TurboMultiplier;
169 
170  } // end else (if(!m_TurboModeEnabled))
171 
172  // If smoothing mode is not enabled...
173  if(!m_DriveTrain_Smoothing_Enabled) {
174 
175  // We'll just use the "turboed" y value
176  y_Smoothed = y_Turboed;
177  // We'll just use the "turboed" x value
178  x_Smoothed = x_Turboed;
179  }
180 
181  // Smoothing is enabled...
182  else {
183 
184  // The smoothed y value is calculated using the formula for
185  // an infinite impulse repsponse digital low pass filter,
186  // which is:
187  // y_curr_filt = d*y_curr_unfilt + (1-d)*y_past_filt
188  y_Smoothed =
189  (k_DriveTrain_IIR_DLPF_d * y_Turboed) +
190  ((1.0 - k_DriveTrain_IIR_DLPF_d) * m_speed1_past_filtered_output);
191 
192  // Save the current filtered y output for next time
193  m_speed1_past_filtered_output = y_Smoothed;
194 
195  // The smoothed x value is calculated using the formula for
196  // an infinite impulse repsponse digital low pass filter,
197  // which is:
198  // x_curr_filt = d*x_curr_unfilt + (1-d)*x_past_filt
199  x_Smoothed =
200  (k_DriveTrain_IIR_DLPF_d * x_Turboed) +
201  ((1.0 - k_DriveTrain_IIR_DLPF_d) * m_speed2_past_filtered_output);
202 
203  // Save the current filtered x output for next time
204  m_speed2_past_filtered_output = x_Smoothed;
205 
206  } // end else (if(!m_DriveTrain_Smoothing_Enabled))
207 
208  // Use arcade drive using the passed in values after:
209  // * Nulling, desensitizing, and speed limiting
210  // * Being Turboed (on or off)
211  // * Being Smoothed (on or off)
212  // * not squaring inputs
213  m_differentialDrive.ArcadeDrive(
214  y_Smoothed, x_Smoothed, k_DoNOTUseBuiltInSquaring);
215 
216 } // end SubSysDriveTrain::DriveArcadeStyle()
