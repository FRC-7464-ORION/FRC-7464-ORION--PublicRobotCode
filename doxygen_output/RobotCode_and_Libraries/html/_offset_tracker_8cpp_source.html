10 OffsetTracker::OffsetTracker(int history_length) {
11  history_len = history_length;
12  value_history = new float[history_len];
13  for ( int i = 0; i < history_len; i++ ) {
14  value_history[i] = 0.0f;

16  next_value_history_index = 0;

18 }

20 void OffsetTracker::UpdateHistory(float curr_value) {
21  if (next_value_history_index >= history_len) {
22  next_value_history_index = 0;
23  }
24  value_history[next_value_history_index] = curr_value;

28 double OffsetTracker::GetAverageFromHistory() {
