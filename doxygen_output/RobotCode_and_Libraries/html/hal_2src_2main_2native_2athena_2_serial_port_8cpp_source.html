73 HAL_SerialPortHandle HAL_InitializeSerialPortDirect(HAL_SerialPort port,
74  const char* portName,
75  int32_t* status) {
76  HAL_SerialPortHandle handle;
77  auto serialPort =
78  serialPortHandles->Allocate(static_cast<int16_t>(port), &handle, status);
79 
80  if (*status != 0) {
81  return HAL_kInvalidHandle;
82  }
83 
84  serialPort->portId = open(portName, O_RDWR | O_NOCTTY);
85  if (serialPort->portId < 0) {
86  *status = -errno;
87  if (*status == EACCES) {
88  *status = HAL_CONSOLE_OUT_ENABLED_ERROR;
89  }
90  serialPortHandles->Free(handle);
91  return HAL_kInvalidHandle;
92  }
93 
94  std::memset(&serialPort->tty, 0, sizeof(serialPort->tty));
95 
96  serialPort->baudRate = B9600;
97  if (cfsetospeed(&serialPort->tty,
98  static_cast<speed_t>(serialPort->baudRate)) != 0) {
99  *status = -errno;
100  close(serialPort->portId);
101  serialPortHandles->Free(handle);
102  return HAL_kInvalidHandle;
103  }
104  if (cfsetispeed(&serialPort->tty,
105  static_cast<speed_t>(serialPort->baudRate)) != 0) {
106  *status = -errno;
107  close(serialPort->portId);
108  serialPortHandles->Free(handle);
109  return HAL_kInvalidHandle;
110  }
111 
112  serialPort->tty.c_cflag &= ~PARENB;
113  serialPort->tty.c_cflag &= ~CSTOPB;
114  serialPort->tty.c_cflag &= ~CSIZE;
115  serialPort->tty.c_cflag |= CS8;
116 
117  serialPort->tty.c_cc[VMIN] = 0;
118  serialPort->tty.c_cc[VTIME] = 10;
119 
120  serialPort->tty.c_cflag |= CREAD | CLOCAL;
121 
122  serialPort->tty.c_lflag &= ~(ICANON | ECHO | ISIG);
123  serialPort->tty.c_iflag &= ~(IXON | IXOFF | IXANY);
124  /* Raw output mode, sends the raw and unprocessed data (send as it is).
125  * If it is in canonical mode and sending new line char then CR
126  * will be added as prefix and send as CR LF
127  */
128  serialPort->tty.c_oflag = ~OPOST;
129 
130  if (tcflush(serialPort->portId, TCIOFLUSH) != 0) {
131  *status = -errno;
132  close(serialPort->portId);
133  serialPortHandles->Free(handle);
134  return HAL_kInvalidHandle;
135  }
136  if (tcsetattr(serialPort->portId, TCSANOW, &serialPort->tty) != 0) {
137  *status = -errno;
138  close(serialPort->portId);
139  serialPortHandles->Free(handle);
140  return HAL_kInvalidHandle;
141  }
142  return handle;
143 }

435 int32_t HAL_ReadSerial(HAL_SerialPortHandle handle, char* buffer, int32_t count,
436  int32_t* status) {
437  // Don't do anything if 0 bytes were requested
438  if (count == 0) {
439  return 0;
440  }
441 
442  auto port = serialPortHandles->Get(handle);
443  if (!port) {
444  *status = HAL_HANDLE_ERROR;
445  return -1;
446  }
447 
448  int n = 0, loc = 0;
449  char buf = '\0';
450  std::memset(buffer, '\0', count);
451  *status = 0;
452 
453  do {
454  n = read(port->portId, &buf, 1);
455  if (n == 1) {
456  buffer[loc] = buf;
457  loc++;
458  // If buffer is full, return
459  if (loc == count) {
460  return loc;
461  }
462  // If terminating, and termination was hit return;
463  if (port->termination && buf == port->terminationChar) {
464  return loc;
465  }
466  } else if (n == -1) {
467  // ERROR
468  *status = errno;
469  return loc;
470  } else {
471  // If nothing read, timeout
472  return loc;
473  }
474  } while (true);
475 }

477 int32_t HAL_WriteSerial(HAL_SerialPortHandle handle, const char* buffer,
478  int32_t count, int32_t* status) {
479  auto port = serialPortHandles->Get(handle);
480  if (!port) {
481  *status = HAL_HANDLE_ERROR;
482  return -1;
483  }
484 
485  int written = 0, spot = 0;
486  do {
487  written = write(port->portId, buffer + spot, count - spot);
488  if (written < 0) {
489  *status = errno;
490  return spot;
491  }
492  spot += written;
493  } while (spot < count);
494  return spot;
