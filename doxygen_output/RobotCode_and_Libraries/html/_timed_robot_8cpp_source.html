21 void TimedRobot::StartCompetition() {
22  RobotInit();
23 
24  if constexpr (IsSimulation()) {
25  SimulationInit();
26  }
27 
28  // Tell the DS that the robot is ready to be enabled
29  std::puts("\n********** Robot program startup complete **********");
30  HAL_ObserveUserProgramStarting();
31 
32  // Loop forever, calling the appropriate mode-dependent function
33  while (true) {
34  // We don't have to check there's an element in the queue first because
35  // there's always at least one (the constructor adds one). It's reenqueued
36  // at the end of the loop.
37  auto callback = m_callbacks.pop();
38 
39  int32_t status = 0;
40  HAL_UpdateNotifierAlarm(
41  m_notifier, static_cast<uint64_t>(callback.expirationTime * 1e6),
42  &status);
43  FRC_CheckErrorStatus(status, "{}", "UpdateNotifierAlarm");
44 
45  uint64_t curTime = HAL_WaitForNotifierAlarm(m_notifier, &status);
46  if (curTime == 0 || status != 0) {
47  break;
48  }
49 
50  callback.func();
51 
52  callback.expirationTime += callback.period;
53  m_callbacks.push(std::move(callback));
54 
55  // Process all other callbacks that are ready to run
56  while (static_cast<uint64_t>(m_callbacks.top().expirationTime * 1e6) <=
57  curTime) {
58  callback = m_callbacks.pop();
59 
60  callback.func();
61 
62  callback.expirationTime += callback.period;
63  m_callbacks.push(std::move(callback));
64  }
65  }
66 }
