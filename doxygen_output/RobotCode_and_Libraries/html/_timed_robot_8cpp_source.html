24 void TimedRobot::StartCompetition() {
25  RobotInit();
26 
27  if (IsSimulation()) {
28  SimulationInit();
29  }
30 
31  // Tell the DS that the robot is ready to be enabled
32  HAL_ObserveUserProgramStarting();
33 
34  m_expirationTime = units::second_t{Timer::GetFPGATimestamp()} + m_period;
35  UpdateAlarm();
36 
37  // Loop forever, calling the appropriate mode-dependent function
38  while (true) {
39  int32_t status = 0;
40  uint64_t curTime = HAL_WaitForNotifierAlarm(m_notifier, &status);
41  if (curTime == 0 || status != 0) break;
42 
43  m_expirationTime += m_period;
44 
45  UpdateAlarm();
46 
47  // Call callback
48  LoopFunc();
49  }
50 }
