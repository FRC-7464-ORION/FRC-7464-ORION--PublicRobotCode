20 bool HolonomicDriveController::AtReference() const {
21  const auto& eTranslate = m_poseError.Translation();
22  const auto& eRotate = m_rotationError;
23  const auto& tolTranslate = m_poseTolerance.Translation();
24  const auto& tolRotate = m_poseTolerance.Rotation();
25  return units::math::abs(eTranslate.X()) < tolTranslate.X() &&
26  units::math::abs(eTranslate.Y()) < tolTranslate.Y() &&
27  units::math::abs(eRotate.Radians()) < tolRotate.Radians();
28 }

34 ChassisSpeeds HolonomicDriveController::Calculate(
35  const Pose2d& currentPose, const Pose2d& poseRef,
36  units::meters_per_second_t linearVelocityRef, const Rotation2d& angleRef) {
37  // If this is the first run, then we need to reset the theta controller to the
38  // current pose's heading.
39  if (m_firstRun) {
40  m_thetaController.Reset(currentPose.Rotation().Radians());
41  m_firstRun = false;
42  }
43 
44  // Calculate feedforward velocities (field-relative)
45  auto xFF = linearVelocityRef * poseRef.Rotation().Cos();
46  auto yFF = linearVelocityRef * poseRef.Rotation().Sin();
47  auto thetaFF = units::radians_per_second_t(m_thetaController.Calculate(
48  currentPose.Rotation().Radians(), angleRef.Radians()));
49 
50  m_poseError = poseRef.RelativeTo(currentPose);
51  m_rotationError = angleRef - currentPose.Rotation();
52 
53  if (!m_enabled) {
54  return ChassisSpeeds::FromFieldRelativeSpeeds(xFF, yFF, thetaFF,
55  currentPose.Rotation());
56  }
57 
58  // Calculate feedback velocities (based on position error).
59  auto xFeedback = units::meters_per_second_t(
60  m_xController.Calculate(currentPose.X().value(), poseRef.X().value()));
61  auto yFeedback = units::meters_per_second_t(
62  m_yController.Calculate(currentPose.Y().value(), poseRef.Y().value()));
63 
64  // Return next output.
65  return ChassisSpeeds::FromFieldRelativeSpeeds(
66  xFF + xFeedback, yFF + yFeedback, thetaFF, currentPose.Rotation());
67 }
