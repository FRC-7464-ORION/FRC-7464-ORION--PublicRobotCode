105  static diyfp mul(const diyfp& x, const diyfp& y) noexcept
106  {
107  static_assert(kPrecision == 64, "internal error");
108 
109  // Computes:
110  // f = round((x.f * y.f) / 2^q)
111  // e = x.e + y.e + q
112 
113  // Emulate the 64-bit * 64-bit multiplication:
114  //
115  // p = u * v
116  // = (u_lo + 2^32 u_hi) (v_lo + 2^32 v_hi)
117  // = (u_lo v_lo ) + 2^32 ((u_lo v_hi ) + (u_hi v_lo )) + 2^64 (u_hi v_hi )
118  // = (p0 ) + 2^32 ((p1 ) + (p2 )) + 2^64 (p3 )
119  // = (p0_lo + 2^32 p0_hi) + 2^32 ((p1_lo + 2^32 p1_hi) + (p2_lo + 2^32 p2_hi)) + 2^64 (p3 )
120  // = (p0_lo ) + 2^32 (p0_hi + p1_lo + p2_lo ) + 2^64 (p1_hi + p2_hi + p3)
121  // = (p0_lo ) + 2^32 (Q ) + 2^64 (H )
122  // = (p0_lo ) + 2^32 (Q_lo + 2^32 Q_hi ) + 2^64 (H )
123  //
124  // (Since Q might be larger than 2^32 - 1)
125  //
126  // = (p0_lo + 2^32 Q_lo) + 2^64 (Q_hi + H)
127  //
128  // (Q_hi + H does not overflow a 64-bit int)
129  //
130  // = p_lo + 2^64 p_hi
131 
132  const uint64_t u_lo = x.f & 0xFFFFFFFF;
133  const uint64_t u_hi = x.f >> 32;
134  const uint64_t v_lo = y.f & 0xFFFFFFFF;
135  const uint64_t v_hi = y.f >> 32;
136 
137  const uint64_t p0 = u_lo * v_lo;
138  const uint64_t p1 = u_lo * v_hi;
139  const uint64_t p2 = u_hi * v_lo;
140  const uint64_t p3 = u_hi * v_hi;
141 
142  const uint64_t p0_hi = p0 >> 32;
143  const uint64_t p1_lo = p1 & 0xFFFFFFFF;
144  const uint64_t p1_hi = p1 >> 32;
145  const uint64_t p2_lo = p2 & 0xFFFFFFFF;
146  const uint64_t p2_hi = p2 >> 32;
147 
148  uint64_t Q = p0_hi + p1_lo + p2_lo;
149 
150  // The full product might now be computed as
151  //
152  // p_hi = p3 + p2_hi + p1_hi + (Q >> 32)
153  // p_lo = p0_lo + (Q << 32)
154  //
155  // But in this particular case here, the full p_lo is not required.
156  // Effectively we only need to add the highest bit in p_lo to p_hi (and
157  // Q_hi + 1 does not overflow).
158 
159  Q += uint64_t{1} << (64 - 32 - 1); // round, ties up
160 
161  const uint64_t h = p3 + p2_hi + p1_hi + (Q >> 32);
162 
163  return diyfp(h, x.e + y.e + 64);
164  }

170  static diyfp normalize(diyfp x) noexcept
171  {
172  assert(x.f != 0);
173 
174  while ((x.f >> 63) == 0)
175  {
176  x.f <<= 1;
177  x.e--;
178  }
179 
180  return x;
181  }

187  static diyfp normalize_to(const diyfp& x, const int target_exponent) noexcept
188  {
189  const int delta = x.e - target_exponent;
190 
191  assert(delta >= 0);
192  assert(((x.f << delta) >> delta) == x.f);
193 
194  return diyfp(x.f << delta, target_exponent);
195  }

211 template <typename FloatType>
212 boundaries compute_boundaries(FloatType value)
213 {
214  assert(std::isfinite(value));
215  assert(value > 0);
216 
217  // Convert the IEEE representation into a diyfp.
218  //
219  // If v is denormal:
220  // value = 0.F * 2^(1 - bias) = ( F) * 2^(1 - bias - (p-1))
221  // If v is normalized:
222  // value = 1.F * 2^(E - bias) = (2^(p-1) + F) * 2^(E - bias - (p-1))
223 
224  static_assert(std::numeric_limits<FloatType>::is_iec559,
225  "internal error: dtoa_short requires an IEEE-754 floating-point implementation");
226 
227  constexpr int kPrecision = std::numeric_limits<FloatType>::digits; // = p (includes the hidden bit)
228  constexpr int kBias = std::numeric_limits<FloatType>::max_exponent - 1 + (kPrecision - 1);
229  constexpr int kMinExp = 1 - kBias;
230  constexpr uint64_t kHiddenBit = uint64_t{1} << (kPrecision - 1); // = 2^(p-1)
231 
232  using bits_type = typename std::conditional< kPrecision == 24, uint32_t, uint64_t >::type;
233 
234  const uint64_t bits = reinterpret_bits<bits_type>(value);
235  const uint64_t E = bits >> (kPrecision - 1);
236  const uint64_t F = bits & (kHiddenBit - 1);
237 
238  const bool is_denormal = (E == 0);
239  const diyfp v = is_denormal
240  ? diyfp(F, kMinExp)
241  : diyfp(F + kHiddenBit, static_cast<int>(E) - kBias);
242 
243  // Compute the boundaries m- and m+ of the floating-point value
244  // v = f * 2^e.
245  //
246  // Determine v- and v+, the floating-point predecessor and successor if v,
247  // respectively.
248  //
249  // v- = v - 2^e if f != 2^(p-1) or e == e_min (A)
250  // = v - 2^(e-1) if f == 2^(p-1) and e > e_min (B)
251  //
252  // v+ = v + 2^e
253  //
254  // Let m- = (v- + v) / 2 and m+ = (v + v+) / 2. All real numbers _strictly_
255  // between m- and m+ round to v, regardless of how the input rounding
256  // algorithm breaks ties.
257  //
258  // ---+-------------+-------------+-------------+-------------+--- (A)
259  // v- m- v m+ v+
260  //
261  // -----------------+------+------+-------------+-------------+--- (B)
262  // v- m- v m+ v+
263 
264  const bool lower_boundary_is_closer = (F == 0 and E > 1);
265  const diyfp m_plus = diyfp(2 * v.f + 1, v.e - 1);
266  const diyfp m_minus = lower_boundary_is_closer
267  ? diyfp(4 * v.f - 1, v.e - 2) // (B)
268  : diyfp(2 * v.f - 1, v.e - 1); // (A)
269 
270  // Determine the normalized w+ = m+.
271  const diyfp w_plus = diyfp::normalize(m_plus);
272 
273  // Determine w- = m- such that e_(w-) = e_(w+).
274  const diyfp w_minus = diyfp::normalize_to(m_minus, w_plus.e);
275 
276  return {diyfp::normalize(v), w_minus, w_plus};
277 }

279 // Given normalized diyfp w, Grisu needs to find a (normalized) cached
280 // power-of-ten c, such that the exponent of the product c * w = f * 2^e lies
281 // within a certain range [alpha, gamma] (Definition 3.2 from [1])
282 //
283 // alpha <= e = e_c + e_w + q <= gamma
284 //
285 // or
286 //
287 // f_c * f_w * 2^alpha <= f_c 2^(e_c) * f_w 2^(e_w) * 2^q
288 // <= f_c * f_w * 2^gamma
289 //
290 // Since c and w are normalized, i.e. 2^(q-1) <= f < 2^q, this implies
291 //
292 // 2^(q-1) * 2^(q-1) * 2^alpha <= c * w * 2^q < 2^q * 2^q * 2^gamma
293 //
294 // or
295 //
296 // 2^(q - 2 + alpha) <= c * w < 2^(q + gamma)
297 //
298 // The choice of (alpha,gamma) determines the size of the table and the form of
299 // the digit generation procedure. Using (alpha,gamma)=(-60,-32) works out well
300 // in practice:
301 //
302 // The idea is to cut the number c * w = f * 2^e into two parts, which can be
303 // processed independently: An integral part p1, and a fractional part p2:
304 //
305 // f * 2^e = ( (f div 2^-e) * 2^-e + (f mod 2^-e) ) * 2^e
306 // = (f div 2^-e) + (f mod 2^-e) * 2^e
307 // = p1 + p2 * 2^e
308 //
309 // The conversion of p1 into decimal form requires a series of divisions and
310 // modulos by (a power of) 10. These operations are faster for 32-bit than for
311 // 64-bit integers, so p1 should ideally fit into a 32-bit integer. This can be
312 // achieved by choosing
313 //
314 // -e >= 32 or e <= -32 := gamma
315 //
316 // In order to convert the fractional part
317 //
318 // p2 * 2^e = p2 / 2^-e = d[-1] / 10^1 + d[-2] / 10^2 + ...
319 //
320 // into decimal form, the fraction is repeatedly multiplied by 10 and the digits
321 // d[-i] are extracted in order:
322 //
323 // (10 * p2) div 2^-e = d[-1]
324 // (10 * p2) mod 2^-e = d[-2] / 10^1 + ...
325 //
326 // The multiplication by 10 must not overflow. It is sufficient to choose
327 //
328 // 10 * p2 < 16 * p2 = 2^4 * p2 <= 2^64.
329 //
330 // Since p2 = f mod 2^-e < 2^-e,
331 //
332 // -e <= 60 or e >= -60 := alpha
333 
334 constexpr int kAlpha = -60;
335 constexpr int kGamma = -32;

351 inline cached_power get_cached_power_for_binary_exponent(int e)
352 {
353  // Now
354  //
355  // alpha <= e_c + e + q <= gamma (1)
356  // ==> f_c * 2^alpha <= c * 2^e * 2^q
357  //
358  // and since the c's are normalized, 2^(q-1) <= f_c,
359  //
360  // ==> 2^(q - 1 + alpha) <= c * 2^(e + q)
361  // ==> 2^(alpha - e - 1) <= c
362  //
363  // If c were an exakt power of ten, i.e. c = 10^k, one may determine k as
364  //
365  // k = ceil( log_10( 2^(alpha - e - 1) ) )
366  // = ceil( (alpha - e - 1) * log_10(2) )
367  //
368  // From the paper:
369  // "In theory the result of the procedure could be wrong since c is rounded,
370  // and the computation itself is approximated [...]. In practice, however,
371  // this simple function is sufficient."
372  //
373  // For IEEE double precision floating-point numbers converted into
374  // normalized diyfp's w = f * 2^e, with q = 64,
375  //
376  // e >= -1022 (min IEEE exponent)
377  // -52 (p - 1)
378  // -52 (p - 1, possibly normalize denormal IEEE numbers)
379  // -11 (normalize the diyfp)
380  // = -1137
381  //
382  // and
383  //
384  // e <= +1023 (max IEEE exponent)
385  // -52 (p - 1)
386  // -11 (normalize the diyfp)
387  // = 960
388  //
389  // This binary exponent range [-1137,960] results in a decimal exponent
390  // range [-307,324]. One does not need to store a cached power for each
391  // k in this range. For each such k it suffices to find a cached power
392  // such that the exponent of the product lies in [alpha,gamma].
393  // This implies that the difference of the decimal exponents of adjacent
394  // table entries must be less than or equal to
395  //
396  // floor( (gamma - alpha) * log_10(2) ) = 8.
397  //
398  // (A smaller distance gamma-alpha would require a larger table.)

571 inline void grisu2_round(char* buf, int len, uint64_t dist, uint64_t delta,
572  uint64_t rest, uint64_t ten_k)
573 {
574  assert(len >= 1);
575  assert(dist <= delta);
576  assert(rest <= delta);
577  assert(ten_k > 0);
578 
579  // <--------------------------- delta ---->
580  // <---- dist --------->
581  // --------------[------------------+-------------------]--------------
582  // M- w M+
583  //
584  // ten_k
585  // <------>
586  // <---- rest ---->
587  // --------------[------------------+----+--------------]--------------
588  // w V
589  // = buf * 10^k
590  //
591  // ten_k represents a unit-in-the-last-place in the decimal representation
592  // stored in buf.
593  // Decrement buf by ten_k while this takes buf closer to w.
594 
595  // The tests are written in this order to avoid overflow in unsigned
596  // integer arithmetic.
597 
598  while (rest < dist
599  and delta - rest >= ten_k
600  and (rest + ten_k < dist or dist - rest > rest + ten_k - dist))
601  {
602  assert(buf[len - 1] != '0');
603  buf[len - 1]--;
604  rest += ten_k;
605  }
606 }

612 inline void grisu2_digit_gen(char* buffer, int& length, int& decimal_exponent,
613  diyfp M_minus, diyfp w, diyfp M_plus)
614 {
615  static_assert(kAlpha >= -60, "internal error");
616  static_assert(kGamma <= -32, "internal error");
617 
618  // Generates the digits (and the exponent) of a decimal floating-point
619  // number V = buffer * 10^decimal_exponent in the range [M-, M+]. The diyfp's
620  // w, M- and M+ share the same exponent e, which satisfies alpha <= e <= gamma.
621  //
622  // <--------------------------- delta ---->
623  // <---- dist --------->
624  // --------------[------------------+-------------------]--------------
625  // M- w M+
626  //
627  // Grisu2 generates the digits of M+ from left to right and stops as soon as
628  // V is in [M-,M+].
629 
630  assert(M_plus.e >= kAlpha);
631  assert(M_plus.e <= kGamma);
632 
633  uint64_t delta = diyfp::sub(M_plus, M_minus).f; // (significand of (M+ - M-), implicit exponent is e)
634  uint64_t dist = diyfp::sub(M_plus, w ).f; // (significand of (M+ - w ), implicit exponent is e)
635 
636  // Split M+ = f * 2^e into two parts p1 and p2 (note: e < 0):
637  //
638  // M+ = f * 2^e
639  // = ((f div 2^-e) * 2^-e + (f mod 2^-e)) * 2^e
640  // = ((p1 ) * 2^-e + (p2 )) * 2^e
641  // = p1 + p2 * 2^e
642 
643  const diyfp one(uint64_t{1} << -M_plus.e, M_plus.e);
644 
645  uint32_t p1 = static_cast<uint32_t>(M_plus.f >> -one.e); // p1 = f div 2^-e (Since -e >= 32, p1 fits into a 32-bit int.)
646  uint64_t p2 = M_plus.f & (one.f - 1); // p2 = f mod 2^-e
647 
648  // 1)
649  //
650  // Generate the digits of the integral part p1 = d[n-1]...d[1]d[0]
651 
652  assert(p1 > 0);
653 
654  uint32_t pow10;
655  const int k = find_largest_pow10(p1, pow10);
656 
657  // 10^(k-1) <= p1 < 10^k, pow10 = 10^(k-1)
658  //
659  // p1 = (p1 div 10^(k-1)) * 10^(k-1) + (p1 mod 10^(k-1))
660  // = (d[k-1] ) * 10^(k-1) + (p1 mod 10^(k-1))
661  //
662  // M+ = p1 + p2 * 2^e
663  // = d[k-1] * 10^(k-1) + (p1 mod 10^(k-1)) + p2 * 2^e
664  // = d[k-1] * 10^(k-1) + ((p1 mod 10^(k-1)) * 2^-e + p2) * 2^e
665  // = d[k-1] * 10^(k-1) + ( rest) * 2^e
666  //
667  // Now generate the digits d[n] of p1 from left to right (n = k-1,...,0)
668  //
669  // p1 = d[k-1]...d[n] * 10^n + d[n-1]...d[0]
670  //
671  // but stop as soon as
672  //
673  // rest * 2^e = (d[n-1]...d[0] * 2^-e + p2) * 2^e <= delta * 2^e
674 
675  int n = k;
676  while (n > 0)
677  {
678  // Invariants:
679  // M+ = buffer * 10^n + (p1 + p2 * 2^e) (buffer = 0 for n = k)
680  // pow10 = 10^(n-1) <= p1 < 10^n
681  //
682  const uint32_t d = p1 / pow10; // d = p1 div 10^(n-1)
683  const uint32_t r = p1 % pow10; // r = p1 mod 10^(n-1)
684  //
685  // M+ = buffer * 10^n + (d * 10^(n-1) + r) + p2 * 2^e
686  // = (buffer * 10 + d) * 10^(n-1) + (r + p2 * 2^e)
687  //
688  assert(d <= 9);
689  buffer[length++] = static_cast<char>('0' + d); // buffer := buffer * 10 + d
690  //
691  // M+ = buffer * 10^(n-1) + (r + p2 * 2^e)
692  //
693  p1 = r;
694  n--;
695  //
696  // M+ = buffer * 10^n + (p1 + p2 * 2^e)
697  // pow10 = 10^n
698  //
699 
700  // Now check if enough digits have been generated.
701  // Compute
702  //
703  // p1 + p2 * 2^e = (p1 * 2^-e + p2) * 2^e = rest * 2^e
704  //
705  // Note:
706  // Since rest and delta share the same exponent e, it suffices to
707  // compare the significands.
708  const uint64_t rest = (uint64_t{p1} << -one.e) + p2;
709  if (rest <= delta)
710  {
711  // V = buffer * 10^n, with M- <= V <= M+.
712 
713  decimal_exponent += n;
714 
715  // We may now just stop. But instead look if the buffer could be
716  // decremented to bring V closer to w.
717  //
718  // pow10 = 10^n is now 1 ulp in the decimal representation V.
719  // The rounding procedure works with diyfp's with an implicit
720  // exponent of e.
721  //
722  // 10^n = (10^n * 2^-e) * 2^e = ulp * 2^e
723  //
724  const uint64_t ten_n = uint64_t{pow10} << -one.e;
725  grisu2_round(buffer, length, dist, delta, rest, ten_n);
726 
727  return;
728  }
729 
730  pow10 /= 10;
731  //
732  // pow10 = 10^(n-1) <= p1 < 10^n
733  // Invariants restored.
734  }
735 
736  // 2)
737  //
738  // The digits of the integral part have been generated:
739  //
740  // M+ = d[k-1]...d[1]d[0] + p2 * 2^e
741  // = buffer + p2 * 2^e
742  //
743  // Now generate the digits of the fractional part p2 * 2^e.
744  //
745  // Note:
746  // No decimal point is generated: the exponent is adjusted instead.
747  //
748  // p2 actually represents the fraction
749  //
750  // p2 * 2^e
751  // = p2 / 2^-e
752  // = d[-1] / 10^1 + d[-2] / 10^2 + ...
753  //
754  // Now generate the digits d[-m] of p1 from left to right (m = 1,2,...)
755  //
756  // p2 * 2^e = d[-1]d[-2]...d[-m] * 10^-m
757  // + 10^-m * (d[-m-1] / 10^1 + d[-m-2] / 10^2 + ...)
758  //
759  // using
760  //
761  // 10^m * p2 = ((10^m * p2) div 2^-e) * 2^-e + ((10^m * p2) mod 2^-e)
762  // = ( d) * 2^-e + ( r)
763  //
764  // or
765  // 10^m * p2 * 2^e = d + r * 2^e
766  //
767  // i.e.
768  //
769  // M+ = buffer + p2 * 2^e
770  // = buffer + 10^-m * (d + r * 2^e)
771  // = (buffer * 10^m + d) * 10^-m + 10^-m * r * 2^e
772  //
773  // and stop as soon as 10^-m * r * 2^e <= delta * 2^e
774 
775  assert(p2 > delta);
776 
777  int m = 0;
778  for (;;)
779  {
780  // Invariant:
781  // M+ = buffer * 10^-m + 10^-m * (d[-m-1] / 10 + d[-m-2] / 10^2 + ...) * 2^e
782  // = buffer * 10^-m + 10^-m * (p2 ) * 2^e
783  // = buffer * 10^-m + 10^-m * (1/10 * (10 * p2) ) * 2^e
784  // = buffer * 10^-m + 10^-m * (1/10 * ((10*p2 div 2^-e) * 2^-e + (10*p2 mod 2^-e)) * 2^e
785  //
786  assert(p2 <= UINT64_MAX / 10);
787  p2 *= 10;
788  const uint64_t d = p2 >> -one.e; // d = (10 * p2) div 2^-e
789  const uint64_t r = p2 & (one.f - 1); // r = (10 * p2) mod 2^-e
790  //
791  // M+ = buffer * 10^-m + 10^-m * (1/10 * (d * 2^-e + r) * 2^e
792  // = buffer * 10^-m + 10^-m * (1/10 * (d + r * 2^e))
793  // = (buffer * 10 + d) * 10^(-m-1) + 10^(-m-1) * r * 2^e
794  //
795  assert(d <= 9);
796  buffer[length++] = static_cast<char>('0' + d); // buffer := buffer * 10 + d
797  //
798  // M+ = buffer * 10^(-m-1) + 10^(-m-1) * r * 2^e
799  //
800  p2 = r;
801  m++;
802  //
803  // M+ = buffer * 10^-m + 10^-m * p2 * 2^e
804  // Invariant restored.
805 
806  // Check if enough digits have been generated.
807  //
808  // 10^-m * p2 * 2^e <= delta * 2^e
809  // p2 * 2^e <= 10^m * delta * 2^e
810  // p2 <= 10^m * delta
811  delta *= 10;
812  dist *= 10;
813  if (p2 <= delta)
814  {
815  break;
816  }
817  }
818 
819  // V = buffer * 10^-m, with M- <= V <= M+.
820 
821  decimal_exponent -= m;
822 
823  // 1 ulp in the decimal representation is now 10^-m.
824  // Since delta and dist are now scaled by 10^m, we need to do the
825  // same with ulp in order to keep the units in sync.
826  //
827  // 10^m * 10^-m = 1 = 2^-e * 2^e = ten_m * 2^e
828  //
829  const uint64_t ten_m = one.f;
830  grisu2_round(buffer, length, dist, delta, p2, ten_m);
831 
832  // By construction this algorithm generates the shortest possible decimal
833  // number (Loitsch, Theorem 6.2) which rounds back to w.
834  // For an input number of precision p, at least
835  //
836  // N = 1 + ceil(p * log_10(2))
837  //
838  // decimal digits are sufficient to identify all binary floating-point
839  // numbers (Matula, "In-and-Out conversions").
840  // This implies that the algorithm does not produce more than N decimal
841  // digits.
842  //
843  // N = 17 for p = 53 (IEEE double precision)
844  // N = 9 for p = 24 (IEEE single precision)
845 }

852 inline void grisu2(char* buf, int& len, int& decimal_exponent,
853  diyfp m_minus, diyfp v, diyfp m_plus)
854 {
855  assert(m_plus.e == m_minus.e);
856  assert(m_plus.e == v.e);
857 
858  // --------(-----------------------+-----------------------)-------- (A)
859  // m- v m+
860  //
861  // --------------------(-----------+-----------------------)-------- (B)
862  // m- v m+
863  //
864  // First scale v (and m- and m+) such that the exponent is in the range
865  // [alpha, gamma].
866 
867  const cached_power cached = get_cached_power_for_binary_exponent(m_plus.e);
868 
869  const diyfp c_minus_k(cached.f, cached.e); // = c ~= 10^-k
870 
871  // The exponent of the products is = v.e + c_minus_k.e + q and is in the range [alpha,gamma]
872  const diyfp w = diyfp::mul(v, c_minus_k);
873  const diyfp w_minus = diyfp::mul(m_minus, c_minus_k);
874  const diyfp w_plus = diyfp::mul(m_plus, c_minus_k);
875 
876  // ----(---+---)---------------(---+---)---------------(---+---)----
877  // w- w w+
878  // = c*m- = c*v = c*m+
879  //
880  // diyfp::mul rounds its result and c_minus_k is approximated too. w, w- and
881  // w+ are now off by a small amount.
882  // In fact:
883  //
884  // w - v * 10^k < 1 ulp
885  //
886  // To account for this inaccuracy, add resp. subtract 1 ulp.
887  //
888  // --------+---[---------------(---+---)---------------]---+--------
889  // w- M- w M+ w+
890  //
891  // Now any number in [M-, M+] (bounds included) will round to w when input,
892  // regardless of how the input rounding algorithm breaks ties.
893  //
894  // And digit_gen generates the shortest possible such number in [M-, M+].
895  // Note that this does not mean that Grisu2 always generates the shortest
896  // possible number in the interval (m-, m+).
897  const diyfp M_minus(w_minus.f + 1, w_minus.e);
898  const diyfp M_plus (w_plus.f - 1, w_plus.e );
899 
900  decimal_exponent = -cached.k; // = -(-k) = k
901 
902  grisu2_digit_gen(buf, len, decimal_exponent, M_minus, w, M_plus);
903 }

1000  int min_exp, int max_exp)
1001 {
1002  assert(min_exp < 0);
1003  assert(max_exp > 0);
1004 
1005  const int k = len;
1006  const int n = len + decimal_exponent;
1007 
1008  // v = buf * 10^(n-k)
1009  // k is the length of the buffer (number of decimal digits)
1010  // n is the position of the decimal point relative to the start of the buffer.
1011

1013  {
1014  // digits[000]
1015  // len <= max_exp + 2
1016 
1017  std::memset(buf + k, '0', static_cast<size_t>(n - k));
1018  // Make it look like a floating-point number (#362, #378)
1019  buf[n + 0] = '.';
1020  buf[n + 1] = '0';
1021  return buf + (n + 2);
1022  }
1023 
1024  if (0 < n and n <= max_exp)
1025  {
1026  // dig.its
1027  // len <= max_digits10 + 1
1028 
1029  assert(k > n);
1030 
1031  std::memmove(buf + (n + 1), buf + n, static_cast<size_t>(k - n));
1032  buf[n] = '.';
1033  return buf + (k + 1);
1034  }
1035 
1036  if (min_exp < n and n <= 0)
1037  {
1038  // 0.[000]digits
1039  // len <= 2 + (-min_exp - 1) + max_digits10
1040 
1041  std::memmove(buf + (2 + -n), buf, static_cast<size_t>(k));
1042  buf[0] = '0';
1043  buf[1] = '.';
1044  std::memset(buf + 2, '0', static_cast<size_t>(-n));
1045  return buf + (2 + (-n) + k);
1046  }
1047 
1048  if (k == 1)
1049  {
1050  // dE+123
1051  // len <= 1 + 5
1052 
1053  buf += 1;
1054  }
1055  else
1056  {
1057  // d.igitsE+123
1058  // len <= max_digits10 + 1 + 5
1059 
1060  std::memmove(buf + 2, buf + 1, static_cast<size_t>(k - 1));
1061  buf[1] = '.';
1062  buf += 1 + k;
1063  }
1064 
1065  *buf++ = 'e';
1066  return append_exponent(buf, n - 1);
1067 }
