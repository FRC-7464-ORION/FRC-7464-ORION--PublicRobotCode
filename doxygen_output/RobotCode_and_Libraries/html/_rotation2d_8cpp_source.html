21 Rotation2d::Rotation2d(double x, double y) {
22  const auto magnitude = std::hypot(x, y);
23  if (magnitude > 1e-6) {
24  m_sin = y / magnitude;
25  m_cos = x / magnitude;
26  } else {
27  m_sin = 0.0;
28  m_cos = 1.0;
29  }
30  m_value = units::radian_t(std::atan2(m_sin, m_cos));
31 }
