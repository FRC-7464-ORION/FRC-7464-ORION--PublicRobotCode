26 AnalogInput::AnalogInput(int channel) {
27  if (!SensorUtil::CheckAnalogInputChannel(channel)) {
28  wpi_setWPIErrorWithContext(ChannelIndexOutOfRange,
29  "Analog Input " + wpi::Twine(channel));
30  return;
31  }
32 
33  m_channel = channel;
34 
35  HAL_PortHandle port = HAL_GetPort(channel);
36  int32_t status = 0;
37  m_port = HAL_InitializeAnalogInputPort(port, &status);
38  if (status != 0) {
39  wpi_setHALErrorWithRange(status, 0, HAL_GetNumAnalogInputs(), channel);
40  m_channel = std::numeric_limits<int>::max();
41  m_port = HAL_kInvalidHandle;
42  return;
43  }
44 
45  HAL_Report(HALUsageReporting::kResourceType_AnalogChannel, channel + 1);
46 
47  SendableRegistry::GetInstance().AddLW(this, "AnalogInput", channel);
48 }

155 void AnalogInput::ResetAccumulator() {
156  if (StatusIsFatal()) return;
157  int32_t status = 0;
158  HAL_ResetAccumulator(m_port, &status);
159  wpi_setHALError(status);
160 
161  if (!StatusIsFatal()) {
162  // Wait until the next sample, so the next call to GetAccumulator*()
163  // won't have old values.
164  const double sampleTime = 1.0 / GetSampleRate();
165  const double overSamples = 1 << GetOversampleBits();
166  const double averageSamples = 1 << GetAverageBits();
167  Wait(sampleTime * overSamples * averageSamples);
168  }
169 }
