54  ySpeed = ApplyDeadband(ySpeed, m_deadband);
55  xSpeed = ApplyDeadband(xSpeed, m_deadband);

57  auto [frontLeft, frontRight, rearLeft, rearRight] =

89 MecanumDrive::WheelSpeeds MecanumDrive::DriveCartesianIK(double ySpeed,
90  double xSpeed,
91  double zRotation,
92  double gyroAngle) {
93  ySpeed = std::clamp(ySpeed, -1.0, 1.0);
94  xSpeed = std::clamp(xSpeed, -1.0, 1.0);
95 
96  // Compensate for gyro angle.
97  Vector2d input{ySpeed, xSpeed};
98  input.Rotate(-gyroAngle);
99 
100  double wheelSpeeds[4];
101  wheelSpeeds[kFrontLeft] = input.x + input.y + zRotation;
102  wheelSpeeds[kFrontRight] = input.x - input.y - zRotation;
103  wheelSpeeds[kRearLeft] = input.x - input.y + zRotation;
104  wheelSpeeds[kRearRight] = input.x + input.y - zRotation;
105 
106  Desaturate(wheelSpeeds);
107 
108  return {wheelSpeeds[kFrontLeft], wheelSpeeds[kFrontRight],
109  wheelSpeeds[kRearLeft], wheelSpeeds[kRearRight]};
110 }
