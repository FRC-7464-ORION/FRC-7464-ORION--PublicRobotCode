67 units::turns_per_second_t Tachometer::GetRevolutionsPerSecond() const {
68  auto period = GetPeriod();
69  if (period.to<double>() == 0) {
70  return units::turns_per_second_t{0.0};
71  }
72  int edgesPerRevolution = GetEdgesPerRevolution();
73  if (edgesPerRevolution == 0) {
74  return units::turns_per_second_t{0.0};
75  }
76  auto rotationHz = ((1.0 / edgesPerRevolution) / period);
77  return units::turns_per_second_t{rotationHz.to<double>()};
78 }
