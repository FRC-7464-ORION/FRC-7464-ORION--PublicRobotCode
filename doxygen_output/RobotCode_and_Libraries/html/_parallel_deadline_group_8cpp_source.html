24 void ParallelDeadlineGroup::Execute() {

56 void ParallelDeadlineGroup::AddCommands(
57  std::vector<std::unique_ptr<Command>>&& commands) {
58  if (!RequireUngrouped(commands)) {
59  return;
60  }
61 
62  if (!m_finished) {
63  throw FRC_MakeError(frc::err::CommandIllegalUse, "{}",
64  "Commands cannot be added to a CommandGroup "
65  "while the group is running");
66  }
67 
68  for (auto&& command : commands) {
69  if (RequirementsDisjoint(this, command.get())) {
70  command->SetGrouped(true);
71  AddRequirements(command->GetRequirements());
72  m_runWhenDisabled &= command->RunsWhenDisabled();
73  m_commands.emplace_back(std::move(command), false);
74  } else {
75  throw FRC_MakeError(frc::err::CommandIllegalUse, "{}",
76  "Multiple commands in a parallel group cannot "
77  "require the same subsystems");
78  }
79  }
80 }
