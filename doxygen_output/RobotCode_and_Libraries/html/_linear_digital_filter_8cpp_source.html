99 double LinearDigitalFilter::Get() const {
100  double retVal = 0.0;
101 
102  // Calculate the new value
103  for (size_t i = 0; i < m_inputGains.size(); i++) {
104  retVal += m_inputs[i] * m_inputGains[i];
105  }
106  for (size_t i = 0; i < m_outputGains.size(); i++) {
107  retVal -= m_outputs[i] * m_outputGains[i];
108  }
109 
110  return retVal;
111 }
