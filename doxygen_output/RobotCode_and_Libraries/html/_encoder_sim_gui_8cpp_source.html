128  static void DistancePerPulseCallbackFunc(const char*, void* param,
129  const HAL_Value* value) {
130  if (value->type == HAL_DOUBLE) {
131  auto self = static_cast<EncoderSimModel*>(param);
132  double distPerPulse = value->data.v_double;
133  self->m_distancePerPulse.SetValue(distPerPulse);
134  self->m_distance.SetValue(self->m_count.GetValue() * distPerPulse);
135  double period = self->m_period.GetValue();
136  if (period == 0) {
137  self->m_rate.SetValue(std::numeric_limits<double>::infinity());
138  } else if (period == std::numeric_limits<double>::infinity()) {
139  self->m_rate.SetValue(0);
140  } else {
141  self->m_rate.SetValue(static_cast<float>(distPerPulse / period));
142  }
143  }
144  }
145 
146  static void CountCallbackFunc(const char*, void* param,
147  const HAL_Value* value) {
148  if (value->type == HAL_INT) {
149  auto self = static_cast<EncoderSimModel*>(param);
150  double count = value->data.v_int;
151  self->m_count.SetValue(count);
152  self->m_distance.SetValue(count * self->m_distancePerPulse.GetValue());
153  }
154  }
155 
156  static void PeriodCallbackFunc(const char*, void* param,
157  const HAL_Value* value) {
158  if (value->type == HAL_DOUBLE) {
159  auto self = static_cast<EncoderSimModel*>(param);
160  double period = value->data.v_double;
161  self->m_period.SetValue(period);
162  if (period == 0) {
163  self->m_rate.SetValue(std::numeric_limits<double>::infinity());
164  } else if (period == std::numeric_limits<double>::infinity()) {
165  self->m_rate.SetValue(0);
166  } else {
167  self->m_rate.SetValue(
168  static_cast<float>(self->m_distancePerPulse.GetValue() / period));
169  }
170  }
171  }
172 
173  static void DirectionCallbackFunc(const char*, void* param,
174  const HAL_Value* value) {
175  if (value->type == HAL_BOOLEAN) {
176  static_cast<EncoderSimModel*>(param)->m_direction.SetValue(
177  value->data.v_boolean);
178  }
179  }
