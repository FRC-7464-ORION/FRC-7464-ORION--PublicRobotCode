136  m_sortedHubPath = m_unsortedHubPath;
137  std::sort(m_sortedHubPath.begin(), m_sortedHubPath.end(),
138  [](const wpi::SmallVectorImpl<char>& lhs,
139  const wpi::SmallVectorImpl<char>& rhs) -> int {
140  std::string_view lhsRef(lhs.begin(), lhs.size());
141  std::string_view rhsRef(rhs.begin(), rhs.size());
142  return lhsRef.compare(rhsRef);
143  });

146 void SerialHelper::CoiteratedSort(
147  wpi::SmallVectorImpl<wpi::SmallString<16>>& vec) {
148  wpi::SmallVector<wpi::SmallString<16>, 4> sortedVec;
149  for (auto& str : m_sortedHubPath) {
150  for (size_t i = 0; i < m_unsortedHubPath.size(); i++) {
151  if (wpi::equals(std::string_view{m_unsortedHubPath[i].begin(),
152  m_unsortedHubPath[i].size()},
153  std::string_view{str.begin(), str.size()})) {
154  sortedVec.push_back(vec[i]);
155  break;
156  }
157  }
158  }
159  vec.swap(sortedVec);
160 }

162 void SerialHelper::QueryHubPaths(int32_t* status) {
163  // VISA resource matching string
164  const char* str = "?*";
165  // Items needed for VISA
166  ViUInt32 retCnt = 0;
167  ViFindList viList = 0;
168  ViChar desc[VI_FIND_BUFLEN];
169  *status = viFindRsrc(m_resourceHandle, const_cast<char*>(str), &viList,
170  &retCnt, desc);
171 
172  if (*status < 0) {
173  // Handle the bad status elsewhere
174  // Note let positive statii (warnings) continue
175  goto done;
176  }
177  // Status might be positive, so reset it to 0
178  *status = 0;
179 
180  // Storage buffer for Visa call
181  char osName[256];
182 
183  // Loop through all returned VISA objects.
184  // Increment the internal VISA ptr every loop
185  for (size_t i = 0; i < retCnt; i++, viFindNext(viList, desc)) {
186  // Ignore any matches to the 2 onboard ports
187  if (std::strcmp(OnboardResourceVISA, desc) == 0 ||
188  std::strcmp(MxpResourceVISA, desc) == 0) {
189  continue;
190  }
191 
192  // Open the resource, grab its interface name, and close it.
193  ViSession vSession;
194  *status = viOpen(m_resourceHandle, desc, VI_NULL, VI_NULL, &vSession);
195  if (*status < 0)
196  goto done;
197  *status = 0;
198 
199  *status = viGetAttribute(vSession, VI_ATTR_INTF_INST_NAME, &osName);
200  // Ignore an error here, as we want to close the session on an error
201  // Use a separate close variable so we can check
202  ViStatus closeStatus = viClose(vSession);
203  if (*status < 0)
204  goto done;
205  if (closeStatus < 0)
206  goto done;
207  *status = 0;
208 
209  // split until (/dev/
210  std::string_view devNameRef = wpi::split(osName, "(/dev/").second;
211  // String not found, continue
212  if (wpi::equals(devNameRef, ""))
213  continue;
214 
215  // Split at )
216  std::string_view matchString = wpi::split(devNameRef, ')').first;
217  if (wpi::equals(matchString, devNameRef))
218  continue;
219 
220  // Search directories to get a list of system accessors
221  // The directories we need are not symbolic, so we can safely
222  // disable symbolic links.
223  std::error_code ec;
224  for (auto& p : fs::recursive_directory_iterator("/sys/devices/soc0", ec)) {
225  if (ec)
226  break;
227  std::string path = p.path();
228  if (path.find("amba") == std::string::npos)
229  continue;
230  if (path.find("usb") == std::string::npos)
231  continue;
232  if (path.find(matchString) == std::string::npos)
233  continue;
234 
235  wpi::SmallVector<std::string_view, 16> pathSplitVec;
236  // Split path into individual directories
237  wpi::split(path, pathSplitVec, '/', -1, false);
238 
239  // Find each individual item index
240  int findusb = -1;
241  int findtty = -1;
242  int findregex = -1;
243  for (size_t i = 0; i < pathSplitVec.size(); i++) {
244  if (findusb == -1 && wpi::equals(pathSplitVec[i], "usb1")) {
245  findusb = i;
246  }
247  if (findtty == -1 && wpi::equals(pathSplitVec[i], "tty")) {
248  findtty = i;
249  }
250  if (findregex == -1 && wpi::equals(pathSplitVec[i], matchString)) {
251  findregex = i;
252  }
253  }
254 
255  // Get the index for our device
256  int hubIndex = findtty;
257  if (findtty == -1)
258  hubIndex = findregex;
259 
260  int devStart = findusb + 1;
261 
262  if (hubIndex < devStart)
263  continue;
264 
265  // Add our devices to our list
266  m_unsortedHubPath.emplace_back(
267  std::string_view{pathSplitVec[hubIndex - 2]});
268  m_visaResource.emplace_back(desc);
269  m_osResource.emplace_back(
270  wpi::split(wpi::split(osName, "(").second, ")").first);
271  break;
272  }
273  }
274 
275  SortHubPathVector();
276 
277  CoiteratedSort(m_visaResource);
278  CoiteratedSort(m_osResource);
279 done:
280  viClose(viList);
281 }

283 int32_t SerialHelper::GetIndexForPort(HAL_SerialPort port, int32_t* status) {
284  // Hold lock whenever we're using the names array
285  std::scoped_lock lock(m_nameMutex);
286 
287  std::string portString = m_usbNames[port - 2];
288 
289  wpi::SmallVector<int32_t, 4> indices;
290 
291  // If port has not been assigned, find the one to assign
292  if (portString.empty()) {
293  for (size_t i = 0; i < 2; i++) {
294  // Remove all used ports
295  auto idx = std::find_if(
296  m_sortedHubPath.begin(), m_sortedHubPath.end(),
297  [&](const auto& s) { return wpi::equals(s, m_usbNames[i]); });
298  if (idx != m_sortedHubPath.end()) {
299  // found
300  m_sortedHubPath.erase(idx);
301  }
302  if (m_usbNames[i] == "") {
303  indices.push_back(i);
304  }
305  }
306 
307  int32_t idx = -1;
308  for (size_t i = 0; i < indices.size(); i++) {
309  if (indices[i] == port - 2) {
310  idx = i;
311  break;
312  }
313  }
314 
315  if (idx == -1) {
316  *status = HAL_SERIAL_PORT_NOT_FOUND;
317  return -1;
318  }
319 
320  if (idx >= static_cast<int32_t>(m_sortedHubPath.size())) {
321  *status = HAL_SERIAL_PORT_NOT_FOUND;
322  return -1;
323  }
324 
325  portString = m_sortedHubPath[idx].str();
326  m_usbNames[port - 2] = portString;
327  }
328 
329  int retIndex = -1;
330 
331  for (size_t i = 0; i < m_sortedHubPath.size(); i++) {
332  if (wpi::equals(m_sortedHubPath[i], portString)) {
333  retIndex = i;
334  break;
335  }
336  }
337 
338  return retIndex;
339 }
