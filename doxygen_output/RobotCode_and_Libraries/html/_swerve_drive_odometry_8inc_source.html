22 template <size_t NumModules>
23 template <typename... ModuleStates>
24 const Pose2d& frc::SwerveDriveOdometry<NumModules>::UpdateWithTime(
25  units::second_t currentTime, const Rotation2d& gyroAngle,
26  ModuleStates&&... moduleStates) {
27  units::second_t deltaTime =
28  (m_previousTime >= 0_s) ? currentTime - m_previousTime : 0_s;
29  m_previousTime = currentTime;
30 
31  auto angle = gyroAngle + m_gyroOffset;
32 
33  auto [dx, dy, dtheta] = m_kinematics.ToChassisSpeeds(moduleStates...);
34  static_cast<void>(dtheta);
35 
36  auto newPose = m_pose.Exp(
37  {dx * deltaTime, dy * deltaTime, (angle - m_previousAngle).Radians()});
38 
39  m_previousAngle = angle;
40  m_pose = {newPose.Translation(), angle};
41 
42  return m_pose;
43 }
