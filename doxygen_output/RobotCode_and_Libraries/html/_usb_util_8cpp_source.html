23 static std::string GetUsbNameFromFile(int vendor, int product) {
24  int fd = open("/var/lib/usbutils/usb.ids", O_RDONLY);
25  if (fd < 0) {
26  return {};
27  }
28 
29  wpi::SmallString<128> buf;
30  wpi::raw_fd_istream is{fd, true};
31 
32  // build vendor and product 4-char hex strings
33  auto vendorStr = fmt::format("{:04x}", vendor);
34  auto productStr = fmt::format("{:04x}", product);
35 
36  // scan file
37  wpi::SmallString<128> lineBuf;
38  bool foundVendor = false;
39  for (;;) {
40  auto line = is.getline(lineBuf, 4096);
41  if (is.has_error()) {
42  break;
43  }
44 
45  if (line.empty()) {
46  continue;
47  }
48 
49  // look for vendor at start of line
50  if (wpi::starts_with(line, vendorStr)) {
51  foundVendor = true;
52  buf += wpi::trim(wpi::substr(line, 5));
53  buf += ' ';
54  continue;
55  }
56 
57  if (foundVendor) {
58  // next vendor, but didn't match product?
59  if (line[0] != '\t') {
60  buf += "Unknown";
61  return buf;
62  }
63 
64  // look for product
65  if (wpi::starts_with(wpi::substr(line, 1), productStr)) {
66  buf += wpi::trim(wpi::substr(line, 6));
67  return buf;
68  }
69  }
70  }
71 
72  return {};
73 }
