69 ADIS16470_IMU::ADIS16470_IMU(IMUAxis yaw_axis, SPI::Port port,
70  CalibrationTime cal_time)
71  : m_yaw_axis(yaw_axis),
72  m_spi_port(port),
73  m_calibration_time(static_cast<uint16_t>(cal_time)),
74  m_simDevice("Gyro:ADIS16470", port) {
75  if (m_simDevice) {
76  m_simGyroAngleX =
77  m_simDevice.CreateDouble("gyro_angle_x", hal::SimDevice::kInput, 0.0);
78  m_simGyroAngleY =
79  m_simDevice.CreateDouble("gyro_angle_y", hal::SimDevice::kInput, 0.0);
80  m_simGyroAngleZ =
81  m_simDevice.CreateDouble("gyro_angle_z", hal::SimDevice::kInput, 0.0);
82  m_simGyroRateX =
83  m_simDevice.CreateDouble("gyro_rate_x", hal::SimDevice::kInput, 0.0);
84  m_simGyroRateY =
85  m_simDevice.CreateDouble("gyro_rate_y", hal::SimDevice::kInput, 0.0);
86  m_simGyroRateZ =
87  m_simDevice.CreateDouble("gyro_rate_z", hal::SimDevice::kInput, 0.0);
88  m_simAccelX =
89  m_simDevice.CreateDouble("accel_x", hal::SimDevice::kInput, 0.0);
90  m_simAccelY =
91  m_simDevice.CreateDouble("accel_y", hal::SimDevice::kInput, 0.0);
92  m_simAccelZ =
93  m_simDevice.CreateDouble("accel_z", hal::SimDevice::kInput, 0.0);
94  }
95 
96  if (!m_simDevice) {
97  // Force the IMU reset pin to toggle on startup (doesn't require DS enable)
98  // Relies on the RIO hardware by default configuring an output as low
99  // and configuring an input as high Z. The 10k pull-up resistor internal to
100  // the IMU then forces the reset line high for normal operation.
101  DigitalOutput* m_reset_out =
102  new DigitalOutput(27); // Drive SPI CS2 (IMU RST) low
103  Wait(10_ms); // Wait 10ms
104  delete m_reset_out;
105  new DigitalInput(27); // Set SPI CS2 (IMU RST) high
106  Wait(500_ms); // Wait 500ms for reset to complete
107 
108  // Configure standard SPI
109  if (!SwitchToStandardSPI()) {
110  return;
111  }
112 
113  // Set IMU internal decimation to 4 (output data rate of 2000 SPS / (4 + 1)
114  // = 400Hz)
115  WriteRegister(DEC_RATE, 0x0004);
116  // Set data ready polarity (HIGH = Good Data), Disable gSense Compensation
117  // and PoP
118  WriteRegister(MSC_CTRL, 0x0001);
119  // Configure IMU internal Bartlett filter
120  WriteRegister(FILT_CTRL, 0x0000);
121  // Configure continuous bias calibration time based on user setting
122  WriteRegister(NULL_CNFG, m_calibration_time | 0x700);
123 
124  // Notify DS that IMU calibration delay is active
125  REPORT_WARNING(
126  "ADIS16470 IMU Detected. Starting initial calibration delay.");
127 
128  // Wait for samples to accumulate internal to the IMU (110% of user-defined
129  // time)
130  Wait(units::second_t{pow(2, m_calibration_time) / 2000 * 64 * 1.1});
131 
132  // Write offset calibration command to IMU
133  WriteRegister(GLOB_CMD, 0x0001);
134 
135  // Configure and enable auto SPI
136  if (!SwitchToAutoSPI()) {
137  return;
138  }
139 
140  // Let the user know the IMU was initiallized successfully
141  REPORT_WARNING("ADIS16470 IMU Successfully Initialized!");
142 
143  // Drive SPI CS3 (IMU ready LED) low (active low)
144  new DigitalOutput(28);
145  }
146 
147  // Report usage and post data to DS
148  HAL_Report(HALUsageReporting::kResourceType_ADIS16470, 0);
149 
150  wpi::SendableRegistry::AddLW(this, "ADIS16470", port);
151 }

166 bool ADIS16470_IMU::SwitchToStandardSPI() {
167  // Check to see whether the acquire thread is active. If so, wait for it to
168  // stop producing data.
169  if (m_thread_active) {
170  m_thread_active = false;
171  while (!m_thread_idle) {
172  Wait(10_ms);
173  }
174  std::cout << "Paused the IMU processing thread successfully!" << std::endl;
175  // Maybe we're in auto SPI mode? If so, kill auto SPI, and then SPI.
176  if (m_spi != nullptr && m_auto_configured) {
177  m_spi->StopAuto();
178  // We need to get rid of all the garbage left in the auto SPI buffer after
179  // stopping it. Sometimes data magically reappears, so we have to check
180  // the buffer size a couple of times
181  // to be sure we got it all. Yuck.
182  uint32_t trashBuffer[200];
183  Wait(100_ms);
184  int data_count = m_spi->ReadAutoReceivedData(trashBuffer, 0, 0_s);
185  while (data_count > 0) {
186  /* Receive data, max of 200 words at a time (prevent potential segfault)
187  */
188  m_spi->ReadAutoReceivedData(trashBuffer, std::min(data_count, 200),
189  0_s);
190  /*Get the reamining data count */
191  data_count = m_spi->ReadAutoReceivedData(trashBuffer, 0, 0_s);
192  }
193  std::cout << "Paused the auto SPI successfully!" << std::endl;
194  }
195  }
196  // There doesn't seem to be a SPI port active. Let's try to set one up
197  if (m_spi == nullptr) {
198  std::cout << "Setting up a new SPI port." << std::endl;
199  m_spi = new SPI(m_spi_port);
200  m_spi->SetClockRate(2000000);
201  m_spi->SetMSBFirst();
202  m_spi->SetSampleDataOnTrailingEdge();
203  m_spi->SetClockActiveLow();
204  m_spi->SetChipSelectActiveLow();
205  ReadRegister(PROD_ID); // Dummy read
206 
207  // Validate the product ID
208  uint16_t prod_id = ReadRegister(PROD_ID);
209  if (prod_id != 16982 && prod_id != 16470) {
210  REPORT_ERROR("Could not find ADIS16470!");
211  Close();
212  return false;
213  }
214  return true;
215  } else {
216  // Maybe the SPI port is active, but not in auto SPI mode? Try to read the
217  // product ID.
218  ReadRegister(PROD_ID); // Dummy read
219  uint16_t prod_id = ReadRegister(PROD_ID);
220  if (prod_id != 16982 && prod_id != 16470) {
221  REPORT_ERROR("Could not find ADIS16470!");
222  Close();
223  return false;
224  } else {
225  return true;
226  }
227  }
228 }

246 bool ADIS16470_IMU::SwitchToAutoSPI() {
247  // No SPI port has been set up. Go set one up first.
248  if (m_spi == nullptr) {
249  if (!SwitchToStandardSPI()) {
250  REPORT_ERROR("Failed to start/restart auto SPI");
251  return false;
252  }
253  }
254  // Only set up the interrupt if needed.
255  if (m_auto_interrupt == nullptr) {
256  m_auto_interrupt = new DigitalInput(26);
257  }
258  // The auto SPI controller gets angry if you try to set up two instances on
259  // one bus.
260  if (!m_auto_configured) {
261  m_spi->InitAuto(8200);
262  m_auto_configured = true;
263  }
264  // Do we need to change auto SPI settings?
265  switch (m_yaw_axis) {
266  case kX:
267  m_spi->SetAutoTransmitData(m_autospi_x_packet, 2);
268  break;
269  case kY:
270  m_spi->SetAutoTransmitData(m_autospi_y_packet, 2);
271  break;
272  default:
273  m_spi->SetAutoTransmitData(m_autospi_z_packet, 2);
274  break;
275  }
276  // Configure auto stall time
277  m_spi->ConfigureAutoStall(HAL_SPI_kOnboardCS0, 5, 1000, 1);
278  // Kick off DMA SPI (Note: Device configration impossible after SPI DMA is
279  // activated) DR High = Data good (data capture should be triggered on the
280  // rising edge)
281  m_spi->StartAutoTrigger(*m_auto_interrupt, true, false);
282  // Check to see if the acquire thread is running. If not, kick one off.
283  if (!m_thread_idle) {
284  m_first_run = true;
285  m_thread_active = true;
286  m_acquire_task = std::thread(&ADIS16470_IMU::Acquire, this);
287  std::cout << "New IMU Processing thread activated!" << std::endl;
288  } else {
289  m_first_run = true;
290  m_thread_active = true;
291  std::cout << "Old IMU Processing thread re-activated!" << std::endl;
292  }
293  // Looks like the thread didn't start for some reason. Abort.
294  /*
295  if(!m_thread_idle) {
296  REPORT_ERROR("Failed to start/restart the acquire() thread.");
297  Close();
298  return false;
299  }
300  */
301  return true;
302 }

494 void ADIS16470_IMU::Acquire() {
495  // Set data packet length
496  const int dataset_len = 19; // 18 data points + timestamp
497 
498  /* Fixed buffer size */
499  const int BUFFER_SIZE = 4000;
500 
501  // This buffer can contain many datasets
502  uint32_t buffer[BUFFER_SIZE];
503  int data_count = 0;
504  int data_remainder = 0;
505  int data_to_read = 0;
506  uint32_t previous_timestamp = 0;
507  double delta_angle = 0.0;
508  double gyro_rate_x = 0.0;
509  double gyro_rate_y = 0.0;
510  double gyro_rate_z = 0.0;
511  double accel_x = 0.0;
512  double accel_y = 0.0;
513  double accel_z = 0.0;
514  double gyro_rate_x_si = 0.0;
515  double gyro_rate_y_si = 0.0;
516  // double gyro_rate_z_si = 0.0;
517  double accel_x_si = 0.0;
518  double accel_y_si = 0.0;
519  double accel_z_si = 0.0;
520  double compAngleX = 0.0;
521  double compAngleY = 0.0;
522  double accelAngleX = 0.0;
523  double accelAngleY = 0.0;
524 
525  while (true) {
526  // Sleep loop for 10ms (wait for data)
527  Wait(10_ms);
528 
529  if (m_thread_active) {
530  m_thread_idle = false;
531 
532  data_count = m_spi->ReadAutoReceivedData(
533  buffer, 0,
534  0_s); // Read number of bytes currently stored in the buffer
535  data_remainder =
536  data_count % dataset_len; // Check if frame is incomplete. Add 1
537  // because of timestamp
538  data_to_read = data_count -
539  data_remainder; // Remove incomplete data from read count
540  /* Want to cap the data to read in a single read at the buffer size */
541  if (data_to_read > BUFFER_SIZE) {
542  REPORT_WARNING(
543  "ADIS16470 data processing thread overrun has occurred!");
544  data_to_read = BUFFER_SIZE - (BUFFER_SIZE % dataset_len);
545  }
546  m_spi->ReadAutoReceivedData(
547  buffer, data_to_read,
548  0_s); // Read data from DMA buffer (only complete sets)
549 
550  /*
551  // DEBUG: Print buffer size and contents to terminal
552  std::cout << "Start - " << data_count << "," << data_remainder << "," <<
553  data_to_read << std::endl; for (int m = 0; m < data_to_read - 1; m++ )
554  {
555  std::cout << buffer[m] << ",";
556  }
557  std::cout << " " << std::endl;
558  std::cout << "End" << std::endl;
559  std::cout << "Reading " << data_count << " bytes." << std::endl;
560  */
561 
562  // Could be multiple data sets in the buffer. Handle each one.
563  for (int i = 0; i < data_to_read; i += dataset_len) {
564  // Timestamp is at buffer[i]
565  m_dt = (buffer[i] - previous_timestamp) / 1000000.0;
566  /* Get delta angle value for selected yaw axis and scale by the elapsed
567  * time (based on timestamp) */
568  delta_angle = (ToInt(&buffer[i + 3]) * delta_angle_sf) /
569  (m_scaled_sample_rate / (buffer[i] - previous_timestamp));
570  gyro_rate_x = (BuffToShort(&buffer[i + 7]) / 10.0);
571  gyro_rate_y = (BuffToShort(&buffer[i + 9]) / 10.0);
572  gyro_rate_z = (BuffToShort(&buffer[i + 11]) / 10.0);
573  accel_x = (BuffToShort(&buffer[i + 13]) / 800.0);
574  accel_y = (BuffToShort(&buffer[i + 15]) / 800.0);
575  accel_z = (BuffToShort(&buffer[i + 17]) / 800.0);
576 
577  // Convert scaled sensor data to SI units
578  gyro_rate_x_si = gyro_rate_x * deg_to_rad;
579  gyro_rate_y_si = gyro_rate_y * deg_to_rad;
580  // gyro_rate_z_si = gyro_rate_z * deg_to_rad;
581  accel_x_si = accel_x * grav;
582  accel_y_si = accel_y * grav;
583  accel_z_si = accel_z * grav;
584 
585  // Store timestamp for next iteration
586  previous_timestamp = buffer[i];
587 
588  /*
589  // DEBUG: Print timestamp and delta values
590  std::cout << previous_timestamp << "," << delta_x << "," << delta_y <<
591  "," << delta_z << std::endl;
592  */
593 
594  m_alpha = m_tau / (m_tau + m_dt);
595 
596  if (m_first_run) {
597  accelAngleX = atan2f(accel_x_si, sqrtf((accel_y_si * accel_y_si) +
598  (accel_z_si * accel_z_si)));
599  accelAngleY = atan2f(accel_y_si, sqrtf((accel_x_si * accel_x_si) +
600  (accel_z_si * accel_z_si)));
601  compAngleX = accelAngleX;
602  compAngleY = accelAngleY;
603  } else {
604  // Process X angle
605  accelAngleX = atan2f(accel_x_si, sqrtf((accel_y_si * accel_y_si) +
606  (accel_z_si * accel_z_si)));
607  accelAngleY = atan2f(accel_y_si, sqrtf((accel_x_si * accel_x_si) +
608  (accel_z_si * accel_z_si)));
609  accelAngleX = FormatAccelRange(accelAngleX, accel_z_si);
610  accelAngleY = FormatAccelRange(accelAngleY, accel_z_si);
611  compAngleX =
612  CompFilterProcess(compAngleX, accelAngleX, -gyro_rate_y_si);
613  compAngleY =
614  CompFilterProcess(compAngleY, accelAngleY, gyro_rate_x_si);
615  }
616 
617  // DEBUG: Print accumulated values
618  // std::cout << m_compAngleX << "," << m_compAngleY << std::endl;
619 
620  {
621  std::scoped_lock sync(m_mutex);
622  /* Push data to global variables */
623  if (m_first_run) {
624  /* Don't accumulate first run. previous_timestamp will be "very" old
625  * and the integration will end up way off */
626  m_integ_angle = 0.0;
627  } else {
628  m_integ_angle += delta_angle;
629  }
630  m_gyro_rate_x = gyro_rate_x;
631  m_gyro_rate_y = gyro_rate_y;
632  m_gyro_rate_z = gyro_rate_z;
633  m_accel_x = accel_x;
634  m_accel_y = accel_y;
635  m_accel_z = accel_z;
636  m_compAngleX = compAngleX * rad_to_deg;
637  m_compAngleY = compAngleY * rad_to_deg;
638  m_accelAngleX = accelAngleX * rad_to_deg;
639  m_accelAngleY = accelAngleY * rad_to_deg;
640  }
641  m_first_run = false;
642  }
643  } else {
644  m_thread_idle = true;
645  data_count = 0;
646  data_remainder = 0;
647  data_to_read = 0;
648  previous_timestamp = 0.0;
649  delta_angle = 0.0;
650  gyro_rate_x = 0.0;
651  gyro_rate_y = 0.0;
652  gyro_rate_z = 0.0;
653  accel_x = 0.0;
654  accel_y = 0.0;
655  accel_z = 0.0;
656  gyro_rate_x_si = 0.0;
657  gyro_rate_y_si = 0.0;
658  // gyro_rate_z_si = 0.0;
659  accel_x_si = 0.0;
660  accel_y_si = 0.0;
661  accel_z_si = 0.0;
662  compAngleX = 0.0;
663  compAngleY = 0.0;
664  accelAngleX = 0.0;
665  accelAngleY = 0.0;
666  }
667  }
668 }

670 /* Complementary filter functions */
671 double ADIS16470_IMU::FormatFastConverge(double compAngle, double accAngle) {
672  if (compAngle > accAngle + wpi::numbers::pi) {
673  compAngle = compAngle - 2.0 * wpi::numbers::pi;
674  } else if (accAngle > compAngle + wpi::numbers::pi) {
675  compAngle = compAngle + 2.0 * wpi::numbers::pi;
676  }
677  return compAngle;
678 }

680 double ADIS16470_IMU::FormatRange0to2PI(double compAngle) {
681  while (compAngle >= 2 * wpi::numbers::pi) {
682  compAngle = compAngle - 2.0 * wpi::numbers::pi;
683  }
684  while (compAngle < 0.0) {
685  compAngle = compAngle + 2.0 * wpi::numbers::pi;
686  }
687  return compAngle;
688 }

690 double ADIS16470_IMU::FormatAccelRange(double accelAngle, double accelZ) {
691  if (accelZ < 0.0) {
692  accelAngle = wpi::numbers::pi - accelAngle;
693  } else if (accelZ > 0.0 && accelAngle < 0.0) {
694  accelAngle = 2.0 * wpi::numbers::pi + accelAngle;
695  }
696  return accelAngle;
697 }

699 double ADIS16470_IMU::CompFilterProcess(double compAngle, double accelAngle,
700  double omega) {
701  compAngle = FormatFastConverge(compAngle, accelAngle);
702  compAngle =
703  m_alpha * (compAngle + omega * m_dt) + (1.0 - m_alpha) * accelAngle;
704  compAngle = FormatRange0to2PI(compAngle);
705  if (compAngle > wpi::numbers::pi) {
706  compAngle = compAngle - 2.0 * wpi::numbers::pi;
707  }
708  return compAngle;
709 }

711 units::degree_t ADIS16470_IMU::GetAngle() const {
712  switch (m_yaw_axis) {
713  case kX:
714  if (m_simGyroAngleX) {
715  return units::degree_t{m_simGyroAngleX.Get()};
716  }

718  case kY:
