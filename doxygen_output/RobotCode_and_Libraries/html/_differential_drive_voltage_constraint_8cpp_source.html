30 TrajectoryConstraint::MinMax
31 DifferentialDriveVoltageConstraint::MinMaxAcceleration(
32  const Pose2d& pose, curvature_t curvature,
33  units::meters_per_second_t speed) {
34  auto wheelSpeeds =
35  m_kinematics.ToWheelSpeeds({speed, 0_mps, speed * curvature});
36 
37  auto maxWheelSpeed = std::max(wheelSpeeds.left, wheelSpeeds.right);
38  auto minWheelSpeed = std::min(wheelSpeeds.left, wheelSpeeds.right);
39 
40  // Calculate maximum/minimum possible accelerations from motor dynamics
41  // and max/min wheel speeds
42  auto maxWheelAcceleration =
43  m_feedforward.MaxAchievableAcceleration(m_maxVoltage, maxWheelSpeed);
44  auto minWheelAcceleration =
45  m_feedforward.MinAchievableAcceleration(m_maxVoltage, minWheelSpeed);
46 
47  // Robot chassis turning on radius = 1/|curvature|. Outer wheel has radius
48  // increased by half of the trackwidth T. Inner wheel has radius decreased
49  // by half of the trackwidth. Achassis / radius = Aouter / (radius + T/2), so
50  // Achassis = Aouter * radius / (radius + T/2) = Aouter / (1 +
51  // |curvature|T/2). Inner wheel is similar.
52 
53  // sgn(speed) term added to correctly account for which wheel is on
54  // outside of turn:
55  // If moving forward, max acceleration constraint corresponds to wheel on
56  // outside of turn If moving backward, max acceleration constraint corresponds
57  // to wheel on inside of turn
58 
59  // When velocity is zero, then wheel velocities are uniformly zero (robot
60  // cannot be turning on its center) - we have to treat this as a special case,
61  // as it breaks the signum function. Both max and min acceleration are
62  // *reduced in magnitude* in this case.
63 
64  units::meters_per_second_squared_t maxChassisAcceleration;
65  units::meters_per_second_squared_t minChassisAcceleration;
66 
67  if (speed == 0_mps) {
68  maxChassisAcceleration =
69  maxWheelAcceleration /
70  (1 + m_kinematics.trackWidth * units::math::abs(curvature) / (2_rad));
71  minChassisAcceleration =
72  minWheelAcceleration /
73  (1 + m_kinematics.trackWidth * units::math::abs(curvature) / (2_rad));
74  } else {
75  maxChassisAcceleration =
76  maxWheelAcceleration /
77  (1 + m_kinematics.trackWidth * units::math::abs(curvature) *
78  wpi::sgn(speed) / (2_rad));
79  minChassisAcceleration =
80  minWheelAcceleration /
81  (1 - m_kinematics.trackWidth * units::math::abs(curvature) *
82  wpi::sgn(speed) / (2_rad));
83  }
84 
85  // When turning about a point inside of the wheelbase (i.e. radius less than
86  // half the trackwidth), the inner wheel's direction changes, but the
87  // magnitude remains the same. The formula above changes sign for the inner
88  // wheel when this happens. We can accurately account for this by simply
89  // negating the inner wheel.
90 
91  if ((m_kinematics.trackWidth / 2) > 1_rad / units::math::abs(curvature)) {
92  if (speed > 0_mps) {
93  minChassisAcceleration = -minChassisAcceleration;
94  } else if (speed < 0_mps) {
95  maxChassisAcceleration = -maxChassisAcceleration;
96  }
97  }
98 
99  return {minChassisAcceleration, maxChassisAcceleration};
100 }
