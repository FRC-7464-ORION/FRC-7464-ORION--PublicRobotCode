41 // Mutexes to synchronize installing error handlers and calling error handlers.
42 // Do not use ManagedStatic, or that may allocate memory while attempting to
43 // report an OOM.
44 //
45 // This usage of std::mutex has to be conditionalized behind ifdefs because
46 // of this script:
47 // compiler-rt/lib/sanitizer_common/symbolizer/scripts/build_symbolizer.sh
48 // That script attempts to statically link the LLVM symbolizer library with the
49 // STL and hide all of its symbols with 'opt -internalize'. To reduce size, it
50 // cuts out the threading portions of the hermetic copy of libc++ that it
51 // builds. We can remove these ifdefs if that script goes away.
52 static std::mutex ErrorHandlerMutex;
53 static std::mutex BadAllocErrorHandlerMutex;

55 void wpi::install_fatal_error_handler(fatal_error_handler_t handler,
56  void *user_data) {
57  std::scoped_lock Lock(ErrorHandlerMutex);
58  assert(!ErrorHandler && "Error handler already registered!\n");
59  ErrorHandler = handler;
60  ErrorHandlerUserData = user_data;
61 }

77 void wpi::report_fatal_error(std::string_view Reason, bool GenCrashDiag) {
78  wpi::fatal_error_handler_t handler = nullptr;
79  void* handlerData = nullptr;
80  {
81  // Only acquire the mutex while reading the handler, so as not to invoke a
82  // user-supplied callback under a lock.
83  std::scoped_lock Lock(ErrorHandlerMutex);
84  handler = ErrorHandler;
85  handlerData = ErrorHandlerUserData;
86  }
87 
88  if (handler) {
89  handler(handlerData, std::string{Reason}, GenCrashDiag);
90  } else {
91  fmt::print(stderr, "LLVM ERROR: {}\n", Reason);
92  }
93 
94  exit(1);
95 }

111 void wpi::report_bad_alloc_error(const char *Reason, bool GenCrashDiag) {
112  fatal_error_handler_t Handler = nullptr;
113  void *HandlerData = nullptr;
114  {
115  // Only acquire the mutex while reading the handler, so as not to invoke a
116  // user-supplied callback under a lock.
117  std::scoped_lock Lock(BadAllocErrorHandlerMutex);
118  Handler = BadAllocErrorHandler;
119  HandlerData = BadAllocErrorHandlerUserData;
120  }
121 
122  if (Handler) {
123  Handler(HandlerData, Reason, GenCrashDiag);
124  wpi_unreachable("bad alloc handler should not return");
125  }
126 
127  // Don't call the normal error handler. It may allocate memory. Directly write
128  // an OOM to stderr and abort.
129  char OOMMessage[] = "LLVM ERROR: out of memory\n";
130 #ifdef _WIN32
131  int written = ::_write(2, OOMMessage, strlen(OOMMessage));
132 #else
133  ssize_t written = ::write(2, OOMMessage, strlen(OOMMessage));
134 #endif
135  (void)written;
136  abort();
137 }

165 void wpi::wpi_unreachable_internal(const char *msg, const char *file,
166  unsigned line) {
167  // This code intentionally doesn't call the ErrorHandler callback, because
168  // wpi_unreachable is intended to be used to indicate "impossible"
169  // situations, and not legitimate runtime errors.
170  if (msg)
171  fmt::print(stderr, "{}\n", msg);
172  std::fputs("UNREACHABLE executed", stderr);
173  if (file)
174  fmt::print(stderr, " at {}:{}", file, line);
175  fmt::print(stderr, "{}", "!\n");
176  abort();
177 #ifdef LLVM_BUILTIN_UNREACHABLE
178  // Windows systems and possibly others don't declare abort() to be noreturn,
179  // so use the unreachable builtin to avoid a Clang self-host warning.
180  LLVM_BUILTIN_UNREACHABLE;
181 #endif
182 }
