16 units::meters_per_second_t CentripetalAccelerationConstraint::MaxVelocity(
17  const Pose2d& pose, curvature_t curvature,
18  units::meters_per_second_t velocity) {
19  // ac = v^2 / r
20  // k (curvature) = 1 / r
21 
22  // therefore, ac = v^2 * k
23  // ac / k = v^2
24  // v = std::sqrt(ac / k)
25 
26  // We have to multiply by 1_rad here to get the units to cancel out nicely.
27  // The units library defines a unit for radians although it is technically
28  // unitless.
29  return units::math::sqrt(m_maxCentripetalAcceleration /
30  units::math::abs(curvature) * 1_rad);
31 }
