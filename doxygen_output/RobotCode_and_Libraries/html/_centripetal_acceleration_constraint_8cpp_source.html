15 units::meters_per_second_t CentripetalAccelerationConstraint::MaxVelocity(
16  const Pose2d& pose, units::curvature_t curvature,
17  units::meters_per_second_t velocity) const {
18  // ac = v^2 / r
19  // k (curvature) = 1 / r
20 
21  // therefore, ac = v^2 * k
22  // ac / k = v^2
23  // v = std::sqrt(ac / k)
24 
25  // We have to multiply by 1_rad here to get the units to cancel out nicely.
26  // The units library defines a unit for radians although it is technically
27  // unitless.
28  return units::math::sqrt(m_maxCentripetalAcceleration /
29  units::math::abs(curvature) * 1_rad);
30 }
