105 void nt::ConvertToC(const Value& in, NT_Value* out) {
106  out->type = NT_UNASSIGNED;
107  switch (in.type()) {
108  case NT_UNASSIGNED:
109  return;
110  case NT_BOOLEAN:
111  out->data.v_boolean = in.GetBoolean() ? 1 : 0;
112  break;
113  case NT_DOUBLE:
114  out->data.v_double = in.GetDouble();
115  break;
116  case NT_STRING:
117  ConvertToC(in.GetString(), &out->data.v_string);
118  break;
119  case NT_RAW:
120  ConvertToC(in.GetRaw(), &out->data.v_raw);
121  break;
122  case NT_RPC:
123  ConvertToC(in.GetRpc(), &out->data.v_raw);
124  break;
125  case NT_BOOLEAN_ARRAY: {
126  auto v = in.GetBooleanArray();
127  out->data.arr_boolean.arr =
128  static_cast<int*>(wpi::safe_malloc(v.size() * sizeof(int)));
129  out->data.arr_boolean.size = v.size();
130  std::copy(v.begin(), v.end(), out->data.arr_boolean.arr);
131  break;
132  }
133  case NT_DOUBLE_ARRAY: {
134  auto v = in.GetDoubleArray();
135  out->data.arr_double.arr =
136  static_cast<double*>(wpi::safe_malloc(v.size() * sizeof(double)));
137  out->data.arr_double.size = v.size();
138  std::copy(v.begin(), v.end(), out->data.arr_double.arr);
139  break;
140  }
141  case NT_STRING_ARRAY: {
142  auto v = in.GetStringArray();
143  out->data.arr_string.arr = static_cast<NT_String*>(
144  wpi::safe_malloc(v.size() * sizeof(NT_String)));
145  for (size_t i = 0; i < v.size(); ++i) {
146  ConvertToC(v[i], &out->data.arr_string.arr[i]);
147  }
148  out->data.arr_string.size = v.size();
149  break;
150  }
151  default:
152  // assert(false && "unknown value type");
153  return;
154  }
155  out->type = in.type();
156 }

165 std::shared_ptr<Value> nt::ConvertFromC(const NT_Value& value) {
166  switch (value.type) {
167  case NT_UNASSIGNED:
168  return nullptr;
169  case NT_BOOLEAN:
170  return Value::MakeBoolean(value.data.v_boolean != 0);
171  case NT_DOUBLE:
172  return Value::MakeDouble(value.data.v_double);
173  case NT_STRING:
174  return Value::MakeString(ConvertFromC(value.data.v_string));
175  case NT_RAW:
176  return Value::MakeRaw(ConvertFromC(value.data.v_raw));
177  case NT_RPC:
178  return Value::MakeRpc(ConvertFromC(value.data.v_raw));
179  case NT_BOOLEAN_ARRAY:
180  return Value::MakeBooleanArray(
181  wpi::span(value.data.arr_boolean.arr, value.data.arr_boolean.size));
182  case NT_DOUBLE_ARRAY:
183  return Value::MakeDoubleArray(
184  wpi::span(value.data.arr_double.arr, value.data.arr_double.size));
185  case NT_STRING_ARRAY: {
186  std::vector<std::string> v;
187  v.reserve(value.data.arr_string.size);
188  for (size_t i = 0; i < value.data.arr_string.size; ++i) {
189  v.emplace_back(ConvertFromC(value.data.arr_string.arr[i]));
190  }
191  return Value::MakeStringArray(std::move(v));
192  }
193  default:
194  // assert(false && "unknown value type");
195  return nullptr;
196  }
197 }

199 bool nt::operator==(const Value& lhs, const Value& rhs) {
200  if (lhs.type() != rhs.type()) {
201  return false;
202  }
203  switch (lhs.type()) {
204  case NT_UNASSIGNED:
205  return true; // XXX: is this better being false instead?
206  case NT_BOOLEAN:
207  return lhs.m_val.data.v_boolean == rhs.m_val.data.v_boolean;
208  case NT_DOUBLE:
209  return lhs.m_val.data.v_double == rhs.m_val.data.v_double;
210  case NT_STRING:
211  case NT_RAW:
212  case NT_RPC:
213  return lhs.m_string == rhs.m_string;
214  case NT_BOOLEAN_ARRAY:
215  if (lhs.m_val.data.arr_boolean.size != rhs.m_val.data.arr_boolean.size) {
216  return false;
217  }
218  return std::memcmp(lhs.m_val.data.arr_boolean.arr,
219  rhs.m_val.data.arr_boolean.arr,
220  lhs.m_val.data.arr_boolean.size *
221  sizeof(lhs.m_val.data.arr_boolean.arr[0])) == 0;
222  case NT_DOUBLE_ARRAY:
223  if (lhs.m_val.data.arr_double.size != rhs.m_val.data.arr_double.size) {
224  return false;
225  }
226  return std::memcmp(lhs.m_val.data.arr_double.arr,
227  rhs.m_val.data.arr_double.arr,
228  lhs.m_val.data.arr_double.size *
229  sizeof(lhs.m_val.data.arr_double.arr[0])) == 0;
230  case NT_STRING_ARRAY:
231  return lhs.m_string_array == rhs.m_string_array;
232  default:
233  // assert(false && "unknown value type");
234  return false;
235  }
