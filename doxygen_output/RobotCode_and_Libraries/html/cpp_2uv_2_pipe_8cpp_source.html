24 void Pipe::Reuse(std::function<void()> callback, bool ipc) {
25  if (IsClosing()) {
26  return;
27  }
28  if (!m_reuseData) {
29  m_reuseData = std::make_unique<ReuseData>();
30  }
31  m_reuseData->callback = std::move(callback);
32  m_reuseData->ipc = ipc;
33  uv_close(GetRawHandle(), [](uv_handle_t* handle) {
34  Pipe& h = *static_cast<Pipe*>(handle->data);
35  if (!h.m_reuseData) {
36  return;
37  }
38  auto data = std::move(h.m_reuseData);
39  auto err =
40  uv_pipe_init(h.GetLoopRef().GetRaw(), h.GetRaw(), data->ipc ? 1 : 0);
41  if (err < 0) {
42  h.ReportError(err);
43  return;
44  }
45  data->callback();
46  });
47 }

70 void Pipe::Connect(std::string_view name,
71  const std::shared_ptr<PipeConnectReq>& req) {
72  SmallString<128> nameBuf{name};
73  uv_pipe_connect(req->GetRaw(), GetRaw(), nameBuf.c_str(),
74  [](uv_connect_t* req, int status) {
75  auto& h = *static_cast<PipeConnectReq*>(req->data);
76  if (status < 0) {
77  h.ReportError(status);

79  h.connected();
80  }

92 std::string Pipe::GetSock() {
93  // Per libuv docs, the returned buffer is NOT null terminated.
94  // common case should be small
95  char buf[128];
96  size_t size = 128;
97  int r = uv_pipe_getsockname(GetRaw(), buf, &size);
98  if (r == 0) {
99  return std::string{buf, size};
100  } else if (r == UV_ENOBUFS) {
101  // need to allocate a big enough buffer
102  char* buf2 = static_cast<char*>(std::malloc(size));
103  r = uv_pipe_getsockname(GetRaw(), buf2, &size);
104  if (r == 0) {
105  std::string out{buf2, size};
106  std::free(buf2);
107  return out;
108  }
109  std::free(buf2);
110  }
111  ReportError(r);
112  return std::string{};
113 }
