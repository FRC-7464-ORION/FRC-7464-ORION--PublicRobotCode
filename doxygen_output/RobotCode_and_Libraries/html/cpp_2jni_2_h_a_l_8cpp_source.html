225 {
226  HAL_JoystickAxes axes;
227  HAL_GetJoystickAxes(joystickNum, &axes);
228

231  ThrowIllegalArgumentException(
232  env,

235  static_cast<int>(axes.count), static_cast<int>(javaSize)));
236  return 0;
237  }
238 
239  env->SetFloatArrayRegion(axesArray, 0, axes.count, axes.axes);
240 
241  return axes.count;
242 }
