73 void HALSimWSClientConnection::OnSimValueChanged(const wpi::json& msg) {
74  if (msg.empty()) {
75  return;
76  }
77  wpi::SmallVector<uv::Buffer, 4> sendBufs;
78  wpi::raw_uv_ostream os{sendBufs, [this]() -> uv::Buffer {
79  std::lock_guard lock(m_buffers_mutex);
80  return m_buffers.Allocate();
81  }};

83  os << msg;
84 
85  // Call the websocket send function on the uv loop
86  m_client->GetExec().Send([self = shared_from_this(), sendBufs] {
87  self->m_websocket->SendText(sendBufs,
88  [self](auto bufs, wpi::uv::Error err) {

90  std::lock_guard lock(self->m_buffers_mutex);
