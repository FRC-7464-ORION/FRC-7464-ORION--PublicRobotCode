104 template <typename O, typename I>
105 static O* ConvertToC(const std::vector<I>& in, size_t* out_len) {
106  if (!out_len) {
107  return nullptr;
108  }
109  *out_len = in.size();
110  if (in.empty()) {
111  return nullptr;
112  }
113  O* out = static_cast<O*>(wpi::safe_malloc(sizeof(O) * in.size()));
114  for (size_t i = 0; i < in.size(); ++i) {
115  ConvertToC(in[i], &out[i]);
116  }
117  return out;
118 }

152 static RpcDefinition ConvertFromC(const NT_RpcDefinition& in) {
153  RpcDefinition out;
154  out.version = in.version;
155  out.name = ConvertFromC(in.name);
156 
157  out.params.reserve(in.num_params);
158  for (size_t i = 0; i < in.num_params; ++i) {
159  out.params.push_back(ConvertFromC(in.params[i]));
160  }
161 
162  out.results.reserve(in.num_results);
163  for (size_t i = 0; i < in.num_results; ++i) {
164  out.results.push_back(ConvertFromC(in.results[i]));
165  }
166 
167  return out;
168 }
