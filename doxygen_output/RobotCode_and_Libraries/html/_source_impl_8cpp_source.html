184 bool SourceImpl::SetConfigJson(const wpi::json& config, CS_Status* status) {
185  VideoMode mode;
186 
187  // pixel format
188  if (config.count("pixel format") != 0) {
189  try {
190  auto str = config.at("pixel format").get<std::string>();
191  if (wpi::equals_lower(str, "mjpeg")) {
192  mode.pixelFormat = cs::VideoMode::kMJPEG;
193  } else if (wpi::equals_lower(str, "yuyv")) {
194  mode.pixelFormat = cs::VideoMode::kYUYV;
195  } else if (wpi::equals_lower(str, "rgb565")) {
196  mode.pixelFormat = cs::VideoMode::kRGB565;
197  } else if (wpi::equals_lower(str, "bgr")) {
198  mode.pixelFormat = cs::VideoMode::kBGR;
199  } else if (wpi::equals_lower(str, "gray")) {
200  mode.pixelFormat = cs::VideoMode::kGray;
201  } else {
202  SWARNING("SetConfigJson: could not understand pixel format value '{}'",
203  str);
204  }
205  } catch (const wpi::json::exception& e) {
206  SWARNING("SetConfigJson: could not read pixel format: {}", e.what());
207  }
208  }
209 
210  // width
211  if (config.count("width") != 0) {
212  try {
213  mode.width = config.at("width").get<unsigned int>();
214  } catch (const wpi::json::exception& e) {
215  SWARNING("SetConfigJson: could not read width: {}", e.what());
216  }
217  }
218 
219  // height
220  if (config.count("height") != 0) {
221  try {
222  mode.height = config.at("height").get<unsigned int>();
223  } catch (const wpi::json::exception& e) {
224  SWARNING("SetConfigJson: could not read height: {}", e.what());
225  }
226  }
227 
228  // fps
229  if (config.count("fps") != 0) {
230  try {
231  mode.fps = config.at("fps").get<unsigned int>();
232  } catch (const wpi::json::exception& e) {
233  SWARNING("SetConfigJson: could not read fps: {}", e.what());
234  }
235  }
236 
237  // if all of video mode is set, use SetVideoMode, otherwise piecemeal it
238  if (mode.pixelFormat != VideoMode::kUnknown && mode.width != 0 &&
239  mode.height != 0 && mode.fps != 0) {
240  SINFO(
241  "SetConfigJson: setting video mode to pixelFormat {}, width {}, height "
242  "{}, fps {}",
243  mode.pixelFormat, mode.width, mode.height, mode.fps);
244  SetVideoMode(mode, status);
245  } else {
246  if (mode.pixelFormat != cs::VideoMode::kUnknown) {
247  SINFO("SetConfigJson: setting pixelFormat {}", mode.pixelFormat);
248  SetPixelFormat(static_cast<cs::VideoMode::PixelFormat>(mode.pixelFormat),
249  status);
250  }
251  if (mode.width != 0 && mode.height != 0) {
252  SINFO("SetConfigJson: setting width {}, height {}", mode.width,
253  mode.height);
254  SetResolution(mode.width, mode.height, status);
255  }
256  if (mode.fps != 0) {
257  SINFO("SetConfigJson: setting fps {}", mode.fps);
258  SetFPS(mode.fps, status);
259  }
260  }
261 
262  // brightness
263  if (config.count("brightness") != 0) {
264  try {
265  int val = config.at("brightness").get<int>();
266  SINFO("SetConfigJson: setting brightness to {}", val);
267  SetBrightness(val, status);
268  } catch (const wpi::json::exception& e) {
269  SWARNING("SetConfigJson: could not read brightness: {}", e.what());
270  }
271  }
272 
273  // white balance
274  if (config.count("white balance") != 0) {
275  try {
276  auto& setting = config.at("white balance");
277  if (setting.is_string()) {
278  auto str = setting.get<std::string>();
279  if (wpi::equals_lower(str, "auto")) {
280  SINFO("SetConfigJson: setting white balance to {}", "auto");
281  SetWhiteBalanceAuto(status);
282  } else if (wpi::equals_lower(str, "hold")) {
283  SINFO("SetConfigJson: setting white balance to {}", "hold current");
284  SetWhiteBalanceHoldCurrent(status);
285  } else {
286  SWARNING(
287  "SetConfigJson: could not understand white balance value '{}'",
288  str);
289  }
290  } else {
291  int val = setting.get<int>();
292  SINFO("SetConfigJson: setting white balance to {}", val);
293  SetWhiteBalanceManual(val, status);
294  }
295  } catch (const wpi::json::exception& e) {
296  SWARNING("SetConfigJson: could not read white balance: {}", e.what());
297  }
298  }
299 
300  // exposure
301  if (config.count("exposure") != 0) {
302  try {
303  auto& setting = config.at("exposure");
304  if (setting.is_string()) {
305  auto str = setting.get<std::string>();
306  if (wpi::equals_lower(str, "auto")) {
307  SINFO("SetConfigJson: setting exposure to {}", "auto");
308  SetExposureAuto(status);
309  } else if (wpi::equals_lower(str, "hold")) {
310  SINFO("SetConfigJson: setting exposure to {}", "hold current");
311  SetExposureHoldCurrent(status);
312  } else {
313  SWARNING("SetConfigJson: could not understand exposure value '{}'",
314  str);
315  }
316  } else {
317  int val = setting.get<int>();
318  SINFO("SetConfigJson: setting exposure to {}", val);
319  SetExposureManual(val, status);
320  }
321  } catch (const wpi::json::exception& e) {
322  SWARNING("SetConfigJson: could not read exposure: {}", e.what());
323  }
324  }
325 
326  // properties
327  if (config.count("properties") != 0) {
328  SetPropertiesJson(config.at("properties"), m_logger, GetName(), status);
329  }
330 
331  return true;
332 }

405 std::unique_ptr<Image> SourceImpl::AllocImage(
406  VideoMode::PixelFormat pixelFormat, int width, int height, size_t size) {
407  std::unique_ptr<Image> image;
408  {
409  std::scoped_lock lock{m_poolMutex};
410  // find the smallest existing frame that is at least big enough.
411  int found = -1;
412  for (size_t i = 0; i < m_imagesAvail.size(); ++i) {
413  // is it big enough?
414  if (m_imagesAvail[i] && m_imagesAvail[i]->capacity() >= size) {
415  // is it smaller than the last found?
416  if (found < 0 ||
417  m_imagesAvail[i]->capacity() < m_imagesAvail[found]->capacity()) {
418  // yes, update
419  found = i;
420  }
421  }
422  }
423 
424  // if nothing found, allocate a new buffer
425  if (found < 0) {
426  image = std::make_unique<Image>(size);
427  } else {
428  image = std::move(m_imagesAvail[found]);
429  }
430  }
431 
432  // Initialize image
433  image->SetSize(size);
434  image->pixelFormat = pixelFormat;
435  image->width = width;
436  image->height = height;
437 
438  return image;
439 }

512 void SourceImpl::ReleaseImage(std::unique_ptr<Image> image) {
513  std::scoped_lock lock{m_poolMutex};
514  if (m_destroyFrames) {
515  return;
516  }
517  // Return the frame to the pool. First try to find an empty slot, otherwise
518  // add it to the end.
519  auto it = std::find(m_imagesAvail.begin(), m_imagesAvail.end(), nullptr);
520  if (it != m_imagesAvail.end()) {
521  *it = std::move(image);
522  } else if (m_imagesAvail.size() > kMaxImagesAvail) {
523  // Replace smallest buffer; don't need to check for null because the above
524  // find would have found it.
525  auto it2 = std::min_element(
526  m_imagesAvail.begin(), m_imagesAvail.end(),
527  [](const std::unique_ptr<Image>& a, const std::unique_ptr<Image>& b) {
528  return a->capacity() < b->capacity();
529  });
530  if ((*it2)->capacity() < image->capacity()) {
531  *it2 = std::move(image);
532  }
533  } else {
534  m_imagesAvail.emplace_back(std::move(image));
535  }
536 }
