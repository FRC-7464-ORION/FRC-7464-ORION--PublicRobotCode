106 void CommandScheduler::Schedule(bool interruptible, Command* command) {
107  if (m_impl->inRunLoop) {
108  m_impl->toSchedule.try_emplace(command, interruptible);
109  return;
110  }
111 
112  if (command->IsGrouped()) {
113  throw FRC_MakeError(frc::err::CommandIllegalUse, "{}",
114  "A command that is part of a command group "
115  "cannot be independently scheduled");
116  return;
117  }
118  if (m_impl->disabled ||
119  (frc::RobotState::IsDisabled() && !command->RunsWhenDisabled()) ||
120  ContainsKey(m_impl->scheduledCommands, command)) {
121  return;
122  }
123 
124  const auto& requirements = command->GetRequirements();
125 
126  wpi::SmallVector<Command*, 8> intersection;
127 
128  bool isDisjoint = true;
129  bool allInterruptible = true;
130  for (auto&& i1 : m_impl->requirements) {
131  if (requirements.find(i1.first) != requirements.end()) {
132  isDisjoint = false;
133  allInterruptible &=
134  m_impl->scheduledCommands[i1.second].IsInterruptible();
135  intersection.emplace_back(i1.second);
136  }
137  }
138 
139  if (isDisjoint || allInterruptible) {
140  if (allInterruptible) {
141  for (auto&& cmdToCancel : intersection) {
142  Cancel(cmdToCancel);
143  }
144  }
145  command->Initialize();
146  m_impl->scheduledCommands[command] = CommandState{interruptible};
147  for (auto&& requirement : requirements) {
148  m_impl->requirements[requirement] = command;
149  }
150  for (auto&& action : m_impl->initActions) {
151  action(*command);
152  }
153  m_watchdog.AddEpoch(command->GetName() + ".Initialize()");
154  }
155 }

187 void CommandScheduler::Run() {
188  if (m_impl->disabled) {
189  return;
190  }
191 
192  m_watchdog.Reset();
193 
194  // Run the periodic method of all registered subsystems.
195  for (auto&& subsystem : m_impl->subsystems) {
196  subsystem.getFirst()->Periodic();
197  if constexpr (frc::RobotBase::IsSimulation()) {
198  subsystem.getFirst()->SimulationPeriodic();
199  }
200  m_watchdog.AddEpoch("Subsystem Periodic()");
201  }
202 
203  // Poll buttons for new commands to add.
204  for (auto&& button : m_impl->buttons) {
205  button();
206  }
207  m_watchdog.AddEpoch("buttons.Run()");
208 
209  m_impl->inRunLoop = true;
210  // Run scheduled commands, remove finished commands.
211  for (auto iterator = m_impl->scheduledCommands.begin();
212  iterator != m_impl->scheduledCommands.end(); iterator++) {
213  Command* command = iterator->getFirst();
214 
215  if (!command->RunsWhenDisabled() && frc::RobotState::IsDisabled()) {
216  Cancel(command);
217  continue;
218  }
219 
220  command->Execute();
221  for (auto&& action : m_impl->executeActions) {
222  action(*command);
223  }
224  m_watchdog.AddEpoch(command->GetName() + ".Execute()");
225 
226  if (command->IsFinished()) {
227  command->End(false);
228  for (auto&& action : m_impl->finishActions) {
229  action(*command);
230  }
231 
232  for (auto&& requirement : command->GetRequirements()) {
233  m_impl->requirements.erase(requirement);
234  }
235 
236  m_impl->scheduledCommands.erase(iterator);
237  m_watchdog.AddEpoch(command->GetName() + ".End(false)");
238  }
239  }
240  m_impl->inRunLoop = false;
241 
242  for (auto&& commandInterruptible : m_impl->toSchedule) {
243  Schedule(commandInterruptible.second, commandInterruptible.first);
244  }
245 
246  for (auto&& command : m_impl->toCancel) {
247  Cancel(command);
248  }
249 
250  m_impl->toSchedule.clear();
251  m_impl->toCancel.clear();
252 
253  // Add default commands for un-required registered subsystems.
254  for (auto&& subsystem : m_impl->subsystems) {
255  auto s = m_impl->requirements.find(subsystem.getFirst());
256  if (s == m_impl->requirements.end() && subsystem.getSecond()) {
257  Schedule({subsystem.getSecond().get()});
258  }
259  }
260 
261  m_watchdog.Disable();
262  if (m_watchdog.IsExpired()) {
263  m_watchdog.PrintEpochs();
264  }
265 }

315 void CommandScheduler::Cancel(Command* command) {
316  if (!m_impl) {
317  return;
318  }
319 
320  if (m_impl->inRunLoop) {
321  m_impl->toCancel.emplace_back(command);
322  return;
323  }
324 
325  auto find = m_impl->scheduledCommands.find(command);
326  if (find == m_impl->scheduledCommands.end()) {
327  return;
328  }
329  command->End(true);
330  for (auto&& action : m_impl->interruptActions) {
331  action(*command);
332  }
333  m_watchdog.AddEpoch(command->GetName() + ".End(true)");
334  m_impl->scheduledCommands.erase(find);
335  for (auto&& requirement : m_impl->requirements) {
336  if (requirement.second == command) {
337  m_impl->requirements.erase(requirement.first);
338  }
339  }
340 }

429 void CommandScheduler::InitSendable(nt::NTSendableBuilder& builder) {
430  builder.SetSmartDashboardType("Scheduler");
431  auto namesEntry = builder.GetEntry("Names");
432  auto idsEntry = builder.GetEntry("Ids");
433  auto cancelEntry = builder.GetEntry("Cancel");
434 
435  builder.SetUpdateTable([=] {
436  double tmp[1];
437  tmp[0] = 0;
438  auto toCancel = cancelEntry.GetDoubleArray(tmp);
439  for (auto cancel : toCancel) {
440  uintptr_t ptrTmp = static_cast<uintptr_t>(cancel);
441  Command* command = reinterpret_cast<Command*>(ptrTmp);
442  if (m_impl->scheduledCommands.find(command) !=
443  m_impl->scheduledCommands.end()) {
444  Cancel(command);
445  }
446  nt::NetworkTableEntry(cancelEntry).SetDoubleArray({});
447  }
448 
449  wpi::SmallVector<std::string, 8> names;
450  wpi::SmallVector<double, 8> ids;
451  for (auto&& command : m_impl->scheduledCommands) {
452  names.emplace_back(command.first->GetName());
453  uintptr_t ptrTmp = reinterpret_cast<uintptr_t>(command.first);
454  ids.emplace_back(static_cast<double>(ptrTmp));
455  }
456  nt::NetworkTableEntry(namesEntry).SetStringArray(names);
457  nt::NetworkTableEntry(idsEntry).SetDoubleArray(ids);
458  });
459 }
