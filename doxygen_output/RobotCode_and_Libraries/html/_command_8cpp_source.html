35 Command::Command(const wpi::Twine& name, double timeout) {
36  // We use -1.0 to indicate no timeout.
37  if (timeout < 0.0 && timeout != -1.0)
38  wpi_setWPIErrorWithContext(ParameterOutOfRange, "timeout < 0.0");
39 
40  m_timeout = timeout;
41 
42  // If name contains an empty string
43  if (name.isTriviallyEmpty() ||
44  (name.isSingleStringRef() && name.getSingleStringRef().empty())) {
45  SendableRegistry::GetInstance().Add(
46  this, "Command_" + wpi::Twine(typeid(*this).name()));
47  } else {
48  SendableRegistry::GetInstance().Add(this, name);
49  }
50 }

93 bool Command::Run() {
94  if (!m_runWhenDisabled && m_parent == nullptr && RobotState::IsDisabled())
95  Cancel();
96 
97  if (IsCanceled()) return false;
98 
99  if (!m_initialized) {
100  m_initialized = true;
101  StartTiming();
102  _Initialize();
103  Initialize();
104  }
105  _Execute();
106  Execute();
107  return !IsFinished();
108 }
