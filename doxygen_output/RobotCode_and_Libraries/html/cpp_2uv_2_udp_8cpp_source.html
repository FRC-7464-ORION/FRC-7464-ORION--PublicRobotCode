165 void Udp::StartRecv() {
166  Invoke(&uv_udp_recv_start, GetRaw(), &AllocBuf,
167  [](uv_udp_t* handle, ssize_t nread, const uv_buf_t* buf,
168  const sockaddr* addr, unsigned flags) {
169  auto& h = *static_cast<Udp*>(handle->data);
170  Buffer data = *buf;
171 
172  // nread=0 is simply ignored
173  if (nread > 0) {
174  h.received(data, static_cast<size_t>(nread), *addr, flags);
175  } else if (nread < 0) {
176  h.ReportError(nread);
177  }
178 
179  // free the buffer
180  h.FreeBuf(data);
181  });
182 }
