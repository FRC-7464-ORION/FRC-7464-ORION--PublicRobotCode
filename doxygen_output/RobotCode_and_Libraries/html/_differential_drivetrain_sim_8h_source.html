256  static DifferentialDrivetrainSim CreateKitbotSim(
257  frc::DCMotor motor, double gearing, units::meter_t wheelSize,
258  const std::array<double, 7>& measurementStdDevs = {}) {
259  // MOI estimation -- note that I = m r^2 for point masses
260  units::kilogram_square_meter_t batteryMoi = 12.5_lb * 10_in * 10_in;
261  units::kilogram_square_meter_t gearboxMoi = (2.8_lb + 2.0_lb) *
262  2 // CIM plus toughbox per side
263  * (26_in / 2) * (26_in / 2);
264 
265  return DifferentialDrivetrainSim{
266  motor, gearing, batteryMoi + gearboxMoi, 60_lb,
267  wheelSize / 2.0, 26_in, measurementStdDevs};
268  }
