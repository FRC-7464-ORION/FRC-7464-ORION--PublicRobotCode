18 void check_stabilizable(const Eigen::Ref<const Eigen::MatrixXd>& A,
19  const Eigen::Ref<const Eigen::MatrixXd>& B) {
20  // This function checks if (A,B) is a stabilizable pair.
21  // (A,B) is stabilizable if and only if the uncontrollable eigenvalues of
22  // A, if any, have absolute values less than one, where an eigenvalue is
23  // uncontrollable if Rank[lambda * I - A, B] < n.
24  int n = B.rows(), m = B.cols();
25  Eigen::EigenSolver<Eigen::MatrixXd> es(A);
26  for (int i = 0; i < n; i++) {
27  if (es.eigenvalues()[i].real() * es.eigenvalues()[i].real() +
28  es.eigenvalues()[i].imag() * es.eigenvalues()[i].imag() <
29  1)
30  continue;
31  Eigen::MatrixXcd E(n, n + m);
32  E << es.eigenvalues()[i] * Eigen::MatrixXcd::Identity(n, n) - A, B;
33  Eigen::ColPivHouseholderQR<Eigen::MatrixXcd> qr(E);
34  DRAKE_THROW_UNLESS(qr.rank() == n);
35  }
36 }
37 void check_detectable(const Eigen::Ref<const Eigen::MatrixXd>& A,
38  const Eigen::Ref<const Eigen::MatrixXd>& Q) {
39  // This function check if (A,C) is a detectable pair, where Q = C' * C.
40  // (A,C) is detectable if and only if the unobservable eigenvalues of A,
41  // if any, have absolute values less than one, where an eigenvalue is
42  // unobservable if Rank[lambda * I - A; C] < n.
43  // Also, (A,C) is detectable if and only if (A',C') is stabilizable.
44  int n = A.rows();
45  Eigen::LDLT<Eigen::MatrixXd> ldlt(Q);
46  Eigen::MatrixXd L = ldlt.matrixL();
47  Eigen::MatrixXd D = ldlt.vectorD();
48  Eigen::MatrixXd D_sqrt = Eigen::MatrixXd::Zero(n, n);
49  for (int i = 0; i < n; i++) {
50  D_sqrt(i, i) = sqrt(D(i));
51  }
52  Eigen::MatrixXd C = L * D_sqrt;
53  check_stabilizable(A.transpose(), C.transpose());
54 }

56 // "Givens rotation" computes an orthogonal 2x2 matrix R such that
57 // it eliminates the 2nd coordinate of the vector [a,b]', i.e.,
58 // R * [ a ] = [ a_hat ]
59 // [ b ] [ 0 ]
60 // The implementation is based on
61 // https://en.wikipedia.org/wiki/Givens_rotation#Stable_calculation
62 void Givens_rotation(double a, double b, Eigen::Ref<Eigen::Matrix2d> R,
63  double eps = 1e-10) {
64  double c, s;
65  if (fabs(b) < eps) {
66  c = (a < -eps ? -1 : 1);
67  s = 0;
68  } else if (fabs(a) < eps) {
69  c = 0;
70  s = -sgn(b);
71  } else if (fabs(a) > fabs(b)) {
72  double t = b / a;
73  double u = sgn(a) * fabs(sqrt(1 + t * t));
74  c = 1 / u;
75  s = -c * t;
76  } else {
77  double t = a / b;
78  double u = sgn(b) * fabs(sqrt(1 + t * t));
79  s = -1 / u;
80  c = -s * t;
81  }
82  R(0, 0) = c, R(0, 1) = -s, R(1, 0) = s, R(1, 1) = c;
83 }

86 void swap_block_11(Eigen::Ref<Eigen::MatrixXd> S, Eigen::Ref<Eigen::MatrixXd> T,
87  Eigen::Ref<Eigen::MatrixXd> Z, int p) {
88  // Dooren, Case I, p124-125.
89  int n2 = S.rows();
90  Eigen::Matrix2d A = S.block<2, 2>(p, p);
91  Eigen::Matrix2d B = T.block<2, 2>(p, p);
92  Eigen::MatrixXd Z1 = Eigen::MatrixXd::Identity(n2, n2);
93  Eigen::Matrix2d H = A(1, 1) * B - B(1, 1) * A;
94  Givens_rotation(H(0, 1), H(0, 0), Z1.block<2, 2>(p, p));
95  S = (S * Z1).eval();
96  T = (T * Z1).eval();
97  Z = (Z * Z1).eval();
98  Eigen::MatrixXd Q = Eigen::MatrixXd::Identity(n2, n2);
99  Givens_rotation(T(p, p), T(p + 1, p), Q.block<2, 2>(p, p));
100  S = (Q * S).eval();
101  T = (Q * T).eval();
102  S(p + 1, p) = 0;
103  T(p + 1, p) = 0;
104 }

107 void swap_block_21(Eigen::Ref<Eigen::MatrixXd> S, Eigen::Ref<Eigen::MatrixXd> T,
108  Eigen::Ref<Eigen::MatrixXd> Z, int p) {
109  // Dooren, Case II, p126-127.
110  int n2 = S.rows();
111  Eigen::Matrix3d A = S.block<3, 3>(p, p);
112  Eigen::Matrix3d B = T.block<3, 3>(p, p);
113  // Compute H and eliminate H(1,0) by row operation.
114  Eigen::Matrix3d H = A(2, 2) * B - B(2, 2) * A;
115  Eigen::Matrix3d R = Eigen::MatrixXd::Identity(3, 3);
116  Givens_rotation(H(0, 0), H(1, 0), R.block<2, 2>(0, 0));
117  H = (R * H).eval();
118  Eigen::MatrixXd Z1 = Eigen::MatrixXd::Identity(n2, n2);
119  Eigen::MatrixXd Z2 = Eigen::MatrixXd::Identity(n2, n2);
120  Eigen::MatrixXd Q1 = Eigen::MatrixXd::Identity(n2, n2);
121  Eigen::MatrixXd Q2 = Eigen::MatrixXd::Identity(n2, n2);
122  // Compute Z1, Z2, Q1, Q2.
123  Givens_rotation(H(1, 2), H(1, 1), Z1.block<2, 2>(p + 1, p + 1));
124  H = (H * Z1.block<3, 3>(p, p)).eval();
125  Givens_rotation(H(0, 1), H(0, 0), Z2.block<2, 2>(p, p));
126  S = (S * Z1).eval();
127  T = (T * Z1).eval();
128  Z = (Z * Z1 * Z2).eval();
129  Givens_rotation(T(p + 1, p + 1), T(p + 2, p + 1),
130  Q1.block<2, 2>(p + 1, p + 1));
131  S = (Q1 * S * Z2).eval();
132  T = (Q1 * T * Z2).eval();
133  Givens_rotation(T(p, p), T(p + 1, p), Q2.block<2, 2>(p, p));
134  S = (Q2 * S).eval();
135  T = (Q2 * T).eval();
136  S(p + 1, p) = 0;
137  S(p + 2, p) = 0;
138  T(p + 1, p) = 0;
139  T(p + 2, p) = 0;
140  T(p + 2, p + 1) = 0;
141 }

144 void swap_block_12(Eigen::Ref<Eigen::MatrixXd> S, Eigen::Ref<Eigen::MatrixXd> T,
145  Eigen::Ref<Eigen::MatrixXd> Z, int p) {
146  int n2 = S.rows();
147  // Swap the role of S and T.
148  Eigen::MatrixXd Z1 = Eigen::MatrixXd::Identity(n2, n2);
149  Eigen::MatrixXd Z2 = Eigen::MatrixXd::Identity(n2, n2);
150  Eigen::MatrixXd Q0 = Eigen::MatrixXd::Identity(n2, n2);
151  Eigen::MatrixXd Q1 = Eigen::MatrixXd::Identity(n2, n2);
152  Eigen::MatrixXd Q2 = Eigen::MatrixXd::Identity(n2, n2);
153  Eigen::MatrixXd Q3 = Eigen::MatrixXd::Identity(n2, n2);
154  Givens_rotation(S(p + 1, p + 1), S(p + 2, p + 1),
155  Q0.block<2, 2>(p + 1, p + 1));
156  S = (Q0 * S).eval();
157  T = (Q0 * T).eval();
158  Eigen::Matrix3d A = S.block<3, 3>(p, p);
159  Eigen::Matrix3d B = T.block<3, 3>(p, p);
160  // Compute H and eliminate H(2,1) by column operation.
161  Eigen::Matrix3d H = B(0, 0) * A - A(0, 0) * B;
162  Eigen::Matrix3d R = Eigen::MatrixXd::Identity(3, 3);
163  Givens_rotation(H(2, 2), H(2, 1), R.block<2, 2>(1, 1));
164  H = (H * R).eval();
165  // Compute Q1, Q2, Z1, Z2.
166  Givens_rotation(H(0, 1), H(1, 1), Q1.block<2, 2>(p, p));
167  H = (Q1.block<3, 3>(p, p) * H).eval();
168  Givens_rotation(H(1, 2), H(2, 2), Q2.block<2, 2>(p + 1, p + 1));
169  S = (Q1 * S).eval();
170  T = (Q1 * T).eval();
171  Givens_rotation(S(p + 1, p + 1), S(p + 1, p), Z1.block<2, 2>(p, p));
172  S = (Q2 * S * Z1).eval();
173  T = (Q2 * T * Z1).eval();
174  Givens_rotation(S(p + 2, p + 2), S(p + 2, p + 1),
175  Z2.block<2, 2>(p + 1, p + 1));
176  S = (S * Z2).eval();
177  T = (T * Z2).eval();
178  Z = (Z * Z1 * Z2).eval();
179  // Swap back the role of S and T.
180  Givens_rotation(T(p, p), T(p + 1, p), Q3.block<2, 2>(p, p));
181  S = (Q3 * S).eval();
182  T = (Q3 * T).eval();
183  S(p + 2, p) = 0;
184  S(p + 2, p + 1) = 0;
185  T(p + 1, p) = 0;
186  T(p + 2, p) = 0;
187  T(p + 2, p + 1) = 0;
188 }

191 void swap_block_22(Eigen::Ref<Eigen::MatrixXd> S, Eigen::Ref<Eigen::MatrixXd> T,
192  Eigen::Ref<Eigen::MatrixXd> Z, int p) {
193  // Direct Swapping Algorithm based on
194  // "Numerical Methods for General and Structured Eigenvalue Problems" by
195  // Daniel Kressner, p108-111.
196  // ( http://sma.epfl.ch/~anchpcommon/publications/kressner_eigenvalues.pdf ).
197  // Also relevant but not applicable here:
198  // "On Swapping Diagonal Blocks in Real Schur Form" by Zhaojun Bai and James
199  // W. Demmelt;
200  int n2 = S.rows();
201  Eigen::MatrixXd A = S.block<4, 4>(p, p);
202  Eigen::MatrixXd B = T.block<4, 4>(p, p);
203  // Solve
204  // A11 * X - Y A22 = A12
205  // B11 * X - Y B22 = B12
206  // Reduce to solve Cx=D, where x=[x1;...;x4;y1;...;y4].
207  Eigen::Matrix<double, 8, 8> C = Eigen::Matrix<double, 8, 8>::Zero();
208  Eigen::Matrix<double, 8, 1> D;
209  // clang-format off
210  C << A(0, 0), 0, A(0, 1), 0, -A(2, 2), -A(3, 2), 0, 0,
211  0, A(0, 0), 0, A(0, 1), -A(2, 3), -A(3, 3), 0, 0,
212  A(1, 0), 0, A(1, 1), 0, 0, 0, -A(2, 2), -A(3, 2),
213  0, A(1, 0), 0, A(1, 1), 0, 0, -A(2, 3), -A(3, 3),
214  B(0, 0), 0, B(0, 1), 0, -B(2, 2), -B(3, 2), 0, 0,
215  0, B(0, 0), 0, B(0, 1), -B(2, 3), -B(3, 3), 0, 0,
216  B(1, 0), 0, B(1, 1), 0, 0, 0, -B(2, 2), -B(3, 2),
217  0, B(1, 0), 0, B(1, 1), 0, 0, -B(2, 3), -B(3, 3);
218  // clang-format on
219  D << A(0, 2), A(0, 3), A(1, 2), A(1, 3), B(0, 2), B(0, 3), B(1, 2), B(1, 3);
220  Eigen::MatrixXd x = C.colPivHouseholderQr().solve(D);
221  // Q * [ -Y ] = [ R_Y ] , Z' * [ -X ] = [ R_X ] .
222  // [ I ] [ 0 ] [ I ] = [ 0 ]
223  Eigen::Matrix<double, 4, 2> X, Y;
224  X << -x(0, 0), -x(1, 0), -x(2, 0), -x(3, 0), Eigen::Matrix2d::Identity();
225  Y << -x(4, 0), -x(5, 0), -x(6, 0), -x(7, 0), Eigen::Matrix2d::Identity();
226  Eigen::MatrixXd Q1 = Eigen::MatrixXd::Identity(n2, n2);
227  Eigen::MatrixXd Z1 = Eigen::MatrixXd::Identity(n2, n2);
228  Eigen::ColPivHouseholderQR<Eigen::Matrix<double, 4, 2> > qr1(X);
229  Z1.block<4, 4>(p, p) = qr1.householderQ();
230  Eigen::ColPivHouseholderQR<Eigen::Matrix<double, 4, 2> > qr2(Y);
231  Q1.block<4, 4>(p, p) = qr2.householderQ().adjoint();
232  // Apply transform Q1 * (S,T) * Z1.
233  S = (Q1 * S * Z1).eval();
234  T = (Q1 * T * Z1).eval();
235  Z = (Z * Z1).eval();
236  // Eliminate the T(p+3,p+2) entry.
237  Eigen::MatrixXd Q2 = Eigen::MatrixXd::Identity(n2, n2);
238  Givens_rotation(T(p + 2, p + 2), T(p + 3, p + 2),
239  Q2.block<2, 2>(p + 2, p + 2));
240  S = (Q2 * S).eval();
241  T = (Q2 * T).eval();
242  // Eliminate the T(p+1,p) entry.
243  Eigen::MatrixXd Q3 = Eigen::MatrixXd::Identity(n2, n2);
244  Givens_rotation(T(p, p), T(p + 1, p), Q3.block<2, 2>(p, p));
245  S = (Q3 * S).eval();
246  T = (Q3 * T).eval();
247  S(p + 2, p) = 0;
248  S(p + 2, p + 1) = 0;
249  S(p + 3, p) = 0;
250  S(p + 3, p + 1) = 0;
251  T(p + 1, p) = 0;
252  T(p + 2, p) = 0;
253  T(p + 2, p + 1) = 0;
254  T(p + 3, p) = 0;
255  T(p + 3, p + 1) = 0;
256  T(p + 3, p + 2) = 0;
257 }

259 // Functionality of "swap_block" function:
260 // swap the 1x1 or 2x2 blocks pointed by p and q.
261 // There are four cases: swapping 1x1 and 1x1 matrices, swapping 2x2 and 1x1
262 // matrices, swapping 1x1 and 2x2 matrices, and swapping 2x2 and 2x2 matrices.
263 // Algorithms are described in the papers
264 // "A generalized eigenvalue approach for solving Riccati equations" by P. Van
265 // Dooren, 1981 ( http://epubs.siam.org/doi/pdf/10.1137/0902010 ), and
266 // "Numerical Methods for General and Structured Eigenvalue Problems" by
267 // Daniel Kressner, 2005.
268 void swap_block(Eigen::Ref<Eigen::MatrixXd> S, Eigen::Ref<Eigen::MatrixXd> T,
269  Eigen::Ref<Eigen::MatrixXd> Z, int p, int q, int q_block_size,
270  double eps = 1e-10) {
271  int p_tmp = q, p_block_size;
272  while (p_tmp-- > p) {
273  p_block_size = 1;
274  if (p_tmp >= 1 && fabs(S(p_tmp, p_tmp - 1)) > eps) {
275  p_block_size = 2;
276  p_tmp--;
277  }
278  switch (p_block_size * 10 + q_block_size) {
279  case 11:
280  swap_block_11(S, T, Z, p_tmp);
281  break;
282  case 21:
283  swap_block_21(S, T, Z, p_tmp);
284  break;
285  case 12:
286  swap_block_12(S, T, Z, p_tmp);
287  break;
288  case 22:
289  swap_block_22(S, T, Z, p_tmp);
290  break;
291  }
292  }
293 }

295 // Functionality of "reorder_eigen" function:
296 // Reorder the eigenvalues of (S,T) such that the top-left n by n matrix has
297 // stable eigenvalues by multiplying Q's and Z's on the left and the right,
298 // respectively.
299 // Stable eigenvalues are inside the unit disk.
300 //
301 // Algorithm:
302 // Go along the diagonals of (S,T) from the top left to the bottom right.
303 // Once find a stable eigenvalue, push it to top left.
304 // In implementation, use two pointers, p and q.
305 // p points to the current block (1x1 or 2x2) and q points to the block with the
306 // stable eigenvalue(s).
307 // Push the block pointed by q to the position pointed by p.
308 // Finish when n stable eigenvalues are placed at the top-left n by n matrix.
309 // The algorithm for swapping blocks is described in the papers
310 // "A generalized eigenvalue approach for solving Riccati equations" by P. Van
311 // Dooren, 1981, and "Numerical Methods for General and Structured Eigenvalue
312 // Problems" by Daniel Kressner, 2005.
313 void reorder_eigen(Eigen::Ref<Eigen::MatrixXd> S, Eigen::Ref<Eigen::MatrixXd> T,
314  Eigen::Ref<Eigen::MatrixXd> Z, double eps = 1e-10) {
315  // abs(a) < eps => a = 0
316  int n2 = S.rows();
317  int n = n2 / 2, p = 0, q = 0;
318 
319  // Find the first unstable p block.
320  while (p < n) {
321  if (fabs(S(p + 1, p)) < eps) { // p block size = 1
322  if (fabs(T(p, p)) > eps && fabs(S(p, p)) <= fabs(T(p, p))) { // stable
323  p++;
324  continue;
325  }
326  } else { // p block size = 2
327  const double det_T =
328  T(p, p) * T(p + 1, p + 1) - T(p + 1, p) * T(p, p + 1);
329  if (fabs(det_T) > eps) {
330  const double det_S =
331  S(p, p) * S(p + 1, p + 1) - S(p + 1, p) * S(p, p + 1);
332  if (fabs(det_S) <= fabs(det_T)) { // stable
333  p += 2;
334  continue;
335  }
336  }
337  }
338  break;
339  }
340  q = p;
341 
342  // Make the first n generalized eigenvalues stable.
343  while (p < n && q < n2) {
344  // Update q.
345  int q_block_size = 0;
346  while (q < n2) {
347  if (q == n2 - 1 || fabs(S(q + 1, q)) < eps) { // block size = 1
348  if (fabs(T(q, q)) > eps && fabs(S(q, q)) <= fabs(T(q, q))) {
349  q_block_size = 1;
350  break;
351  }
352  q++;
353  } else { // block size = 2
354  const double det_T =
355  T(q, q) * T(q + 1, q + 1) - T(q + 1, q) * T(q, q + 1);
356  if (fabs(det_T) > eps) {
357  const double det_S =
358  S(q, q) * S(q + 1, q + 1) - S(q + 1, q) * S(q, q + 1);
359  if (fabs(det_S) <= fabs(det_T)) {
360  q_block_size = 2;
361  break;
362  }
363  }
364  q += 2;
365  }
366  }
367  if (q >= n2)
368  throw std::runtime_error("fail to find enough stable eigenvalues");
369  // Swap blocks pointed by p and q.
370  if (p != q) {
371  swap_block(S, T, Z, p, q, q_block_size);
372  p += q_block_size;
373  q += q_block_size;
374  }
375  }
376  if (p < n && q >= n2)
377  throw std::runtime_error("fail to find enough stable eigenvalues");
378 }

406 Eigen::MatrixXd DiscreteAlgebraicRiccatiEquation(
407  const Eigen::Ref<const Eigen::MatrixXd>& A,
408  const Eigen::Ref<const Eigen::MatrixXd>& B,
409  const Eigen::Ref<const Eigen::MatrixXd>& Q,
410  const Eigen::Ref<const Eigen::MatrixXd>& R) {
411  int n = B.rows(), m = B.cols();
412 
413  DRAKE_DEMAND(m <= n);
414  DRAKE_DEMAND(A.rows() == n && A.cols() == n);
415  DRAKE_DEMAND(Q.rows() == n && Q.cols() == n);
416  DRAKE_DEMAND(R.rows() == m && R.cols() == m);
417  DRAKE_DEMAND(is_approx_equal_abstol(Q, Q.transpose(), 1e-10));
418  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(Q);
419  for (int i = 0; i < n; i++) {
420  DRAKE_THROW_UNLESS(es.eigenvalues()[i] >= 0);
421  }
422  DRAKE_DEMAND(is_approx_equal_abstol(R, R.transpose(), 1e-10));
423  Eigen::LLT<Eigen::MatrixXd> R_cholesky(R);
424  DRAKE_THROW_UNLESS(R_cholesky.info() == Eigen::Success);
425  check_stabilizable(A, B);
426  check_detectable(A, Q);
427 
428  Eigen::MatrixXd M(2 * n, 2 * n), L(2 * n, 2 * n);
429  M << A, Eigen::MatrixXd::Zero(n, n), -Q, Eigen::MatrixXd::Identity(n, n);
430  L << Eigen::MatrixXd::Identity(n, n), B * R.inverse() * B.transpose(),
431  Eigen::MatrixXd::Zero(n, n), A.transpose();
432 
433  // QZ decomposition of M and L
434  // QMZ = S, QLZ = T
435  // where Q and Z are real orthogonal matrixes
436  // T is upper-triangular matrix, and S is upper quasi-triangular matrix
437  Eigen::RealQZ<Eigen::MatrixXd> qz(2 * n);
438  qz.compute(M, L); // M = Q S Z, L = Q T Z (Q and Z computed by Eigen package
439  // are adjoints of Q and Z above)
440  Eigen::MatrixXd S = qz.matrixS(), T = qz.matrixT(),
441  Z = qz.matrixZ().adjoint();
442 
443  // Reorder the generalized eigenvalues of (S,T).
444  Eigen::MatrixXd Z2 = Eigen::MatrixXd::Identity(2 * n, 2 * n);
445  reorder_eigen(S, T, Z2);
446  Z = (Z * Z2).eval();
447 
448  // The first n columns of Z is ( U1 ) .
449  // ( U2 )
450  // -1
451  // X = U2 * U1 is a solution of the discrete time Riccati equation.
452  Eigen::MatrixXd U1 = Z.block(0, 0, n, n), U2 = Z.block(n, 0, n, n);
453  Eigen::MatrixXd X = U2 * U1.inverse();
454  X = (X + X.adjoint().eval()) / 2.0;
455  return X;
456 }

458 Eigen::MatrixXd DiscreteAlgebraicRiccatiEquation(
459  const Eigen::Ref<const Eigen::MatrixXd>& A,
460  const Eigen::Ref<const Eigen::MatrixXd>& B,
461  const Eigen::Ref<const Eigen::MatrixXd>& Q,
462  const Eigen::Ref<const Eigen::MatrixXd>& R,
463  const Eigen::Ref<const Eigen::MatrixXd>& N) {
464  DRAKE_DEMAND(N.rows() == B.rows() && N.cols() == B.cols());
465 
466  // This is a change of variables to make the DARE that includes Q, R, and N
467  // cost matrices fit the form of the DARE that includes only Q and R cost
468  // matrices.
469  Eigen::MatrixXd A2 = A - B * R.llt().solve(N.transpose());
470  Eigen::MatrixXd Q2 = Q - N * R.llt().solve(N.transpose());
471  return DiscreteAlgebraicRiccatiEquation(A2, B, Q2, R);
472 }
