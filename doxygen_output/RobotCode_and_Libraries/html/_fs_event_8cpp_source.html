40 std::string FsEvent::GetPath() {
41  // Per the libuv docs, GetPath() always gives us a null-terminated string.
42  // common case should be small
43  char buf[128];
44  size_t size = 128;
45  int r = uv_fs_event_getpath(GetRaw(), buf, &size);
46  if (r == 0) {
47  return buf;
48  } else if (r == UV_ENOBUFS) {
49  // need to allocate a big enough buffer
50  char* buf2 = static_cast<char*>(std::malloc(size));
51  r = uv_fs_event_getpath(GetRaw(), buf2, &size);
52  if (r == 0) {
53  std::string out{buf2};
54  std::free(buf2);
55  return out;
56  }
57  std::free(buf2);
58  }
59  ReportError(r);
60  return std::string{};
61 }
