34 std::string_view NetworkTable::NormalizeKey(std::string_view key,
35  wpi::SmallVectorImpl<char>& buf,
36  bool withLeadingSlash) {
37  buf.clear();
38  if (withLeadingSlash) {
39  buf.push_back(PATH_SEPARATOR_CHAR);
40  }
41  // for each path element, add it with a slash following
42  wpi::SmallVector<std::string_view, 16> parts;
43  wpi::split(key, parts, PATH_SEPARATOR_CHAR, -1, false);
44  for (auto i = parts.begin(); i != parts.end(); ++i) {
45  buf.append(i->begin(), i->end());
46  buf.push_back(PATH_SEPARATOR_CHAR);
47  }
48  // remove trailing slash if the input key didn't have one
49  if (!key.empty() && key.back() != PATH_SEPARATOR_CHAR) {
50  buf.pop_back();
51  }
52  return {buf.data(), buf.size()};
53 }

55 std::vector<std::string> NetworkTable::GetHierarchy(std::string_view key) {
56  std::vector<std::string> hierarchy;
57  hierarchy.emplace_back(1, PATH_SEPARATOR_CHAR);
58  // for each path element, add it to the end of what we built previously
59  wpi::SmallString<128> path;
60  wpi::SmallVector<std::string_view, 16> parts;
61  wpi::split(key, parts, PATH_SEPARATOR_CHAR, -1, false);
62  if (!parts.empty()) {
63  for (auto i = parts.begin(); i != parts.end(); ++i) {
64  path += PATH_SEPARATOR_CHAR;
65  path += *i;
66  hierarchy.emplace_back(path.str());
67  }
68  // handle trailing slash
69  if (key.back() == PATH_SEPARATOR_CHAR) {
70  path += PATH_SEPARATOR_CHAR;
71  hierarchy.emplace_back(path.str());
72  }
73  }
74  return hierarchy;
75 }

137 NT_EntryListener NetworkTable::AddSubTableListener(TableListener listener,
138  bool localNotify) {
139  size_t prefix_len = m_path.size() + 1;
140 
141  // The lambda needs to be copyable, but StringMap is not, so use
142  // a shared_ptr to it.
143  auto notified_tables = std::make_shared<wpi::StringMap<char>>();
144 
145  unsigned int flags = NT_NOTIFY_NEW | NT_NOTIFY_IMMEDIATE;
146  if (localNotify) {
147  flags |= NT_NOTIFY_LOCAL;
148  }
149  NT_EntryListener id = nt::AddEntryListener(
150  m_inst, fmt::format("{}/", m_path),
151  [=](const EntryNotification& event) {
152  auto relative_key = wpi::substr(event.name, prefix_len);
153  auto end_sub_table = relative_key.find(PATH_SEPARATOR_CHAR);
154  if (end_sub_table == std::string_view::npos) {
155  return;
156  }
157  auto sub_table_key = wpi::substr(relative_key, 0, end_sub_table);
158  if (notified_tables->find(sub_table_key) == notified_tables->end()) {
159  return;
160  }
161  notified_tables->insert(std::make_pair(sub_table_key, '\0'));
162  listener(this, sub_table_key, this->GetSubTable(sub_table_key));
163  },
164  flags);
165  m_listeners.emplace_back(id);
166  return id;
167 }

193 std::vector<std::string> NetworkTable::GetKeys(int types) const {
194  std::vector<std::string> keys;
195  size_t prefix_len = m_path.size() + 1;
196  auto infos = GetEntryInfo(m_inst, fmt::format("{}/", m_path), types);
197  std::scoped_lock lock(m_mutex);
198  for (auto& info : infos) {
199  auto relative_key = wpi::substr(info.name, prefix_len);
200  if (relative_key.find(PATH_SEPARATOR_CHAR) != std::string_view::npos) {
201  continue;
202  }
203  keys.emplace_back(relative_key);
204  m_entries[relative_key] = info.entry;
205  }
206  return keys;
207 }
