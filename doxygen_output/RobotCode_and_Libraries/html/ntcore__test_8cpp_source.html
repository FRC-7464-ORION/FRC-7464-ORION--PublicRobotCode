205 struct NT_RpcDefinition* NT_GetRpcDefinitionForTesting(
206  unsigned int version, const char* name, size_t num_params,
207  const struct NT_RpcParamDef* params, size_t num_results,
208  const struct NT_RpcResultDef* results, int* struct_size) {
209  struct NT_RpcDefinition* def = static_cast<NT_RpcDefinition*>(
210  wpi::safe_calloc(1, sizeof(NT_RpcDefinition)));
211  def->version = version;
212  nt::ConvertToC(name, &def->name);
213  def->num_params = num_params;
214  def->params = static_cast<NT_RpcParamDef*>(
215  wpi::safe_malloc(num_params * sizeof(NT_RpcParamDef)));
216  for (size_t i = 0; i < num_params; ++i) {
217  CopyNtString(&params[i].name, &def->params[i].name);
218  CopyNtValue(&params[i].def_value, &def->params[i].def_value);
219  }
220  def->num_results = num_results;
221  def->results = static_cast<NT_RpcResultDef*>(
222  wpi::safe_malloc(num_results * sizeof(NT_RpcResultDef)));
223  for (size_t i = 0; i < num_results; ++i) {
224  CopyNtString(&results[i].name, &def->results[i].name);
225  def->results[i].type = results[i].type;
226  }
227  *struct_size = sizeof(NT_RpcDefinition);
228  return def;
229 }
