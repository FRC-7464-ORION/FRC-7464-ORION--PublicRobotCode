83 void MotorSafety::Check() {
84  bool enabled;
85  double stopTime;
86 
87  {
88  std::scoped_lock lock(m_thisMutex);
89  enabled = m_enabled;
90  stopTime = m_stopTime;
91  }
92 
93  DriverStation& ds = DriverStation::GetInstance();
94  if (!enabled || ds.IsDisabled() || ds.IsTest()) {
95  return;
96  }
97 
98  if (stopTime < Timer::GetFPGATimestamp()) {
99  wpi::SmallString<128> buf;
100  wpi::raw_svector_ostream desc(buf);
101  GetDescription(desc);
102  desc << "... Output not updated often enough.";
103  wpi_setWPIErrorWithContext(Timeout, desc.str());
104  StopMotor();
105  }
106 }
