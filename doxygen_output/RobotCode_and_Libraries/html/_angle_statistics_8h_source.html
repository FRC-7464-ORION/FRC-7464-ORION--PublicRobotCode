93 template <int CovDim, int States>
94 Eigen::Vector<double, CovDim> AngleMean(
95  const Eigen::Matrix<double, CovDim, 2 * States + 1>& sigmas,
96  const Eigen::Vector<double, 2 * States + 1>& Wm, int angleStatesIdx) {
97  double sumSin = sigmas.row(angleStatesIdx)
98  .unaryExpr([](auto it) { return std::sin(it); })
99  .sum();
100  double sumCos = sigmas.row(angleStatesIdx)
101  .unaryExpr([](auto it) { return std::cos(it); })
102  .sum();
103 
104  Eigen::Vector<double, CovDim> ret = sigmas * Wm;
105  ret[angleStatesIdx] = std::atan2(sumSin, sumCos);
106  return ret;
107 }
