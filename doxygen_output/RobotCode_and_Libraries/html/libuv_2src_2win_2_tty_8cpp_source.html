89 /*
90  * The console virtual window.
91  *
92  * Normally cursor movement in windows is relative to the console screen buffer,
93  * e.g. the application is allowed to overwrite the 'history'. This is very
94  * inconvenient, it makes absolute cursor movement pretty useless. There is
95  * also the concept of 'client rect' which is defined by the actual size of
96  * the console window and the scroll position of the screen buffer, but it's
97  * very volatile because it changes when the user scrolls.
98  *
99  * To make cursor movement behave sensibly we define a virtual window to which
100  * cursor movement is confined. The virtual window is always as wide as the
101  * console screen buffer, but it's height is defined by the size of the
102  * console window. The top of the virtual window aligns with the position
103  * of the caret when the first stdout/err handle is created, unless that would
104  * mean that it would extend beyond the bottom of the screen buffer - in that
105  * that case it's located as far down as possible.
106  *
107  * When the user writes a long text or many newlines, such that the output
108  * reaches beyond the bottom of the virtual window, the virtual window is
109  * shifted downwards, but not resized.
110  *
111  * Since all tty i/o happens on the same console, this window is shared
112  * between all stdout/stderr handles.
113  */

179 int uv_tty_init(uv_loop_t* loop, uv_tty_t* tty, uv_file fd, int unused) {
180  BOOL readable;
181  DWORD NumberOfEvents;
182  HANDLE handle;
183  CONSOLE_SCREEN_BUFFER_INFO screen_buffer_info;
184  (void)unused;
185 
186  uv__once_init();
187  handle = (HANDLE) uv__get_osfhandle(fd);
188  if (handle == INVALID_HANDLE_VALUE)
189  return UV_EBADF;
190 
191  if (fd <= 2) {
192  /* In order to avoid closing a stdio file descriptor 0-2, duplicate the
193  * underlying OS handle and forget about the original fd.
194  * We could also opt to use the original OS handle and just never close it,
195  * but then there would be no reliable way to cancel pending read operations
196  * upon close.
197  */
198  if (!DuplicateHandle(INVALID_HANDLE_VALUE,
199  handle,
200  INVALID_HANDLE_VALUE,
201  &handle,
202  0,
203  FALSE,
204  DUPLICATE_SAME_ACCESS))
205  return uv_translate_sys_error(GetLastError());
206  fd = -1;
207  }
208 
209  readable = GetNumberOfConsoleInputEvents(handle, &NumberOfEvents);
210  if (!readable) {
211  /* Obtain the screen buffer info with the output handle. */
212  if (!GetConsoleScreenBufferInfo(handle, &screen_buffer_info)) {
213  return uv_translate_sys_error(GetLastError());
214  }
215 
216  /* Obtain the tty_output_lock because the virtual window state is shared
217  * between all uv_tty_t handles. */
218  uv_sem_wait(&uv_tty_output_lock);
219 
220  if (uv__vterm_state == UV_UNCHECKED)
221  uv__determine_vterm_state(handle);
222 
223  /* Remember the original console text attributes. */
224  uv_tty_capture_initial_style(&screen_buffer_info);
225 
226  uv_tty_update_virtual_window(&screen_buffer_info);
227 
228  uv_sem_post(&uv_tty_output_lock);
229  }
230 
231 
232  uv_stream_init(loop, (uv_stream_t*) tty, UV_TTY);
233  uv_connection_init((uv_stream_t*) tty);
234 
235  tty->handle = handle;
236  tty->u.fd = fd;
237  tty->reqs_pending = 0;
238  tty->flags |= UV_HANDLE_BOUND;
239 
240  if (readable) {
241  /* Initialize TTY input specific fields. */
242  tty->flags |= UV_HANDLE_TTY_READABLE | UV_HANDLE_READABLE;
243  /* TODO: remove me in v2.x. */
244  tty->tty.rd.unused_ = NULL;
245  tty->tty.rd.read_line_buffer = uv_null_buf_;
246  tty->tty.rd.read_raw_wait = NULL;
247 
248  /* Init keycode-to-vt100 mapper state. */
249  tty->tty.rd.last_key_len = 0;
250  tty->tty.rd.last_key_offset = 0;
251  tty->tty.rd.last_utf16_high_surrogate = 0;
252  memset(&tty->tty.rd.last_input_record, 0, sizeof tty->tty.rd.last_input_record);
253  } else {
254  /* TTY output specific fields. */
255  tty->flags |= UV_HANDLE_WRITABLE;
256 
257  /* Init utf8-to-utf16 conversion state. */
258  tty->tty.wr.utf8_bytes_left = 0;
259  tty->tty.wr.utf8_codepoint = 0;
260 
261  /* Initialize eol conversion state */
262  tty->tty.wr.previous_eol = 0;
263 
264  /* Init ANSI parser state. */
265  tty->tty.wr.ansi_parser_state = ANSI_NORMAL;
266  }
267 
268  return 0;
269 }

275 static void uv_tty_capture_initial_style(CONSOLE_SCREEN_BUFFER_INFO* info) {
276  static int style_captured = 0;
277 
278  /* Only do this once.
279  Assumption: Caller has acquired uv_tty_output_lock. */
280  if (style_captured)
281  return;
282 
283  /* Save raw win32 attributes. */
284  uv_tty_default_text_attributes = info->wAttributes;
285 
286  /* Convert black text on black background to use white text. */
287  if (uv_tty_default_text_attributes == 0)
288  uv_tty_default_text_attributes = 7;
289 
290  /* Convert Win32 attributes to ANSI colors. */
291  uv_tty_default_fg_color = 0;
292  uv_tty_default_bg_color = 0;
293  uv_tty_default_fg_bright = 0;
294  uv_tty_default_bg_bright = 0;
295  uv_tty_default_inverse = 0;
296 
297  if (uv_tty_default_text_attributes & FOREGROUND_RED)
298  uv_tty_default_fg_color |= 1;
299 
300  if (uv_tty_default_text_attributes & FOREGROUND_GREEN)
301  uv_tty_default_fg_color |= 2;
302 
303  if (uv_tty_default_text_attributes & FOREGROUND_BLUE)
304  uv_tty_default_fg_color |= 4;
305 
306  if (uv_tty_default_text_attributes & BACKGROUND_RED)
307  uv_tty_default_bg_color |= 1;
308 
309  if (uv_tty_default_text_attributes & BACKGROUND_GREEN)
310  uv_tty_default_bg_color |= 2;
311 
312  if (uv_tty_default_text_attributes & BACKGROUND_BLUE)
313  uv_tty_default_bg_color |= 4;
314 
315  if (uv_tty_default_text_attributes & FOREGROUND_INTENSITY)
316  uv_tty_default_fg_bright = 1;
317 
318  if (uv_tty_default_text_attributes & BACKGROUND_INTENSITY)
319  uv_tty_default_bg_bright = 1;
320 
321  if (uv_tty_default_text_attributes & COMMON_LVB_REVERSE_VIDEO)
322  uv_tty_default_inverse = 1;
323 
324  style_captured = 1;
325 }

328 int uv_tty_set_mode(uv_tty_t* tty, uv_tty_mode_t mode) {
329  DWORD flags;
330  unsigned char was_reading;
331  uv_alloc_cb alloc_cb;
332  uv_read_cb read_cb;
333  int err;
334 
335  if (!(tty->flags & UV_HANDLE_TTY_READABLE)) {
336  return UV_EINVAL;
337  }
338 
339  if (!!mode == !!(tty->flags & UV_HANDLE_TTY_RAW)) {
340  return 0;
341  }
342 
343  switch (mode) {
344  case UV_TTY_MODE_NORMAL:
345  flags = ENABLE_ECHO_INPUT | ENABLE_LINE_INPUT | ENABLE_PROCESSED_INPUT;
346  break;
347  case UV_TTY_MODE_RAW:
348  flags = ENABLE_WINDOW_INPUT;
349  break;
350  case UV_TTY_MODE_IO:
351  return UV_ENOTSUP;
352  default:
353  return UV_EINVAL;
354  }
355 
356  /* If currently reading, stop, and restart reading. */
357  if (tty->flags & UV_HANDLE_READING) {
358  was_reading = 1;
359  alloc_cb = tty->alloc_cb;
360  read_cb = tty->read_cb;
361  err = uv_tty_read_stop(tty);
362  if (err) {
363  return uv_translate_sys_error(err);
364  }
365  } else {
366  was_reading = 0;
367  alloc_cb = NULL;
368  read_cb = NULL;
369  }
370 
371  uv_sem_wait(&uv_tty_output_lock);
372  if (!SetConsoleMode(tty->handle, flags)) {
373  err = uv_translate_sys_error(GetLastError());
374  uv_sem_post(&uv_tty_output_lock);
375  return err;
376  }
377  uv_sem_post(&uv_tty_output_lock);
378 
379  /* Update flag. */
380  tty->flags &= ~UV_HANDLE_TTY_RAW;
381  tty->flags |= mode ? UV_HANDLE_TTY_RAW : 0;
382 
383  /* If we just stopped reading, restart. */
384  if (was_reading) {
385  err = uv_tty_read_start(tty, alloc_cb, read_cb);
386  if (err) {
387  return uv_translate_sys_error(err);
388  }
389  }
390 
391  return 0;
392 }

433 static void uv_tty_queue_read_raw(uv_loop_t* loop, uv_tty_t* handle) {
434  uv_read_t* req;
435  BOOL r;
436 
437  assert(handle->flags & UV_HANDLE_READING);
438  assert(!(handle->flags & UV_HANDLE_READ_PENDING));
439 
440  assert(handle->handle && handle->handle != INVALID_HANDLE_VALUE);
441 
442  handle->tty.rd.read_line_buffer = uv_null_buf_;
443 
444  req = &handle->read_req;
445  memset(&req->u.io.overlapped, 0, sizeof(req->u.io.overlapped));
446 
447  r = RegisterWaitForSingleObject(&handle->tty.rd.read_raw_wait,
448  handle->handle,
449  uv_tty_post_raw_read,
450  (void*) req,
451  INFINITE,
452  WT_EXECUTEINWAITTHREAD | WT_EXECUTEONLYONCE);
453  if (!r) {
454  handle->tty.rd.read_raw_wait = NULL;
455  SET_REQ_ERROR(req, GetLastError());
456  uv_insert_pending_req(loop, (uv_req_t*)req);
457  }
458 
459  handle->flags |= UV_HANDLE_READ_PENDING;
460  handle->reqs_pending++;
461 }

464 static DWORD CALLBACK uv_tty_line_read_thread(void* data) {
465  uv_loop_t* loop;
466  uv_tty_t* handle;
467  uv_req_t* req;
468  DWORD bytes, read_bytes;
469  WCHAR utf16[MAX_INPUT_BUFFER_LENGTH / 3];
470  DWORD chars, read_chars;
471  LONG status;
472  COORD pos;
473  BOOL read_console_success;
474 
475  assert(data);
476 
477  req = (uv_req_t*) data;
478  handle = (uv_tty_t*) req->data;
479  loop = handle->loop;
480 
481  assert(handle->tty.rd.read_line_buffer.base != NULL);
482  assert(handle->tty.rd.read_line_buffer.len > 0);
483 
484  /* ReadConsole can't handle big buffers. */
485  if (handle->tty.rd.read_line_buffer.len < MAX_INPUT_BUFFER_LENGTH) {
486  bytes = handle->tty.rd.read_line_buffer.len;
487  } else {
488  bytes = MAX_INPUT_BUFFER_LENGTH;
489  }
490 
491  /* At last, unicode! One utf-16 codeunit never takes more than 3 utf-8
492  * codeunits to encode. */
493  chars = bytes / 3;
494 
495  status = InterlockedExchange(&uv__read_console_status, IN_PROGRESS);
496  if (status == TRAP_REQUESTED) {
497  SET_REQ_SUCCESS(req);
498  req->u.io.overlapped.InternalHigh = 0;
499  POST_COMPLETION_FOR_REQ(loop, req);
500  return 0;
501  }
502 
503  read_console_success = ReadConsoleW(handle->handle,
504  (void*) utf16,
505  chars,
506  &read_chars,
507  NULL);
508 
509  if (read_console_success) {
510  read_bytes = WideCharToMultiByte(CP_UTF8,
511  0,
512  utf16,
513  read_chars,
514  handle->tty.rd.read_line_buffer.base,
515  bytes,
516  NULL,
517  NULL);
518  SET_REQ_SUCCESS(req);
519  req->u.io.overlapped.InternalHigh = read_bytes;
520  } else {
521  SET_REQ_ERROR(req, GetLastError());
522  }
523 
524  status = InterlockedExchange(&uv__read_console_status, COMPLETED);
525 
526  if (status == TRAP_REQUESTED) {
527  /* If we canceled the read by sending a VK_RETURN event, restore the
528  screen state to undo the visual effect of the VK_RETURN */
529  if (read_console_success && InterlockedOr(&uv__restore_screen_state, 0)) {
530  HANDLE active_screen_buffer;
531  active_screen_buffer = CreateFileA("conout$",
532  GENERIC_READ | GENERIC_WRITE,
533  FILE_SHARE_READ | FILE_SHARE_WRITE,
534  NULL,
535  OPEN_EXISTING,
536  FILE_ATTRIBUTE_NORMAL,
537  NULL);
538  if (active_screen_buffer != INVALID_HANDLE_VALUE) {
539  pos = uv__saved_screen_state.dwCursorPosition;
540 
541  /* If the cursor was at the bottom line of the screen buffer, the
542  VK_RETURN would have caused the buffer contents to scroll up by one
543  line. The right position to reset the cursor to is therefore one line
544  higher */
545  if (pos.Y == uv__saved_screen_state.dwSize.Y - 1)
546  pos.Y--;
547 
548  SetConsoleCursorPosition(active_screen_buffer, pos);
549  CloseHandle(active_screen_buffer);
550  }
551  }
552  uv_sem_post(&uv_tty_output_lock);
553  }
554  POST_COMPLETION_FOR_REQ(loop, req);
555  return 0;
556 }

559 static void uv_tty_queue_read_line(uv_loop_t* loop, uv_tty_t* handle) {
560  uv_read_t* req;
561  BOOL r;
562 
563  assert(handle->flags & UV_HANDLE_READING);
564  assert(!(handle->flags & UV_HANDLE_READ_PENDING));
565  assert(handle->handle && handle->handle != INVALID_HANDLE_VALUE);
566 
567  req = &handle->read_req;
568  memset(&req->u.io.overlapped, 0, sizeof(req->u.io.overlapped));
569 
570  handle->tty.rd.read_line_buffer = uv_buf_init(NULL, 0);
571  handle->alloc_cb((uv_handle_t*) handle, 8192, &handle->tty.rd.read_line_buffer);
572  if (handle->tty.rd.read_line_buffer.base == NULL ||
573  handle->tty.rd.read_line_buffer.len == 0) {
574  handle->read_cb((uv_stream_t*) handle,
575  UV_ENOBUFS,
576  &handle->tty.rd.read_line_buffer);
577  return;
578  }
579  assert(handle->tty.rd.read_line_buffer.base != NULL);
580 
581  /* Reset flags No locking is required since there cannot be a line read
582  in progress. We are also relying on the memory barrier provided by
583  QueueUserWorkItem*/
584  uv__restore_screen_state = FALSE;
585  uv__read_console_status = NOT_STARTED;
586  r = QueueUserWorkItem(uv_tty_line_read_thread,
587  (void*) req,
588  WT_EXECUTELONGFUNCTION);
589  if (!r) {
590  SET_REQ_ERROR(req, GetLastError());
591  uv_insert_pending_req(loop, (uv_req_t*)req);
592  }
593 
594  handle->flags |= UV_HANDLE_READ_PENDING;
595  handle->reqs_pending++;
596 }

608 static const char* get_vt100_fn_key(DWORD code, char shift, char ctrl,
609  size_t* len) {
610 #define VK_CASE(vk, normal_str, shift_str, ctrl_str, shift_ctrl_str) \
611  case (vk): \
612  if (shift && ctrl) { \
613  *len = sizeof shift_ctrl_str; \
614  return "\033" shift_ctrl_str; \
615  } else if (shift) { \
616  *len = sizeof shift_str ; \
617  return "\033" shift_str; \
618  } else if (ctrl) { \
619  *len = sizeof ctrl_str; \
620  return "\033" ctrl_str; \
621  } else { \
622  *len = sizeof normal_str; \
623  return "\033" normal_str; \
624  }
625 
626  switch (code) {
627  /* These mappings are the same as Cygwin's. Unmodified and alt-modified
628  * keypad keys comply with linux console, modifiers comply with xterm
629  * modifier usage. F1. f12 and shift-f1. f10 comply with linux console, f6.
630  * f12 with and without modifiers comply with rxvt. */
631  VK_CASE(VK_INSERT, "[2~", "[2;2~", "[2;5~", "[2;6~")
632  VK_CASE(VK_END, "[4~", "[4;2~", "[4;5~", "[4;6~")
633  VK_CASE(VK_DOWN, "[B", "[1;2B", "[1;5B", "[1;6B")
634  VK_CASE(VK_NEXT, "[6~", "[6;2~", "[6;5~", "[6;6~")
635  VK_CASE(VK_LEFT, "[D", "[1;2D", "[1;5D", "[1;6D")

674 void uv_process_tty_read_raw_req(uv_loop_t* loop, uv_tty_t* handle,
675  uv_req_t* req) {
676  /* Shortcut for handle->tty.rd.last_input_record.Event.KeyEvent. */
677 #define KEV handle->tty.rd.last_input_record.Event.KeyEvent
678 
679  DWORD records_left, records_read;
680  uv_buf_t buf;
681  off_t buf_used;
682 
683  assert(handle->type == UV_TTY);
684  assert(handle->flags & UV_HANDLE_TTY_READABLE);
685  handle->flags &= ~UV_HANDLE_READ_PENDING;
686 
687  if (!(handle->flags & UV_HANDLE_READING) ||
688  !(handle->flags & UV_HANDLE_TTY_RAW)) {
689  goto out;
690  }
691 
692  if (!REQ_SUCCESS(req)) {
693  /* An error occurred while waiting for the event. */
694  if ((handle->flags & UV_HANDLE_READING)) {
695  handle->flags &= ~UV_HANDLE_READING;
696  handle->read_cb((uv_stream_t*)handle,
697  uv_translate_sys_error(GET_REQ_ERROR(req)),
698  &uv_null_buf_);
699  }
700  goto out;
701  }
702 
703  /* Fetch the number of events */
704  if (!GetNumberOfConsoleInputEvents(handle->handle, &records_left)) {
705  handle->flags &= ~UV_HANDLE_READING;
706  DECREASE_ACTIVE_COUNT(loop, handle);
707  handle->read_cb((uv_stream_t*)handle,
708  uv_translate_sys_error(GetLastError()),
709  &uv_null_buf_);
710  goto out;
711  }
712 
713  /* Windows sends a lot of events that we're not interested in, so buf will be
714  * allocated on demand, when there's actually something to emit. */
715  buf = uv_null_buf_;
716  buf_used = 0;
717 
718  while ((records_left > 0 || handle->tty.rd.last_key_len > 0) &&
719  (handle->flags & UV_HANDLE_READING)) {
720  if (handle->tty.rd.last_key_len == 0) {
721  /* Read the next input record */
722  if (!ReadConsoleInputW(handle->handle,
723  &handle->tty.rd.last_input_record,
724  1,
725  &records_read)) {
726  handle->flags &= ~UV_HANDLE_READING;
727  DECREASE_ACTIVE_COUNT(loop, handle);
728  handle->read_cb((uv_stream_t*) handle,
729  uv_translate_sys_error(GetLastError()),
730  &buf);
731  goto out;
732  }
733  records_left--;
734 
735  /* Ignore other events that are not key events. */
736  if (handle->tty.rd.last_input_record.EventType != KEY_EVENT) {
737  continue;
738  }
739 
740  /* Ignore keyup events, unless the left alt key was held and a valid
741  * unicode character was emitted. */
742  if (!KEV.bKeyDown &&
743  (KEV.wVirtualKeyCode != VK_MENU ||
744  KEV.uChar.UnicodeChar == 0)) {
745  continue;
746  }
747 
748  /* Ignore keypresses to numpad number keys if the left alt is held
749  * because the user is composing a character, or windows simulating this.
750  */
751  if ((KEV.dwControlKeyState & LEFT_ALT_PRESSED) &&
752  !(KEV.dwControlKeyState & ENHANCED_KEY) &&
753  (KEV.wVirtualKeyCode == VK_INSERT ||
754  KEV.wVirtualKeyCode == VK_END ||
755  KEV.wVirtualKeyCode == VK_DOWN ||
756  KEV.wVirtualKeyCode == VK_NEXT ||
757  KEV.wVirtualKeyCode == VK_LEFT ||
758  KEV.wVirtualKeyCode == VK_CLEAR ||
759  KEV.wVirtualKeyCode == VK_RIGHT ||
760  KEV.wVirtualKeyCode == VK_HOME ||
761  KEV.wVirtualKeyCode == VK_UP ||
762  KEV.wVirtualKeyCode == VK_PRIOR ||
763  KEV.wVirtualKeyCode == VK_NUMPAD0 ||
764  KEV.wVirtualKeyCode == VK_NUMPAD1 ||
765  KEV.wVirtualKeyCode == VK_NUMPAD2 ||
766  KEV.wVirtualKeyCode == VK_NUMPAD3 ||
767  KEV.wVirtualKeyCode == VK_NUMPAD4 ||
768  KEV.wVirtualKeyCode == VK_NUMPAD5 ||
769  KEV.wVirtualKeyCode == VK_NUMPAD6 ||
770  KEV.wVirtualKeyCode == VK_NUMPAD7 ||
771  KEV.wVirtualKeyCode == VK_NUMPAD8 ||
772  KEV.wVirtualKeyCode == VK_NUMPAD9)) {
773  continue;
774  }
775 
776  if (KEV.uChar.UnicodeChar != 0) {
777  int prefix_len, char_len;
778 
779  /* Character key pressed */
780  if (KEV.uChar.UnicodeChar >= 0xD800 &&
781  KEV.uChar.UnicodeChar < 0xDC00) {
782  /* UTF-16 high surrogate */
783  handle->tty.rd.last_utf16_high_surrogate = KEV.uChar.UnicodeChar;
784  continue;
785  }
786 
787  /* Prefix with \u033 if alt was held, but alt was not used as part a
788  * compose sequence. */
789  if ((KEV.dwControlKeyState & (LEFT_ALT_PRESSED | RIGHT_ALT_PRESSED))
790  && !(KEV.dwControlKeyState & (LEFT_CTRL_PRESSED |
791  RIGHT_CTRL_PRESSED)) && KEV.bKeyDown) {
792  handle->tty.rd.last_key[0] = '\033';
793  prefix_len = 1;
794  } else {
795  prefix_len = 0;
796  }
797 
798  if (KEV.uChar.UnicodeChar >= 0xDC00 &&
799  KEV.uChar.UnicodeChar < 0xE000) {
800  /* UTF-16 surrogate pair */
801  WCHAR utf16_buffer[2];
802  utf16_buffer[0] = handle->tty.rd.last_utf16_high_surrogate;
803  utf16_buffer[1] = KEV.uChar.UnicodeChar;
804  char_len = WideCharToMultiByte(CP_UTF8,
805  0,
806  utf16_buffer,
807  2,
808  &handle->tty.rd.last_key[prefix_len],
809  sizeof handle->tty.rd.last_key,
810  NULL,
811  NULL);
812  } else {
813  /* Single UTF-16 character */
814  char_len = WideCharToMultiByte(CP_UTF8,
815  0,
816  &KEV.uChar.UnicodeChar,
817  1,
818  &handle->tty.rd.last_key[prefix_len],
819  sizeof handle->tty.rd.last_key,
820  NULL,
821  NULL);
822  }
823 
824  /* Whatever happened, the last character wasn't a high surrogate. */
825  handle->tty.rd.last_utf16_high_surrogate = 0;
826 
827  /* If the utf16 character(s) couldn't be converted something must be
828  * wrong. */
829  if (!char_len) {
830  handle->flags &= ~UV_HANDLE_READING;
831  DECREASE_ACTIVE_COUNT(loop, handle);
832  handle->read_cb((uv_stream_t*) handle,
833  uv_translate_sys_error(GetLastError()),
834  &buf);
835  goto out;
836  }
837 
838  handle->tty.rd.last_key_len = (unsigned char) (prefix_len + char_len);
839  handle->tty.rd.last_key_offset = 0;
840  continue;
841 
842  } else {
843  /* Function key pressed */
844  const char* vt100;
845  size_t prefix_len, vt100_len;
846 
847  vt100 = get_vt100_fn_key(KEV.wVirtualKeyCode,
848  !!(KEV.dwControlKeyState & SHIFT_PRESSED),
849  !!(KEV.dwControlKeyState & (
850  LEFT_CTRL_PRESSED |
851  RIGHT_CTRL_PRESSED)),
852  &vt100_len);
853 
854  /* If we were unable to map to a vt100 sequence, just ignore. */
855  if (!vt100) {
856  continue;
857  }
858 
859  /* Prefix with \x033 when the alt key was held. */
860  if (KEV.dwControlKeyState & (LEFT_ALT_PRESSED | RIGHT_ALT_PRESSED)) {
861  handle->tty.rd.last_key[0] = '\033';
862  prefix_len = 1;
863  } else {
864  prefix_len = 0;
865  }
866 
867  /* Copy the vt100 sequence to the handle buffer. */
868  assert(prefix_len + vt100_len < sizeof handle->tty.rd.last_key);
869  memcpy(&handle->tty.rd.last_key[prefix_len], vt100, vt100_len);
870 
871  handle->tty.rd.last_key_len = (unsigned char) (prefix_len + vt100_len);
872  handle->tty.rd.last_key_offset = 0;
873  continue;
874  }
875  } else {
876  /* Copy any bytes left from the last keypress to the user buffer. */
877  if (handle->tty.rd.last_key_offset < handle->tty.rd.last_key_len) {
878  /* Allocate a buffer if needed */
879  if (buf_used == 0) {
880  buf = uv_buf_init(NULL, 0);
881  handle->alloc_cb((uv_handle_t*) handle, 1024, &buf);
882  if (buf.base == NULL || buf.len == 0) {
883  handle->read_cb((uv_stream_t*) handle, UV_ENOBUFS, &buf);
884  goto out;
885  }
886  assert(buf.base != NULL);
887  }
888 
889  buf.base[buf_used++] = handle->tty.rd.last_key[handle->tty.rd.last_key_offset++];
890 
891  /* If the buffer is full, emit it */
892  if ((size_t) buf_used == buf.len) {
893  handle->read_cb((uv_stream_t*) handle, buf_used, &buf);
894  buf = uv_null_buf_;
895  buf_used = 0;
896  }
897 
898  continue;
899  }
900 
901  /* Apply dwRepeat from the last input record. */
902  if (--KEV.wRepeatCount > 0) {
903  handle->tty.rd.last_key_offset = 0;
904  continue;
905  }
906 
907  handle->tty.rd.last_key_len = 0;
908  continue;
909  }
910  }
911 
912  /* Send the buffer back to the user */
913  if (buf_used > 0) {
914  handle->read_cb((uv_stream_t*) handle, buf_used, &buf);
915  }
916 
917  out:
918  /* Wait for more input events. */
919  if ((handle->flags & UV_HANDLE_READING) &&
920  !(handle->flags & UV_HANDLE_READ_PENDING)) {
921  uv_tty_queue_read(loop, handle);
922  }
923 
924  DECREASE_PENDING_REQ_COUNT(handle);
925 
926 #undef KEV
927 }

1055 static int uv__cancel_read_console(uv_tty_t* handle) {

1061 
1062  assert(!(handle->flags & UV_HANDLE_CANCELLATION_PENDING));
1063

1650  unsigned int utf8_codepoint = handle->tty.wr.utf8_codepoint;

1652  unsigned char ansi_parser_state = handle->tty.wr.ansi_parser_state;
1653

1656  *error = ERROR_SUCCESS;
1657

1822  if (utf8_codepoint >= '0' && utf8_codepoint <= '9') {
1823  /* Parsing a numerical argument */

1826  /* We were not currently parsing a number */
1827

1830  ansi_parser_state |= ANSI_IGNORE;

1835  handle->tty.wr.ansi_csi_argc++;
1836  handle->tty.wr.ansi_csi_argv[handle->tty.wr.ansi_csi_argc - 1] =
1837  (unsigned short) utf8_codepoint - '0';

1843

1863  * to 0. */
