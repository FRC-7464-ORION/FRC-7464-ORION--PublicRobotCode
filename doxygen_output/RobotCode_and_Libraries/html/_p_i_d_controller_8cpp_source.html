79 bool PIDController::AtSetpoint() const {
80  double positionError;
81  if (m_continuous) {
82  double errorBound = (m_maximumInput - m_minimumInput) / 2.0;
83  positionError =
84  frc::InputModulus(m_setpoint - m_measurement, -errorBound, errorBound);
85  } else {
86  positionError = m_setpoint - m_measurement;
87  }
88 
89  double velocityError = (positionError - m_prevError) / m_period.value();
90 
91  return std::abs(positionError) < m_positionTolerance &&
92  std::abs(velocityError) < m_velocityTolerance;
93 }

130 double PIDController::Calculate(double measurement) {
131  m_measurement = measurement;
132  m_prevError = m_positionError;
133 
134  if (m_continuous) {
135  double errorBound = (m_maximumInput - m_minimumInput) / 2.0;
136  m_positionError =
137  frc::InputModulus(m_setpoint - m_measurement, -errorBound, errorBound);
138  } else {
139  m_positionError = m_setpoint - measurement;
140  }
141 
142  m_velocityError = (m_positionError - m_prevError) / m_period.value();
143 
144  if (m_Ki != 0) {
145  m_totalError =
146  std::clamp(m_totalError + m_positionError * m_period.value(),
147  m_minimumIntegral / m_Ki, m_maximumIntegral / m_Ki);
148  }
149 
150  return m_Kp * m_positionError + m_Ki * m_totalError + m_Kd * m_velocityError;
151 }
