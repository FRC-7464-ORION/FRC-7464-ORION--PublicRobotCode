40 SparkMaxAlternateEncoder::SparkMaxAlternateEncoder(CANSparkMax& device,
41  Type type, int countsPerRev)
42  : RelativeEncoder(), m_device(&device), m_countsPerRev(countsPerRev) {
43  if (m_countsPerRev < 1) {
44  throw std::invalid_argument("countsPerRev must be a positive number");
45  }
46 
47  if (c_SparkMax_AttemptToSetDataPortConfig(
48  static_cast<c_SparkMax_handle>(m_device->m_sparkMaxHandle),
49  c_SparkMax_kDataPortConfigAltEncoder) ==
50  c_REVLibError_SparkMaxDataPortAlreadyConfiguredDifferently) {
51  throw std::runtime_error(fmt::format(
52  "An alternate encoder cannot be used on SPARK MAX #{}, because it "
53  "has a limit switch configured",
54  m_device->m_deviceID));
55  }
56 
57  c_SparkMax_SetAltEncoderCountsPerRevolution(
58  static_cast<c_SparkMax_handle>(m_device->m_sparkMaxHandle),
59  countsPerRev);
60 }
