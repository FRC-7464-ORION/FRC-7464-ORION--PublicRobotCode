184 // Called repeatedly when this Command is scheduled to run
185 void CmdAutoTurnAngle::Execute() {
186 
187  // Set the left and right motors' speeds
188  // NOTE: m_RotateToAngleSpeed is updated by PIDWrite()
189  // NOTE: For tank drive, to turn, left and right motors must
190  // go in opposite directions, hence the negative
191  // multiplier on the right motors' speed
192  m_left_motor_speed = m_RotateToAngleSpeed;
193  m_right_motor_speed = -1.0 * m_RotateToAngleSpeed;
194 
195  // In the AHRS class, which inherits from a PIDSource, the
196  // function PIDGet() returns the current yaw. This is where
197  // we get the current yaw into the PID controller. A
198  // class inheriting from a PIDSource is expected to have
199  // a PIDGet function.
200 
201  // Turn the robot
202  Robot::m_subSysDriveTrain->DriveTankStyle(m_left_motor_speed,
203  m_right_motor_speed);
204 
205  // Indicate our turn is completed when we are on target
206  m_turn_completed = m_turnController->OnTarget();
207 
208 } // end CmdAutoTurnAngle::Execute()
