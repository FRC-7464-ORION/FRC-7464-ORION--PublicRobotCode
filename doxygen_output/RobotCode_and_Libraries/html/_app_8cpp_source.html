66 static void FindDevices() {
67  WPI_EventHandle resolveEvent = multicastResolver->GetEventHandle();
68 
69  bool timedOut = 0;
70  if (wpi::WaitForObject(resolveEvent, 0, &timedOut)) {
71  auto allData = multicastResolver->GetData();
72 
73  for (auto&& data : allData) {
74  // search for MAC
75  auto macKey =
76  std::find_if(data.txt.begin(), data.txt.end(),
77  [](const auto& a) { return a.first == "MAC"; });
78  if (macKey != data.txt.end()) {
79  auto& mac = macKey->second;
80  foundDevices[mac] = std::make_pair(data.ipv4Address, data.hostName);
81  }
82  }
83  }
84 }
