19 static double Sinc(double x) {
20  if (std::abs(x) < 1e-9) {
21  return 1.0 - 1.0 / 6.0 * x * x;
22  } else {
23  return std::sin(x) / x;

44 ChassisSpeeds RamseteController::Calculate(
45  const Pose2d& currentPose, const Pose2d& poseRef,
46  units::meters_per_second_t linearVelocityRef,
47  units::radians_per_second_t angularVelocityRef) {
48  if (!m_enabled) {
49  return ChassisSpeeds{linearVelocityRef, 0_mps, angularVelocityRef};
50  }
51 
52  m_poseError = poseRef.RelativeTo(currentPose);
53 
54  // Aliases for equation readability
55  double eX = m_poseError.Translation().X().to<double>();
56  double eY = m_poseError.Translation().Y().to<double>();
57  double eTheta = m_poseError.Rotation().Radians().to<double>();
58  double vRef = linearVelocityRef.to<double>();
59  double omegaRef = angularVelocityRef.to<double>();
60 
61  double k =
62  2.0 * m_zeta * std::sqrt(std::pow(omegaRef, 2) + m_b * std::pow(vRef, 2));
63 
64  units::meters_per_second_t v{vRef * m_poseError.Rotation().Cos() + k * eX};
65  units::radians_per_second_t omega{omegaRef + k * eTheta +
66  m_b * vRef * Sinc(eTheta) * eY};
67  return ChassisSpeeds{v, 0_mps, omega};
68 }
