82 void SinkImpl::SetSource(std::shared_ptr<SourceImpl> source) {
83  {
84  std::scoped_lock lock(m_mutex);
85  if (m_source == source) {
86  return;
87  }
88  if (m_source) {
89  if (m_enabledCount > 0) {
90  m_source->DisableSink();
91  }
92  m_source->RemoveSink();
93  }
94  m_source = source;
95  if (m_source) {
96  m_source->AddSink();
97  if (m_enabledCount > 0) {
98  m_source->EnableSink();
99  }
100  }
101  }
102  SetSourceImpl(source);
103 }
