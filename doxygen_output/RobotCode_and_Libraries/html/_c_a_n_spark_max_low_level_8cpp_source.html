49 CANSparkMaxLowLevel::CANSparkMaxLowLevel(int deviceID, MotorType type)

51  if (c_SparkMax_RegisterId(deviceID) == c_REVLibError_DuplicateCANId) {

55  deviceID));
56  }
57  m_sparkMaxHandle = static_cast<void*>(
