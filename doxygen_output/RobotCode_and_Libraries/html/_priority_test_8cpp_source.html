13 TEST(PriorityTest, SetThreadPriority) {
14  bool exited = false;
15  wpi::condition_variable cond;
16  wpi::mutex mutex;
17  std::thread thr{[&] {
18  std::unique_lock lock{mutex};
19  cond.wait(lock, [&] { return exited; });
20  }};
21 
22  // Non-RT thread has priority of zero
23  bool isRealTime = false;
24  EXPECT_EQ(0, frc::GetThreadPriority(thr, &isRealTime));
25  EXPECT_FALSE(isRealTime);
26 
27  // Set thread priority to 15
28  EXPECT_TRUE(frc::SetThreadPriority(thr, true, 15));
29 
30  // Verify thread was given priority 15
31  EXPECT_EQ(15, frc::GetThreadPriority(thr, &isRealTime));
32  EXPECT_TRUE(isRealTime);
33 
34  // Set thread back to non-RT
35  EXPECT_TRUE(frc::SetThreadPriority(thr, false, 0));
36 
37  // Verify thread is now non-RT
38  EXPECT_EQ(0, frc::GetThreadPriority(thr, &isRealTime));
39  EXPECT_FALSE(isRealTime);
40 
41  {
42  std::scoped_lock lock{mutex};
43  exited = true;
44  }
45  cond.notify_one();
46  thr.join();
47 }
