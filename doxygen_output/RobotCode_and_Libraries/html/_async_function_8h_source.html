73  static std::shared_ptr<AsyncFunction> Create(
74  const std::shared_ptr<Loop>& loop,
75  std::function<void(promise<R>, T...)> func = nullptr) {
76  auto h =
77  std::make_shared<AsyncFunction>(loop, std::move(func), private_init{});
78  int err =
79  uv_async_init(loop->GetRaw(), h->GetRaw(), [](uv_async_t* handle) {
80  auto& h = *static_cast<AsyncFunction*>(handle->data);
81  std::unique_lock lock(h.m_mutex);
82 
83  if (!h.m_params.empty()) {
84  // for each set of parameters in the input queue, call the wakeup
85  // function and put the result in the output queue if the caller is
86  // waiting for it
87  for (auto&& v : h.m_params) {
88  auto p = h.m_promises.CreatePromise(v.first);
89  if (h.wakeup) {
90  std::apply(h.wakeup,
91  std::tuple_cat(std::make_tuple(std::move(p)),
92  std::move(v.second)));
93  }
94  }
95  h.m_params.clear();
96  // wake up any threads that might be waiting for the result
97  lock.unlock();
98  h.m_promises.Notify();
99  }
100  });
101  if (err < 0) {
102  loop->ReportError(err);
103  return nullptr;
104  }
105  h->Keep();
106  return h;
107  }

119  template <typename... U>
120  future<R> Call(U&&... u) {
121  // create the future
122  uint64_t req = m_promises.CreateRequest();
123 
124  auto loop = m_loop.lock();
125  if (loop && loop->GetThreadId() == std::this_thread::get_id()) {
126  // called from within the loop, just call the function directly
127  wakeup(m_promises.CreatePromise(req), std::forward<U>(u)...);
128  return m_promises.CreateFuture(req);
129  }
130 
131  // add the parameters to the input queue
132  {
133  std::scoped_lock lock(m_mutex);
134  m_params.emplace_back(std::piecewise_construct,
135  std::forward_as_tuple(req),
136  std::forward_as_tuple(std::forward<U>(u)...));
137  }
138 
139  // signal the loop
140  if (loop) {
141  this->Invoke(&uv_async_send, this->GetRaw());
142  }
143 
144  // return future
145  return m_promises.CreateFuture(req);
146  }
147 
148  template <typename... U>
149  future<R> operator()(U&&... u) {
150  return Call(std::forward<U>(u)...);
151  }
