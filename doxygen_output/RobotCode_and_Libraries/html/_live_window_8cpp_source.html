99 void LiveWindow::SetEnabled(bool enabled) {
100  std::scoped_lock lock(m_impl->mutex);
101  if (m_impl->liveWindowEnabled == enabled) return;
102  m_impl->startLiveWindow = enabled;
103  m_impl->liveWindowEnabled = enabled;
104  // Force table generation now to make sure everything is defined
105  UpdateValuesUnsafe();
106  if (enabled) {
107  if (this->enabled) this->enabled();
108  } else {
109  m_impl->registry.ForeachLiveWindow(m_impl->dataHandle, [&](auto& cbdata) {
110  cbdata.builder.StopLiveWindowMode();
111  });
112  if (this->disabled) this->disabled();
113  }
114  m_impl->enabledEntry.SetBoolean(enabled);
115 }

122 void LiveWindow::UpdateValuesUnsafe() {
123  // Only do this if either LiveWindow mode or telemetry is enabled.
124  if (!m_impl->liveWindowEnabled && !m_impl->telemetryEnabled) return;
125 
126  m_impl->registry.ForeachLiveWindow(m_impl->dataHandle, [&](auto& cbdata) {
127  if (!cbdata.sendable || cbdata.parent) return;
128 
129  if (!cbdata.data) cbdata.data = std::make_shared<Impl::Component>();
130 
131  auto& comp = *std::static_pointer_cast<Impl::Component>(cbdata.data);
132 
133  if (!m_impl->liveWindowEnabled && !comp.telemetryEnabled) return;
134 
135  if (comp.firstTime) {
136  // By holding off creating the NetworkTable entries, it allows the
137  // components to be redefined. This allows default sensor and actuator
138  // values to be created that are replaced with the custom names from
139  // users calling setName.
140  if (cbdata.name.empty()) return;
141  auto ssTable = m_impl->liveWindowTable->GetSubTable(cbdata.subsystem);
142  std::shared_ptr<NetworkTable> table;
143  // Treat name==subsystem as top level of subsystem
144  if (cbdata.name == cbdata.subsystem)
145  table = ssTable;
146  else
147  table = ssTable->GetSubTable(cbdata.name);
148  table->GetEntry(".name").SetString(cbdata.name);
149  cbdata.builder.SetTable(table);
150  cbdata.sendable->InitSendable(cbdata.builder);
151  ssTable->GetEntry(".type").SetString("LW Subsystem");
152 
153  comp.firstTime = false;
154  }
155 
156  if (m_impl->startLiveWindow) cbdata.builder.StartLiveWindowMode();
157  cbdata.builder.UpdateTable();
158  });
159 
160  m_impl->startLiveWindow = false;
161 }
