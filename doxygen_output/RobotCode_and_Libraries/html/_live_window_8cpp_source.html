112 void LiveWindow::SetEnabled(bool enabled) {
113  auto& inst = ::GetInstance();
114  std::scoped_lock lock(inst.mutex);
115  if (inst.liveWindowEnabled == enabled) {
116  return;
117  }
118  inst.startLiveWindow = enabled;
119  inst.liveWindowEnabled = enabled;
120  // Force table generation now to make sure everything is defined
121  UpdateValuesUnsafe();
122  if (enabled) {
123  if (inst.enabled) {
124  inst.enabled();
125  }
126  } else {
127  wpi::SendableRegistry::ForeachLiveWindow(
128  inst.dataHandle, [&](auto& cbdata) {
129  static_cast<SendableBuilderImpl&>(cbdata.builder)
130  .StopLiveWindowMode();
131  });
132  if (inst.disabled) {
133  inst.disabled();
134  }
135  }
136  inst.enabledEntry.SetBoolean(enabled);
137 }

145 void LiveWindow::UpdateValuesUnsafe() {
146  auto& inst = ::GetInstance();
147  // Only do this if either LiveWindow mode or telemetry is enabled.
148  if (!inst.liveWindowEnabled && !inst.telemetryEnabled) {
149  return;
150  }
151 
152  wpi::SendableRegistry::ForeachLiveWindow(inst.dataHandle, [&](auto& cbdata) {
153  if (!cbdata.sendable || cbdata.parent) {
154  return;
155  }
156 
157  if (!cbdata.data) {
158  cbdata.data = std::make_shared<Component>();
159  }
160 
161  auto& comp = *std::static_pointer_cast<Component>(cbdata.data);
162 
163  if (!inst.liveWindowEnabled && !comp.telemetryEnabled) {
164  return;
165  }
166 
167  if (comp.firstTime) {
168  // By holding off creating the NetworkTable entries, it allows the
169  // components to be redefined. This allows default sensor and actuator
170  // values to be created that are replaced with the custom names from
171  // users calling setName.
172  if (cbdata.name.empty()) {
173  return;
174  }
175  auto ssTable = inst.liveWindowTable->GetSubTable(cbdata.subsystem);
176  std::shared_ptr<nt::NetworkTable> table;
177  // Treat name==subsystem as top level of subsystem
178  if (cbdata.name == cbdata.subsystem) {
179  table = ssTable;
180  } else {
181  table = ssTable->GetSubTable(cbdata.name);
182  }
183  table->GetEntry(".name").SetString(cbdata.name);
184  static_cast<SendableBuilderImpl&>(cbdata.builder).SetTable(table);
185  cbdata.sendable->InitSendable(cbdata.builder);
186  ssTable->GetEntry(".type").SetString("LW Subsystem");
187 
188  comp.firstTime = false;
189  }
190 
191  if (inst.startLiveWindow) {
192  static_cast<SendableBuilderImpl&>(cbdata.builder).StartLiveWindowMode();
193  }
194  cbdata.builder.Update();
195  });
196 
197  inst.startLiveWindow = false;
198 }
