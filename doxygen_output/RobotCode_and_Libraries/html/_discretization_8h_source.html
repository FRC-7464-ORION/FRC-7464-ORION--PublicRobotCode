39 template <int States, int Inputs>
40 void DiscretizeAB(const Eigen::Matrix<double, States, States>& contA,
41  const Eigen::Matrix<double, States, Inputs>& contB,
42  units::second_t dt,
43  Eigen::Matrix<double, States, States>* discA,
44  Eigen::Matrix<double, States, Inputs>* discB) {
45  // Matrices are blocked here to minimize matrix exponentiation calculations
46  Eigen::Matrix<double, States + Inputs, States + Inputs> Mcont;
47  Mcont.setZero();
48  Mcont.template block<States, States>(0, 0) = contA * dt.value();
49  Mcont.template block<States, Inputs>(0, States) = contB * dt.value();
50 
51  // Discretize A and B with the given timestep
52  Eigen::Matrix<double, States + Inputs, States + Inputs> Mdisc = Mcont.exp();
53  *discA = Mdisc.template block<States, States>(0, 0);
54  *discB = Mdisc.template block<States, Inputs>(0, States);
55 }

67 template <int States>
68 void DiscretizeAQ(const Eigen::Matrix<double, States, States>& contA,
69  const Eigen::Matrix<double, States, States>& contQ,
70  units::second_t dt,
71  Eigen::Matrix<double, States, States>* discA,
72  Eigen::Matrix<double, States, States>* discQ) {
73  // Make continuous Q symmetric if it isn't already
74  Eigen::Matrix<double, States, States> Q = (contQ + contQ.transpose()) / 2.0;
75 
76  // Set up the matrix M = [[-A, Q], [0, A.T]]
77  Eigen::Matrix<double, 2 * States, 2 * States> M;
78  M.template block<States, States>(0, 0) = -contA;
79  M.template block<States, States>(0, States) = Q;
80  M.template block<States, States>(States, 0).setZero();
81  M.template block<States, States>(States, States) = contA.transpose();
82 
83  Eigen::Matrix<double, 2 * States, 2 * States> phi = (M * dt.value()).exp();
84 
85  // Phi12 = phi[0:States, States:2*States]
86  // Phi22 = phi[States:2*States, States:2*States]
87  Eigen::Matrix<double, States, States> phi12 =
88  phi.block(0, States, States, States);
89  Eigen::Matrix<double, States, States> phi22 =
90  phi.block(States, States, States, States);
91 
92  *discA = phi22.transpose();
93 
94  Q = *discA * phi12;
95 
96  // Make discrete Q symmetric if it isn't already
97  *discQ = (Q + Q.transpose()) / 2.0;
98 }

119 template <int States>
120 void DiscretizeAQTaylor(const Eigen::Matrix<double, States, States>& contA,
121  const Eigen::Matrix<double, States, States>& contQ,
122  units::second_t dt,
123  Eigen::Matrix<double, States, States>* discA,
124  Eigen::Matrix<double, States, States>* discQ) {
125  // Make continuous Q symmetric if it isn't already
126  Eigen::Matrix<double, States, States> Q = (contQ + contQ.transpose()) / 2.0;
127 
128  Eigen::Matrix<double, States, States> lastTerm = Q;
129  double lastCoeff = dt.value();
130 
131  // Aᵀⁿ
132  Eigen::Matrix<double, States, States> Atn = contA.transpose();
133 
134  Eigen::Matrix<double, States, States> phi12 = lastTerm * lastCoeff;
135 
136  // i = 6 i.e. 5th order should be enough precision
137  for (int i = 2; i < 6; ++i) {
138  lastTerm = -contA * lastTerm + Q * Atn;
139  lastCoeff *= dt.value() / static_cast<double>(i);
140 
141  phi12 += lastTerm * lastCoeff;
142 
143  Atn *= contA.transpose();
144  }
145 
146  DiscretizeA<States>(contA, dt, discA);
147  Q = *discA * phi12;
148 
149  // Make discrete Q symmetric if it isn't already
150  *discQ = (Q + Q.transpose()) / 2.0;
151 }
