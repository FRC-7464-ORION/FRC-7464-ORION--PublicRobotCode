255  double Calculate(Distance_t measurement) {
256  if (m_controller.IsContinuousInputEnabled()) {
257  // Get error which is smallest distance between goal and measurement
258  auto errorBound = (m_maximumInput - m_minimumInput) / 2.0;
259  auto goalMinDistance = frc::InputModulus<Distance_t>(
260  m_goal.position - measurement, -errorBound, errorBound);
261  auto setpointMinDistance = frc::InputModulus<Distance_t>(
262  m_setpoint.position - measurement, -errorBound, errorBound);
263 
264  // Recompute the profile goal with the smallest error, thus giving the
265  // shortest path. The goal may be outside the input range after this
266  // operation, but that's OK because the controller will still go there and
267  // report an error of zero. In other words, the setpoint only needs to be
268  // offset from the measurement by the input range modulus; they don't need
269  // to be equal.
270  m_goal.position = goalMinDistance + measurement;
271  m_setpoint.position = setpointMinDistance + measurement;
272  }
273 
274  frc::TrapezoidProfile<Distance> profile{m_constraints, m_goal, m_setpoint};
275  m_setpoint = profile.Calculate(GetPeriod());
276  return m_controller.Calculate(measurement.value(),
277  m_setpoint.position.value());
278  }
