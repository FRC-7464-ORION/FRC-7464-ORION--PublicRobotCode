252  /*
253  @brief get next character from the input
254 
255  This function provides the interface to the used input adapter. It does
256  not throw in case the input reached EOF, but returns a
257  `std::char_traits<char>::eof()` in that case. Stores the scanned characters
258  for use in error messages.
259 
260  @return character read from the input
261  */
262  std::char_traits<char>::int_type get()
263  {
264  ++chars_read;
265  if (JSON_UNLIKELY(!unget_chars.empty()))
266  {
267  current = unget_chars.back();
268  unget_chars.pop_back();
269  token_string.push_back(current);
270  return current;
271  }
272  char c;
273  is.read(c);
274  if (JSON_UNLIKELY(is.has_error()))
275  {
276  current = std::char_traits<char>::eof();
277  }
278  else
279  {
280  current = std::char_traits<char>::to_int_type(c);
281  token_string.push_back(c);
282  }
283  return current;
284  }
285 
287  void unget()
288  {
289  --chars_read;
290  if (JSON_LIKELY(current != std::char_traits<char>::eof()))
291  {
292  unget_chars.emplace_back(current);
293  assert(token_string.size() != 0);
294  token_string.pop_back();
295  if (!token_string.empty())
296  {
297  current = token_string.back();
298  }
299  }
300  }
301 
303  void putback(std::char_traits<char>::int_type c)
304  {
305  --chars_read;
306  unget_chars.emplace_back(c);
307  }

310  void add(int c)
311  {
312  token_buffer.push_back(std::char_traits<char>::to_char_type(c));

524 const char* json::lexer::token_type_name(const token_type t) noexcept
525 {
526  switch (t)
527  {
528  case token_type::uninitialized:
529  return "<uninitialized>";
530  case token_type::literal_true:
531  return "true literal";
532  case token_type::literal_false:
533  return "false literal";
534  case token_type::literal_null:
535  return "null literal";
536  case token_type::value_string:
537  return "string literal";
538  case lexer::token_type::value_unsigned:
539  case lexer::token_type::value_integer:
540  case lexer::token_type::value_float:
541  return "number literal";
542  case token_type::begin_array:
543  return "'['";
544  case token_type::begin_object:
545  return "'{'";
546  case token_type::end_array:
547  return "']'";
548  case token_type::end_object:
549  return "'}'";
550  case token_type::name_separator:
551  return "':'";
552  case token_type::value_separator:
553  return "','";
554  case token_type::parse_error:
555  return "<parse error>";
556  case token_type::end_of_input:
557  return "end of input";
558  case token_type::literal_or_value:
559  return "'[', '{', or a literal";
560  default: // catch non-enum values
561  return "unknown token"; // LCOV_EXCL_LINE
562  }

565 json::lexer::lexer(raw_istream& s)
566  : is(s), decimal_point_char(get_decimal_point())
567 {
568  // skip byte order mark
569  std::char_traits<char>::int_type c;
570  if ((c = get()) == 0xEF)
571  {
572  if ((c = get()) == 0xBB)
573  {
574  if ((c = get()) == 0xBF)
575  {
576  chars_read = 0;
577  return; // Ignore BOM
578  }
579  else if (c != std::char_traits<char>::eof())
580  {
581  unget();
582  }
583  putback('\xBB');
584  }
585  else if (c != std::char_traits<char>::eof())
586  {
587  unget();
588  }
589  putback('\xEF');
590  }
591  unget(); // no byte order mark; process as usual
592 }

594 int json::lexer::get_codepoint()
595 {
596  // this function only makes sense after reading `\u`
597  assert(current == 'u');
598  int codepoint = 0;
599 
600  const auto factors = { 12, 8, 4, 0 };
601  for (const auto factor : factors)
602  {
603  get();
604 
605  if (current >= '0' and current <= '9')
606  {
607  codepoint += ((current - 0x30) << factor);
608  }
609  else if (current >= 'A' and current <= 'F')
610  {
611  codepoint += ((current - 0x37) << factor);
612  }
613  else if (current >= 'a' and current <= 'f')
614  {
615  codepoint += ((current - 0x57) << factor);
616  }
617  else
618  {
619  return -1;
620  }
621  }
622 
623  assert(0x0000 <= codepoint and codepoint <= 0xFFFF);
624  return codepoint;
625 }

627 json::lexer::token_type json::lexer::scan_string()
628 {
629  // reset token_buffer (ignore opening quote)
630  reset();
631 
632  // we entered the function by reading an open quote
633  assert(current == '\"');
634 
635  while (true)
636  {
637  // get next character
638  switch (get())
639  {
640  // end of file while parsing string
641  case std::char_traits<char>::eof():
642  {
643  error_message = "invalid string: missing closing quote";
644  return token_type::parse_error;
645  }
646 
647  // closing quote
648  case '\"':
649  {
650  return token_type::value_string;
651  }
652 
653  // escapes
654  case '\\':
655  {
656  switch (get())
657  {
658  // quotation mark
659  case '\"':
660  add('\"');
661  break;
662  // reverse solidus
663  case '\\':
664  add('\\');
665  break;
666  // solidus
667  case '/':
668  add('/');
669  break;
670  // backspace
671  case 'b':
672  add('\b');
673  break;
674  // form feed
675  case 'f':
676  add('\f');
677  break;
678  // line feed
679  case 'n':
680  add('\n');
681  break;
682  // carriage return
683  case 'r':
684  add('\r');
685  break;
686  // tab
687  case 't':
688  add('\t');
689  break;
690 
691  // unicode escapes
692  case 'u':
693  {
694  const int codepoint1 = get_codepoint();
695  int codepoint = codepoint1; // start with codepoint1
696 
697  if (JSON_UNLIKELY(codepoint1 == -1))
698  {
699  error_message = "invalid string: '\\u' must be followed by 4 hex digits";
700  return token_type::parse_error;
701  }
702 
703  // check if code point is a high surrogate
704  if (0xD800 <= codepoint1 and codepoint1 <= 0xDBFF)
705  {
706  // expect next \uxxxx entry
707  if (JSON_LIKELY(get() == '\\' and get() == 'u'))
708  {
709  const int codepoint2 = get_codepoint();
710 
711  if (JSON_UNLIKELY(codepoint2 == -1))
712  {
713  error_message = "invalid string: '\\u' must be followed by 4 hex digits";
714  return token_type::parse_error;
715  }
716 
717  // check if codepoint2 is a low surrogate
718  if (JSON_LIKELY(0xDC00 <= codepoint2 and codepoint2 <= 0xDFFF))
719  {
720  // overwrite codepoint
721  codepoint =
722  // high surrogate occupies the most significant 22 bits
723  (codepoint1 << 10)
724  // low surrogate occupies the least significant 15 bits
725  + codepoint2
726  // there is still the 0xD800, 0xDC00 and 0x10000 noise
727  // in the result so we have to subtract with:
728  // (0xD800 << 10) + DC00 - 0x10000 = 0x35FDC00
729  - 0x35FDC00;
730  }
731  else
732  {
733  error_message = "invalid string: surrogate U+DC00..U+DFFF must be followed by U+DC00..U+DFFF";
734  return token_type::parse_error;
735  }
736  }
737  else
738  {
739  error_message = "invalid string: surrogate U+DC00..U+DFFF must be followed by U+DC00..U+DFFF";
740  return token_type::parse_error;
741  }
742  }
743  else
744  {
745  if (JSON_UNLIKELY(0xDC00 <= codepoint1 and codepoint1 <= 0xDFFF))
746  {
747  error_message = "invalid string: surrogate U+DC00..U+DFFF must follow U+D800..U+DBFF";
748  return token_type::parse_error;
749  }
750  }
751 
752  // result of the above calculation yields a proper codepoint
753  assert(0x00 <= codepoint and codepoint <= 0x10FFFF);
754 
755  // translate codepoint into bytes
756  if (codepoint < 0x80)
757  {
758  // 1-byte characters: 0xxxxxxx (ASCII)
759  add(codepoint);
760  }
761  else if (codepoint <= 0x7FF)
762  {
763  // 2-byte characters: 110xxxxx 10xxxxxx
764  add(0xC0 | (codepoint >> 6));
765  add(0x80 | (codepoint & 0x3F));
766  }
767  else if (codepoint <= 0xFFFF)
768  {
769  // 3-byte characters: 1110xxxx 10xxxxxx 10xxxxxx
770  add(0xE0 | (codepoint >> 12));
771  add(0x80 | ((codepoint >> 6) & 0x3F));
772  add(0x80 | (codepoint & 0x3F));
773  }
774  else
775  {
776  // 4-byte characters: 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
777  add(0xF0 | (codepoint >> 18));
778  add(0x80 | ((codepoint >> 12) & 0x3F));
779  add(0x80 | ((codepoint >> 6) & 0x3F));
780  add(0x80 | (codepoint & 0x3F));
781  }
782 
783  break;
784  }
785 
786  // other characters after escape
787  default:
788  error_message = "invalid string: forbidden character after backslash";
789  return token_type::parse_error;
790  }
791 
792  break;
793  }

1534  {
1535  if (callback)
1536  {

1539

1555  {
1556  result.m_value.destroy(result.m_type);
1557  result.m_type = value_t::discarded;
1558  }
1559  break;
1560  }

1748  }
1749
