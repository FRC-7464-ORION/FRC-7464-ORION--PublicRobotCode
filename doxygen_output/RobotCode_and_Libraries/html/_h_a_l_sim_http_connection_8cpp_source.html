34 void HALSimHttpConnection::ProcessWsUpgrade() {
35  m_websocket->open.connect_extended([this](auto conn, auto) {
36  conn.disconnect(); // one-shot
37 
38  if (!m_server->RegisterWebsocket(shared_from_this())) {
39  Log(409);
40  m_websocket->Fail(409, "Only a single simulation websocket is allowed");
41  return;
42  }
43 
44  Log(200);
45  m_isWsConnected = true;
46  std::fputs("HALWebSim: websocket connected\n", stderr);
47  });
48 
49  // parse incoming JSON, dispatch to parent
50  m_websocket->text.connect([this](auto msg, bool) {
51  if (!m_isWsConnected) {
52  return;
53  }
54 
55  wpi::json j;
56  try {
57  j = wpi::json::parse(msg);
58  } catch (const wpi::json::parse_error& e) {
59  std::string err("JSON parse failed: ");
60  err += e.what();
61  m_websocket->Fail(400, err);
62  return;
63  }
64  m_server->OnNetValueChanged(j);
65  });
66 
67  m_websocket->closed.connect([this](uint16_t, auto) {
68  // unset the global, allow another websocket to connect
69  if (m_isWsConnected) {
70  std::fputs("HALWebSim: websocket disconnected\n", stderr);
71  m_isWsConnected = false;
72 
73  m_server->CloseWebsocket(shared_from_this());
74  }
75  });
76 }

104 void HALSimHttpConnection::SendFileResponse(int code, std::string_view codeText,
105  std::string_view contentType,
106  std::string_view filename,
107  std::string_view extraHeader) {
108  std::error_code ec;
109 
110  // get file size
111  auto size = fs::file_size(filename, ec);
112  if (ec) {
113  MySendError(404, "error getting file size");
114  return;
115  }
116 
117  // open file
118  wpi::raw_fd_istream is{filename, ec, true};
119  if (ec) {
120  MySendError(404, "error opening file");
121  return;
122  }
123 
124  wpi::SmallVector<uv::Buffer, 4> toSend;
125  wpi::raw_uv_ostream os{toSend, 4096};
126  BuildHeader(os, code, codeText, contentType, size, extraHeader);
127  SendData(os.bufs(), false);
128 
129  Log(code);
130 
131  // Read the file byte by byte
132  wpi::SmallVector<uv::Buffer, 4> bodyData;
133  wpi::raw_uv_ostream bodyOs{bodyData, 4096};
134 
135  std::string fileBuf;
136  size_t oldSize = 0;
137 
138  while (fileBuf.size() < size) {
139  oldSize = fileBuf.size();
140  fileBuf.resize(oldSize + 1);
141  is.read(&(*fileBuf.begin()) + oldSize, 1);
142  }
143 
144  bodyOs << fileBuf;
145 
146  SendData(bodyOs.bufs(), false);
147  if (!m_keepAlive) {
148  m_stream.Close();
149  }
150 }

152 void HALSimHttpConnection::ProcessRequest() {
153  wpi::UrlParser url{m_request.GetUrl(),
154  m_request.GetMethod() == wpi::HTTP_CONNECT};
155  if (!url.IsValid()) {
156  // failed to parse URL
157  MySendError(400, "Invalid URL");
158  return;
159  }
160 
161  std::string_view path;
162  if (url.HasPath()) {
163  path = url.GetPath();
164  }
165 
166  if (m_request.GetMethod() == wpi::HTTP_GET && wpi::starts_with(path, '/') &&
167  !wpi::contains(path, "..") && !wpi::contains(path, "//")) {
168  // convert to fs native representation
169  fs::path nativePath;
170  if (wpi::starts_with(path, "/user/")) {
171  nativePath =
172  fs::path{m_server->GetWebrootSys()} /
173  fs::path{wpi::drop_front(path, 6), fs::path::format::generic_format};
174  } else {
175  nativePath =
176  fs::path{m_server->GetWebrootSys()} /
177  fs::path{wpi::drop_front(path, 1), fs::path::format::generic_format};
178  }
179 
180  if (fs::is_directory(nativePath)) {
181  nativePath.append("index.html");
182  }
183 
184  if (!fs::exists(nativePath) || fs::is_directory(nativePath)) {
185  MySendError(404, fmt::format("Resource '{}' not found", path));
186  } else {
187  auto contentType = wpi::MimeTypeFromPath(nativePath.string());
188  SendFileResponse(200, "OK", contentType, nativePath.string());
189  }
190  } else {
191  MySendError(404, "Resource not found");
192  }
193 }
