24 void Subsystem::SetDefaultCommand(Command* command) {
25  if (command == nullptr) {
26  m_defaultCommand = nullptr;
27  } else {
28  const auto& reqs = command->GetRequirements();
29  if (std::find(reqs.begin(), reqs.end(), this) == reqs.end()) {
30  wpi_setWPIErrorWithContext(
31  CommandIllegalUse, "A default command must require the subsystem");
32  return;
33  }
34 
35  m_defaultCommand = command;
36  }
37 }
