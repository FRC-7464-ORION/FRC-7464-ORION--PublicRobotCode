28 void CvSourceImpl::PutFrame(cv::Mat& image) {
29  // We only support 8-bit images; convert if necessary.
30  cv::Mat finalImage;
31  if (image.depth() == CV_8U) {
32  finalImage = image;
33  } else {
34  image.convertTo(finalImage, CV_8U);
35  }
36 
37  std::unique_ptr<Image> dest;
38  switch (image.channels()) {
39  case 1:
40  dest =
41  AllocImage(VideoMode::kGray, image.cols, image.rows, image.total());
42  finalImage.copyTo(dest->AsMat());
43  break;
44  case 3:
45  dest = AllocImage(VideoMode::kBGR, image.cols, image.rows,
46  image.total() * 3);
47  finalImage.copyTo(dest->AsMat());
48  break;
49  case 4:
50  dest = AllocImage(VideoMode::kBGR, image.cols, image.rows,
51  image.total() * 3);
52  cv::cvtColor(finalImage, dest->AsMat(), cv::COLOR_BGRA2BGR);
53  break;
54  default:
55  SERROR("PutFrame: {}-channel images not supported", image.channels());
56  return;
57  }
58  SourceImpl::PutFrame(std::move(dest), wpi::Now());
59 }

141 void SetSourceEnumPropertyChoices(CS_Source source, CS_Property property,
142  wpi::span<const std::string> choices,
143  CS_Status* status) {
144  auto data = Instance::GetInstance().GetSource(source);
145  if (!data || (data->kind & SourceMask) == 0) {
146  *status = CS_INVALID_HANDLE;
147  return;
148  }
149 
150  // Get property index; also validate the source owns this property
151  Handle handle{property};
152  int i = handle.GetParentIndex();
153  if (i < 0) {
154  *status = CS_INVALID_HANDLE;
155  return;
156  }
157  auto data2 = Instance::GetInstance().GetSource(Handle{i, Handle::kSource});
158  if (!data2 || data->source.get() != data2->source.get()) {
159  *status = CS_INVALID_HANDLE;
160  return;
161  }
162  int propertyIndex = handle.GetIndex();
163  static_cast<CvSourceImpl&>(*data->source)
164  .SetEnumPropertyChoices(propertyIndex, choices, status);
165 }
