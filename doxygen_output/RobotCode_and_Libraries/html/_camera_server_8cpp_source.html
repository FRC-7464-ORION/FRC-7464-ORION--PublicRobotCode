61 static std::string_view MakeSourceValue(CS_Source source,
62  wpi::SmallVectorImpl<char>& buf) {
63  CS_Status status = 0;
64  buf.clear();
65  switch (cs::GetSourceKind(source, &status)) {
66  case CS_SOURCE_USB: {
67  std::string_view prefix{"usb:"};
68  buf.append(prefix.begin(), prefix.end());
69  auto path = cs::GetUsbCameraPath(source, &status);
70  buf.append(path.begin(), path.end());
71  break;
72  }
73  case CS_SOURCE_HTTP: {
74  std::string_view prefix{"ip:"};
75  buf.append(prefix.begin(), prefix.end());
76  auto urls = cs::GetHttpCameraUrls(source, &status);
77  if (!urls.empty()) {
78  buf.append(urls[0].begin(), urls[0].end());
79  }
80  break;
81  }
82  case CS_SOURCE_CV:
83  return "cv:";
84  default:
85  return "unknown:";
86  }
87 
88  return {buf.begin(), buf.size()};
89 }

100 std::vector<std::string> Instance::GetSinkStreamValues(CS_Sink sink) {
101  CS_Status status = 0;
102 
103  // Ignore all but MjpegServer
104  if (cs::GetSinkKind(sink, &status) != CS_SINK_MJPEG) {
105  return {};
106  }
107 
108  // Get port
109  int port = cs::GetMjpegServerPort(sink, &status);
110 
111  // Generate values
112  std::vector<std::string> values;
113  auto listenAddress = cs::GetMjpegServerListenAddress(sink, &status);
114  if (!listenAddress.empty()) {
115  // If a listen address is specified, only use that
116  values.emplace_back(MakeStreamValue(listenAddress, port));
117  } else {
118  // Otherwise generate for hostname and all interface addresses
119  values.emplace_back(MakeStreamValue(cs::GetHostname() + ".local", port));
120 
121  for (const auto& addr : m_addresses) {
122  if (addr == "127.0.0.1") {
123  continue; // ignore localhost
124  }
125  values.emplace_back(MakeStreamValue(addr, port));
126  }
127  }
128 
129  return values;
130 }

132 std::vector<std::string> Instance::GetSourceStreamValues(CS_Source source) {
133  CS_Status status = 0;
134 
135  // Ignore all but HttpCamera
136  if (cs::GetSourceKind(source, &status) != CS_SOURCE_HTTP) {
137  return {};
138  }
139 
140  // Generate values
141  auto values = cs::GetHttpCameraUrls(source, &status);
142  for (auto& value : values) {
143  value = "mjpg:" + value;
144  }
145 
146 #ifdef __FRC_ROBORIO__
147  // Look to see if we have a passthrough server for this source
148  // Only do this on the roboRIO
149  for (const auto& i : m_sinks) {
150  CS_Sink sink = i.second.GetHandle();
151  CS_Source sinkSource = cs::GetSinkSource(sink, &status);
152  if (source == sinkSource &&
153  cs::GetSinkKind(sink, &status) == CS_SINK_MJPEG) {
154  // Add USB-only passthrough
155  int port = cs::GetMjpegServerPort(sink, &status);
156  values.emplace_back(MakeStreamValue("172.22.11.2", port));
157  break;
158  }
159  }
160 #endif
161 
162  // Set table value
163  return values;
164 }

166 void Instance::UpdateStreamValues() {
167  std::scoped_lock lock(m_mutex);
168  // Over all the sinks...
169  for (const auto& i : m_sinks) {
170  CS_Status status = 0;
171  CS_Sink sink = i.second.GetHandle();
172 
173  // Get the source's subtable (if none exists, we're done)
174  CS_Source source = m_fixedSources.lookup(sink);
175  if (source == 0) {
176  source = cs::GetSinkSource(sink, &status);
177  }
178  if (source == 0) {
179  continue;
180  }
181  auto table = m_tables.lookup(source);
182  if (table) {
183  // Don't set stream values if this is a HttpCamera passthrough
184  if (cs::GetSourceKind(source, &status) == CS_SOURCE_HTTP) {
185  continue;
186  }
187 
188  // Set table value
189  auto values = GetSinkStreamValues(sink);
190  if (!values.empty()) {
191  table->GetEntry("streams").SetStringArray(values);
192  }
193  }
194  }
195 
196  // Over all the sources...
197  for (const auto& i : m_sources) {
198  CS_Source source = i.second.GetHandle();
199 
200  // Get the source's subtable (if none exists, we're done)
201  auto table = m_tables.lookup(source);
202  if (table) {
203  // Set table value
204  auto values = GetSourceStreamValues(source);
205  if (!values.empty()) {
206  table->GetEntry("streams").SetStringArray(values);
207  }
208  }
209  }
210 }

243 static void PutSourcePropertyValue(nt::NetworkTable* table,
244  const cs::VideoEvent& event, bool isNew) {
245  std::string_view namePrefix;
246  std::string_view infoPrefix;
247  if (wpi::starts_with(event.name, "raw_")) {
248  namePrefix = "RawProperty";
249  infoPrefix = "RawPropertyInfo";
250  } else {
251  namePrefix = "Property";
252  infoPrefix = "PropertyInfo";
253  }
254 
255  wpi::SmallString<64> buf;
256  CS_Status status = 0;
257  nt::NetworkTableEntry entry =
258  table->GetEntry(fmt::format("{}/{}", namePrefix, event.name));
259  switch (event.propertyKind) {
260  case CS_PROP_BOOLEAN:
261  if (isNew) {
262  entry.SetDefaultBoolean(event.value != 0);
263  } else {
264  entry.SetBoolean(event.value != 0);
265  }
266  break;
267  case CS_PROP_INTEGER:
268  case CS_PROP_ENUM:
269  if (isNew) {
270  entry.SetDefaultDouble(event.value);
271  table->GetEntry(fmt::format("{}/{}/min", infoPrefix, event.name))
272  .SetDouble(cs::GetPropertyMin(event.propertyHandle, &status));
273  table->GetEntry(fmt::format("{}/{}/max", infoPrefix, event.name))
274  .SetDouble(cs::GetPropertyMax(event.propertyHandle, &status));
275  table->GetEntry(fmt::format("{}/{}/step", infoPrefix, event.name))
276  .SetDouble(cs::GetPropertyStep(event.propertyHandle, &status));
277  table->GetEntry(fmt::format("{}/{}/default", infoPrefix, event.name))
278  .SetDouble(cs::GetPropertyDefault(event.propertyHandle, &status));
279  } else {
280  entry.SetDouble(event.value);
281  }
282  break;
283  case CS_PROP_STRING:
284  if (isNew) {
285  entry.SetDefaultString(event.valueStr);
286  } else {
287  entry.SetString(event.valueStr);
288  }
289  break;
290  default:
291  break;
292  }

295 Instance::Instance() {
296  // We publish sources to NetworkTables using the following structure:
297  // "/CameraPublisher/{Source.Name}/" - root
298  // - "source" (string): Descriptive, prefixed with type (e.g. "usb:0")
299  // - "streams" (string array): URLs that can be used to stream data
300  // - "description" (string): Description of the source
301  // - "connected" (boolean): Whether source is connected
302  // - "mode" (string): Current video mode
303  // - "modes" (string array): Available video modes
304  // - "Property/{Property}" - Property values
305  // - "PropertyInfo/{Property}" - Property supporting information
306 
307  // Listener for video events
308  m_videoListener = cs::VideoListener{
309  [=](const cs::VideoEvent& event) {
310  CS_Status status = 0;
311  switch (event.kind) {
312  case cs::VideoEvent::kSourceCreated: {
313  // Create subtable for the camera
314  auto table = m_publishTable->GetSubTable(event.name);
315  {
316  std::scoped_lock lock(m_mutex);
317  m_tables.insert(std::make_pair(event.sourceHandle, table));
318  }
319  wpi::SmallString<64> buf;
320  table->GetEntry("source").SetString(
321  MakeSourceValue(event.sourceHandle, buf));
322  wpi::SmallString<64> descBuf;
323  table->GetEntry("description")
324  .SetString(cs::GetSourceDescription(event.sourceHandle, descBuf,
325  &status));
326  table->GetEntry("connected")
327  .SetBoolean(cs::IsSourceConnected(event.sourceHandle, &status));
328  table->GetEntry("streams").SetStringArray(
329  GetSourceStreamValues(event.sourceHandle));
330  auto mode = cs::GetSourceVideoMode(event.sourceHandle, &status);
331  table->GetEntry("mode").SetDefaultString(VideoModeToString(mode));
332  table->GetEntry("modes").SetStringArray(
333  GetSourceModeValues(event.sourceHandle));
334  break;
335  }
336  case cs::VideoEvent::kSourceDestroyed: {
337  auto table = GetSourceTable(event.sourceHandle);
338  if (table) {
339  table->GetEntry("source").SetString("");
340  table->GetEntry("streams").SetStringArray(
341  std::vector<std::string>{});
342  table->GetEntry("modes").SetStringArray(
343  std::vector<std::string>{});
344  }
345  break;
346  }
347  case cs::VideoEvent::kSourceConnected: {
348  auto table = GetSourceTable(event.sourceHandle);
349  if (table) {
350  // update the description too (as it may have changed)
351  wpi::SmallString<64> descBuf;
352  table->GetEntry("description")
353  .SetString(cs::GetSourceDescription(event.sourceHandle,
354  descBuf, &status));
355  table->GetEntry("connected").SetBoolean(true);
356  }
357  break;
358  }
359  case cs::VideoEvent::kSourceDisconnected: {
360  auto table = GetSourceTable(event.sourceHandle);
361  if (table) {
362  table->GetEntry("connected").SetBoolean(false);
363  }
364  break;
365  }
366  case cs::VideoEvent::kSourceVideoModesUpdated: {
367  auto table = GetSourceTable(event.sourceHandle);
368  if (table) {
369  table->GetEntry("modes").SetStringArray(
370  GetSourceModeValues(event.sourceHandle));
371  }
372  break;
373  }
374  case cs::VideoEvent::kSourceVideoModeChanged: {
375  auto table = GetSourceTable(event.sourceHandle);
376  if (table) {
377  table->GetEntry("mode").SetString(VideoModeToString(event.mode));
378  }
379  break;
380  }
381  case cs::VideoEvent::kSourcePropertyCreated: {
382  auto table = GetSourceTable(event.sourceHandle);
383  if (table) {
384  PutSourcePropertyValue(table.get(), event, true);
385  }
386  break;
387  }
388  case cs::VideoEvent::kSourcePropertyValueUpdated: {
389  auto table = GetSourceTable(event.sourceHandle);
390  if (table) {
391  PutSourcePropertyValue(table.get(), event, false);
392  }
393  break;
394  }
395  case cs::VideoEvent::kSourcePropertyChoicesUpdated: {
396  auto table = GetSourceTable(event.sourceHandle);
397  if (table) {
398  auto choices =
399  cs::GetEnumPropertyChoices(event.propertyHandle, &status);
400  table
401  ->GetEntry(fmt::format("PropertyInfo/{}/choices", event.name))
402  .SetStringArray(choices);
403  }
404  break;
405  }
406  case cs::VideoEvent::kSinkSourceChanged:
407  case cs::VideoEvent::kSinkCreated:
408  case cs::VideoEvent::kSinkDestroyed:
409  case cs::VideoEvent::kNetworkInterfacesChanged: {
410  m_addresses = cs::GetNetworkInterfaces();
411  UpdateStreamValues();
412  break;
413  }
414  default:
415  break;
416  }
417  },
418  0x4fff, true};
419 
420  // Listener for NetworkTable events
421  // We don't currently support changing settings via NT due to
422  // synchronization issues, so just update to current setting if someone
423  // else tries to change it.
424  wpi::SmallString<64> buf;
425  m_tableListener = nt::NetworkTableInstance::GetDefault().AddEntryListener(
426  fmt::format("{}/", kPublishName),
427  [=](const nt::EntryNotification& event) {
428  auto relativeKey = wpi::drop_front(
429  event.name, std::string_view{kPublishName}.size() + 1);
430 
431  // get source (sourceName/...)
432  auto subKeyIndex = relativeKey.find('/');
433  if (subKeyIndex == std::string_view::npos) {
434  return;
435  }
436  auto sourceName = wpi::slice(relativeKey, 0, subKeyIndex);
437  auto sourceIt = m_sources.find(sourceName);
438  if (sourceIt == m_sources.end()) {
439  return;
440  }
441 
442  // get subkey
443  relativeKey.remove_prefix(subKeyIndex + 1);
444 
445  // handle standard names
446  std::string_view propName;
447  nt::NetworkTableEntry entry{event.entry};
448  if (relativeKey == "mode") {
449  // reset to current mode
450  entry.SetString(VideoModeToString(sourceIt->second.GetVideoMode()));
451  return;
452  } else if (wpi::starts_with(relativeKey, "Property/")) {
453  propName = wpi::substr(relativeKey, 9);
454  } else if (wpi::starts_with(relativeKey, "RawProperty/")) {
455  propName = wpi::substr(relativeKey, 12);
456  } else {
457  return; // ignore
458  }
459 
460  // everything else is a property
461  auto property = sourceIt->second.GetProperty(propName);
462  switch (property.GetKind()) {
463  case cs::VideoProperty::kNone:
464  return;
465  case cs::VideoProperty::kBoolean:
466  entry.SetBoolean(property.Get() != 0);
467  return;
468  case cs::VideoProperty::kInteger:
469  case cs::VideoProperty::kEnum:
470  entry.SetDouble(property.Get());
471  return;
472  case cs::VideoProperty::kString:
473  entry.SetString(property.GetString());
474  return;
475  default:
476  return;
477  }
478  },
479  NT_NOTIFY_IMMEDIATE | NT_NOTIFY_UPDATE);
480 }

585 cs::CvSink CameraServer::GetVideo() {
586  auto& inst = ::GetInstance();
587  cs::VideoSource source;
588  {
589  auto csShared = GetCameraServerShared();
590  std::scoped_lock lock(inst.m_mutex);
591  if (inst.m_primarySourceName.empty()) {
592  csShared->SetCameraServerError("no camera available");
593  return cs::CvSink{};
594  }
595  auto it = inst.m_sources.find(inst.m_primarySourceName);
596  if (it == inst.m_sources.end()) {
597  csShared->SetCameraServerError("no camera available");
598  return cs::CvSink{};
599  }
600  source = it->second;
601  }
602  return GetVideo(std::move(source));
603 }
