227  void Predict(const Eigen::Vector<double, Inputs>& u, units::second_t dt) {
228  m_dt = dt;
229 
230  // Discretize Q before projecting mean and covariance forward
231  Eigen::Matrix<double, States, States> contA =
232  NumericalJacobianX<States, States, Inputs>(m_f, m_xHat, u);
233  Eigen::Matrix<double, States, States> discA;
234  Eigen::Matrix<double, States, States> discQ;
235  DiscretizeAQTaylor<States>(contA, m_contQ, dt, &discA, &discQ);
236 
237  Eigen::Matrix<double, States, 2 * States + 1> sigmas =
238  m_pts.SigmaPoints(m_xHat, m_P);
239 
240  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
241  Eigen::Vector<double, States> x = sigmas.template block<States, 1>(0, i);
242  m_sigmasF.template block<States, 1>(0, i) = RK4(m_f, x, u, dt);
243  }
244 
245  auto ret = UnscentedTransform<States, States>(
246  m_sigmasF, m_pts.Wm(), m_pts.Wc(), m_meanFuncX, m_residualFuncX);
247  m_xHat = std::get<0>(ret);
248  m_P = std::get<1>(ret);
249 
250  m_P += discQ;
251  }

322  template <int Rows>
323  void Correct(const Eigen::Vector<double, Inputs>& u,
324  const Eigen::Vector<double, Rows>& y,
325  std::function<Eigen::Vector<double, Rows>(
326  const Eigen::Vector<double, States>&,
327  const Eigen::Vector<double, Inputs>&)>
328  h,
329  const Eigen::Matrix<double, Rows, Rows>& R,
330  std::function<Eigen::Vector<double, Rows>(
331  const Eigen::Matrix<double, Rows, 2 * States + 1>&,
332  const Eigen::Vector<double, 2 * States + 1>&)>
333  meanFuncY,
334  std::function<Eigen::Vector<double, Rows>(
335  const Eigen::Vector<double, Rows>&,
336  const Eigen::Vector<double, Rows>&)>
337  residualFuncY,
338  std::function<Eigen::Vector<double, States>(
339  const Eigen::Vector<double, States>&,
340  const Eigen::Vector<double, States>&)>
341  residualFuncX,
342  std::function<Eigen::Vector<double, States>(
343  const Eigen::Vector<double, States>&,
344  const Eigen::Vector<double, States>)>
345  addFuncX) {
346  const Eigen::Matrix<double, Rows, Rows> discR = DiscretizeR<Rows>(R, m_dt);
347 
348  // Transform sigma points into measurement space
349  Eigen::Matrix<double, Rows, 2 * States + 1> sigmasH;
350  Eigen::Matrix<double, States, 2 * States + 1> sigmas =
351  m_pts.SigmaPoints(m_xHat, m_P);
352  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
353  sigmasH.template block<Rows, 1>(0, i) =
354  h(sigmas.template block<States, 1>(0, i), u);
355  }
356 
357  // Mean and covariance of prediction passed through UT
358  auto [yHat, Py] = UnscentedTransform<Rows, States>(
359  sigmasH, m_pts.Wm(), m_pts.Wc(), meanFuncY, residualFuncY);
360  Py += discR;
361 
362  // Compute cross covariance of the state and the measurements
363  Eigen::Matrix<double, States, Rows> Pxy;
364  Pxy.setZero();
365  for (int i = 0; i < m_pts.NumSigmas(); ++i) {
366  // Pxy += (sigmas_f[:, i] - x̂)(sigmas_h[:, i] - ŷ)ᵀ W_c[i]
367  Pxy +=
368  m_pts.Wc(i) *
369  (residualFuncX(m_sigmasF.template block<States, 1>(0, i), m_xHat)) *
370  (residualFuncY(sigmasH.template block<Rows, 1>(0, i), yHat))
371  .transpose();
372  }
373 
374  // K = P_{xy} P_y⁻¹
375  // Kᵀ = P_yᵀ⁻¹ P_{xy}ᵀ
376  // P_yᵀKᵀ = P_{xy}ᵀ
377  // Kᵀ = P_yᵀ.solve(P_{xy}ᵀ)
378  // K = (P_yᵀ.solve(P_{xy}ᵀ)ᵀ
379  Eigen::Matrix<double, States, Rows> K =
380  Py.transpose().ldlt().solve(Pxy.transpose()).transpose();
381 
382  // x̂ₖ₊₁⁺ = x̂ₖ₊₁⁻ + K(y − ŷ)
383  m_xHat = addFuncX(m_xHat, K * residualFuncY(y, yHat));
384 
385  // Pₖ₊₁⁺ = Pₖ₊₁⁻ − KP_yKᵀ
386  m_P -= K * Py * K.transpose();
387  }
