100 void SendableBuilderImpl::AddBooleanProperty(std::string_view key,
101  std::function<bool()> getter,
102  std::function<void(bool)> setter) {
103  m_properties.emplace_back(*m_table, key);
104  if (getter) {
105  m_properties.back().update = [=](nt::NetworkTableEntry entry,
106  uint64_t time) {
107  entry.SetValue(nt::Value::MakeBoolean(getter(), time));
108  };
109  }
110  if (setter) {
111  m_properties.back().createListener =
112  [=](nt::NetworkTableEntry entry) -> NT_EntryListener {
113  return entry.AddListener(
114  [=](const nt::EntryNotification& event) {
115  if (!event.value->IsBoolean()) {
116  return;
117  }
118  SmartDashboard::PostListenerTask(
119  [=] { setter(event.value->GetBoolean()); });
120  },
121  NT_NOTIFY_IMMEDIATE | NT_NOTIFY_NEW | NT_NOTIFY_UPDATE);
122  };
123  }
124 }
