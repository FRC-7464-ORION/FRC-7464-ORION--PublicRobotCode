65 units::ampere_t ElevatorSim::GetCurrentDraw() const {
66  // I = V / R - omega / (Kv * R)
67  // Reductions are greater than 1, so a reduction of 10:1 would mean the motor
68  // is spinning 10x faster than the output.
69 
70  // v = r w, so w = v / r
71  units::meters_per_second_t velocity{m_x(1)};
72  units::radians_per_second_t motorVelocity =
73  velocity / m_drumRadius * m_gearing * 1_rad;
74 
75  // Perform calculation and return.
76  return m_gearbox.Current(motorVelocity, units::volt_t{m_u(0)}) *
77  wpi::sgn(m_u(0));
78 }

84 Eigen::Vector<double, 2> ElevatorSim::UpdateX(
85  const Eigen::Vector<double, 2>& currentXhat,
86  const Eigen::Vector<double, 1>& u, units::second_t dt) {
87  auto updatedXhat = RKDP(
88  [&](const Eigen::Vector<double, 2>& x,
89  const Eigen::Vector<double, 1>& u_) -> Eigen::Vector<double, 2> {
90  return m_plant.A() * x + m_plant.B() * u_ +
91  Eigen::Vector<double, 2>{0.0, -9.8};
92  },
93  currentXhat, u, dt);
94  // Check for collision after updating x-hat.
95  if (WouldHitLowerLimit(units::meter_t(updatedXhat(0)))) {
96  return Eigen::Vector<double, 2>{m_minHeight.value(), 0.0};
97  }
98  if (WouldHitUpperLimit(units::meter_t(updatedXhat(0)))) {
99  return Eigen::Vector<double, 2>{m_maxHeight.value(), 0.0};
100  }
101  return updatedXhat;
102 }
