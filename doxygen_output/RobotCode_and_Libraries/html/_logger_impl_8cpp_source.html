12 static void DefaultLogger(unsigned int level, const char* file,
13  unsigned int line, const char* msg) {
14  if (level == 20) {
15  fmt::print(stderr, "NT: {}\n", msg);
16  return;
17  }
18 
19  std::string_view levelmsg;
20  if (level >= 50) {
21  levelmsg = "CRITICAL";
22  } else if (level >= 40) {
23  levelmsg = "ERROR";
24  } else if (level >= 30) {
25  levelmsg = "WARNING";
26  } else {
27  return;
28  }
29  fmt::print(stderr, "NT: {}: {} ({}:{})\n", levelmsg, msg, file, line);
30 }

50 unsigned int LoggerImpl::GetMinLevel() {
51  auto thr = GetThread();
52  if (!thr) {
53  return NT_LOG_INFO;
54  }
55  unsigned int level = NT_LOG_INFO;
56  for (size_t i = 0; i < thr->m_listeners.size(); ++i) {
57  const auto& listener = thr->m_listeners[i];
58  if (listener && listener.min_level < level) {
59  level = listener.min_level;
60  }
61  }
62  return level;
63 }

65 void LoggerImpl::Log(unsigned int level, const char* file, unsigned int line,
66  const char* msg) {
67  auto filename = fs::path{file}.filename();
68  {
69  auto thr = GetThread();
70  if (!thr || thr->m_listeners.empty()) {
71  DefaultLogger(level, filename.string().c_str(), line, msg);
72  }
73  }
74  Send(UINT_MAX, 0, level, filename.string(), line, msg);
75 }
