17 SerialPort::SerialPort(int baudRate, Port port, int dataBits,
18  SerialPort::Parity parity,
19  SerialPort::StopBits stopBits) {
20  int32_t status = 0;
21 
22  m_portHandle =
23  HAL_InitializeSerialPort(static_cast<HAL_SerialPort>(port), &status);
24  wpi_setHALError(status);
25  // Don't continue if initialization failed
26  if (status < 0) return;
27  HAL_SetSerialBaudRate(m_portHandle, baudRate, &status);
28  wpi_setHALError(status);
29  HAL_SetSerialDataBits(m_portHandle, dataBits, &status);
30  wpi_setHALError(status);
31  HAL_SetSerialParity(m_portHandle, parity, &status);
32  wpi_setHALError(status);
33  HAL_SetSerialStopBits(m_portHandle, stopBits, &status);
34  wpi_setHALError(status);
35 
36  // Set the default timeout to 5 seconds.
37  SetTimeout(5.0);
38 
39  // Don't wait until the buffer is full to transmit.
40  SetWriteBufferMode(kFlushOnAccess);
41 
42  DisableTermination();
43 
44  HAL_Report(HALUsageReporting::kResourceType_SerialPort,
45  static_cast<uint8_t>(port) + 1);
46 }
