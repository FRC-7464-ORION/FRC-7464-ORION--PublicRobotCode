38 bool HALSimWeb::Initialize() {
39  if (!m_server || !m_exec) {
40  return false;
41  }
42 
43  // determine where to get static content from
44  fs::path path;
45  const char* webroot_sys = std::getenv("HALSIMWS_SYSROOT");
46  if (webroot_sys != nullptr) {
47  path = webroot_sys;
48  } else {
49  path = fs::current_path() / "sim";
50  }
51  m_webroot_sys = fs::absolute(path).string();
52 
53  const char* webroot_user = std::getenv("HALSIMWS_USERROOT");
54  if (webroot_user != nullptr) {
55  path = webroot_sys;
56  } else {
57  path = fs::current_path() / "sim" / "user";
58  }
59  m_webroot_user = fs::absolute(path).string();
60 
61  const char* uri = std::getenv("HALSIMWS_URI");
62  if (uri != nullptr) {
63  m_uri = uri;
64  } else {
65  m_uri = "/wpilibws";
66  }
67 
68  const char* port = std::getenv("HALSIMWS_PORT");
69  if (port != nullptr) {
70  try {
71  m_port = std::stoi(port);
72  } catch (const std::invalid_argument& err) {
73  fmt::print(stderr, "Error decoding HALSIMWS_PORT ({})\n", err.what());
74  return false;
75  }
76  } else {
77  m_port = 3300;
78  }
79 
80  return true;

83 void HALSimWeb::Start() {
84  m_server->Bind("", m_port);
85 
86  // when we get a connection, accept it and start reading
87  m_server->connection.connect([this, srv = m_server.get()] {
88  auto tcp = srv->Accept();
89  if (!tcp) {
90  return;
91  }
92 
93  tcp->SetNoDelay(true);
94 
95  auto conn = std::make_shared<HALSimHttpConnection>(shared_from_this(), tcp);
96  tcp->SetData(conn);
97  });
98 
99  // start listening for incoming connections
100  m_server->Listen();
101  fmt::print("Listening at http://localhost:{}\n", m_port);
102  fmt::print("WebSocket URI: {}\n", m_uri);
103 }
