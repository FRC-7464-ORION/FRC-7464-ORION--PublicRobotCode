413 #define CRC7_POLY 0x91

415  static void buildCRCLookupTable( uint8_t* table, size_t length )
416  {
417  size_t crc;
418  size_t i, j;
419  if ( length == 256 ) {
420  for ( i = 0; i < length; i++ ) {
421  crc = (uint8_t)i;
422  for (j = 0; j < 8; j++) {
423  if (crc & 1) {
424  crc ^= CRC7_POLY;
425  }
426  crc >>= 1;
427  }
428  table[i] = crc;
429  }
430  }
431  }

433  static inline uint8_t getCRCWithTable( uint8_t* table, uint8_t message[], uint8_t length )
434  {
435  uint8_t i, crc = 0;
436 
437  for (i = 0; i < length; i++)
438  {
439  crc ^= message[i];
