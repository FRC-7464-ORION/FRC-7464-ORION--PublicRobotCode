219  @SuppressWarnings({"MethodName", "PMD.CyclomaticComplexity", "PMD.NPathComplexity"})
220  void _execute() {
221  Entry entry = null;
222  Command cmd = null;
223  boolean firstRun = false;
224  if (m_currentCommandIndex == -1) {
225  firstRun = true;
226  m_currentCommandIndex = 0;
227  }
228 
229  while (m_currentCommandIndex < m_commands.size()) {
230  if (cmd != null) {
231  if (entry.isTimedOut()) {
232  cmd._cancel();
233  }
234  if (cmd.run()) {
235  break;
236  } else {
237  cmd.removed();
238  m_currentCommandIndex++;
239  firstRun = true;
240  cmd = null;
241  continue;
242  }
243  }
244 
245  entry = m_commands.elementAt(m_currentCommandIndex);
246  cmd = null;
247 
248  switch (entry.m_state) {
249  case Entry.IN_SEQUENCE:
250  cmd = entry.m_command;
251  if (firstRun) {
252  cmd.startRunning();
253  cancelConflicts(cmd);
254  }
255  firstRun = false;
256  break;
257  case Entry.BRANCH_PEER:
258  m_currentCommandIndex++;
259  entry.m_command.start();
260  break;
261  case Entry.BRANCH_CHILD:
262  m_currentCommandIndex++;
263  cancelConflicts(entry.m_command);
264  entry.m_command.startRunning();
265  m_children.addElement(entry);
266  break;
267  default:
268  break;
269  }
270  }
271 
272  // Run Children
273  for (int i = 0; i < m_children.size(); i++) {
274  entry = m_children.elementAt(i);
275  Command child = entry.m_command;
276  if (entry.isTimedOut()) {
277  child._cancel();
278  }
279  if (!child.run()) {
280  child.removed();
281  m_children.removeElementAt(i--);
282  }
283  }
284  }

375  private void cancelConflicts(Command command) {
376  for (int i = 0; i < m_children.size(); i++) {
377  Command child = m_children.elementAt(i).m_command;
378 
379  Enumeration requirements = command.getRequirements();
380 
381  while (requirements.hasMoreElements()) {
382  Object requirement = requirements.nextElement();
383  if (child.doesRequire((Subsystem) requirement)) {
384  child._cancel();
385  child.removed();
386  m_children.removeElementAt(i--);
387  break;
388  }
389  }
