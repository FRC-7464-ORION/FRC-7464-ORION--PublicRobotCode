152  @SuppressWarnings({"MethodName", "PMD.CyclomaticComplexity"})
153  private void _add(Command command) {
154  if (command == null) {
155  return;
156  }
157 
158  // Check to make sure no adding during adding
159  if (m_adding) {
160  System.err.println("WARNING: Can not start command from cancel method. Ignoring:" + command);
161  return;
162  }
163 
164  // Only add if not already in
165  if (!m_commandTable.containsKey(command)) {
166  // Check that the requirements can be had
167  Enumeration requirements = command.getRequirements();
168  while (requirements.hasMoreElements()) {
169  Subsystem lock = (Subsystem) requirements.nextElement();
170  if (lock.getCurrentCommand() != null && !lock.getCurrentCommand().isInterruptible()) {
171  return;
172  }
173  }
174 
175  // Give it the requirements
176  m_adding = true;
177  requirements = command.getRequirements();
178  while (requirements.hasMoreElements()) {
179  Subsystem lock = (Subsystem) requirements.nextElement();
180  if (lock.getCurrentCommand() != null) {
181  lock.getCurrentCommand().cancel();
182  remove(lock.getCurrentCommand());
183  }
184  lock.setCurrentCommand(command);
185  }
186  m_adding = false;
187 
188  // Add it to the list
189  LinkedListElement element = new LinkedListElement();
190  element.setData(command);
191  if (m_firstCommand == null) {
192  m_firstCommand = m_lastCommand = element;
193  } else {
194  m_lastCommand.add(element);
195  m_lastCommand = element;
196  }
197  m_commandTable.put(command, element);
198 
199  m_runningCommandsChanged = true;
200 
201  command.startRunning();
202  }
203  }

212  @SuppressWarnings({"PMD.CyclomaticComplexity", "PMD.NPathComplexity"})
213  public void run() {
214  m_runningCommandsChanged = false;
215 
216  if (m_disabled) {
217  return;
218  } // Don't run when m_disabled
219 
220  // Get button input (going backwards preserves button priority)
221  if (m_buttons != null) {
222  for (int i = m_buttons.size() - 1; i >= 0; i--) {
223  m_buttons.elementAt(i).execute();
224  }
225  }
226 
227  // Call every subsystem's periodic method
228  Enumeration subsystems = m_subsystems.getElements();
229  while (subsystems.hasMoreElements()) {
230  ((Subsystem) subsystems.nextElement()).periodic();
231  }
232 
233  // Loop through the commands
234  LinkedListElement element = m_firstCommand;
235  while (element != null) {
236  Command command = element.getData();
237  element = element.getNext();
238  if (!command.run()) {
239  remove(command);
240  m_runningCommandsChanged = true;
241  }
242  }
243 
244  // Add the new things
245  for (int i = 0; i < m_additions.size(); i++) {
246  _add(m_additions.elementAt(i));
247  }
248  m_additions.removeAllElements();
249 
250  // Add in the defaults
251  Enumeration locks = m_subsystems.getElements();
252  while (locks.hasMoreElements()) {
253  Subsystem lock = (Subsystem) locks.nextElement();
254  if (lock.getCurrentCommand() == null) {
255  _add(lock.getDefaultCommand());
256  }
257  lock.confirmCommand();
258  }
259  }

279  void remove(Command command) {
280  if (command == null || !m_commandTable.containsKey(command)) {
281  return;
282  }
283  LinkedListElement element = m_commandTable.get(command);
284  m_commandTable.remove(command);
285 
286  if (element.equals(m_lastCommand)) {
287  m_lastCommand = element.getPrevious();
288  }
289  if (element.equals(m_firstCommand)) {
290  m_firstCommand = element.getNext();
291  }
292  element.remove();
293 
294  Enumeration requirements = command.getRequirements();
295  while (requirements.hasMoreElements()) {
296  ((Subsystem) requirements.nextElement()).setCurrentCommand(null);
297  }
298 
299  command.removed();
300  }

336  if (toCancel.length > 0) {
337  for (LinkedListElement e = m_firstCommand; e != null; e = e.getNext()) {
338  for (double d : toCancel) {
339  if (e.getData().hashCode() == d) {
340  e.getData().cancel();
341  }
342  }
343  }
344  cancelEntry.setDoubleArray(new double[0]);
345  }
346 
347  if (m_runningCommandsChanged) {
348  // Set the the running commands
349  int number = 0;
350  for (LinkedListElement e = m_firstCommand; e != null; e = e.getNext()) {
351  number++;
352  }
353  String[] commands = new String[number];
354  double[] ids = new double[number];
355  number = 0;
356  for (LinkedListElement e = m_firstCommand; e != null; e = e.getNext()) {
357  commands[number] = e.getData().getName();
358  ids[number] = e.getData().hashCode();
359  number++;
360  }
361  namesEntry.setStringArray(commands);
362  idsEntry.setDoubleArray(ids);
363  }
364  }
