33 void Poll::Reuse(int fd, std::function<void()> callback) {
34  if (IsClosing()) {
35  return;
36  }
37  if (!m_reuseData) {
38  m_reuseData = std::make_unique<ReuseData>();
39  }
40  m_reuseData->callback = std::move(callback);
41  m_reuseData->isSocket = false;
42  m_reuseData->fd = fd;
43  uv_close(GetRawHandle(), [](uv_handle_t* handle) {
44  Poll& h = *static_cast<Poll*>(handle->data);
45  if (!h.m_reuseData || h.m_reuseData->isSocket) {
46  return; // just in case
47  }
48  auto data = std::move(h.m_reuseData);
49  int err = uv_poll_init(h.GetLoopRef().GetRaw(), h.GetRaw(), data->fd);
50  if (err < 0) {
51  h.ReportError(err);
52  return;
53  }
