69 bool ReadCameraConfig(const wpi::json& config) {
70  CameraConfig c;
71 
72  // name
73  try {
74  c.name = config.at("name").get<std::string>();
75  } catch (const wpi::json::exception& e) {
76  ParseError() << "could not read camera name: " << e.what() << '\n';
77  return false;
78  }
79 
80  // path
81  try {
82  c.path = config.at("path").get<std::string>();
83  } catch (const wpi::json::exception& e) {
84  ParseError() << "camera '" << c.name
85  << "': could not read path: " << e.what() << '\n';
86  return false;
87  }

89  c.config = config;
90 
91  cameras.emplace_back(std::move(c));
92  return true;

95 bool ReadConfig() {
96  // open config file
97  std::error_code ec;
98  wpi::raw_fd_istream is(configFile, ec);
99  if (ec) {
100  wpi::errs() << "could not open '" << configFile << "': " << ec.message()
101  << '\n';
102  return false;
103  }
104 
105  // parse file
106  wpi::json j;
107  try {
108  j = wpi::json::parse(is);
109  } catch (const wpi::json::parse_error& e) {
110  fmt::print(ParseError(), "byte {}: {}\n", e.byte, e.what());
111  return false;
112  }
113 
114  // top level must be an object
115  if (!j.is_object()) {
116  ParseError() << "must be JSON object\n";
117  return false;
118  }
119 
120  // team number
121  try {
122  team = j.at("team").get<unsigned int>();
123  } catch (const wpi::json::exception& e) {
124  ParseError() << "could not read team number: " << e.what() << '\n';
125  return false;
126  }
127 
128  // ntmode (optional)
129  if (j.count("ntmode") != 0) {
130  try {
131  auto str = j.at("ntmode").get<std::string>();
132  if (wpi::equals_lower(str, "client")) {
133  server = false;
134  } else if (wpi::equals_lower(str, "server")) {
135  server = true;
136  } else {
137  ParseError() << "could not understand ntmode value '" << str << "'\n";
138  }
139  } catch (const wpi::json::exception& e) {
140  ParseError() << "could not read ntmode: " << e.what() << '\n';
141  }
142  }
143 
144  // cameras
145  try {
146  for (auto&& camera : j.at("cameras")) {
147  if (!ReadCameraConfig(camera)) {
148  return false;
149  }
150  }
151  } catch (const wpi::json::exception& e) {
152  ParseError() << "could not read cameras: " << e.what() << '\n';
153  return false;
154  }
155 
156  return true;
157 }
