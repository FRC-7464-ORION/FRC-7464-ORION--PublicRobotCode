74 units::second_t Timer::Get() const {
75  units::second_t result;
76  units::second_t currentTime = GetFPGATimestamp();
77 
78  std::scoped_lock lock(m_mutex);
79  if (m_running) {
80  // If the current time is before the start time, then the FPGA clock rolled
81  // over. Compensate by adding the ~71 minutes that it takes to roll over to
82  // the current time.
83  if (currentTime < m_startTime) {
84  currentTime += kRolloverTime;
85  }
86 
87  result = (currentTime - m_startTime) + m_accumulatedTime;
88  } else {
89  result = m_accumulatedTime;
90  }
91 
92  return result;
93 }
