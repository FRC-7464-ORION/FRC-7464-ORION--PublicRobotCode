41 int32_t SpiReadAutoReceiveBufferCallbackStore::performCallback(
42  const char* name, uint32_t* buffer, int32_t numToRead) {
43  JNIEnv* env;
44  JavaVM* vm = sim::GetJVM();
45  bool didAttachThread = false;
46  int tryGetEnv = vm->GetEnv(reinterpret_cast<void**>(&env), JNI_VERSION_1_6);
47  if (tryGetEnv == JNI_EDETACHED) {
48  // Thread not attached
49  didAttachThread = true;
50  if (vm->AttachCurrentThread(reinterpret_cast<void**>(&env), nullptr) != 0) {
51  // Failed to attach, log and return
52  std::puts("Failed to attach");
53  std::fflush(stdout);
54  return -1;
55  }
56  } else if (tryGetEnv == JNI_EVERSION) {
57  std::puts("Invalid JVM Version requested");
58  std::fflush(stdout);
59  }
60 
61  auto toCallbackArr = MakeJIntArray(
62  env, wpi::span<const uint32_t>{buffer, static_cast<size_t>(numToRead)});
63 
64  jint ret = env->CallIntMethod(m_call, sim::GetBufferCallback(),
65  MakeJString(env, name), toCallbackArr,
66  static_cast<jint>(numToRead));
67 
68  jint* fromCallbackArr = reinterpret_cast<jint*>(
69  env->GetPrimitiveArrayCritical(toCallbackArr, nullptr));
70 
71  for (int i = 0; i < ret; i++) {
72  buffer[i] = fromCallbackArr[i];
73  }
74 
75  env->ReleasePrimitiveArrayCritical(toCallbackArr, fromCallbackArr, JNI_ABORT);
76 
77  if (env->ExceptionCheck()) {
78  env->ExceptionDescribe();
79  }
80 
81  if (didAttachThread) {
82  vm->DetachCurrentThread();
83  }
84  return ret;
85 }

91 SIM_JniHandle sim::AllocateSpiBufferCallback(
92  JNIEnv* env, jint index, jobject callback,
93  RegisterSpiBufferCallbackFunc createCallback) {
94  auto callbackStore =
95  std::make_shared<SpiReadAutoReceiveBufferCallbackStore>();
96 
97  auto handle = callbackHandles->Allocate(callbackStore);
98 
99  if (handle == HAL_kInvalidHandle) {
100  return -1;
101  }
102 
103  uintptr_t handleAsPtr = static_cast<uintptr_t>(handle);
104  void* handleAsVoidPtr = reinterpret_cast<void*>(handleAsPtr);
105 
106  callbackStore->create(env, callback);
107 
108  auto callbackFunc = [](const char* name, void* param, uint32_t* buffer,
109  int32_t numToRead, int32_t* outputCount) {
110  uintptr_t handleTmp = reinterpret_cast<uintptr_t>(param);
111  SIM_JniHandle handle = static_cast<SIM_JniHandle>(handleTmp);
112  auto data = callbackHandles->Get(handle);
113  if (!data) {
114  return;
115  }
116 
117  *outputCount = data->performCallback(name, buffer, numToRead);
118  };
119 
120  auto id = createCallback(index, callbackFunc, handleAsVoidPtr);
121 
122  callbackStore->setCallbackId(id);
123 
124  return handle;
125 }
