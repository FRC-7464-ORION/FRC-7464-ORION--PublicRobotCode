22 HAL_AnalogInputHandle HAL_InitializeAnalogInputPort(
23  HAL_PortHandle portHandle, const char* allocationLocation,
24  int32_t* status) {
25  hal::init::CheckInit();
26  int16_t channel = getPortHandleChannel(portHandle);
27  if (channel == InvalidHandleIndex || channel >= kNumAnalogInputs) {
28  *status = RESOURCE_OUT_OF_RANGE;
29  hal::SetLastErrorIndexOutOfRange(status, "Invalid Index for Analog Input",
30  0, kNumAnalogInputs, channel);
31  return HAL_kInvalidHandle;
32  }
33 
34  HAL_AnalogInputHandle handle;
35  auto analog_port = analogInputHandles->Allocate(channel, &handle, status);
36 
37  if (*status != 0) {
38  if (analog_port) {
39  hal::SetLastErrorPreviouslyAllocated(status, "Analog Input", channel,
40  analog_port->previousAllocation);
41  } else {
42  hal::SetLastErrorIndexOutOfRange(status, "Invalid Index for Analog Input",
43  0, kNumAnalogInputs, channel);
44  }
45  return HAL_kInvalidHandle; // failed to allocate. Pass error back.
46  }
47 
48  analog_port->channel = static_cast<uint8_t>(channel);
49  if (HAL_IsAccumulatorChannel(handle, status)) {
50  analog_port->isAccumulator = true;
51  } else {
52  analog_port->isAccumulator = false;
53  }
54 
55  SimAnalogInData[channel].initialized = true;
56  SimAnalogInData[channel].accumulatorInitialized = false;
57  SimAnalogInData[channel].simDevice = 0;
58 
59  analog_port->previousAllocation =
60  allocationLocation ? allocationLocation : "";
61 
62  return handle;
63 }

154 int32_t HAL_GetAnalogVoltsToValue(HAL_AnalogInputHandle analogPortHandle,
155  double voltage, int32_t* status) {
156  if (voltage > 5.0) {
157  voltage = 5.0;
158  *status = VOLTAGE_OUT_OF_RANGE;
159  }
160  if (voltage < 0.0) {
161  voltage = 0.0;
162  *status = VOLTAGE_OUT_OF_RANGE;
163  }
164  int32_t LSBWeight = HAL_GetAnalogLSBWeight(analogPortHandle, status);
165  int32_t offset = HAL_GetAnalogOffset(analogPortHandle, status);
166  int32_t value =
167  static_cast<int32_t>((voltage + offset * 1.0e-9) / (LSBWeight * 1.0e-9));
168  return value;
169 }

181 double HAL_GetAnalogValueToVolts(HAL_AnalogInputHandle analogPortHandle,
182  int32_t rawValue, int32_t* status) {
183  int32_t LSBWeight = HAL_GetAnalogLSBWeight(analogPortHandle, status);
184  int32_t offset = HAL_GetAnalogOffset(analogPortHandle, status);
185  double voltage = LSBWeight * 1.0e-9 * rawValue - offset * 1.0e-9;
