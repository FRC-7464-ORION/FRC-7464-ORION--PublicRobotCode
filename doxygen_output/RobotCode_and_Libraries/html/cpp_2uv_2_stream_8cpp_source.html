60 void Stream::StartRead() {
61  Invoke(&uv_read_start, GetRawStream(), &Handle::AllocBuf,
62  [](uv_stream_t* stream, ssize_t nread, const uv_buf_t* buf) {
63  auto& h = *static_cast<Stream*>(stream->data);
64  Buffer data = *buf;
65 
66  // nread=0 is simply ignored
67  if (nread == UV_EOF) {
68  h.end();
69  } else if (nread > 0) {
70  h.data(data, static_cast<size_t>(nread));
71  } else if (nread < 0) {
72  h.ReportError(nread);
73  }
74 
75  // free the buffer
76  h.FreeBuf(data);
77  });
78 }
