14 static void def_log_func(unsigned int level, const char* file,
15  unsigned int line, const char* msg) {
16  if (level == 20) {
17  fmt::print(stderr, "CS: {}\n", msg);
18  return;
19  }
20 
21  std::string_view levelmsg;
22  if (level >= 50) {
23  levelmsg = "CRITICAL";
24  } else if (level >= 40) {
25  levelmsg = "ERROR";
26  } else if (level >= 30) {
27  levelmsg = "WARNING";
28  } else {
29  return;
30  }
31  fmt::print(stderr, "CS: {}: {} ({}:{})\n", levelmsg, msg,
32  fs::path{file}.filename().string(), line);
33 }
