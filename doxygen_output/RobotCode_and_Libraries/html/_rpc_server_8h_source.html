59  void DoCallback(std::function<void(const RpcAnswer& call)> callback,
60  const RpcNotifierData& data) {
61  DEBUG4("rpc calling {}", data.name);
62  unsigned int local_id = Handle{data.entry}.GetIndex();
63  unsigned int call_uid = Handle{data.call}.GetIndex();
64  RpcIdPair lookup_uid{local_id, call_uid};
65  callback(data);
66  {
67  std::scoped_lock lock(m_mutex);
68  auto i = m_response_map.find(lookup_uid);
69  if (i != m_response_map.end()) {
70  // post an empty response and erase it
71  (i->getSecond())("");
72  m_response_map.erase(i);
73  }
74  }
75  }
