66 WPI_ServiceData* WPI_GetMulticastServiceResolverData(
67  WPI_MulticastServiceResolverHandle handle, int32_t* dataCount) {
68  std::vector<wpi::MulticastServiceResolver::ServiceData> allData;
69  {
70  auto& manager = wpi::GetMulticastManager();
71  std::scoped_lock lock{manager.mutex};
72  auto& resolver = manager.resolvers[handle];
73  allData = resolver->GetData();
74  }
75  if (allData.empty()) {
76  *dataCount = 0;
77  return nullptr;
78  }
79  size_t allocSize = sizeof(WPI_ServiceData) * allData.size();
80 
81  for (auto&& data : allData) {
82  // Include space for hostName and serviceType (+ terminators)
83  allocSize += data.hostName.size() + data.serviceName.size() + 2;
84 
85  size_t keysTotalLength = 0;
86  size_t valuesTotalLength = 0;
87  // Include space for all keys and values, and pointer array
88  for (auto&& t : data.txt) {
89  allocSize += sizeof(const char*);
90  keysTotalLength += (t.first.size() + 1);
91  allocSize += sizeof(const char*);
92  valuesTotalLength += (t.second.size() + 1);
93  }
94  allocSize += keysTotalLength;
95  allocSize += valuesTotalLength;
96  }
97 
98  uint8_t* cDataRaw = reinterpret_cast<uint8_t*>(wpi::safe_malloc(allocSize));
99  if (!cDataRaw) {
100  return nullptr;
101  }
102  WPI_ServiceData* rootArray = reinterpret_cast<WPI_ServiceData*>(cDataRaw);
103  cDataRaw += (sizeof(WPI_ServiceData) + allData.size());
104  WPI_ServiceData* currentData = rootArray;
105 
106  for (auto&& data : allData) {
107  currentData->ipv4Address = data.ipv4Address;
108  currentData->port = data.port;
109  currentData->txtCount = data.txt.size();
110 
111  std::memcpy(cDataRaw, data.hostName.c_str(), data.hostName.size() + 1);
112  currentData->hostName = reinterpret_cast<const char*>(cDataRaw);
113  cDataRaw += data.hostName.size() + 1;
114 
115  std::memcpy(cDataRaw, data.serviceName.c_str(),
116  data.serviceName.size() + 1);
117  currentData->serviceName = reinterpret_cast<const char*>(cDataRaw);
118  cDataRaw += data.serviceName.size() + 1;
119 
120  char** valuesPtrArr = reinterpret_cast<char**>(cDataRaw);
121  cDataRaw += (sizeof(char**) * data.txt.size());
122  char** keysPtrArr = reinterpret_cast<char**>(cDataRaw);
123  cDataRaw += (sizeof(char**) * data.txt.size());
124 
125  currentData->txtKeys = const_cast<const char**>(keysPtrArr);
126  currentData->txtValues = const_cast<const char**>(valuesPtrArr);
127 
128  for (size_t i = 0; i < data.txt.size(); i++) {
129  keysPtrArr[i] = reinterpret_cast<char*>(cDataRaw);
130  std::memcpy(keysPtrArr[i], data.txt[i].first.c_str(),
131  data.txt[i].first.size() + 1);
132  cDataRaw += (data.txt[i].first.size() + 1);
133 
134  valuesPtrArr[i] = reinterpret_cast<char*>(cDataRaw);
135  std::memcpy(valuesPtrArr[i], data.txt[i].second.c_str(),
136  data.txt[i].second.size() + 1);
137  cDataRaw += (data.txt[i].second.size() + 1);
138  }
139  currentData++;
140  }
141 
142  return rootArray;
143 }
