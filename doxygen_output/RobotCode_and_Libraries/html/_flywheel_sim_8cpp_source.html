31 units::ampere_t FlywheelSim::GetCurrentDraw() const {
32  // I = V / R - omega / (Kv * R)
33  // Reductions are greater than 1, so a reduction of 10:1 would mean the motor
34  // is spinning 10x faster than the output.
35  return m_gearbox.Current(GetAngularVelocity() * m_gearing,
36  units::volt_t{m_u(0)}) *
37  wpi::sgn(m_u(0));
38 }
