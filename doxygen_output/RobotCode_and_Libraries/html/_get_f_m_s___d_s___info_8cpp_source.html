49

53

56 
57  // Switch on the alliance color
58  switch(alliance_color)
59  {
60  // If it's red...

65

76  break;
77 
78  default :
79  // Make it unknown
80  alliance = mk_UnknownAlliance;
81 
82  } // end switch(alliance_color)
83 
84  // Get the station location number
85  station_location = frc::DriverStation::GetLocation();
86 
87  // Switch on the station location
88  switch(station_location)
89  {
90 
91  // If we get #0 station
92  case 0 :
93  // Make it invalid
94  station_number = mk_StationNumber0;
95  break;
96 
97  // If we get #1 station
98  case 1 :
99  // Make it #1 (LEFT)
100  station_number = mk_StationNumber1;
101  break;
102 
103  // If we get #2 station
104  case 2 :
105  // Make it #2 (CENTER)
106  station_number = mk_StationNumber2;
107  break;
108 
109  // If we get #3 station
110  case 3 :
111  // Make it #3 (RIGHT)
112  station_number = mk_StationNumber3;
113  break;
114 
115  default :
116  // Make it unknown
117  station_number = mk_StationNumberUnknown;
118 
119  } // end switch(station_location)
120 
121  // If the alliance color was INVALID or UNKNOWN
122  if( (alliance.compare(mk_InvalidAlliance) == 0)
123  || (alliance.compare(mk_UnknownAlliance) == 0) )
124  // Just use the alliance color, as a station # makes no sense
125  m_AllianceColorStationNumber = alliance;
126  else

128  m_AllianceColorStationNumber =
