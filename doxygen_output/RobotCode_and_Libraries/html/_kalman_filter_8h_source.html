47 template <int States, int Inputs, int Outputs>
48 class KalmanFilterImpl {
49  public:
58  KalmanFilterImpl(LinearSystem<States, Inputs, Outputs>& plant,
59  const wpi::array<double, States>& stateStdDevs,
60  const wpi::array<double, Outputs>& measurementStdDevs,
61  units::second_t dt) {
62  m_plant = &plant;
63 
64  auto contQ = MakeCovMatrix(stateStdDevs);
65  auto contR = MakeCovMatrix(measurementStdDevs);
66 
67  Eigen::Matrix<double, States, States> discA;
68  Eigen::Matrix<double, States, States> discQ;
69  DiscretizeAQTaylor<States>(plant.A(), contQ, dt, &discA, &discQ);
70 
71  auto discR = DiscretizeR<Outputs>(contR, dt);
72 
73  const auto& C = plant.C();
74 
75  if (!IsDetectable<States, Outputs>(discA, C)) {
76  std::string msg = fmt::format(
77  "The system passed to the Kalman filter is "
78  "unobservable!\n\nA =\n{}\nC =\n{}\n",
79  discA, C);
80 
81  wpi::math::MathSharedStore::ReportError(msg);
82  throw std::invalid_argument(msg);
83  }
84 
85  Eigen::Matrix<double, States, States> P =
86  drake::math::DiscreteAlgebraicRiccatiEquation(
87  discA.transpose(), C.transpose(), discQ, discR);
88 
89  // S = CPCᵀ + R
90  Eigen::Matrix<double, Outputs, Outputs> S = C * P * C.transpose() + discR;
91 
92  // We want to put K = PCᵀS⁻¹ into Ax = b form so we can solve it more
93  // efficiently.
94  //
95  // K = PCᵀS⁻¹
96  // KS = PCᵀ
97  // (KS)ᵀ = (PCᵀ)ᵀ
98  // SᵀKᵀ = CPᵀ
99  //
100  // The solution of Ax = b can be found via x = A.solve(b).
101  //
102  // Kᵀ = Sᵀ.solve(CPᵀ)
103  // K = (Sᵀ.solve(CPᵀ))ᵀ
104  m_K = S.transpose().ldlt().solve(C * P.transpose()).transpose();
105 
106  Reset();
107  }

110  KalmanFilterImpl& operator=(KalmanFilterImpl&&) = default;

115  const Eigen::Matrix<double, States, Outputs>& K() const { return m_K; }

123  double K(int i, int j) const { return m_K(i, j); }

163  void Predict(const Eigen::Vector<double, Inputs>& u, units::second_t dt) {
164  m_xHat = m_plant->CalculateX(m_xHat, u, dt);
165  }
166 
173  void Correct(const Eigen::Vector<double, Inputs>& u,
174  const Eigen::Vector<double, Outputs>& y) {
175  // x̂ₖ₊₁⁺ = x̂ₖ₊₁⁻ + K(y − (Cx̂ₖ₊₁⁻ + Duₖ₊₁))
176  m_xHat += m_K * (y - (m_plant->C() * m_xHat + m_plant->D() * u));
177  }

179  private:

181
