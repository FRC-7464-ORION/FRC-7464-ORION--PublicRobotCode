89 void RamseteCommand::Initialize() {
90  m_prevTime = 0_s;
91  auto initialState = m_trajectory.Sample(0_s);
92  m_prevSpeeds = m_kinematics.ToWheelSpeeds(
93  frc::ChassisSpeeds{initialState.velocity, 0_mps,
94  initialState.velocity * initialState.curvature});
95  m_timer.Reset();
96  m_timer.Start();
97  if (m_usePID) {
98  m_leftController->Reset();
99  m_rightController->Reset();
100  }
101 }

103 void RamseteCommand::Execute() {
104  auto curTime = m_timer.Get();
105  auto dt = curTime - m_prevTime;
106 
107  auto targetWheelSpeeds = m_kinematics.ToWheelSpeeds(
108  m_controller.Calculate(m_pose(), m_trajectory.Sample(curTime)));
109 
110  if (m_usePID) {
111  auto leftFeedforward = m_feedforward.Calculate(
112  targetWheelSpeeds.left,
113  (targetWheelSpeeds.left - m_prevSpeeds.left) / dt);
114 
115  auto rightFeedforward = m_feedforward.Calculate(
116  targetWheelSpeeds.right,
117  (targetWheelSpeeds.right - m_prevSpeeds.right) / dt);
118 
119  auto leftOutput = volt_t(m_leftController->Calculate(
120  m_speeds().left.to<double>(),
121  targetWheelSpeeds.left.to<double>())) +
122  leftFeedforward;
123 
124  auto rightOutput = volt_t(m_rightController->Calculate(
125  m_speeds().right.to<double>(),
126  targetWheelSpeeds.right.to<double>())) +
127  rightFeedforward;
128 
129  m_outputVolts(leftOutput, rightOutput);
130  } else {
131  m_outputVel(targetWheelSpeeds.left, targetWheelSpeeds.right);
132  }
133 
134  m_prevTime = curTime;
135  m_prevSpeeds = targetWheelSpeeds;
136 }
