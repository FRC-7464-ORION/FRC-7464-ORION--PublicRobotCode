129 // Standard header to send along with other header information like mimetype.
130 //
131 // The parameters should ensure the browser does not cache our answer.
132 // A browser should connect for each file and not serve files from its cache.
133 // Using cached pictures would lead to showing old/outdated pictures.
134 // Many browsers seem to ignore, or at least not always obey, those headers.
135 static void SendHeader(wpi::raw_ostream& os, int code,
136  std::string_view codeText, std::string_view contentType,
137  std::string_view extra = {}) {
138  fmt::print(os, "HTTP/1.0 {} {}\r\n", code, codeText);
139  os << "Connection: close\r\n"
140  "Server: CameraServer/1.0\r\n"
141  "Cache-Control: no-store, no-cache, must-revalidate, pre-check=0, "
142  "post-check=0, max-age=0\r\n"
143  "Pragma: no-cache\r\n"
144  "Expires: Mon, 3 Jan 2000 12:34:56 GMT\r\n";
145  os << "Content-Type: " << contentType << "\r\n";
146  os << "Access-Control-Allow-Origin: *\r\nAccess-Control-Allow-Methods: *\r\n";
147  if (!extra.empty()) {
148  os << extra << "\r\n";
149  }
150  os << "\r\n"; // header ends with a blank line
151 }

153 // Send error header and message
154 // @param code HTTP error code (e.g. 404)
155 // @param message Additional message text
156 static void SendError(wpi::raw_ostream& os, int code,
157  std::string_view message) {
158  std::string_view codeText, extra, baseMessage;
159  switch (code) {
160  case 401:
161  codeText = "Unauthorized";
162  extra = "WWW-Authenticate: Basic realm=\"CameraServer\"";
163  baseMessage = "401: Not Authenticated!";
164  break;
165  case 404:
166  codeText = "Not Found";
167  baseMessage = "404: Not Found!";
168  break;
169  case 500:
170  codeText = "Internal Server Error";
171  baseMessage = "500: Internal Server Error!";
172  break;
173  case 400:
174  codeText = "Bad Request";
175  baseMessage = "400: Not Found!";
176  break;
177  case 403:
178  codeText = "Forbidden";
179  baseMessage = "403: Forbidden!";
180  break;
181  case 503:
182  codeText = "Service Unavailable";
183  baseMessage = "503: Service Unavailable";
184  break;
185  default:
186  code = 501;
187  codeText = "Not Implemented";
188  baseMessage = "501: Not Implemented!";
189  break;
190  }
191  SendHeader(os, code, codeText, "text/plain", extra);
192  os << baseMessage << "\r\n" << message;
193 }

195 // Perform a command specified by HTTP GET parameters.
196 bool MjpegServerImpl::ConnThread::ProcessCommand(wpi::raw_ostream& os,
197  SourceImpl& source,
198  std::string_view parameters,
199  bool respond) {
200  wpi::SmallString<256> responseBuf;
201  wpi::raw_svector_ostream response{responseBuf};
202  // command format: param1=value1&param2=value2...
203  while (!parameters.empty()) {
204  // split out next param and value
205  std::string_view rawParam, rawValue;
206  std::tie(rawParam, parameters) = wpi::split(parameters, '&');
207  if (rawParam.empty()) {
208  continue; // ignore "&&"
209  }
210  std::tie(rawParam, rawValue) = wpi::split(rawParam, '=');
211  if (rawParam.empty() || rawValue.empty()) {
212  continue; // ignore "param="
213  }
214  SDEBUG4("HTTP parameter \"{}\" value \"{}\"", rawParam, rawValue);
215 
216  // unescape param
217  bool error = false;
218  wpi::SmallString<64> paramBuf;
219  std::string_view param = wpi::UnescapeURI(rawParam, paramBuf, &error);
220  if (error) {
221  auto estr = fmt::format("could not unescape parameter \"{}\"", rawParam);
222  SendError(os, 500, estr);
223  SDEBUG("{}", estr);
224  return false;
225  }
226 
227  // unescape value
228  wpi::SmallString<64> valueBuf;
229  std::string_view value = wpi::UnescapeURI(rawValue, valueBuf, &error);
230  if (error) {
231  auto estr = fmt::format("could not unescape value \"{}\"", rawValue);
232  SendError(os, 500, estr);
233  SDEBUG("{}", estr);
234  return false;
235  }
236 
237  // Handle resolution, compression, and FPS. These are handled locally
238  // rather than passed to the source.
239  if (param == "resolution") {
240  auto [widthStr, heightStr] = wpi::split(value, 'x');
241  int width = wpi::parse_integer<int>(widthStr, 10).value_or(-1);
242  int height = wpi::parse_integer<int>(heightStr, 10).value_or(-1);
243  if (width < 0) {
244  response << param << ": \"width is not an integer\"\r\n";
245  SWARNING("HTTP parameter \"{}\" width \"{}\" is not an integer", param,
246  widthStr);
247  continue;
248  }
249  if (height < 0) {
250  response << param << ": \"height is not an integer\"\r\n";
251  SWARNING("HTTP parameter \"{}\" height \"{}\" is not an integer", param,
252  heightStr);
253  continue;
254  }
255  m_width = width;
256  m_height = height;
257  response << param << ": \"ok\"\r\n";
258  continue;
259  }
260 
261  if (param == "fps") {
262  if (auto v = wpi::parse_integer<int>(value, 10)) {
263  m_fps = v.value();
264  response << param << ": \"ok\"\r\n";
265  } else {
266  response << param << ": \"invalid integer\"\r\n";
267  SWARNING("HTTP parameter \"{}\" value \"{}\" is not an integer", param,
268  value);
269  }
270  continue;
271  }
272 
273  if (param == "compression") {
274  if (auto v = wpi::parse_integer<int>(value, 10)) {
275  m_compression = v.value();
276  response << param << ": \"ok\"\r\n";
277  } else {
278  response << param << ": \"invalid integer\"\r\n";
279  SWARNING("HTTP parameter \"{}\" value \"{}\" is not an integer", param,
280  value);
281  }
282  continue;
283  }
284 
285  // ignore name parameter
286  if (param == "name") {
287  continue;
288  }
289 
290  // try to assign parameter
291  auto prop = source.GetPropertyIndex(param);
292  if (!prop) {
293  response << param << ": \"ignored\"\r\n";
294  SWARNING("ignoring HTTP parameter \"{}\"", param);
295  continue;
296  }
297 
298  CS_Status status = 0;
299  auto kind = source.GetPropertyKind(prop);
300  switch (kind) {
301  case CS_PROP_BOOLEAN:
302  case CS_PROP_INTEGER:
303  case CS_PROP_ENUM: {
304  if (auto v = wpi::parse_integer<int>(value, 10)) {
305  fmt::print(response, "{}: {}\r\n", param, v.value());
306  SDEBUG4("HTTP parameter \"{}\" value {}", param, value);
307  source.SetProperty(prop, v.value(), &status);
308  } else {
309  response << param << ": \"invalid integer\"\r\n";
310  SWARNING("HTTP parameter \"{}\" value \"{}\" is not an integer",
311  param, value);
312  }
313  break;
314  }
315  case CS_PROP_STRING: {
316  response << param << ": \"ok\"\r\n";
317  SDEBUG4("HTTP parameter \"{}\" value \"{}\"", param, value);
318  source.SetStringProperty(prop, value, &status);
319  break;
320  }
321  default:
322  break;
323  }
324  }
325 
326  // Send HTTP response
327  if (respond) {
328  SendHeader(os, 200, "OK", "text/plain");
329  os << response.str() << "\r\n";
330  }
331 
332  return true;
333 }

341 // Send the root html file with controls for all the settable properties.
342 void MjpegServerImpl::ConnThread::SendHTML(wpi::raw_ostream& os,
343  SourceImpl& source, bool header) {
344  if (header) {
345  SendHeader(os, 200, "OK", "text/html");
346  }
347 
348  SendHTMLHeadTitle(os);
349  os << startRootPage;
350  wpi::SmallVector<int, 32> properties_vec;
351  CS_Status status = 0;
352  for (auto prop : source.EnumerateProperties(properties_vec, &status)) {
353  wpi::SmallString<128> name_buf;
354  auto name = source.GetPropertyName(prop, name_buf, &status);
355  if (wpi::starts_with(name, "raw_")) {
356  continue;
357  }
358  auto kind = source.GetPropertyKind(prop);
359  fmt::print(os, "<p /><label for=\"{0}\">{0}</label>\n", name);
360  switch (kind) {
361  case CS_PROP_BOOLEAN:
362  fmt::print(os,
363  "<input id=\"{0}\" type=\"checkbox\" "
364  "onclick=\"update('{0}', this.checked ? 1 : 0)\" ",
365  name);
366  if (source.GetProperty(prop, &status) != 0) {
367  os << "checked />\n";
368  } else {
369  os << " />\n";
370  }
371  break;
372  case CS_PROP_INTEGER: {
373  auto valI = source.GetProperty(prop, &status);
374  auto min = source.GetPropertyMin(prop, &status);
375  auto max = source.GetPropertyMax(prop, &status);
376  auto step = source.GetPropertyStep(prop, &status);
377  fmt::print(os,
378  "<input type=\"range\" min=\"{1}\" max=\"{2}\" "
379  "value=\"{3}\" id=\"{0}\" step=\"{4}\" "
380  "oninput=\"updateInt('#{0}op', '{0}', value)\" />\n",
381  name, min, max, valI, step);
382  fmt::print(os, "<output for=\"{0}\" id=\"{0}op\">{1}</output>\n", name,
383  valI);
384  break;
385  }
386  case CS_PROP_ENUM: {
387  auto valE = source.GetProperty(prop, &status);
388  auto choices = source.GetEnumPropertyChoices(prop, &status);
389  int j = 0;
390  for (auto choice = choices.begin(), end = choices.end(); choice != end;
391  ++j, ++choice) {
392  if (choice->empty()) {
393  continue; // skip empty choices
394  }
395  // replace any non-printable characters in name with spaces

397  for (char ch : *choice) {
398  ch_name.push_back(wpi::isPrint(ch) ? ch : ' ');
399  }
400  fmt::print(os,
401  "<input id=\"{0}{1}\" type=\"radio\" name=\"{0}\" "
402  "value=\"{2}\" onclick=\"update('{0}', {1})\"",
403  name, j, ch_name);
404  if (j == valE) {
405  os << " checked";
406  }
407  fmt::print(os, " /><label for=\"{}{}\">{}</label>\n", name, j,
408  ch_name);
409  }
410  break;
411  }
412  case CS_PROP_STRING: {
413  wpi::SmallString<128> strval_buf;
414  fmt::print(os,
415  "<input type=\"text\" id=\"{0}box\" name=\"{0}\" "
416  "value=\"{1}\" />\n",
417  name, source.GetStringProperty(prop, strval_buf, &status));
418  fmt::print(os,
419  "<input type=\"button\" value =\"Submit\" "
420  "onclick=\"update('{0}', {0}box.value)\" />\n",
421  name);
422  break;
423  }
424  default:
425  break;
426  }
427  }
428 
429  status = 0;
430  auto info = GetUsbCameraInfo(Instance::GetInstance().FindSource(source).first,
431  &status);
432  if (status == CS_OK) {
433  os << "<p>USB device path: " << info.path << '\n';
434  for (auto&& path : info.otherPaths) {
435  os << "<p>Alternate device path: " << path << '\n';
436  }
437  }
438 
439  os << "<p>Supported Video Modes:</p>\n";
440  os << "<table cols=\"4\" style=\"border: 1px solid black\">\n";
441  os << "<tr><th>Pixel Format</th>"
442  << "<th>Width</th>"
443  << "<th>Height</th>"
444  << "<th>FPS</th></tr>";
445  for (auto mode : source.EnumerateVideoModes(&status)) {
446  os << "<tr><td>";
447  switch (mode.pixelFormat) {
448  case VideoMode::kMJPEG:
449  os << "MJPEG";
450  break;
451  case VideoMode::kYUYV:
452  os << "YUYV";
453  break;
454  case VideoMode::kRGB565:
455  os << "RGB565";
456  break;
457  case VideoMode::kBGR:
458  os << "BGR";
459  break;
460  case VideoMode::kGray:
461  os << "gray";
462  break;
463  default:
464  os << "unknown";
465  break;
466  }
467  fmt::print(os, "</td><td>{}</td><td>{}</td><td>{}</td></tr>", mode.width,
468  mode.height, mode.fps);
469  }
470  os << "</table>\n";
471  os << endRootPage << "\r\n";
472  os.flush();
473 }

623 void MjpegServerImpl::Stop() {
624  m_active = false;
625 
626  // wake up server thread by shutting down the socket
627  m_acceptor->shutdown();
628 
629  // join server thread
630  if (m_serverThread.joinable()) {
631  m_serverThread.join();
632  }
633 
634  // close streams
635  for (auto& connThread : m_connThreads) {
636  if (auto thr = connThread.GetThread()) {
637  if (thr->m_stream) {
638  thr->m_stream->close();
639  }
640  }
641  connThread.Stop();
642  }
643 
644  // wake up connection threads by forcing an empty frame to be sent
645  if (auto source = GetSource()) {
646  source->Wakeup();
647  }
648 }

650 // Send HTTP response and a stream of JPG-frames
651 void MjpegServerImpl::ConnThread::SendStream(wpi::raw_socket_ostream& os) {
652  if (m_noStreaming) {
653  SERROR("{}", "Too many simultaneous client streams");
654  SendError(os, 503, "Too many simultaneous streams");
655  return;
656  }
657 
658  os.SetUnbuffered();
659 
660  wpi::SmallString<256> header;
661  wpi::raw_svector_ostream oss{header};
662 
663  SendHeader(oss, 200, "OK", "multipart/x-mixed-replace;boundary=" BOUNDARY);
664  os << oss.str();
665 
666  SDEBUG("{}", "Headers send, sending stream now");
667 
668  Frame::Time lastFrameTime = 0;
669  Frame::Time timePerFrame = 0;
670  if (m_fps != 0) {
671  timePerFrame = 1000000.0 / m_fps;
672  }
673  Frame::Time averageFrameTime = 0;
674  Frame::Time averagePeriod = 1000000; // 1 second window
675  if (averagePeriod < timePerFrame) {
676  averagePeriod = timePerFrame * 10;
677  }
678 
679  StartStream();
680  while (m_active && !os.has_error()) {
681  auto source = GetSource();
682  if (!source) {
683  // Source disconnected; sleep so we don't consume all processor time.
684  os << "\r\n"; // Keep connection alive
685  std::this_thread::sleep_for(std::chrono::milliseconds(200));
686  continue;
687  }
688  SDEBUG4("{}", "waiting for frame");
689  Frame frame = source->GetNextFrame(0.225); // blocks
690  if (!m_active) {
691  break;
692  }
693  if (!frame) {
694  // Bad frame; sleep for 20 ms so we don't consume all processor time.
695  os << "\r\n"; // Keep connection alive
696  std::this_thread::sleep_for(std::chrono::milliseconds(20));
697  continue;
698  }
699 
700  auto thisFrameTime = frame.GetTime();
701  if (thisFrameTime != 0 && timePerFrame != 0 && lastFrameTime != 0) {
702  Frame::Time deltaTime = thisFrameTime - lastFrameTime;
703 
704  // drop frame if it is early compared to the desired frame rate AND
705  // the current average is higher than the desired average
706  if (deltaTime < timePerFrame && averageFrameTime < timePerFrame) {
707  // sleep for 1 ms so we don't consume all processor time
708  std::this_thread::sleep_for(std::chrono::milliseconds(1));
709  continue;
710  }
711 
712  // update average
713  if (averageFrameTime != 0) {
714  averageFrameTime =
715  averageFrameTime * (averagePeriod - timePerFrame) / averagePeriod +
716  deltaTime * timePerFrame / averagePeriod;
717  } else {
718  averageFrameTime = deltaTime;
719  }
720  }
721 
722  int width = m_width != 0 ? m_width : frame.GetOriginalWidth();
723  int height = m_height != 0 ? m_height : frame.GetOriginalHeight();
724  Image* image = frame.GetImageMJPEG(
725  width, height, m_compression,
726  m_compression == -1 ? m_defaultCompression : m_compression);
727  if (!image) {
728  // Shouldn't happen, but just in case...
729  std::this_thread::sleep_for(std::chrono::milliseconds(20));
730  continue;
731  }
732 
733  const char* data = image->data();
734  size_t size = image->size();
735  bool addDHT = false;
736  size_t locSOF = size;
737  switch (image->pixelFormat) {
738  case VideoMode::kMJPEG:
739  // Determine if we need to add DHT to it, and allocate enough space
740  // for adding it if required.
741  addDHT = JpegNeedsDHT(data, &size, &locSOF);
742  break;
743  case VideoMode::kYUYV:
744  case VideoMode::kRGB565:
745  default:
746  // Bad frame; sleep for 10 ms so we don't consume all processor time.
747  std::this_thread::sleep_for(std::chrono::milliseconds(10));
748  continue;
749  }
750 
751  SDEBUG4("sending frame size={} addDHT={}", size, addDHT);
752 
753  // print the individual mimetype and the length
754  // sending the content-length fixes random stream disruption observed
755  // with firefox
756  lastFrameTime = thisFrameTime;
757  double timestamp = lastFrameTime / 1000000.0;
758  header.clear();
759  oss << "\r\n--" BOUNDARY "\r\n"
760  << "Content-Type: image/jpeg\r\n";
761  fmt::print(oss, "Content-Length: {}\r\n", size);
762  fmt::print(oss, "X-Timestamp: {}\r\n", timestamp);
763  oss << "\r\n";
764  os << oss.str();
765  if (addDHT) {
766  // Insert DHT data immediately before SOF
767  os << std::string_view(data, locSOF);
768  os << JpegGetDHT();
769  os << std::string_view(data + locSOF, image->size() - locSOF);
770  } else {
771  os << std::string_view(data, size);
772  }
773  // os.flush();
774  }
775  StopStream();
776 }

778 void MjpegServerImpl::ConnThread::ProcessRequest() {
779  wpi::raw_socket_istream is{*m_stream};
780  wpi::raw_socket_ostream os{*m_stream, true};
781 
782  // Read the request string from the stream
783  wpi::SmallString<128> reqBuf;
784  std::string_view req = is.getline(reqBuf, 4096);
785  if (is.has_error()) {
786  SDEBUG("{}", "error getting request string");
787  return;
788  }
789 
790  enum { kCommand, kStream, kGetSettings, kGetSourceConfig, kRootPage } kind;
791  std::string_view parameters;
792  size_t pos;
793 
794  SDEBUG("HTTP request: '{}'\n", req);
795 
796  // Determine request kind. Most of these are for mjpgstreamer
797  // compatibility, others are for Axis camera compatibility.
798  if ((pos = req.find("POST /stream")) != std::string_view::npos) {
799  kind = kStream;
800  parameters = wpi::substr(wpi::substr(req, req.find('?', pos + 12)), 1);
801  } else if ((pos = req.find("GET /?action=stream")) !=
802  std::string_view::npos) {
803  kind = kStream;
804  parameters = wpi::substr(wpi::substr(req, req.find('&', pos + 19)), 1);
805  } else if ((pos = req.find("GET /stream.mjpg")) != std::string_view::npos) {
806  kind = kStream;
807  parameters = wpi::substr(wpi::substr(req, req.find('?', pos + 16)), 1);
808  } else if (req.find("GET /settings") != std::string_view::npos &&
809  req.find(".json") != std::string_view::npos) {
810  kind = kGetSettings;
811  } else if (req.find("GET /config") != std::string_view::npos &&
812  req.find(".json") != std::string_view::npos) {
813  kind = kGetSourceConfig;
814  } else if (req.find("GET /input") != std::string_view::npos &&
815  req.find(".json") != std::string_view::npos) {
816  kind = kGetSettings;
817  } else if (req.find("GET /output") != std::string_view::npos &&
818  req.find(".json") != std::string_view::npos) {
819  kind = kGetSettings;
820  } else if ((pos = req.find("GET /?action=command")) !=
821  std::string_view::npos) {
822  kind = kCommand;
823  parameters = wpi::substr(wpi::substr(req, req.find('&', pos + 20)), 1);
824  } else if (req.find("GET / ") != std::string_view::npos || req == "GET /\n") {
825  kind = kRootPage;
826  } else {
827  SDEBUG("{}", "HTTP request resource not found");
828  SendError(os, 404, "Resource not found");
829  return;
830  }
831 
832  // Parameter can only be certain characters. This also strips the EOL.
833  pos = parameters.find_first_not_of(
834  "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ_"
835  "-=&1234567890%./");
836  parameters = wpi::substr(parameters, 0, pos);
837  SDEBUG("command parameters: \"{}\"", parameters);
838 
839  // Read the rest of the HTTP request.
840  // The end of the request is marked by a single, empty line
841  wpi::SmallString<128> lineBuf;
842  for (;;) {
843  if (wpi::starts_with(is.getline(lineBuf, 4096), "\n")) {
844  break;
845  }
846  if (is.has_error()) {
847  return;
848  }
849  }
850 
851  // Send response
852  switch (kind) {
853  case kStream:
854  if (auto source = GetSource()) {
855  SDEBUG("request for stream {}", source->GetName());
856  if (!ProcessCommand(os, *source, parameters, false)) {
857  return;
858  }
859  }
860  SendStream(os);
861  break;
862  case kCommand:
863  if (auto source = GetSource()) {
864  ProcessCommand(os, *source, parameters, true);
865  } else {
866  SendHeader(os, 200, "OK", "text/plain");
867  os << "Ignored due to no connected source."
868  << "\r\n";
869  SDEBUG("{}", "Ignored due to no connected source.");
870  }
871  break;
872  case kGetSettings:
873  SDEBUG("{}", "request for JSON file");
874  if (auto source = GetSource()) {
875  SendJSON(os, *source, true);
876  } else {
877  SendError(os, 404, "Resource not found");
878  }
879  break;
880  case kGetSourceConfig:
881  SDEBUG("{}", "request for JSON file");
882  if (auto source = GetSource()) {
883  SendHeader(os, 200, "OK", "application/json");
884  CS_Status status = CS_OK;
885  os << source->GetConfigJson(&status);
886  os.flush();
887  } else {
888  SendError(os, 404, "Resource not found");
889  }
890  break;
891  case kRootPage:
892  SDEBUG("{}", "request for root page");
893  SendHeader(os, 200, "OK", "text/html");
894  if (auto source = GetSource()) {
895  SendHTML(os, *source, false);
896  } else {
897  SendHTMLHeadTitle(os);
898  os << emptyRootPage << "\r\n";
899  }
900  break;
901  }
902 
903  SDEBUG("{}", "leaving HTTP client thread");
904 }

923 // Main server thread
924 void MjpegServerImpl::ServerThreadMain() {
925  if (m_acceptor->start() != 0) {
926  m_active = false;
927  return;
928  }
929 
930  SDEBUG("{}", "waiting for clients to connect");
931  while (m_active) {
932  auto stream = m_acceptor->accept();
933  if (!stream) {
934  m_active = false;
935  return;
936  }
937  if (!m_active) {
938  return;
939  }
940 
941  SDEBUG("client connection from {}", stream->getPeerIP());
942 
943  auto source = GetSource();
944 
945  std::scoped_lock lock(m_mutex);
946  // Find unoccupied worker thread, or create one if necessary
947  auto it = std::find_if(m_connThreads.begin(), m_connThreads.end(),
948  [](const wpi::SafeThreadOwner<ConnThread>& owner) {
949  auto thr = owner.GetThread();
950  return !thr || !thr->m_stream;
951  });
952  if (it == m_connThreads.end()) {
953  m_connThreads.emplace_back();
954  it = std::prev(m_connThreads.end());
955  }
956 
957  // Start it if not already started
958  it->Start(GetName(), m_logger);
959 
960  auto nstreams =
961  std::count_if(m_connThreads.begin(), m_connThreads.end(),
962  [](const wpi::SafeThreadOwner<ConnThread>& owner) {
963  auto thr = owner.GetThread();
964  return thr && thr->m_streaming;
965  });
966 
967  // Hand off connection to it
968  auto thr = it->GetThread();
969  thr->m_stream = std::move(stream);
970  thr->m_source = source;
971  thr->m_noStreaming = nstreams >= 10;
972  thr->m_width = GetProperty(m_widthProp)->value;
973  thr->m_height = GetProperty(m_heightProp)->value;
974  thr->m_compression = GetProperty(m_compressionProp)->value;
975  thr->m_defaultCompression = GetProperty(m_defaultCompressionProp)->value;
976  thr->m_fps = GetProperty(m_fpsProp)->value;
977  thr->m_cond.notify_one();
978  }
979 
980  SDEBUG("{}", "leaving server thread");
981 }
