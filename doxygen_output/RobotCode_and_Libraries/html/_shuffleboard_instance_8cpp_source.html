43 void ShuffleboardInstance::Update() {
44  if (m_impl->tabsChanged) {
45  wpi::SmallVector<std::string, 16> tabTitles;
46  for (auto& entry : m_impl->tabs) {
47  tabTitles.emplace_back(entry.second->GetTitle());
48  }
49  m_impl->rootMetaTable->GetEntry("Tabs").ForceSetStringArray(tabTitles);
50  m_impl->tabsChanged = false;
51  }
52  for (auto& entry : m_impl->tabs) {
53  auto& tab = *entry.second;
54  tab.BuildInto(m_impl->rootTable,
55  m_impl->rootMetaTable->GetSubTable(tab.GetTitle()));
56  }
57 }
