24 class CubicHermiteSplineTest : public ::testing::Test {
25  protected:
26  static void Run(const Pose2d& a, const std::vector<Translation2d>& waypoints,
27  const Pose2d& b) {
28  // Start the timer.
29  const auto start = std::chrono::high_resolution_clock::now();
30 
31  // Generate and parameterize the spline.
32 
33  const auto [startCV, endCV] =
34  SplineHelper::CubicControlVectorsFromWaypoints(a, waypoints, b);
35 
36  const auto splines =
37  SplineHelper::CubicSplinesFromControlVectors(startCV, waypoints, endCV);
38  std::vector<Spline<3>::PoseWithCurvature> poses;
39 
40  poses.push_back(splines[0].GetPoint(0.0));
41 
42  for (auto&& spline : splines) {
43  auto x = SplineParameterizer::Parameterize(spline);
44  poses.insert(std::end(poses), std::begin(x) + 1, std::end(x));
45  }
46 
47  // End timer.
48  const auto finish = std::chrono::high_resolution_clock::now();
49 
50  // Calculate the duration (used when benchmarking)
51  const auto duration =
52  std::chrono::duration_cast<std::chrono::microseconds>(finish - start);
53 
54  for (unsigned int i = 0; i < poses.size() - 1; i++) {
55  auto& p0 = poses[i];
56  auto& p1 = poses[i + 1];
57 
58  // Make sure the twist is under the tolerance defined by the Spline class.
59  auto twist = p0.first.Log(p1.first);
60  EXPECT_LT(std::abs(twist.dx.to<double>()),
61  SplineParameterizer::kMaxDx.to<double>());
62  EXPECT_LT(std::abs(twist.dy.to<double>()),
63  SplineParameterizer::kMaxDy.to<double>());
64  EXPECT_LT(std::abs(twist.dtheta.to<double>()),
65  SplineParameterizer::kMaxDtheta.to<double>());
66  }
67 
68  // Check first point.
69  EXPECT_NEAR(poses.front().first.Translation().X().to<double>(),
70  a.Translation().X().to<double>(), 1E-9);
71  EXPECT_NEAR(poses.front().first.Translation().Y().to<double>(),
72  a.Translation().Y().to<double>(), 1E-9);
73  EXPECT_NEAR(poses.front().first.Rotation().Radians().to<double>(),
74  a.Rotation().Radians().to<double>(), 1E-9);
75 
76  // Check interior waypoints
77  bool interiorsGood = true;
78  for (auto& waypoint : waypoints) {
79  bool found = false;
80  for (auto& state : poses) {
81  if (std::abs(
82  waypoint.Distance(state.first.Translation()).to<double>()) <
83  1E-9) {
84  found = true;
85  }
86  }
87  interiorsGood &= found;
88  }
89 
90  EXPECT_TRUE(interiorsGood);
91 
92  // Check last point.
93  EXPECT_NEAR(poses.back().first.Translation().X().to<double>(),
94  b.Translation().X().to<double>(), 1E-9);
95  EXPECT_NEAR(poses.back().first.Translation().Y().to<double>(),
96  b.Translation().Y().to<double>(), 1E-9);
97  EXPECT_NEAR(poses.back().first.Rotation().Radians().to<double>(),
98  b.Rotation().Radians().to<double>(), 1E-9);
99 
100  static_cast<void>(duration);
101  }
102 };
103 } // namespace frc
