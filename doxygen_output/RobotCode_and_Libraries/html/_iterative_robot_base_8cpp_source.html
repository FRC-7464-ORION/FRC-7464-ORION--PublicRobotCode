105 void IterativeRobotBase::LoopFunc() {
106  m_watchdog.Reset();
107 
108  // Call the appropriate function depending upon the current robot mode
109  if (IsDisabled()) {
110  // Call DisabledInit() if we are now just entering disabled mode from
111  // either a different mode or from power-on.
112  if (m_lastMode != Mode::kDisabled) {
113  LiveWindow::GetInstance()->SetEnabled(false);
114  Shuffleboard::DisableActuatorWidgets();
115  DisabledInit();
116  m_watchdog.AddEpoch("DisabledInit()");
117  m_lastMode = Mode::kDisabled;
118  }
119 
120  HAL_ObserveUserProgramDisabled();
121  DisabledPeriodic();
122  m_watchdog.AddEpoch("DisabledPeriodic()");
123  } else if (IsAutonomous()) {
124  // Call AutonomousInit() if we are now just entering autonomous mode from
125  // either a different mode or from power-on.
126  if (m_lastMode != Mode::kAutonomous) {
127  LiveWindow::GetInstance()->SetEnabled(false);
128  Shuffleboard::DisableActuatorWidgets();
129  AutonomousInit();
130  m_watchdog.AddEpoch("AutonomousInit()");
131  m_lastMode = Mode::kAutonomous;
132  }
133 
134  HAL_ObserveUserProgramAutonomous();
135  AutonomousPeriodic();
136  m_watchdog.AddEpoch("AutonomousPeriodic()");
137  } else if (IsOperatorControl()) {
138  // Call TeleopInit() if we are now just entering teleop mode from
139  // either a different mode or from power-on.
140  if (m_lastMode != Mode::kTeleop) {
141  LiveWindow::GetInstance()->SetEnabled(false);
142  Shuffleboard::DisableActuatorWidgets();
143  TeleopInit();
144  m_watchdog.AddEpoch("TeleopInit()");
145  m_lastMode = Mode::kTeleop;
146  }
147 
148  HAL_ObserveUserProgramTeleop();
149  TeleopPeriodic();
150  m_watchdog.AddEpoch("TeleopPeriodic()");
151  } else {
152  // Call TestInit() if we are now just entering test mode from
153  // either a different mode or from power-on.
154  if (m_lastMode != Mode::kTest) {
155  LiveWindow::GetInstance()->SetEnabled(true);
156  Shuffleboard::EnableActuatorWidgets();
157  TestInit();
158  m_watchdog.AddEpoch("TestInit()");
159  m_lastMode = Mode::kTest;
160  }
161 
162  HAL_ObserveUserProgramTest();
163  TestPeriodic();
164  m_watchdog.AddEpoch("TestPeriodic()");
165  }
166 
167  RobotPeriodic();
168  m_watchdog.AddEpoch("RobotPeriodic()");
169 
170  SmartDashboard::UpdateValues();
171  m_watchdog.AddEpoch("SmartDashboard::UpdateValues()");
172  LiveWindow::GetInstance()->UpdateValues();
173  m_watchdog.AddEpoch("LiveWindow::UpdateValues()");
174  Shuffleboard::Update();
175  m_watchdog.AddEpoch("Shuffleboard::Update()");
176 
177  if (IsSimulation()) {
178  SimulationPeriodic();
179  m_watchdog.AddEpoch("SimulationPeriodic()");
180  }
181 
182  m_watchdog.Disable();
183 
184  // Warn on loop time overruns
185  if (m_watchdog.IsExpired()) {
186  m_watchdog.PrintEpochs();
187  }
188 }
