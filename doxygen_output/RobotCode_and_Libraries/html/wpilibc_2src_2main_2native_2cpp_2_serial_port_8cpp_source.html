16 SerialPort::SerialPort(int baudRate, Port port, int dataBits,
17  SerialPort::Parity parity,
18  SerialPort::StopBits stopBits) {
19  int32_t status = 0;
20 
21  m_portHandle =
22  HAL_InitializeSerialPort(static_cast<HAL_SerialPort>(port), &status);
23  FRC_CheckErrorStatus(status, "Port {}", port);
24  HAL_SetSerialBaudRate(m_portHandle, baudRate, &status);
25  FRC_CheckErrorStatus(status, "SetSerialBaudRate {}", baudRate);
26  HAL_SetSerialDataBits(m_portHandle, dataBits, &status);
27  FRC_CheckErrorStatus(status, "SetSerialDataBits {}", dataBits);
28  HAL_SetSerialParity(m_portHandle, parity, &status);
29  FRC_CheckErrorStatus(status, "SetSerialParity {}", parity);
30  HAL_SetSerialStopBits(m_portHandle, stopBits, &status);
31  FRC_CheckErrorStatus(status, "SetSerialStopBits {}", stopBits);
32 
33  // Set the default timeout to 5 seconds.
34  SetTimeout(5_s);
35 
36  // Don't wait until the buffer is full to transmit.
37  SetWriteBufferMode(kFlushOnAccess);
38 
39  DisableTermination();
40 
41  HAL_Report(HALUsageReporting::kResourceType_SerialPort,
42  static_cast<uint8_t>(port) + 1);
43 }
