29 Trajectory TrajectoryGenerator::GenerateTrajectory(
30  Spline<3>::ControlVector initial,
31  const std::vector<Translation2d>& interiorWaypoints,
32  Spline<3>::ControlVector end, const TrajectoryConfig& config) {
33  const Transform2d flip{Translation2d(), Rotation2d(180_deg)};
34 
35  // Make theta normal for trajectory generation if path is reversed.
36  // Flip the headings.
37  if (config.IsReversed()) {
38  initial.x[1] *= -1;
39  initial.y[1] *= -1;
40  end.x[1] *= -1;
41  end.y[1] *= -1;
42  }
43 
44  std::vector<frc::SplineParameterizer::PoseWithCurvature> points;
45  try {
46  points =
47  SplinePointsFromSplines(SplineHelper::CubicSplinesFromControlVectors(
48  initial, interiorWaypoints, end));
49  } catch (SplineParameterizer::MalformedSplineException& e) {
50  ReportError(e.what());
51  return kDoNothingTrajectory;
52  }
53 
54  // After trajectory generation, flip theta back so it's relative to the
55  // field. Also fix curvature.
56  if (config.IsReversed()) {
57  for (auto& point : points) {
58  point = {point.first + flip, -point.second};
59  }
60  }
61 
62  return TrajectoryParameterizer::TimeParameterizeTrajectory(
63  points, config.Constraints(), config.StartVelocity(),
64  config.EndVelocity(), config.MaxVelocity(), config.MaxAcceleration(),
65  config.IsReversed());
66 }
