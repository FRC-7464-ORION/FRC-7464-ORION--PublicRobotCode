50 void KilloughDrive::DriveCartesian(double ySpeed, double xSpeed,
51  double zRotation, double gyroAngle) {
52  if (!reported) {
53  HAL_Report(HALUsageReporting::kResourceType_RobotDrive,
54  HALUsageReporting::kRobotDrive2_KilloughCartesian, 3);
55  reported = true;
56  }
57 
58  ySpeed = std::clamp(ySpeed, -1.0, 1.0);
59  ySpeed = ApplyDeadband(ySpeed, m_deadband);
60 
61  xSpeed = std::clamp(xSpeed, -1.0, 1.0);
62  xSpeed = ApplyDeadband(xSpeed, m_deadband);
63 
64  // Compensate for gyro angle.
65  Vector2d input{ySpeed, xSpeed};
66  input.Rotate(-gyroAngle);
67 
68  double wheelSpeeds[3];
69  wheelSpeeds[kLeft] = input.ScalarProject(m_leftVec) + zRotation;
70  wheelSpeeds[kRight] = input.ScalarProject(m_rightVec) + zRotation;
71  wheelSpeeds[kBack] = input.ScalarProject(m_backVec) + zRotation;
72 
73  Normalize(wheelSpeeds);
74 
75  m_leftMotor->Set(wheelSpeeds[kLeft] * m_maxOutput);
76  m_rightMotor->Set(wheelSpeeds[kRight] * m_maxOutput);
77  m_backMotor->Set(wheelSpeeds[kBack] * m_maxOutput);
78 
79  Feed();
80 }
