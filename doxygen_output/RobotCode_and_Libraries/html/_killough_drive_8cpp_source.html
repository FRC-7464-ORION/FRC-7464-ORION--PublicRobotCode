89 KilloughDrive::WheelSpeeds KilloughDrive::DriveCartesianIK(double ySpeed,
90  double xSpeed,
91  double zRotation,
92  double gyroAngle) {
93  ySpeed = std::clamp(ySpeed, -1.0, 1.0);
94  xSpeed = std::clamp(xSpeed, -1.0, 1.0);
95 
96  // Compensate for gyro angle.
97  Vector2d input{ySpeed, xSpeed};
98  input.Rotate(-gyroAngle);
99 
100  double wheelSpeeds[3];
101  wheelSpeeds[kLeft] = input.ScalarProject(m_leftVec) + zRotation;
102  wheelSpeeds[kRight] = input.ScalarProject(m_rightVec) + zRotation;
103  wheelSpeeds[kBack] = input.ScalarProject(m_backVec) + zRotation;
104 
105  Desaturate(wheelSpeeds);
106 
107  return {wheelSpeeds[kLeft], wheelSpeeds[kRight], wheelSpeeds[kBack]};
108 }
