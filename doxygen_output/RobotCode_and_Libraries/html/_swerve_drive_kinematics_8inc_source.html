18 template <size_t NumModules>
19 wpi::array<SwerveModuleState, NumModules>
20 SwerveDriveKinematics<NumModules>::ToSwerveModuleStates(
21  const ChassisSpeeds& chassisSpeeds,
22  const Translation2d& centerOfRotation) const {
23  // We have a new center of rotation. We need to compute the matrix again.
24  if (centerOfRotation != m_previousCoR) {
25  for (size_t i = 0; i < NumModules; i++) {
26  // clang-format off
27  m_inverseKinematics.template block<2, 3>(i * 2, 0) =
28  Eigen::Matrix<double, 2, 3>{
29  {1, 0, (-m_modules[i].Y() + centerOfRotation.Y()).value()},
30  {0, 1, (+m_modules[i].X() - centerOfRotation.X()).value()}};
31  // clang-format on
32  }
33  m_previousCoR = centerOfRotation;
34  }
35 
36  Eigen::Vector3d chassisSpeedsVector{chassisSpeeds.vx.value(),
37  chassisSpeeds.vy.value(),
38  chassisSpeeds.omega.value()};
39 
40  Eigen::Matrix<double, NumModules * 2, 1> moduleStatesMatrix =
41  m_inverseKinematics * chassisSpeedsVector;
42  wpi::array<SwerveModuleState, NumModules> moduleStates{wpi::empty_array};
43 
44  for (size_t i = 0; i < NumModules; i++) {
45  units::meters_per_second_t x{moduleStatesMatrix(i * 2, 0)};
46  units::meters_per_second_t y{moduleStatesMatrix(i * 2 + 1, 0)};
47 
48  auto speed = units::math::hypot(x, y);
49  Rotation2d rotation{x.value(), y.value()};
50 
51  moduleStates[i] = {speed, rotation};
52  }
53 
54  return moduleStates;
55 }

70 template <size_t NumModules>
71 ChassisSpeeds SwerveDriveKinematics<NumModules>::ToChassisSpeeds(
72  wpi::array<SwerveModuleState, NumModules> moduleStates) const {
73  Eigen::Matrix<double, NumModules * 2, 1> moduleStatesMatrix;
74 
75  for (size_t i = 0; i < NumModules; ++i) {
76  SwerveModuleState module = moduleStates[i];
77  moduleStatesMatrix(i * 2, 0) = module.speed.value() * module.angle.Cos();
78  moduleStatesMatrix(i * 2 + 1, 0) =
79  module.speed.value() * module.angle.Sin();
80  }
81 
82  Eigen::Vector3d chassisSpeedsVector =
83  m_forwardKinematics.solve(moduleStatesMatrix);
84 
85  return {units::meters_per_second_t{chassisSpeedsVector(0)},
86  units::meters_per_second_t{chassisSpeedsVector(1)},
87  units::radians_per_second_t{chassisSpeedsVector(2)}};
88 }

90 template <size_t NumModules>
91 void SwerveDriveKinematics<NumModules>::DesaturateWheelSpeeds(
92  wpi::array<SwerveModuleState, NumModules>* moduleStates,
93  units::meters_per_second_t attainableMaxSpeed) {
94  auto& states = *moduleStates;
95  auto realMaxSpeed = std::max_element(states.begin(), states.end(),
96  [](const auto& a, const auto& b) {
97  return units::math::abs(a.speed) <
98  units::math::abs(b.speed);
99  })
100  ->speed;
101 
102  if (realMaxSpeed > attainableMaxSpeed) {
103  for (auto& module : states) {
104  module.speed = module.speed / realMaxSpeed * attainableMaxSpeed;
105  }
106  }
107 }
