12 MecanumDriveWheelSpeeds MecanumDriveKinematics::ToWheelSpeeds(
13  const ChassisSpeeds& chassisSpeeds, const Translation2d& centerOfRotation) {
14  // We have a new center of rotation. We need to compute the matrix again.
15  if (centerOfRotation != m_previousCoR) {
16  auto fl = m_frontLeftWheel - centerOfRotation;
17  auto fr = m_frontRightWheel - centerOfRotation;
18  auto rl = m_rearLeftWheel - centerOfRotation;
19  auto rr = m_rearRightWheel - centerOfRotation;
20 
21  SetInverseKinematics(fl, fr, rl, rr);
22 
23  m_previousCoR = centerOfRotation;
24  }
25 
26  Eigen::Vector3d chassisSpeedsVector;
27  chassisSpeedsVector << chassisSpeeds.vx.to<double>(),
28  chassisSpeeds.vy.to<double>(), chassisSpeeds.omega.to<double>();
29 
30  Eigen::Matrix<double, 4, 1> wheelsMatrix =
31  m_inverseKinematics * chassisSpeedsVector;
32 
33  MecanumDriveWheelSpeeds wheelSpeeds;
34  wheelSpeeds.frontLeft = units::meters_per_second_t{wheelsMatrix(0, 0)};
35  wheelSpeeds.frontRight = units::meters_per_second_t{wheelsMatrix(1, 0)};
36  wheelSpeeds.rearLeft = units::meters_per_second_t{wheelsMatrix(2, 0)};
37  wheelSpeeds.rearRight = units::meters_per_second_t{wheelsMatrix(3, 0)};
38  return wheelSpeeds;
39 }

41 ChassisSpeeds MecanumDriveKinematics::ToChassisSpeeds(
42  const MecanumDriveWheelSpeeds& wheelSpeeds) {
43  Eigen::Matrix<double, 4, 1> wheelSpeedsMatrix;
44  // clang-format off
45  wheelSpeedsMatrix << wheelSpeeds.frontLeft.to<double>(), wheelSpeeds.frontRight.to<double>(),
46  wheelSpeeds.rearLeft.to<double>(), wheelSpeeds.rearRight.to<double>();
47  // clang-format on
48 
49  Eigen::Vector3d chassisSpeedsVector =
50  m_forwardKinematics.solve(wheelSpeedsMatrix);
51 
52  return {units::meters_per_second_t{chassisSpeedsVector(0)},
53  units::meters_per_second_t{chassisSpeedsVector(1)},
54  units::radians_per_second_t{chassisSpeedsVector(2)}};
55 }

57 void MecanumDriveKinematics::SetInverseKinematics(Translation2d fl,
58  Translation2d fr,
59  Translation2d rl,
60  Translation2d rr) {
61  // clang-format off
62  m_inverseKinematics << 1, -1, (-(fl.X() + fl.Y())).template to<double>(),
63  1, 1, (fr.X() - fr.Y()).template to<double>(),
64  1, 1, (rl.X() - rl.Y()).template to<double>(),
65  1, -1, (-(rr.X() + rr.Y())).template to<double>();
66  // clang-format on
67  m_inverseKinematics /= std::sqrt(2);
68 }
