9 MecanumDriveWheelSpeeds MecanumDriveKinematics::ToWheelSpeeds(
10  const ChassisSpeeds& chassisSpeeds,
11  const Translation2d& centerOfRotation) const {
12  // We have a new center of rotation. We need to compute the matrix again.
13  if (centerOfRotation != m_previousCoR) {
14  auto fl = m_frontLeftWheel - centerOfRotation;
15  auto fr = m_frontRightWheel - centerOfRotation;
16  auto rl = m_rearLeftWheel - centerOfRotation;
17  auto rr = m_rearRightWheel - centerOfRotation;
18 
19  SetInverseKinematics(fl, fr, rl, rr);
20 
21  m_previousCoR = centerOfRotation;
22  }

24  Eigen::Vector3d chassisSpeedsVector{chassisSpeeds.vx.value(),
25  chassisSpeeds.vy.value(),
26  chassisSpeeds.omega.value()};
27 
28  Eigen::Vector<double, 4> wheelsVector =
29  m_inverseKinematics * chassisSpeedsVector;

40  const MecanumDriveWheelSpeeds& wheelSpeeds) const {

53 void MecanumDriveKinematics::SetInverseKinematics(Translation2d fl,
54  Translation2d fr,
55  Translation2d rl,
56  Translation2d rr) const {
57  m_inverseKinematics =
58  Eigen::Matrix<double, 4, 3>{{1, -1, (-(fl.X() + fl.Y())).value()},
59  {1, 1, (fr.X() - fr.Y()).value()},
60  {1, 1, (rl.X() - rl.Y()).value()},
61  {1, -1, (-(rr.X() + rr.Y())).value()}};
