23 class QuinticHermiteSplineTest : public ::testing::Test {
24  protected:
25  static void Run(const Pose2d& a, const Pose2d& b) {
26  // Start the timer.
27  const auto start = std::chrono::high_resolution_clock::now();
28 
29  // Generate and parameterize the spline.
30  const auto spline = SplineHelper::QuinticSplinesFromControlVectors(
31  SplineHelper::QuinticControlVectorsFromWaypoints({a, b}))[0];
32  const auto poses = SplineParameterizer::Parameterize(spline);
33 
34  // End timer.
35  const auto finish = std::chrono::high_resolution_clock::now();
36 
37  // Calculate the duration (used when benchmarking)
38  const auto duration =
39  std::chrono::duration_cast<std::chrono::microseconds>(finish - start);
40 
41  for (unsigned int i = 0; i < poses.size() - 1; i++) {
42  auto& p0 = poses[i];
43  auto& p1 = poses[i + 1];
44 
45  // Make sure the twist is under the tolerance defined by the Spline class.
46  auto twist = p0.first.Log(p1.first);
47  EXPECT_LT(std::abs(twist.dx.to<double>()),
48  SplineParameterizer::kMaxDx.to<double>());
49  EXPECT_LT(std::abs(twist.dy.to<double>()),
50  SplineParameterizer::kMaxDy.to<double>());
51  EXPECT_LT(std::abs(twist.dtheta.to<double>()),
52  SplineParameterizer::kMaxDtheta.to<double>());
53  }
54 
55  // Check first point.
56  EXPECT_NEAR(poses.front().first.Translation().X().to<double>(),
57  a.Translation().X().to<double>(), 1E-9);
58  EXPECT_NEAR(poses.front().first.Translation().Y().to<double>(),
59  a.Translation().Y().to<double>(), 1E-9);
60  EXPECT_NEAR(poses.front().first.Rotation().Radians().to<double>(),
61  a.Rotation().Radians().to<double>(), 1E-9);
62 
63  // Check last point.
64  EXPECT_NEAR(poses.back().first.Translation().X().to<double>(),
65  b.Translation().X().to<double>(), 1E-9);
66  EXPECT_NEAR(poses.back().first.Translation().Y().to<double>(),
67  b.Translation().Y().to<double>(), 1E-9);
68  EXPECT_NEAR(poses.back().first.Rotation().Radians().to<double>(),
69  b.Rotation().Radians().to<double>(), 1E-9);
70 
71  static_cast<void>(duration);
72  }
73 };
74 } // namespace frc
