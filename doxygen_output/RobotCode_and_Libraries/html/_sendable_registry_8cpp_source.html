175 bool SendableRegistry::Remove(Sendable* sendable) {
176  auto& inst = GetInstance();
177  std::scoped_lock lock(inst.mutex);
178  auto it = inst.componentMap.find(sendable);
179  if (it == inst.componentMap.end()) {
180  return false;
181  }
182  UID compUid = it->getSecond();
183  inst.components.erase(compUid - 1);
184  inst.componentMap.erase(it);
185  // update any parent pointers
186  for (auto&& comp : inst.components) {
187  if (comp->parent == sendable) {
188  comp->parent = nullptr;
189  }
190  }
191  return true;
192 }

194 void SendableRegistry::Move(Sendable* to, Sendable* from) {
195  auto& inst = GetInstance();
196  std::scoped_lock lock(inst.mutex);
197  auto it = inst.componentMap.find(from);
198  if (it == inst.componentMap.end() || !inst.components[it->getSecond() - 1]) {
199  return;
200  }
201  UID compUid = it->getSecond();
202  inst.componentMap.erase(it);
203  inst.componentMap[to] = compUid;
204  auto& comp = *inst.components[compUid - 1];
205  comp.sendable = to;
206  if (comp.builder && comp.builder->IsPublished()) {
207  // rebuild builder, as lambda captures can point to "from"
208  comp.builder->ClearProperties();
209  to->InitSendable(*comp.builder);
210  }
211  // update any parent pointers
212  for (auto&& comp : inst.components) {
213  if (comp->parent == from) {
214  comp->parent = to;
215  }
216  }
217 }

308 std::shared_ptr<void> SendableRegistry::SetData(Sendable* sendable, int handle,

312  std::scoped_lock lock(inst.mutex);
313  auto it = inst.componentMap.find(sendable);
314  if (it == inst.componentMap.end() || !inst.components[it->getSecond() - 1]) {
315  return nullptr;
316  }
317  auto& comp = *inst.components[it->getSecond() - 1];
318  std::shared_ptr<void> rv;
319  if (static_cast<size_t>(handle) < comp.data.size()) {
320  rv = std::move(comp.data[handle]);
321  } else {
322  comp.data.resize(handle + 1);
323  }
324  comp.data[handle] = std::move(data);
325  return rv;
326 }

414 void SendableRegistry::ForeachLiveWindow(
415  int dataHandle, wpi::function_ref<void(CallbackData& data)> callback) {
416  auto& inst = GetInstance();
417  assert(dataHandle >= 0);
418  std::scoped_lock lock(inst.mutex);
419  wpi::SmallVector<Component*, 128> components;
420  for (auto&& comp : inst.components) {
421  components.emplace_back(comp.get());
422  }
423  for (auto comp : components) {
424  if (comp && comp->builder && comp->sendable && comp->liveWindow) {
425  if (static_cast<size_t>(dataHandle) >= comp->data.size()) {
426  comp->data.resize(dataHandle + 1);
427  }
428  CallbackData cbdata{comp->sendable, comp->name,
429  comp->subsystem, comp->parent,
430  comp->data[dataHandle], *comp->builder};
431  callback(cbdata);
432  }
433  }
