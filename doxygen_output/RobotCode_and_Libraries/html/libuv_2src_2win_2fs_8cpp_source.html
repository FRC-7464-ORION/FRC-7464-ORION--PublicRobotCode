149  const char* new_path, const int copy_path) {
150  char* buf;
151  char* pos;
152  ssize_t buf_sz = 0, path_len = 0, pathw_len = 0, new_pathw_len = 0;
153 
154  /* new_path can only be set if path is also set. */
155  assert(new_path == NULL || path != NULL);
156 
157  if (path != NULL) {
158  pathw_len = MultiByteToWideChar(CP_UTF8,
159  0,
160  path,
161  -1,
162  NULL,
163  0);
164  if (pathw_len == 0) {
165  return GetLastError();
166  }
167 
168  buf_sz += pathw_len * sizeof(WCHAR);
169  }
170 
171  if (path != NULL && copy_path) {
172  path_len = 1 + strlen(path);
173  buf_sz += path_len;
174  }
175 
176  if (new_path != NULL) {
177  new_pathw_len = MultiByteToWideChar(CP_UTF8,
178  0,
179  new_path,
180  -1,
181  NULL,
182  0);
183  if (new_pathw_len == 0) {
184  return GetLastError();
185  }
186 
187  buf_sz += new_pathw_len * sizeof(WCHAR);
188  }
189 
190 
191  if (buf_sz == 0) {
192  req->file.pathw = NULL;
193  req->fs.info.new_pathw = NULL;
194  req->path = NULL;
195  return 0;
196  }
197 
198  buf = (char*) uv__malloc(buf_sz);
199  if (buf == NULL) {
200  return ERROR_OUTOFMEMORY;
201  }
202 
203  pos = buf;
204 
205  if (path != NULL) {
206  DWORD r = MultiByteToWideChar(CP_UTF8,
207  0,
208  path,
209  -1,
210  (WCHAR*) pos,
211  pathw_len);
212  assert(r == (DWORD) pathw_len);
213  req->file.pathw = (WCHAR*) pos;
214  pos += r * sizeof(WCHAR);
215  } else {
216  req->file.pathw = NULL;
217  }
218 
219  if (new_path != NULL) {
220  DWORD r = MultiByteToWideChar(CP_UTF8,
221  0,
222  new_path,
223  -1,
224  (WCHAR*) pos,
225  new_pathw_len);
226  assert(r == (DWORD) new_pathw_len);
227  req->fs.info.new_pathw = (WCHAR*) pos;
228  pos += r * sizeof(WCHAR);
229  } else {
230  req->fs.info.new_pathw = NULL;
231  }
232 
233  req->path = path;
234  if (path != NULL && copy_path) {
235  memcpy(pos, path, path_len);
236  assert(path_len == buf_sz - (pos - buf));
237  req->path = pos;
238  }
239 
240  req->flags |= UV_FS_FREE_PATHS;
241 
242  return 0;
243 }

311 INLINE static int fs__readlink_handle(HANDLE handle, char** target_ptr,
312  uint64_t* target_len_ptr) {
313  char buffer[MAXIMUM_REPARSE_DATA_BUFFER_SIZE];
314  REPARSE_DATA_BUFFER* reparse_data = (REPARSE_DATA_BUFFER*) buffer;
315  WCHAR* w_target;
316  DWORD w_target_len;
317  DWORD bytes;
318 
319  if (!DeviceIoControl(handle,
320  FSCTL_GET_REPARSE_POINT,
321  NULL,
322  0,
323  buffer,
324  sizeof buffer,
325  &bytes,
326  NULL)) {
327  return -1;
328  }
329 
330  if (reparse_data->ReparseTag == IO_REPARSE_TAG_SYMLINK) {
331  /* Real symlink */
332  w_target = reparse_data->SymbolicLinkReparseBuffer.PathBuffer +
333  (reparse_data->SymbolicLinkReparseBuffer.SubstituteNameOffset /
334  sizeof(WCHAR));
335  w_target_len =
336  reparse_data->SymbolicLinkReparseBuffer.SubstituteNameLength /
337  sizeof(WCHAR);
338 
339  /* Real symlinks can contain pretty much everything, but the only thing we
340  * really care about is undoing the implicit conversion to an NT namespaced
341  * path that CreateSymbolicLink will perform on absolute paths. If the path
342  * is win32-namespaced then the user must have explicitly made it so, and
343  * we better just return the unmodified reparse data. */
344  if (w_target_len >= 4 &&
345  w_target[0] == L'\\' &&
346  w_target[1] == L'?' &&
347  w_target[2] == L'?' &&
348  w_target[3] == L'\\') {
349  /* Starts with \??\ */
350  if (w_target_len >= 6 &&
351  ((w_target[4] >= L'A' && w_target[4] <= L'Z') ||
352  (w_target[4] >= L'a' && w_target[4] <= L'z')) &&
353  w_target[5] == L':' &&
354  (w_target_len == 6 || w_target[6] == L'\\')) {
355  /* \??<drive>:\ */
356  w_target += 4;
357  w_target_len -= 4;
358 
359  } else if (w_target_len >= 8 &&
360  (w_target[4] == L'U' || w_target[4] == L'u') &&
361  (w_target[5] == L'N' || w_target[5] == L'n') &&
362  (w_target[6] == L'C' || w_target[6] == L'c') &&
363  w_target[7] == L'\\') {
364  /* \??\UNC<server><share>\ - make sure the final path looks like
365  * \<server><share>\ */
366  w_target += 6;
367  w_target[0] = L'\\';
368  w_target_len -= 6;
369  }
370  }
371 
372  } else if (reparse_data->ReparseTag == IO_REPARSE_TAG_MOUNT_POINT) {
373  /* Junction. */
374  w_target = reparse_data->MountPointReparseBuffer.PathBuffer +
375  (reparse_data->MountPointReparseBuffer.SubstituteNameOffset /
376  sizeof(WCHAR));
377  w_target_len = reparse_data->MountPointReparseBuffer.SubstituteNameLength /
378  sizeof(WCHAR);
379 
380  /* Only treat junctions that look like \??<drive>:\ as symlink. Junctions
381  * can also be used as mount points, like \??\Volume{<guid>}, but that's
382  * confusing for programs since they wouldn't be able to actually
383  * understand such a path when returned by uv_readlink(). UNC paths are
384  * never valid for junctions so we don't care about them. */
385  if (!(w_target_len >= 6 &&
386  w_target[0] == L'\\' &&
387  w_target[1] == L'?' &&
388  w_target[2] == L'?' &&
389  w_target[3] == L'\\' &&
390  ((w_target[4] >= L'A' && w_target[4] <= L'Z') ||
391  (w_target[4] >= L'a' && w_target[4] <= L'z')) &&
392  w_target[5] == L':' &&
393  (w_target_len == 6 || w_target[6] == L'\\'))) {
394  SetLastError(ERROR_SYMLINK_NOT_SUPPORTED);
395  return -1;
396  }
397 
398  /* Remove leading \??\ */
399  w_target += 4;
400  w_target_len -= 4;
401 
402  } else {
403  /* Reparse tag does not indicate a symlink. */
404  SetLastError(ERROR_SYMLINK_NOT_SUPPORTED);
405  return -1;
406  }
407 
408  return fs__wide_to_utf8(w_target, w_target_len, target_ptr, target_len_ptr);
409 }

412 void fs__open(uv_fs_t* req) {
413  DWORD access;
414  DWORD share;
415  DWORD disposition;
416  DWORD attributes = 0;
417  HANDLE file;
418  int fd, current_umask;
419  int flags = req->fs.info.file_flags;
420 
421  /* Obtain the active umask. umask() never fails and returns the previous
422  * umask. */
423  current_umask = umask(0);
424  umask(current_umask);
425 
426  /* convert flags and mode to CreateFile parameters */
427  switch (flags & (UV_FS_O_RDONLY | UV_FS_O_WRONLY | UV_FS_O_RDWR)) {
428  case UV_FS_O_RDONLY:
429  access = FILE_GENERIC_READ;
430  break;
431  case UV_FS_O_WRONLY:
432  access = FILE_GENERIC_WRITE;
433  break;
434  case UV_FS_O_RDWR:
435  access = FILE_GENERIC_READ | FILE_GENERIC_WRITE;
436  break;
437  default:
438  goto einval;
439  }
440 
441  if (flags & UV_FS_O_APPEND) {
442  access &= ~FILE_WRITE_DATA;
443  access |= FILE_APPEND_DATA;
444  }
445 
446  /*
447  * Here is where we deviate significantly from what CRT's _open()
448  * does. We indiscriminately use all the sharing modes, to match
449  * UNIX semantics. In particular, this ensures that the file can
450  * be deleted even whilst it's open, fixing issue #1449.
451  * We still support exclusive sharing mode, since it is necessary
452  * for opening raw block devices, otherwise Windows will prevent
453  * any attempt to write past the master boot record.
454  */
455  if (flags & UV_FS_O_EXLOCK) {
456  share = 0;
457  } else {
458  share = FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE;
459  }
460 
461  switch (flags & (UV_FS_O_CREAT | UV_FS_O_EXCL | UV_FS_O_TRUNC)) {
462  case 0:
463  case UV_FS_O_EXCL:
464  disposition = OPEN_EXISTING;
465  break;
466  case UV_FS_O_CREAT:
467  disposition = OPEN_ALWAYS;
468  break;
469  case UV_FS_O_CREAT | UV_FS_O_EXCL:
470  case UV_FS_O_CREAT | UV_FS_O_TRUNC | UV_FS_O_EXCL:
471  disposition = CREATE_NEW;
472  break;
473  case UV_FS_O_TRUNC:
474  case UV_FS_O_TRUNC | UV_FS_O_EXCL:
475  disposition = TRUNCATE_EXISTING;
476  break;
477  case UV_FS_O_CREAT | UV_FS_O_TRUNC:
478  disposition = CREATE_ALWAYS;
479  break;
480  default:
481  goto einval;
482  }
483 
484  attributes |= FILE_ATTRIBUTE_NORMAL;
485  if (flags & UV_FS_O_CREAT) {
486  if (!((req->fs.info.mode & ~current_umask) & _S_IWRITE)) {
487  attributes |= FILE_ATTRIBUTE_READONLY;
488  }
489  }
490 
491  if (flags & UV_FS_O_TEMPORARY ) {
492  attributes |= FILE_FLAG_DELETE_ON_CLOSE | FILE_ATTRIBUTE_TEMPORARY;
493  access |= DELETE;
494  }
495 
496  if (flags & UV_FS_O_SHORT_LIVED) {
497  attributes |= FILE_ATTRIBUTE_TEMPORARY;
498  }
499 
500  switch (flags & (UV_FS_O_SEQUENTIAL | UV_FS_O_RANDOM)) {
501  case 0:
502  break;
503  case UV_FS_O_SEQUENTIAL:
504  attributes |= FILE_FLAG_SEQUENTIAL_SCAN;
505  break;
506  case UV_FS_O_RANDOM:
507  attributes |= FILE_FLAG_RANDOM_ACCESS;
508  break;
509  default:
510  goto einval;
511  }
512 
513  if (flags & UV_FS_O_DIRECT) {
514  /*
515  * FILE_APPEND_DATA and FILE_FLAG_NO_BUFFERING are mutually exclusive.
516  * Windows returns 87, ERROR_INVALID_PARAMETER if these are combined.
517  *
518  * FILE_APPEND_DATA is included in FILE_GENERIC_WRITE:
519  *
520  * FILE_GENERIC_WRITE = STANDARD_RIGHTS_WRITE |
521  * FILE_WRITE_DATA |
522  * FILE_WRITE_ATTRIBUTES |
523  * FILE_WRITE_EA |
524  * FILE_APPEND_DATA |
525  * SYNCHRONIZE
526  *
527  * Note: Appends are also permitted by FILE_WRITE_DATA.
528  *
529  * In order for direct writes and direct appends to succeed, we therefore
530  * exclude FILE_APPEND_DATA if FILE_WRITE_DATA is specified, and otherwise
531  * fail if the user's sole permission is a direct append, since this
532  * particular combination is invalid.
533  */
534  if (access & FILE_APPEND_DATA) {
535  if (access & FILE_WRITE_DATA) {
536  access &= ~FILE_APPEND_DATA;
537  } else {
538  goto einval;
539  }
540  }
541  attributes |= FILE_FLAG_NO_BUFFERING;
542  }
543 
544  switch (flags & (UV_FS_O_DSYNC | UV_FS_O_SYNC)) {
545  case 0:
546  break;
547  case UV_FS_O_DSYNC:
548  case UV_FS_O_SYNC:
549  attributes |= FILE_FLAG_WRITE_THROUGH;
550  break;
551  default:
552  goto einval;
553  }
554 
555  /* Setting this flag makes it possible to open a directory. */
556  attributes |= FILE_FLAG_BACKUP_SEMANTICS;
557 
558  file = CreateFileW(req->file.pathw,
559  access,
560  share,
561  NULL,
562  disposition,
563  attributes,
564  NULL);
565  if (file == INVALID_HANDLE_VALUE) {
566  DWORD error = GetLastError();
567  if (error == ERROR_FILE_EXISTS && (flags & UV_FS_O_CREAT) &&
568  !(flags & UV_FS_O_EXCL)) {
569  /* Special case: when ERROR_FILE_EXISTS happens and UV_FS_O_CREAT was
570  * specified, it means the path referred to a directory. */
571  SET_REQ_UV_ERROR(req, UV_EISDIR, error);
572  } else {
573  SET_REQ_WIN32_ERROR(req, GetLastError());
574  }
575  return;
576  }
577 
578  fd = _open_osfhandle((intptr_t) file, flags);
579  if (fd < 0) {
580  /* The only known failure mode for _open_osfhandle() is EMFILE, in which
581  * case GetLastError() will return zero. However we'll try to handle other
582  * errors as well, should they ever occur.
583  */
584  if (errno == EMFILE)
585  SET_REQ_UV_ERROR(req, UV_EMFILE, ERROR_TOO_MANY_OPEN_FILES);
586  else if (GetLastError() != ERROR_SUCCESS)
587  SET_REQ_WIN32_ERROR(req, GetLastError());
588  else
589  SET_REQ_WIN32_ERROR(req, UV_UNKNOWN);
590  CloseHandle(file);
591  return;
592  }
593 
594  SET_REQ_RESULT(req, fd);
595  return;
596 
597  einval:
598  SET_REQ_UV_ERROR(req, UV_EINVAL, ERROR_INVALID_PARAMETER);
599 }

624 void fs__read(uv_fs_t* req) {
625  int fd = req->file.fd;
626  int64_t offset = req->fs.info.offset;
627  HANDLE handle;
628  OVERLAPPED overlapped, *overlapped_ptr;
629  LARGE_INTEGER offset_;
630  DWORD bytes;
631  DWORD error;
632  int result;
633  unsigned int index;
634  LARGE_INTEGER original_position;
635  LARGE_INTEGER zero_offset;
636  int restore_position;
637 
638  VERIFY_FD(fd, req);
639 
640  zero_offset.QuadPart = 0;
641  restore_position = 0;
642  handle = uv__get_osfhandle(fd);
643 
644  if (handle == INVALID_HANDLE_VALUE) {
645  SET_REQ_WIN32_ERROR(req, ERROR_INVALID_HANDLE);
646  return;
647  }
648 
649  if (offset != -1) {
650  memset(&overlapped, 0, sizeof overlapped);
651  overlapped_ptr = &overlapped;
652  if (SetFilePointerEx(handle, zero_offset, &original_position,
653  FILE_CURRENT)) {
654  restore_position = 1;
655  }
656  } else {
657  overlapped_ptr = NULL;
658  }
659 
660  index = 0;
661  bytes = 0;
662  do {
663  DWORD incremental_bytes;
664 
665  if (offset != -1) {
666  offset_.QuadPart = offset + bytes;
667  overlapped.Offset = offset_.LowPart;
668  overlapped.OffsetHigh = offset_.HighPart;
669  }
670 
671  result = ReadFile(handle,
672  req->fs.info.bufs[index].base,
673  req->fs.info.bufs[index].len,
674  &incremental_bytes,
675  overlapped_ptr);
676  bytes += incremental_bytes;
677  ++index;
678  } while (result && index < req->fs.info.nbufs);
679 
680  if (restore_position)
681  SetFilePointerEx(handle, original_position, NULL, FILE_BEGIN);
682 
683  if (result || bytes > 0) {

921

924  uv__dirent_t** dirents = NULL;

930

932  * It's important that this buffer can hold at least one entry, regardless
933  * of the length of the file names present in the enumerated directory.
934  * A file name is at most 256 WCHARs long.
935  * According to MSDN, the buffer must be aligned at an 8-byte boundary.
936  */
937 #if _MSC_VER
938  __declspec(align(8)) char buffer[8192];
939 #else
940  __attribute__ ((aligned (8))) char buffer[8192];
941 #endif
942 
943  STATIC_ASSERT(sizeof buffer >=
944  sizeof(FILE_DIRECTORY_INFORMATION) + 256 * sizeof(WCHAR));
945 
946  /* Open the directory. */
947  dir_handle =
948  CreateFileW(req->file.pathw,
949  FILE_LIST_DIRECTORY | SYNCHRONIZE,
950  FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
951  NULL,
952  OPEN_EXISTING,
953  FILE_FLAG_BACKUP_SEMANTICS,
954  NULL);
955  if (dir_handle == INVALID_HANDLE_VALUE)
956  goto win32_error;
957 
958  /* Read the first chunk. */
959  status = pNtQueryDirectoryFile(dir_handle,
960  NULL,
961  NULL,

964  &buffer,
965  sizeof buffer,
966  FileDirectoryInformation,
967  FALSE,
968  NULL,
969  TRUE);
970

972  * This should be reported back as UV_ENOTDIR.
973  */
974  if (status == STATUS_INVALID_PARAMETER)
975  goto not_a_directory_error;
976

979  size_t next_entry_offset = 0;
980

982  FILE_DIRECTORY_INFORMATION* info;
983  uv__dirent_t* dirent;
984 
985  size_t wchar_len;
986  size_t utf8_len;
987

989  position += next_entry_offset;
990  info = (FILE_DIRECTORY_INFORMATION*) position;
991 
992  /* Fetch the offset to the next directory entry. */
993  next_entry_offset = info->NextEntryOffset;
994

996  wchar_len = info->FileNameLength / sizeof info->FileName[0];

999  * the SharePoint driver includes the terminating zero byte in

1003  wchar_len -= 1;

1005  if (wchar_len == 0)
1006  continue;
1007  if (wchar_len == 1 && info->FileName[0] == L'.')
1008  continue;
1009  if (wchar_len == 2 && info->FileName[0] == L'.' &&
1010  info->FileName[1] == L'.')
1011  continue;
1012

1014  utf8_len = WideCharToMultiByte(
1015  CP_UTF8, 0, &info->FileName[0], wchar_len, NULL, 0, NULL, NULL);
1016  if (utf8_len == 0)
1017  goto win32_error;
1018 
1019  /* Resize the dirent array if needed. */
1020  if (dirents_used >= dirents_size) {
1021  size_t new_dirents_size =
1022  dirents_size == 0 ? dirents_initial_size : dirents_size << 1;
1023  uv__dirent_t** new_dirents = (uv__dirent_t**)

1025 
1026  if (new_dirents == NULL)
1027  goto out_of_memory_error;

1029  dirents_size = new_dirents_size;
1030  dirents = new_dirents;
1031  }
1032

1034  * includes room for the first character of the filename, but `utf8_len`

1048  &dirent->d_name[0],
1049  utf8_len,

1058  if (info->FileAttributes & FILE_ATTRIBUTE_DEVICE)
1059  dirent->d_type = UV__DT_CHAR;
1060  else if (info->FileAttributes & FILE_ATTRIBUTE_REPARSE_POINT)
1061  dirent->d_type = UV__DT_LINK;
1062  else if (info->FileAttributes & FILE_ATTRIBUTE_DIRECTORY)
1063  dirent->d_type = UV__DT_DIR;
1064  else
1065  dirent->d_type = UV__DT_FILE;
1066  } while (next_entry_offset != 0);
1067

1069  status = pNtQueryDirectoryFile(dir_handle,
1070  NULL,
1071  NULL,
1072  NULL,
1073  &iosb,
1074  &buffer,
1075  sizeof buffer,
1076  FileDirectoryInformation,
1077  FALSE,
1078  NULL,
1079  FALSE);
1080

1082  * STATUS_SUCCESS even if the buffer was too small to hold at least one
1083  * directory entry.
1084  */
1085  if (status == STATUS_SUCCESS && iosb.Information == 0)
1086  status = STATUS_BUFFER_OVERFLOW;
1087  }
1088 
1089  if (status != STATUS_NO_MORE_FILES)
1090  goto nt_error;
1091 
1092  CloseHandle(dir_handle);
1093 
1094  /* Store the result in the request object. */
1095  req->ptr = dirents;
1096  if (dirents != NULL)
1097  req->flags |= UV_FS_FREE_PTR;
1098 
1099  SET_REQ_RESULT(req, dirents_used);
1100 
1101  /* `nbufs` will be used as index by uv_fs_scandir_next. */
1102  req->fs.info.nbufs = 0;
1103 
1104  return;
1105 
1106 nt_error:
1107  SET_REQ_WIN32_ERROR(req, pRtlNtStatusToDosError(status));
1108  goto cleanup;
1109 
1110 win32_error:
1111  SET_REQ_WIN32_ERROR(req, GetLastError());
1112  goto cleanup;
1113 
1114 not_a_directory_error:
1115  SET_REQ_UV_ERROR(req, UV_ENOTDIR, ERROR_DIRECTORY);
1116  goto cleanup;
1117 
1118 out_of_memory_error:
1119  SET_REQ_UV_ERROR(req, UV_ENOMEM, ERROR_OUTOFMEMORY);
1120  goto cleanup;
1121 
1122 cleanup:
1123  if (dir_handle != INVALID_HANDLE_VALUE)
1124  CloseHandle(dir_handle);
1125  while (dirents_used > 0)
1126  uv__free(dirents[--dirents_used]);
1127  if (dirents != NULL)
1128  uv__free(dirents);
1129 }

1266  FILE_FS_VOLUME_INFORMATION volume_info;
1267  NTSTATUS nt_status;
1268  IO_STATUS_BLOCK io_status;
1269 
1270  nt_status = pNtQueryInformationFile(handle,
1271  &io_status,
1272  &file_info,
1273  sizeof file_info,
1274  FileAllInformation);
1275

1277  if (NT_ERROR(nt_status)) {
1278  SetLastError(pRtlNtStatusToDosError(nt_status));
1279  return -1;
1280  }
1281 
1282  nt_status = pNtQueryVolumeInformationFile(handle,
1283  &io_status,
1284  &volume_info,
1285  sizeof volume_info,
1286  FileFsVolumeInformation);
1287 
1288  /* Buffer overflow (a warning status code) is expected here. */
1289  if (io_status.Status == STATUS_NOT_IMPLEMENTED) {
1290  statbuf->st_dev = 0;
1291  } else if (NT_ERROR(nt_status)) {
1292  SetLastError(pRtlNtStatusToDosError(nt_status));
1293  return -1;
1294  } else {
1295  statbuf->st_dev = volume_info.VolumeSerialNumber;
1296  }
1297 
1298  /* Todo: st_mode should probably always be 0666 for everyone. We might also
1299  * want to report 0777 if the file is a .exe or a directory.
1300  *
1301  * Currently it's based on whether the 'readonly' attribute is set, which
1302  * makes little sense because the semantics are so different: the 'read-only'
1303  * flag is just a way for a user to protect against accidental deletion, and
1304  * serves no security purpose. Windows uses ACLs for that.
1305  *
1306  * Also people now use uv_fs_chmod() to take away the writable bit for good
1307  * reasons. Windows however just makes the file read-only, which makes it
1308  * impossible to delete the file afterwards, since read-only files can't be
1309  * deleted.
1310  *
1311  * IOW it's all just a clusterfuck and we should think of something that
1312  * makes slightly more sense.
1313  *
1314  * And uv_fs_chmod should probably just fail on windows or be a total no-op.
1315  * There's nothing sensible it can do anyway.
1316  */
1317  statbuf->st_mode = 0;
1318 
1319  /*
1320  * On Windows, FILE_ATTRIBUTE_REPARSE_POINT is a general purpose mechanism
1321  * by which filesystem drivers can intercept and alter file system requests.
1322  *
1323  * The only reparse points we care about are symlinks and mount points, both
1324  * of which are treated as POSIX symlinks. Further, we only care when
1325  * invoked via lstat, which seeks information about the link instead of its
1326  * target. Otherwise, reparse points must be treated as regular files.
1327  */
1328  if (do_lstat &&
1329  (file_info.BasicInformation.FileAttributes & FILE_ATTRIBUTE_REPARSE_POINT)) {
1330  /*
1331  * If reading the link fails, the reparse point is not a symlink and needs
1332  * to be treated as a regular file. The higher level lstat function will
1333  * detect this failure and retry without do_lstat if appropriate.
1334  */
1335  if (fs__readlink_handle(handle, NULL, &statbuf->st_size) != 0)
1336  return -1;
1337  statbuf->st_mode |= S_IFLNK;
1338  }

1841 static void fs__create_junction(uv_fs_t* req, const WCHAR* path,
1842  const WCHAR* new_path) {
1843  HANDLE handle = INVALID_HANDLE_VALUE;
1844  REPARSE_DATA_BUFFER *buffer = NULL;
1845  int created = 0;
1846  int target_len;
1847  int is_absolute, is_long_path;
1848  int needed_buf_size, used_buf_size, used_data_size, path_buf_len;
1849  int start, len, i;
1850  int add_slash;
1851  DWORD bytes;
1852  WCHAR* path_buf;
1853 
1854  target_len = wcslen(path);
1855  is_long_path = wcsncmp(path, LONG_PATH_PREFIX, LONG_PATH_PREFIX_LEN) == 0;
1856 
1857  if (is_long_path) {
1858  is_absolute = 1;
1859  } else {
1860  is_absolute = target_len >= 3 && IS_LETTER(path[0]) &&
1861  path[1] == L':' && IS_SLASH(path[2]);
1862  }
1863 
1864  if (!is_absolute) {
1865  /* Not supporting relative paths */
1866  SET_REQ_UV_ERROR(req, UV_EINVAL, ERROR_NOT_SUPPORTED);
1867  return;
1868  }
1869 
1870  /* Do a pessimistic calculation of the required buffer size */
1871  needed_buf_size =
1872  FIELD_OFFSET(REPARSE_DATA_BUFFER, MountPointReparseBuffer.PathBuffer) +
1873  JUNCTION_PREFIX_LEN * sizeof(WCHAR) +
1874  2 * (target_len + 2) * sizeof(WCHAR);
1875 
1876  /* Allocate the buffer */
1877  buffer = (REPARSE_DATA_BUFFER*)uv__malloc(needed_buf_size);
1878  if (!buffer) {
1879  uv_fatal_error(ERROR_OUTOFMEMORY, "uv__malloc");
1880  }
1881 
1882  /* Grab a pointer to the part of the buffer where filenames go */
1883  path_buf = (WCHAR*)&(buffer->MountPointReparseBuffer.PathBuffer);
1884  path_buf_len = 0;
1885 
1886  /* Copy the substitute (internal) target path */
1887  start = path_buf_len;
1888 
1889  wcsncpy((WCHAR*)&path_buf[path_buf_len], JUNCTION_PREFIX,
1890  JUNCTION_PREFIX_LEN);
1891  path_buf_len += JUNCTION_PREFIX_LEN;
1892 
1893  add_slash = 0;
1894  for (i = is_long_path ? LONG_PATH_PREFIX_LEN : 0; path[i] != L'\0'; i++) {
1895  if (IS_SLASH(path[i])) {
1896  add_slash = 1;
1897  continue;
1898  }
1899 
1900  if (add_slash) {
1901  path_buf[path_buf_len++] = L'\\';
1902  add_slash = 0;
1903  }
1904 
1905  path_buf[path_buf_len++] = path[i];
1906  }
1907  path_buf[path_buf_len++] = L'\\';
1908  len = path_buf_len - start;
1909 
1910  /* Set the info about the substitute name */
1911  buffer->MountPointReparseBuffer.SubstituteNameOffset = start * sizeof(WCHAR);
1912  buffer->MountPointReparseBuffer.SubstituteNameLength = len * sizeof(WCHAR);
1913 
1914  /* Insert null terminator */
1915  path_buf[path_buf_len++] = L'\0';
1916 
1917  /* Copy the print name of the target path */
1918  start = path_buf_len;
1919  add_slash = 0;
1920  for (i = is_long_path ? LONG_PATH_PREFIX_LEN : 0; path[i] != L'\0'; i++) {
1921  if (IS_SLASH(path[i])) {
1922  add_slash = 1;
1923  continue;
1924  }
1925 
1926  if (add_slash) {
1927  path_buf[path_buf_len++] = L'\\';
1928  add_slash = 0;
1929  }
1930 
1931  path_buf[path_buf_len++] = path[i];
1932  }
1933  len = path_buf_len - start;
1934  if (len == 2) {
1935  path_buf[path_buf_len++] = L'\\';
1936  len++;
1937  }
1938 
1939  /* Set the info about the print name */
1940  buffer->MountPointReparseBuffer.PrintNameOffset = start * sizeof(WCHAR);
1941  buffer->MountPointReparseBuffer.PrintNameLength = len * sizeof(WCHAR);
1942 
1943  /* Insert another null terminator */
1944  path_buf[path_buf_len++] = L'\0';
1945 
1946  /* Calculate how much buffer space was actually used */
1947  used_buf_size = FIELD_OFFSET(REPARSE_DATA_BUFFER, MountPointReparseBuffer.PathBuffer) +
1948  path_buf_len * sizeof(WCHAR);
1949  used_data_size = used_buf_size -
1950  FIELD_OFFSET(REPARSE_DATA_BUFFER, MountPointReparseBuffer);
1951 
1952  /* Put general info in the data buffer */
1953  buffer->ReparseTag = IO_REPARSE_TAG_MOUNT_POINT;
1954  buffer->ReparseDataLength = used_data_size;
1955  buffer->Reserved = 0;
1956 
1957  /* Create a new directory */
1958  if (!CreateDirectoryW(new_path, NULL)) {
1959  SET_REQ_WIN32_ERROR(req, GetLastError());
1960  goto error;
1961  }
1962  created = 1;
1963 
1964  /* Open the directory */
1965  handle = CreateFileW(new_path,
1966  GENERIC_WRITE,
1967  0,
1968  NULL,
1969  OPEN_EXISTING,
1970  FILE_FLAG_BACKUP_SEMANTICS |
1971  FILE_FLAG_OPEN_REPARSE_POINT,
1972  NULL);
1973  if (handle == INVALID_HANDLE_VALUE) {
1974  SET_REQ_WIN32_ERROR(req, GetLastError());
1975  goto error;
1976  }
1977 
1978  /* Create the actual reparse point */
1979  if (!DeviceIoControl(handle,
1980  FSCTL_SET_REPARSE_POINT,
1981  buffer,
1982  used_buf_size,
1983  NULL,
1984  0,
1985  &bytes,
1986  NULL)) {
1987  SET_REQ_WIN32_ERROR(req, GetLastError());
1988  goto error;
1989  }
1990 
1991  /* Clean up */
1992  CloseHandle(handle);
1993  uv__free(buffer);
1994 
1995  SET_REQ_RESULT(req, 0);
1996  return;
1997 
1998 error:
1999  uv__free(buffer);
