38 SparkMaxRelativeEncoder::SparkMaxRelativeEncoder(CANSparkMax& device, Type type,
39  int countsPerRev)
40  : RelativeEncoder(), m_device(&device), m_countsPerRev(countsPerRev) {
41  bool setCountsPerRevParameter = true;
42 
43  if (type == Type::kHallSensor) {
44  // The CPR for a hall sensor is hardcoded in the firmware as 42
45  setCountsPerRevParameter = false;
46 
47  if (m_device->m_motorType == CANSparkMax::MotorType::kBrushed) {
48  throw std::runtime_error(
49  "A hall sensor cannot be used with a brushed motor");
50  }
51 
52  // Make sure that the correct CPR value was provided
53  if (m_countsPerRev != 42 && m_countsPerRev != 0) {
54  throw std::invalid_argument(
55  "countsPerRev must be 42 when using the hall sensor");
56  }
57  } else { // non-hall sensor
58  if (m_device->m_motorType == CANSparkMax::MotorType::kBrushless) {
59  throw std::runtime_error(
60  "The encoder type must be kHallSensor when the SPARK MAX is "
61  "configured in brushless mode.\nTo use an external quadrature "
62  "encoder with a brushless motor, you must wire it as an "
63  "Alternate Encoder, and then call getAlternateEncoder() on the "
64  "CANSparkMax object.");
65  }
66 
67  if (m_countsPerRev < 1) {
68  throw std::invalid_argument(
69  "countsPerRev must be a positive number");
70  }
71  }
72 
73  // Unhandled REVLibError is returned by the c_SparkMax_SetSensorType and
74  // c_SparkMax_SetCPR calls
75  c_SparkMax_SetSensorType(
76  static_cast<c_SparkMax_handle>(m_device->m_sparkMaxHandle),
77  static_cast<c_SparkMax_EncoderType>(type));
78 
79  if (setCountsPerRevParameter) {
80  c_SparkMax_SetCountsPerRevolution(
81  static_cast<c_SparkMax_handle>(m_device->m_sparkMaxHandle),
82  m_countsPerRev);
83  }
84 }
