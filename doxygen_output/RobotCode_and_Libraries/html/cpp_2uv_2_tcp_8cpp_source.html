24 void Tcp::Reuse(std::function<void()> callback, unsigned int flags) {
25  if (IsClosing()) {
26  return;
27  }
28  if (!m_reuseData) {
29  m_reuseData = std::make_unique<ReuseData>();
30  }
31  m_reuseData->callback = std::move(callback);
32  m_reuseData->flags = flags;
33  uv_close(GetRawHandle(), [](uv_handle_t* handle) {
34  Tcp& h = *static_cast<Tcp*>(handle->data);
35  if (!h.m_reuseData) {
36  return; // just in case
37  }
38  auto data = std::move(h.m_reuseData);
39  int err = uv_tcp_init_ex(h.GetLoopRef().GetRaw(), h.GetRaw(), data->flags);
40  if (err < 0) {
41  h.ReportError(err);
42  return;
43  }
44  data->callback();
45  });
46 }

104 void Tcp::Connect(const sockaddr& addr,
105  const std::shared_ptr<TcpConnectReq>& req) {
106  if (Invoke(&uv_tcp_connect, req->GetRaw(), GetRaw(), &addr,
107  [](uv_connect_t* req, int status) {
108  auto& h = *static_cast<TcpConnectReq*>(req->data);
109  if (status < 0) {
110  h.ReportError(status);
111  } else {
112  h.connected();
113  }
114  h.Release(); // this is always a one-shot
115  })) {
116  req->Keep();
117  }
118 }
