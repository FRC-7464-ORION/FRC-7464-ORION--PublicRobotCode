52 // Additional option flags for various dialog constructors
53 enum class opt : uint8_t
54 {
55  none = 0,
56  // For file open, allow multiselect.
57  multiselect = 0x1,
58  // For file save, force overwrite and disable the confirmation dialog.
59  force_overwrite = 0x2,
60  // For folder select, force path to be the provided argument instead
61  // of the last opened directory, which is the Microsoft-recommended,
62  // user-friendly behaviour.
63  force_path = 0x4,
64 };

66 inline opt operator |(opt a, opt b) { return opt(uint8_t(a) | uint8_t(b)); }
67 inline bool operator &(opt a, opt b) { return bool(uint8_t(a) & uint8_t(b)); }
