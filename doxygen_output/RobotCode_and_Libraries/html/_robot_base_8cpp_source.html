74 static void SetupCameraServerShared() {
75 #ifdef __FRC_ROBORIO__
76 #ifdef DYNAMIC_CAMERA_SERVER
77 #ifdef DYNAMIC_CAMERA_SERVER_DEBUG
78  auto cameraServerLib = dlopen("libcameraserverd.so", RTLD_NOW);
79 #else
80  auto cameraServerLib = dlopen("libcameraserver.so", RTLD_NOW);
81 #endif
82 
83  if (!cameraServerLib) {
84  wpi::outs() << "Camera Server Library Not Found\n";
85  wpi::outs().flush();
86  return;
87  }
88  auto symbol = dlsym(cameraServerLib, "CameraServer_SetCameraServerShared");
89  if (symbol) {
90  auto setCameraServerShared = (SetCameraServerSharedFP)symbol;
91  setCameraServerShared(new WPILibCameraServerShared{});
92  } else {
93  wpi::outs() << "Camera Server Shared Symbol Missing\n";
94  wpi::outs().flush();
95  }
96 #else
97  CameraServer_SetCameraServerShared(new WPILibCameraServerShared{});
98 #endif
99 #else
100  wpi::outs() << "Not loading CameraServerShared\n";
101  wpi::outs().flush();
102 #endif
103 }
