175 void SimIO::Run() {
176  /* IO Loop */
177  // Simulate startup delay
178  delayMillis(50);
179 
180  // Default to connected state
181  is_connected = true;
182  notify_sink->ConnectDetected();
183 
184  long sensor_timestamp = 2000; // NOTE: Simulate a 2-second navX-sensor firmware startup delay
185 
186  // Update all static values
187  notify_sink->SetBoardID(board_id);
188  notify_sink->SetRawData(raw_data_update, sensor_timestamp);
189 
190  // Update AHRS data (portions of which are static; others are updated from sim variables)
191  notify_sink->SetAHRSPosData(ahrs_update, sensor_timestamp);
192 
193  start_seconds = Timer::GetFPGATimestamp();
194 
195  while (!stop) {
196  delayMillis(20);
197  sensor_timestamp += 20;
198  UpdatePeriodicFromSimVariables(sensor_timestamp);
199  }
200 }

206 void SimIO::UpdatePeriodicFromSimVariables(long sensor_timestamp) {
207 
208  if (!sim_device) return;
209 
210  bool curr_is_connected = simConnected.Get();
211 
212  // Update connection state
213  if (curr_is_connected != is_connected) {
214  is_connected = curr_is_connected;
215  if (curr_is_connected) {
216  notify_sink->ConnectDetected();
217  } else {
218  notify_sink->DisconnectDetected();
219  }
220  }
221 
222  if (curr_is_connected) {
223  ahrs_update.yaw = (float)simYaw.Get();
224  ahrs_update.pitch = (float)simPitch.Get();
225  ahrs_update.roll = (float)simRoll.Get();
226  ahrs_update.compass_heading = (float)simCompassHeading.Get();
227  ahrs_update.fused_heading = (float)simFusedHeading.Get();
228  ahrs_update.linear_accel_x = (float)simLinearWorldAccelX.Get();
229  ahrs_update.linear_accel_y = (float)simLinearWorldAccelY.Get();
230  ahrs_update.linear_accel_z = (float)simLinearWorldAccelZ.Get();
231 
232  // Detect motion
233  float last_linear_accel_norm = CalculateNormal(last_linear_world_accel_x, last_linear_world_accel_y);
234  float curr_linear_accel_norm = CalculateNormal(ahrs_update.linear_accel_x, ahrs_update.linear_accel_y);
235  bool curr_is_moving = (fabs(curr_linear_accel_norm - last_linear_accel_norm) >= 0.02f);
236  if (curr_is_moving) {
237  board_state.sensor_status |= NAVX_SENSOR_STATUS_MOVING;
238  } else {
239  board_state.sensor_status &= ~(NAVX_SENSOR_STATUS_MOVING);
240  }
241 
242  // Detect rotation
243  bool curr_is_rotating = fabs(last_yaw - ahrs_update.yaw) >= 0.05f;
244  if (!curr_is_rotating) {
245  board_state.sensor_status |= NAVX_SENSOR_STATUS_YAW_STABLE;
246  } else {
247  board_state.sensor_status &= ~(NAVX_SENSOR_STATUS_YAW_STABLE);
248  }
249 
250  ahrs_update.sensor_status = board_state.sensor_status;
251 
252  // Trigger simulated update to notification sink
253  notify_sink->SetAHRSPosData(ahrs_update, sensor_timestamp);
254 
255  // Update cached state variables
256  last_yaw = ahrs_update.yaw;
257  last_linear_world_accel_x = ahrs_update.linear_accel_x;
258  last_linear_world_accel_y = ahrs_update.linear_accel_y;
259  }
260 }
