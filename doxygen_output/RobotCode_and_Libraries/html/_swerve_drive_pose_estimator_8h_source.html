260  template <typename... ModuleState>
261  Pose2d UpdateWithTime(units::second_t currentTime,
262  const Rotation2d& gyroAngle,
263  ModuleState&&... moduleStates) {
264  auto dt = m_prevTime >= 0_s ? currentTime - m_prevTime : m_nominalDt;
265  m_prevTime = currentTime;
266 
267  auto angle = gyroAngle + m_gyroOffset;
268  auto omega = (angle - m_previousAngle).Radians() / dt;
269 
270  auto chassisSpeeds = m_kinematics.ToChassisSpeeds(moduleStates...);
271  auto fieldRelativeSpeeds =
272  Translation2d(chassisSpeeds.vx * 1_s, chassisSpeeds.vy * 1_s)
273  .RotateBy(angle);
274 
275  Eigen::Vector<double, 3> u{fieldRelativeSpeeds.X().value(),
276  fieldRelativeSpeeds.Y().value(), omega.value()};
277 
278  Eigen::Vector<double, 1> localY{angle.Radians().value()};
279  m_previousAngle = angle;
280 
281  m_latencyCompensator.AddObserverState(m_observer, u, localY, currentTime);
282 
283  m_observer.Predict(u, dt);
284  m_observer.Correct(u, localY);
285 
286  return GetEstimatedPosition();
287  }
