103 template <typename R, typename... T>
104 void RunWorkerThreadRequest(WorkerThreadThread<R, T...>& thr,
105  WorkerThreadRequest<R, T...>& req) {
106  R result = std::apply(req.work, std::move(req.params));
107  if (req.afterWork) {
108  if (auto async = thr.m_async.m_async.lock()) {
109  async->Send(std::move(req.afterWork), std::move(result));
110  }
111  } else {
112  thr.m_promises.SetValue(req.promiseId, std::move(result));
113  }
114 }

129 template <typename R, typename... T>
130 void WorkerThreadThread<R, T...>::Main() {
131  std::vector<Request> requests;
132  while (m_active) {
133  std::unique_lock lock(m_mutex);
134  m_cond.wait(lock, [&] { return !m_active || !m_requests.empty(); });
135  if (!m_active) {
136  break;
137  }
138 
139  // don't want to hold the lock while executing the callbacks
140  requests.swap(m_requests);
141  lock.unlock();
142 
143  for (auto&& req : requests) {
144  if (!m_active) {
145  break; // requests may be long-running
146  }
147  RunWorkerThreadRequest(*this, req);
148  }
149  requests.clear();
150  m_promises.Notify();
151  }
152 }

230  template <typename... U>
231  future<R> QueueWork(WorkFunction work, U&&... u) {
232  if (auto thr = m_owner.GetThread()) {
233  // create the future
234  uint64_t req = thr->m_promises.CreateRequest();
235 
236  // add the parameters to the input queue
237  thr->m_requests.emplace_back(

239 
240  // signal the thread

242 
243  // return future
244  return thr->m_promises.CreateFuture(req);
245  }
