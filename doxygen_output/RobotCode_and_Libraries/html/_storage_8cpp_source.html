52 void Storage::ProcessIncoming(std::shared_ptr<Message> msg,
53  INetworkConnection* conn,
54  std::weak_ptr<INetworkConnection> conn_weak) {
55  switch (msg->type()) {
56  case Message::kKeepAlive:
57  break; // ignore
58  case Message::kClientHello:
59  case Message::kProtoUnsup:
60  case Message::kServerHelloDone:
61  case Message::kServerHello:
62  case Message::kClientHelloDone:
63  // shouldn't get these, but ignore if we do
64  break;
65  case Message::kEntryAssign:
66  ProcessIncomingEntryAssign(std::move(msg), conn);
67  break;
68  case Message::kEntryUpdate:
69  ProcessIncomingEntryUpdate(std::move(msg), conn);
70  break;
71  case Message::kFlagsUpdate:
72  ProcessIncomingFlagsUpdate(std::move(msg), conn);
73  break;
74  case Message::kEntryDelete:
75  ProcessIncomingEntryDelete(std::move(msg), conn);
76  break;
77  case Message::kClearEntries:
78  ProcessIncomingClearEntries(std::move(msg), conn);
79  break;
80  case Message::kExecuteRpc:
81  ProcessIncomingExecuteRpc(std::move(msg), conn, std::move(conn_weak));
82  break;
83  case Message::kRpcResponse:
84  ProcessIncomingRpcResponse(std::move(msg), conn);
85  break;
86  default:
87  break;
88  }
89 }

91 void Storage::ProcessIncomingEntryAssign(std::shared_ptr<Message> msg,
92  INetworkConnection* conn) {
93  std::unique_lock lock(m_mutex);
94  unsigned int id = msg->id();
95  std::string_view name = msg->str();
96  Entry* entry;
97  bool may_need_update = false;
98  SequenceNumber seq_num(msg->seq_num_uid());
99  if (m_server) {
100  // if we're a server, id=0xffff requests are requests for an id
101  // to be assigned, and we need to send the new assignment back to
102  // the sender as well as all other connections.
103  if (id == 0xffff) {
104  entry = GetOrNew(name);
105  // see if it was already assigned; ignore if so.
106  if (entry->id != 0xffff) {
107  return;
108  }
109 
110  entry->flags = msg->flags();
111  entry->seq_num = seq_num;
112  SetEntryValueImpl(entry, msg->value(), lock, false);
113  return;
114  }
115  if (id >= m_idmap.size() || !m_idmap[id]) {
116  // ignore arbitrary entry assignments
117  // this can happen due to e.g. assignment to deleted entry
118  lock.unlock();
119  DEBUG0("{}", "server: received assignment to unknown entry");
120  return;
121  }
122  entry = m_idmap[id];
123  } else {
124  // clients simply accept new assignments
125  if (id == 0xffff) {
126  lock.unlock();
127  DEBUG0("{}", "client: received entry assignment request?");
128  return;
129  }
130  if (id >= m_idmap.size()) {
131  m_idmap.resize(id + 1);
132  }
133  entry = m_idmap[id];
134  if (!entry) {
135  // create local
136  entry = GetOrNew(name);
137  entry->id = id;
138  m_idmap[id] = entry;
139  if (!entry->value) {
140  // didn't exist at all (rather than just being a response to a
141  // id assignment request)
142  entry->value = msg->value();
143  entry->flags = msg->flags();
144  entry->seq_num = seq_num;
145 
146  // notify
147  m_notifier.NotifyEntry(entry->local_id, name, entry->value,
148  NT_NOTIFY_NEW);
149  return;
150  }
151  may_need_update = true; // we may need to send an update message
152 
153  // if the received flags don't match what we sent, we most likely
154  // updated flags locally in the interim; send flags update message.
155  if (msg->flags() != entry->flags) {
156  auto dispatcher = m_dispatcher;
157  auto outmsg = Message::FlagsUpdate(id, entry->flags);
158  lock.unlock();
159  dispatcher->QueueOutgoing(outmsg, nullptr, nullptr);
160  lock.lock();
161  }
162  }
163  }
164 
165  // common client and server handling
166 
167  // already exists; ignore if sequence number not higher than local
168  if (seq_num < entry->seq_num) {
169  if (may_need_update) {
170  auto dispatcher = m_dispatcher;
171  auto outmsg =
172  Message::EntryUpdate(entry->id, entry->seq_num.value(), entry->value);
173  lock.unlock();
174  dispatcher->QueueOutgoing(outmsg, nullptr, nullptr);
175  }
176  return;
177  }
178 
179  // sanity check: name should match id
180  if (msg->str() != entry->name) {
181  lock.unlock();
182  DEBUG0("{}", "entry assignment for same id with different name?");
183  return;
184  }
185 
186  unsigned int notify_flags = NT_NOTIFY_UPDATE;
187 
188  // don't update flags from a <3.0 remote (not part of message)
189  // don't update flags if this is a server response to a client id request
190  if (!may_need_update && conn->proto_rev() >= 0x0300) {
191  // update persistent dirty flag if persistent flag changed
192  if ((entry->flags & NT_PERSISTENT) != (msg->flags() & NT_PERSISTENT)) {
193  m_persistent_dirty = true;
194  }
195  if (entry->flags != msg->flags()) {
196  notify_flags |= NT_NOTIFY_FLAGS;
197  }
198  entry->flags = msg->flags();
199  }
200 
201  // update persistent dirty flag if the value changed and it's persistent
202  if (entry->IsPersistent() && *entry->value != *msg->value()) {
203  m_persistent_dirty = true;
204  }
205 
206  // update local
207  entry->value = msg->value();
208  entry->seq_num = seq_num;
209 
210  // notify
211  m_notifier.NotifyEntry(entry->local_id, name, entry->value, notify_flags);
212 
213  // broadcast to all other connections (note for client there won't
214  // be any other connections, so don't bother)
215  if (m_server && m_dispatcher) {
216  auto dispatcher = m_dispatcher;
217  auto outmsg = Message::EntryAssign(entry->name, id, msg->seq_num_uid(),
218  msg->value(), entry->flags);
219  lock.unlock();
220  dispatcher->QueueOutgoing(outmsg, nullptr, conn);
221  }
222 }

225  INetworkConnection* conn) {

328 void Storage::ProcessIncomingExecuteRpc(
329  std::shared_ptr<Message> msg, INetworkConnection* /*conn*/,
330  std::weak_ptr<INetworkConnection> conn_weak) {
331  std::unique_lock lock(m_mutex);
332  if (!m_server) {
333  return; // only process on server
334  }
335  unsigned int id = msg->id();
336  if (id >= m_idmap.size() || !m_idmap[id]) {
337  // ignore call to non-existent RPC
338  // this can happen due to deleted entries
339  lock.unlock();
340  DEBUG0("{}", "received RPC call to unknown entry");
341  return;
342  }
343  Entry* entry = m_idmap[id];
344  if (!entry->value || !entry->value->IsRpc()) {
345  lock.unlock();
346  DEBUG0("{}", "received RPC call to non-RPC entry");
347  return;
348  }
349  ConnectionInfo conn_info;
350  auto c = conn_weak.lock();
351  if (c) {
352  conn_info = c->info();
353  } else {
354  conn_info.remote_id = "";
355  conn_info.remote_ip = "";
356  conn_info.remote_port = 0;
357  conn_info.last_update = 0;
358  conn_info.protocol_version = 0;
359  }
360  unsigned int call_uid = msg->seq_num_uid();
361  m_rpc_server.ProcessRpc(
362  entry->local_id, call_uid, entry->name, msg->str(), conn_info,
363  [=](std::string_view result) {
364  auto c = conn_weak.lock();
365  if (c) {
366  c->QueueOutgoing(Message::RpcResponse(id, call_uid, result));
367  }
368  },
369  entry->rpc_uid);
370 }

412 void Storage::ApplyInitialAssignments(
413  INetworkConnection& conn, wpi::span<std::shared_ptr<Message>> msgs,
414  bool /*new_server*/, std::vector<std::shared_ptr<Message>>* out_msgs) {
415  std::unique_lock lock(m_mutex);
416  if (m_server) {
417  return; // should not do this on server
418  }
419 
420  conn.set_state(INetworkConnection::kSynchronized);
421 
422  std::vector<std::shared_ptr<Message>> update_msgs;
423 
424  // clear existing id's
425  for (auto& i : m_entries) {
426  i.getValue()->id = 0xffff;
427  }
428 
429  // clear existing idmap
430  m_idmap.resize(0);
431 
432  // apply assignments
433  for (auto& msg : msgs) {
434  if (!msg->Is(Message::kEntryAssign)) {
435  DEBUG0("{}", "client: received non-entry assignment request?");
436  continue;
437  }
438 
439  unsigned int id = msg->id();
440  if (id == 0xffff) {
441  DEBUG0("{}", "client: received entry assignment request?");
442  continue;
443  }
444 
445  SequenceNumber seq_num(msg->seq_num_uid());
446  std::string_view name = msg->str();
447 
448  Entry* entry = GetOrNew(name);
449  entry->seq_num = seq_num;
450  entry->id = id;
451  if (!entry->value) {
452  // doesn't currently exist
453  entry->value = msg->value();
454  entry->flags = msg->flags();
455  // notify
456  m_notifier.NotifyEntry(entry->local_id, name, entry->value,
457  NT_NOTIFY_NEW);
458  } else {
459  // if we have written the value locally and the value is not persistent,
460  // then we don't update the local value and instead send it back to the
461  // server as an update message
462  if (entry->local_write && !entry->IsPersistent()) {
463  ++entry->seq_num;
464  update_msgs.emplace_back(Message::EntryUpdate(
465  entry->id, entry->seq_num.value(), entry->value));
466  } else {
467  entry->value = msg->value();
468  unsigned int notify_flags = NT_NOTIFY_UPDATE;
469  // don't update flags from a <3.0 remote (not part of message)
470  if (conn.proto_rev() >= 0x0300) {
471  if (entry->flags != msg->flags()) {
472  notify_flags |= NT_NOTIFY_FLAGS;
473  }
474  entry->flags = msg->flags();
475  }
476  // notify
477  m_notifier.NotifyEntry(entry->local_id, name, entry->value,
478  notify_flags);
479  }
480  }
481 
482  // save to idmap
483  if (id >= m_idmap.size()) {
484  m_idmap.resize(id + 1);
485  }
486  m_idmap[id] = entry;
487  }
488 
489  // delete or generate assign messages for unassigned local entries
490  DeleteAllEntriesImpl(false, [&](Entry* entry) -> bool {
491  // was assigned by the server, don't delete
492  if (entry->id != 0xffff) {
493  return false;
494  }
495  // if we have written the value locally, we send an assign message to the
496  // server instead of deleting
497  if (entry->local_write) {
498  out_msgs->emplace_back(Message::EntryAssign(entry->name, entry->id,
499  entry->seq_num.value(),
500  entry->value, entry->flags));
501  return false;
502  }
503  // otherwise delete
504  return true;
505  });
506  auto dispatcher = m_dispatcher;
507  lock.unlock();
508  for (auto& msg : update_msgs) {
509  dispatcher->QueueOutgoing(msg, nullptr, nullptr);
510  }
511 }

608 void Storage::SetEntryValueImpl(Entry* entry, std::shared_ptr<Value> value,
609  std::unique_lock<wpi::mutex>& lock,
610  bool local) {
611  if (!value) {
612  return;
613  }
614  auto old_value = entry->value;
615  entry->value = value;
616 
617  // if we're the server, assign an id if it doesn't have one
618  if (m_server && entry->id == 0xffff) {
619  unsigned int id = m_idmap.size();
620  entry->id = id;
621  m_idmap.push_back(entry);
622  }
623 
624  // update persistent dirty flag if value changed and it's persistent
625  if (entry->IsPersistent() && (!old_value || *old_value != *value)) {
626  m_persistent_dirty = true;
627  }
628 
629  // notify
630  if (!old_value) {
631  m_notifier.NotifyEntry(entry->local_id, entry->name, value,
632  NT_NOTIFY_NEW | (local ? NT_NOTIFY_LOCAL : 0));
633  } else if (*old_value != *value) {
634  m_notifier.NotifyEntry(entry->local_id, entry->name, value,
635  NT_NOTIFY_UPDATE | (local ? NT_NOTIFY_LOCAL : 0));
636  }
637 
638  // remember local changes
639  if (local) {
640  entry->local_write = true;
641  }
642 
643  // generate message
644  if (!m_dispatcher || (!local && !m_server)) {
645  return;
646  }
647  auto dispatcher = m_dispatcher;
648  if (!old_value || old_value->type() != value->type()) {
649  if (local) {
650  ++entry->seq_num;
651  }
652  auto msg = Message::EntryAssign(
653  entry->name, entry->id, entry->seq_num.value(), value, entry->flags);
654  lock.unlock();
655  dispatcher->QueueOutgoing(msg, nullptr, nullptr);
656  } else if (*old_value != *value) {
657  if (local) {
658  ++entry->seq_num;
659  }
660  // don't send an update if we don't have an assigned id yet
661  if (entry->id != 0xffff) {
662  auto msg = Message::EntryUpdate(entry->id, entry->seq_num.value(), value);
663  lock.unlock();
664  dispatcher->QueueOutgoing(msg, nullptr, nullptr);
665  }
666  }
667 }

1232  std::string* result, double timeout,
1233  bool* timed_out) {

1259  *timed_out = true;

1265  return false;
