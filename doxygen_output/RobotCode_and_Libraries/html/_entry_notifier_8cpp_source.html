26 bool impl::EntryNotifierThread::Matches(const EntryListenerData& listener,
27  const EntryNotification& data) {
28  if (!data.value) {
29  return false;
30  }
31 
32  // Flags must be within requested flag set for this listener.
33  // Because assign messages can result in both a value and flags update,
34  // we handle that case specially.
35  unsigned int listen_flags =
36  listener.flags & ~(NT_NOTIFY_IMMEDIATE | NT_NOTIFY_LOCAL);
37  unsigned int flags = data.flags & ~(NT_NOTIFY_IMMEDIATE | NT_NOTIFY_LOCAL);
38  unsigned int assign_both = NT_NOTIFY_UPDATE | NT_NOTIFY_FLAGS;
39  if ((flags & assign_both) == assign_both) {
40  if ((listen_flags & assign_both) == 0) {
41  return false;
42  }
43  listen_flags &= ~assign_both;
44  flags &= ~assign_both;
45  }
46  if ((flags & ~listen_flags) != 0) {
47  return false;
48  }
49 
50  // must match local id or prefix
51  if (listener.entry != 0 && data.entry != listener.entry) {
52  return false;
53  }
54  if (listener.entry == 0 && !wpi::starts_with(data.name, listener.prefix)) {
55  return false;
56  }
57 
58  return true;
59 }

97 void EntryNotifier::NotifyEntry(unsigned int local_id, std::string_view name,
98  std::shared_ptr<Value> value,
99  unsigned int flags,
100  unsigned int only_listener) {
101  // optimization: don't generate needless local queue entries if we have
102  // no local listeners (as this is a common case on the server side)
103  if ((flags & NT_NOTIFY_LOCAL) != 0 && !m_local_notifiers) {
104  return;
105  }
106  DEBUG0("notifying '{}' (local={}), flags={}", name, local_id, flags);
107  Send(only_listener, 0, Handle(m_inst, local_id, Handle::kEntry).handle(),
108  name, value, flags);
