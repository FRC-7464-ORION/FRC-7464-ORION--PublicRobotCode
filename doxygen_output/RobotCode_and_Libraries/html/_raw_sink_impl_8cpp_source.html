43 uint64_t RawSinkImpl::GrabFrame(CS_RawFrame& image) {
44  SetEnabled(true);
45 
46  auto source = GetSource();
47  if (!source) {
48  // Source disconnected; sleep for one second
49  std::this_thread::sleep_for(std::chrono::seconds(1));
50  return 0;
51  }
52 
53  auto frame = source->GetNextFrame(); // blocks
54  if (!frame) {
55  // Bad frame; sleep for 20 ms so we don't consume all processor time.
56  std::this_thread::sleep_for(std::chrono::milliseconds(20));
57  return 0; // signal error
58  }
59 
60  return GrabFrameImpl(image, frame);
61 }

83 uint64_t RawSinkImpl::GrabFrameImpl(CS_RawFrame& rawFrame,
84  Frame& incomingFrame) {
85  Image* newImage = nullptr;
86 
87  if (rawFrame.pixelFormat == CS_PixelFormat::CS_PIXFMT_UNKNOWN) {
88  // Always get incoming image directly on unknown
89  newImage = incomingFrame.GetExistingImage(0);
90  } else {
91  // Format is known, ask for it
92  auto width = rawFrame.width;
93  auto height = rawFrame.height;
94  auto pixelFormat =
95  static_cast<VideoMode::PixelFormat>(rawFrame.pixelFormat);
96  if (width <= 0 || height <= 0) {
97  width = incomingFrame.GetOriginalWidth();
98  height = incomingFrame.GetOriginalHeight();
99  }
100  newImage = incomingFrame.GetImage(width, height, pixelFormat);
101  }
102 
103  if (!newImage) {
104  // Shouldn't happen, but just in case...
105  std::this_thread::sleep_for(std::chrono::milliseconds(20));
106  return 0;
107  }
108 
109  CS_AllocateRawFrameData(&rawFrame, newImage->size());
110  rawFrame.height = newImage->height;
111  rawFrame.width = newImage->width;
112  rawFrame.pixelFormat = newImage->pixelFormat;
113  rawFrame.totalData = newImage->size();
114  std::copy(newImage->data(), newImage->data() + rawFrame.totalData,
115  rawFrame.data);
116 
117  return incomingFrame.GetTime();
118 }

121 void RawSinkImpl::ThreadMain() {
122  Enable();
123  while (m_active) {
124  auto source = GetSource();
125  if (!source) {
126  // Source disconnected; sleep for one second
127  std::this_thread::sleep_for(std::chrono::seconds(1));
128  continue;
129  }
130  SDEBUG4("{}", "waiting for frame");
131  Frame frame = source->GetNextFrame(); // blocks
132  if (!m_active) {
133  break;
134  }
135  if (!frame) {
136  // Bad frame; sleep for 10 ms so we don't consume all processor time.
137  std::this_thread::sleep_for(std::chrono::milliseconds(10));
138  continue;
139  }
140  // TODO m_processFrame();
141  }
142  Disable();
143 }
