22 DigitalOutput::DigitalOutput(int channel) {
23  m_pwmGenerator = HAL_kInvalidHandle;
24  if (!SensorUtil::CheckDigitalChannel(channel)) {
25  throw FRC_MakeError(err::ChannelIndexOutOfRange, "Channel {}", channel);
26  }
27  m_channel = channel;
28 
29  int32_t status = 0;
30  std::string stackTrace = wpi::GetStackTrace(1);
31  m_handle = HAL_InitializeDIOPort(HAL_GetPort(channel), false,
32  stackTrace.c_str(), &status);
33  FRC_CheckErrorStatus(status, "Channel {}", channel);
34 
35  HAL_Report(HALUsageReporting::kResourceType_DigitalOutput, channel + 1);
36  wpi::SendableRegistry::AddLW(this, "DigitalOutput", channel);
37 }

98 void DigitalOutput::EnablePWM(double initialDutyCycle) {
99  if (m_pwmGenerator != HAL_kInvalidHandle) {
100  return;
101  }
102 
103  int32_t status = 0;
104 
105  m_pwmGenerator = HAL_AllocateDigitalPWM(&status);
106  FRC_CheckErrorStatus(status, "Channel {}", m_channel);
107 
108  HAL_SetDigitalPWMDutyCycle(m_pwmGenerator, initialDutyCycle, &status);
109  FRC_CheckErrorStatus(status, "Channel {}", m_channel);
110 
111  HAL_SetDigitalPWMOutputChannel(m_pwmGenerator, m_channel, &status);
112  FRC_CheckErrorStatus(status, "Channel {}", m_channel);
113 }

115 void DigitalOutput::DisablePWM() {
116  if (m_pwmGenerator == HAL_kInvalidHandle) {
117  return;
118  }
119 
120  int32_t status = 0;
121 
122  // Disable the output by routing to a dead bit.
123  HAL_SetDigitalPWMOutputChannel(m_pwmGenerator, SensorUtil::kDigitalChannels,
124  &status);
125  FRC_CheckErrorStatus(status, "Channel {}", m_channel);
126 
127  HAL_FreeDigitalPWM(m_pwmGenerator, &status);
128  FRC_CheckErrorStatus(status, "Channel {}", m_channel);
129 
130  m_pwmGenerator = HAL_kInvalidHandle;
131 }
