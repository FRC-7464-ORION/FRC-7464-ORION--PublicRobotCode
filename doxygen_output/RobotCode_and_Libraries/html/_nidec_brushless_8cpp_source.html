14 NidecBrushless::NidecBrushless(int pwmChannel, int dioChannel)
15  : m_dio(dioChannel), m_pwm(pwmChannel) {
16  wpi::SendableRegistry::AddChild(this, &m_dio);
17  wpi::SendableRegistry::AddChild(this, &m_pwm);
18  SetExpiration(0_s);
19  SetSafetyEnabled(false);

24 
25  HAL_Report(HALUsageReporting::kResourceType_NidecBrushless, pwmChannel + 1);

27 }

29 void NidecBrushless::Set(double speed) {

36 }
