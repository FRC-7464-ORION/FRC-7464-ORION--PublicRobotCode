13 std::shared_ptr<Process> Process::SpawnArray(Loop& loop, std::string_view file,
14  span<const Option> options) {
15  // convert Option array to libuv structure
16  uv_process_options_t coptions;
17 
18  coptions.exit_cb = [](uv_process_t* handle, int64_t status, int signal) {
19  auto& h = *static_cast<Process*>(handle->data);
20  h.exited(status, signal);
21  };
22 
23  SmallString<128> fileBuf{file};
24  coptions.file = fileBuf.c_str();
25  coptions.cwd = nullptr;
26  coptions.flags = 0;
27  coptions.uid = 0;
28  coptions.gid = 0;
29 
30  SmallVector<char*, 4> argsBuf;
31  SmallVector<char*, 4> envBuf;
32  struct StdioContainer : public uv_stdio_container_t {
33  StdioContainer() {
34  flags = UV_IGNORE;
35  data.fd = 0;
36  }
37  };
38  SmallVector<StdioContainer, 4> stdioBuf;
39 
40  for (auto&& o : options) {
41  switch (o.m_type) {
42  case Option::kArg:
43  argsBuf.push_back(const_cast<char*>(o.m_data.str));
44  break;
45  case Option::kEnv:
46  envBuf.push_back(const_cast<char*>(o.m_data.str));
47  break;
48  case Option::kCwd:
49  coptions.cwd = o.m_data.str[0] == '\0' ? nullptr : o.m_data.str;
50  break;
51  case Option::kUid:
52  coptions.uid = o.m_data.uid;
53  coptions.flags |= UV_PROCESS_SETUID;
54  break;
55  case Option::kGid:
56  coptions.gid = o.m_data.gid;
57  coptions.flags |= UV_PROCESS_SETGID;
58  break;
59  case Option::kSetFlags:
60  coptions.flags |= o.m_data.flags;
61  break;
62  case Option::kClearFlags:
63  coptions.flags &= ~o.m_data.flags;
64  break;
65  case Option::kStdioIgnore: {
66  size_t index = o.m_data.stdio.index;
67  if (index >= stdioBuf.size()) {
68  stdioBuf.resize(index + 1);
69  }
70  stdioBuf[index].flags = UV_IGNORE;
71  stdioBuf[index].data.fd = 0;
72  break;
73  }
74  case Option::kStdioInheritFd: {
75  size_t index = o.m_data.stdio.index;
76  if (index >= stdioBuf.size()) {
77  stdioBuf.resize(index + 1);
78  }
79  stdioBuf[index].flags = UV_INHERIT_FD;
80  stdioBuf[index].data.fd = o.m_data.stdio.fd;
81  break;
82  }
83  case Option::kStdioInheritPipe: {
84  size_t index = o.m_data.stdio.index;
85  if (index >= stdioBuf.size()) {
86  stdioBuf.resize(index + 1);
87  }
88  stdioBuf[index].flags = UV_INHERIT_STREAM;
89  stdioBuf[index].data.stream = o.m_data.stdio.pipe->GetRawStream();
90  break;
91  }
92  case Option::kStdioCreatePipe: {
93  size_t index = o.m_data.stdio.index;
94  if (index >= stdioBuf.size()) {
95  stdioBuf.resize(index + 1);
96  }
97  stdioBuf[index].flags =
98  static_cast<uv_stdio_flags>(UV_CREATE_PIPE | o.m_data.stdio.flags);
99  stdioBuf[index].data.stream = o.m_data.stdio.pipe->GetRawStream();
100  break;
101  }
102  default:
103  break;
104  }
105  }
106 
107  if (argsBuf.empty()) {
108  argsBuf.push_back(const_cast<char*>(coptions.file));
109  }
110  argsBuf.push_back(nullptr);
111  coptions.args = argsBuf.data();
112 
113  if (envBuf.empty()) {
114  coptions.env = nullptr;
115  } else {
116  envBuf.push_back(nullptr);
117  coptions.env = envBuf.data();
118  }
119 
120  coptions.stdio_count = stdioBuf.size();
121  coptions.stdio = static_cast<uv_stdio_container_t*>(stdioBuf.data());
122 
123  auto h = std::make_shared<Process>(private_init{});
124  int err = uv_spawn(loop.GetRaw(), h->GetRaw(), &coptions);
125  if (err < 0) {
126  loop.ReportError(err);
127  return nullptr;
128  }
129  h->Keep();
130  return h;
131 }
132
