72 ADIS16448_IMU::ADIS16448_IMU(IMUAxis yaw_axis, SPI::Port port,
73  CalibrationTime cal_time)
74  : m_yaw_axis(yaw_axis),
75  m_spi_port(port),
76  m_simDevice("Gyro:ADIS16448", port) {
77  if (m_simDevice) {
78  m_simGyroAngleX =
79  m_simDevice.CreateDouble("gyro_angle_x", hal::SimDevice::kInput, 0.0);
80  m_simGyroAngleY =
81  m_simDevice.CreateDouble("gyro_angle_y", hal::SimDevice::kInput, 0.0);
82  m_simGyroAngleZ =
83  m_simDevice.CreateDouble("gyro_angle_z", hal::SimDevice::kInput, 0.0);
84  m_simGyroRateX =
85  m_simDevice.CreateDouble("gyro_rate_x", hal::SimDevice::kInput, 0.0);
86  m_simGyroRateY =
87  m_simDevice.CreateDouble("gyro_rate_y", hal::SimDevice::kInput, 0.0);
88  m_simGyroRateZ =
89  m_simDevice.CreateDouble("gyro_rate_z", hal::SimDevice::kInput, 0.0);
90  m_simAccelX =
91  m_simDevice.CreateDouble("accel_x", hal::SimDevice::kInput, 0.0);
92  m_simAccelY =
93  m_simDevice.CreateDouble("accel_y", hal::SimDevice::kInput, 0.0);
94  m_simAccelZ =
95  m_simDevice.CreateDouble("accel_z", hal::SimDevice::kInput, 0.0);
96  }
97 
98  if (!m_simDevice) {
99  // Force the IMU reset pin to toggle on startup (doesn't require DS enable)
100  // Relies on the RIO hardware by default configuring an output as low
101  // and configuring an input as high Z. The 10k pull-up resistor internal to
102  // the IMU then forces the reset line high for normal operation.
103  DigitalOutput* m_reset_out = new DigitalOutput(18); // Drive MXP DIO8 low
104  Wait(10_ms); // Wait 10ms
105  delete m_reset_out;
106  new DigitalInput(18); // Set MXP DIO8 high
107  Wait(500_ms); // Wait 500ms for reset to complete
108 
109  ConfigCalTime(cal_time);
110 
111  // Configure standard SPI
112  if (!SwitchToStandardSPI()) {
113  return;
114  }
115 
116  // Set IMU internal decimation to 819.2 SPS
117  WriteRegister(SMPL_PRD, 0x0001);
118  // Enable Data Ready (LOW = Good Data) on DIO1 (PWM0 on MXP)
119  WriteRegister(MSC_CTRL, 0x0016);
120  // Disable IMU internal Bartlett filter
121  WriteRegister(SENS_AVG, 0x0400);
122  // Clear offset registers
123  WriteRegister(XGYRO_OFF, 0x0000);
124  WriteRegister(YGYRO_OFF, 0x0000);
125  WriteRegister(ZGYRO_OFF, 0x0000);
126  // Configure and enable auto SPI
127  if (!SwitchToAutoSPI()) {
128  return;
129  }
130  // Notify DS that IMU calibration delay is active
131  REPORT_WARNING(
132  "ADIS16448 IMU Detected. Starting initial calibration delay.");
133  // Wait for whatever time the user set as the start-up delay
134  Wait(static_cast<double>(m_calibration_time) * 1.2_s);
135  // Execute calibration routine
136  Calibrate();
137  // Reset accumulated offsets
138  Reset();
139  // Tell the acquire loop that we're done starting up
140  m_start_up_mode = false;
141 
142  // Let the user know the IMU was initiallized successfully
143  REPORT_WARNING("ADIS16448 IMU Successfully Initialized!");
144 
145  // TODO: Find what the proper pin is to turn this LED
146  // Drive MXP PWM5 (IMU ready LED) low (active low)
147  new DigitalOutput(19);
148  }
149 
150  // Report usage and post data to DS
151  HAL_Report(HALUsageReporting::kResourceType_ADIS16448, 0);
152 
153  wpi::SendableRegistry::AddLW(this, "ADIS16448", port);
154 }

169 bool ADIS16448_IMU::SwitchToStandardSPI() {
170  // Check to see whether the acquire thread is active. If so, wait for it to
171  // stop producing data.
172  if (m_thread_active) {
173  m_thread_active = false;
174  while (!m_thread_idle) {
175  Wait(10_ms);
176  }
177  std::cout << "Paused the IMU processing thread successfully!" << std::endl;
178  // Maybe we're in auto SPI mode? If so, kill auto SPI, and then SPI.
179  if (m_spi != nullptr && m_auto_configured) {
180  m_spi->StopAuto();
181  // We need to get rid of all the garbage left in the auto SPI buffer after
182  // stopping it. Sometimes data magically reappears, so we have to check
183  // the buffer size a couple of times
184  // to be sure we got it all. Yuck.
185  uint32_t trashBuffer[200];
186  Wait(100_ms);
187  int data_count = m_spi->ReadAutoReceivedData(trashBuffer, 0, 0_s);
188  while (data_count > 0) {
189  /* Dequeue 200 at a time, or the remainder of the buffer if less than
190  * 200 */
191  m_spi->ReadAutoReceivedData(trashBuffer, std::min(200, data_count),
192  0_s);
193  /* Update remaining buffer count */
194  data_count = m_spi->ReadAutoReceivedData(trashBuffer, 0, 0_s);
195  }
196  std::cout << "Paused the auto SPI successfully!" << std::endl;
197  }
198  }
199  // There doesn't seem to be a SPI port active. Let's try to set one up
200  if (m_spi == nullptr) {
201  std::cout << "Setting up a new SPI port." << std::endl;
202  m_spi = new SPI(m_spi_port);
203  m_spi->SetClockRate(1000000);
204  m_spi->SetMSBFirst();
205  m_spi->SetSampleDataOnTrailingEdge();
206  m_spi->SetClockActiveLow();
207  m_spi->SetChipSelectActiveLow();
208  ReadRegister(PROD_ID); // Dummy read
209 
210  // Validate the product ID
211  uint16_t prod_id = ReadRegister(PROD_ID);
212  if (prod_id != 16448) {
213  REPORT_ERROR("Could not find ADIS16448!");
214  Close();
215  return false;
216  }
217  return true;
218  } else {
219  // Maybe the SPI port is active, but not in auto SPI mode? Try to read the
220  // product ID.
221  ReadRegister(PROD_ID); // Dummy read
222  uint16_t prod_id = ReadRegister(PROD_ID);
223  if (prod_id != 16448) {
224  REPORT_ERROR("Could not find ADIS16448!");
225  Close();
226  return false;
227  } else {
228  return true;
229  }
230  }

262 bool ADIS16448_IMU::SwitchToAutoSPI() {
263  // No SPI port has been set up. Go set one up first.
264  if (m_spi == nullptr) {
265  if (!SwitchToStandardSPI()) {
266  REPORT_ERROR("Failed to start/restart auto SPI");
267  return false;
268  }
269  }
270  // Only set up the interrupt if needed.
271  if (m_auto_interrupt == nullptr) {
272  m_auto_interrupt = new DigitalInput(10);
273  }
274  // The auto SPI controller gets angry if you try to set up two instances on
275  // one bus.
276  if (!m_auto_configured) {
277  m_spi->InitAuto(8200);
278  m_auto_configured = true;
279  }
280  // Set auto SPI packet data and size
281  m_spi->SetAutoTransmitData({{GLOB_CMD}}, 27);
282  // Configure auto stall time
283  m_spi->ConfigureAutoStall(HAL_SPI_kMXP, 100, 1000, 255);
284  // Kick off DMA SPI (Note: Device configration impossible after SPI DMA is
285  // activated)
286  m_spi->StartAutoTrigger(*m_auto_interrupt, true, false);
287  // Check to see if the acquire thread is running. If not, kick one off.
288  if (!m_thread_idle) {
289  m_first_run = true;
290  m_thread_active = true;
291  // Set up circular buffer
292  InitOffsetBuffer(m_avg_size);
293  // Kick off acquire thread
294  m_acquire_task = std::thread(&ADIS16448_IMU::Acquire, this);
295  std::cout << "New IMU Processing thread activated!" << std::endl;
296  } else {
297  m_first_run = true;
298  m_thread_active = true;
299  std::cout << "Old IMU Processing thread re-activated!" << std::endl;
300  }
301  // Looks like the thread didn't start for some reason. Abort.
302  /*
303  if(!m_thread_idle) {
304  REPORT_ERROR("Failed to start/restart the acquire() thread.");
305  Close();
306  return false;
307  }
308  */
309  return true;
310 }

329 void ADIS16448_IMU::Calibrate() {
330  std::scoped_lock sync(m_mutex);
331  // Calculate the running average
332  int gyroAverageSize = std::min(m_accum_count, m_avg_size);
333  double accum_gyro_rate_x = 0.0;
334  double accum_gyro_rate_y = 0.0;
335  double accum_gyro_rate_z = 0.0;
336  for (int i = 0; i < gyroAverageSize; i++) {
337  accum_gyro_rate_x += m_offset_buffer[i].gyro_rate_x;
338  accum_gyro_rate_y += m_offset_buffer[i].gyro_rate_y;
339  accum_gyro_rate_z += m_offset_buffer[i].gyro_rate_z;
340  }
341  m_gyro_rate_offset_x = accum_gyro_rate_x / gyroAverageSize;
342  m_gyro_rate_offset_y = accum_gyro_rate_y / gyroAverageSize;
343  m_gyro_rate_offset_z = accum_gyro_rate_z / gyroAverageSize;
344  m_integ_gyro_angle_x = 0.0;
345  m_integ_gyro_angle_y = 0.0;
346  m_integ_gyro_angle_z = 0.0;
347  // std::cout << "Avg Size: " << gyroAverageSize << " X off: " <<
348  // m_gyro_rate_offset_x << " Y off: " << m_gyro_rate_offset_y << " Z off: " <<
349  // m_gyro_rate_offset_z << std::endl;
350 }

359 uint16_t ADIS16448_IMU::ReadRegister(uint8_t reg) {
360  uint8_t buf[2];
361  buf[0] = reg & 0x7f;
362  buf[1] = 0;
363 
364  m_spi->Write(buf, 2);

366

377 void ADIS16448_IMU::WriteRegister(uint8_t reg, uint16_t val) {
378  uint8_t buf[2];
379  buf[0] = 0x80 | reg;
380  buf[1] = val & 0xff;
381  m_spi->Write(buf, 2);
382  buf[0] = 0x81 | reg;
383  buf[1] = val >> 8;
384  m_spi->Write(buf, 2);

398 void ADIS16448_IMU::Close() {
399  if (m_thread_active) {
400  m_thread_active = false;
401  if (m_acquire_task.joinable()) {
402  m_acquire_task.join();
403  }
404  }
405  if (m_spi != nullptr) {
406  if (m_auto_configured) {
407  m_spi->StopAuto();
408  }
409  delete m_spi;
410  m_auto_configured = false;
411  if (m_auto_interrupt != nullptr) {
412  delete m_auto_interrupt;
413  m_auto_interrupt = nullptr;
414  }
415  m_spi = nullptr;
416  }
417  delete[] m_offset_buffer;
418  std::cout << "Finished cleaning up after the IMU driver." << std::endl;
419 }

425 void ADIS16448_IMU::Acquire() {
426  // Set data packet length
427  const int dataset_len = 29; // 18 data points + timestamp
428 
429  const int BUFFER_SIZE = 4000;
430 
431  // This buffer can contain many datasets
432  uint32_t buffer[BUFFER_SIZE];
433  int data_count = 0;
434  int data_remainder = 0;
435  int data_to_read = 0;
436  int bufferAvgIndex = 0;
437  uint32_t previous_timestamp = 0;
438  double gyro_rate_x = 0.0;
439  double gyro_rate_y = 0.0;
440  double gyro_rate_z = 0.0;
441  double accel_x = 0.0;
442  double accel_y = 0.0;
443  double accel_z = 0.0;
444  double mag_x = 0.0;
445  double mag_y = 0.0;
446  double mag_z = 0.0;
447  double baro = 0.0;
448  double temp = 0.0;
449  double gyro_rate_x_si = 0.0;
450  double gyro_rate_y_si = 0.0;
451  // double gyro_rate_z_si = 0.0;
452  double accel_x_si = 0.0;
453  double accel_y_si = 0.0;
454  double accel_z_si = 0.0;
455  double compAngleX = 0.0;
456  double compAngleY = 0.0;
457  double accelAngleX = 0.0;
458  double accelAngleY = 0.0;
459 
460  while (true) {
461  // Sleep loop for 10ms (wait for data)
462  Wait(10_ms);
463 
464  if (m_thread_active) {
465  data_count = m_spi->ReadAutoReceivedData(
466  buffer, 0,
467  0_s); // Read number of bytes currently stored in the buffer
468  data_remainder =
469  data_count % dataset_len; // Check if frame is incomplete
470  data_to_read = data_count -
471  data_remainder; // Remove incomplete data from read count
472  /* Want to cap the data to read in a single read at the buffer size */
473  if (data_to_read > BUFFER_SIZE) {
474  REPORT_WARNING(
475  "ADIS16448 data processing thread overrun has occurred!");
476  data_to_read = BUFFER_SIZE - (BUFFER_SIZE % dataset_len);
477  }
478  m_spi->ReadAutoReceivedData(buffer, data_to_read,
479  0_s); // Read data from DMA buffer
480 
481  // Could be multiple data sets in the buffer. Handle each one.
482  for (int i = 0; i < data_to_read; i += dataset_len) {
483  // Calculate CRC-16 on each data packet
484  uint16_t calc_crc = 0xFFFF; // Starting word
485  uint8_t byte = 0;
486  uint16_t imu_crc = 0;
487  for (int k = 5; k < 27;
488  k += 2) // Cycle through XYZ GYRO, XYZ ACCEL, XYZ MAG, BARO, TEMP
489  // (Ignore Status & CRC)
490  {
491  byte = BuffToUByte(&buffer[i + k + 1]); // Process LSB
492  calc_crc = (calc_crc >> 8) ^ adiscrc[(calc_crc & 0x00FF) ^ byte];
493  byte = BuffToUByte(&buffer[i + k]); // Process MSB
494  calc_crc = (calc_crc >> 8) ^ adiscrc[(calc_crc & 0x00FF) ^ byte];
495  }
496  calc_crc = ~calc_crc; // Complement
497  calc_crc = static_cast<uint16_t>((calc_crc << 8) |
498  (calc_crc >> 8)); // Flip LSB & MSB
499  imu_crc =
500  BuffToUShort(&buffer[i + 27]); // Extract DUT CRC from data buffer
501 
502  // Compare calculated vs read CRC. Don't update outputs or dt if CRC-16
503  // is bad
504  if (calc_crc == imu_crc) {
505  // Timestamp is at buffer[i]
506  m_dt = (buffer[i] - previous_timestamp) / 1000000.0;
507  // Split array and scale data
508  gyro_rate_x = BuffToShort(&buffer[i + 5]) * 0.04;
509  gyro_rate_y = BuffToShort(&buffer[i + 7]) * 0.04;
510  gyro_rate_z = BuffToShort(&buffer[i + 9]) * 0.04;
511  accel_x = BuffToShort(&buffer[i + 11]) * 0.833;
512  accel_y = BuffToShort(&buffer[i + 13]) * 0.833;
513  accel_z = BuffToShort(&buffer[i + 15]) * 0.833;
514  mag_x = BuffToShort(&buffer[i + 17]) * 0.1429;
515  mag_y = BuffToShort(&buffer[i + 19]) * 0.1429;
516  mag_z = BuffToShort(&buffer[i + 21]) * 0.1429;
517  baro = BuffToShort(&buffer[i + 23]) * 0.02;
518  temp = BuffToShort(&buffer[i + 25]) * 0.07386 + 31.0;
519 
520  /*std::cout << BuffToShort(&buffer[i + 3]) << "," <<
521  BuffToShort(&buffer[i + 5]) << "," << BuffToShort(&buffer[i + 7]) <<
522  "," << BuffToShort(&buffer[i + 9]) << "," << BuffToShort(&buffer[i +
523  11]) << "," << BuffToShort(&buffer[i + 13]) << "," <<
524  BuffToShort(&buffer[i + 15]) << "," << BuffToShort(&buffer[i + 17]) <<
525  "," << BuffToShort(&buffer[i + 19]) << "," << BuffToShort(&buffer[i +
526  21]) << "," << BuffToShort(&buffer[i + 23]) << "," <<
527  BuffToShort(&buffer[i + 25]) << "," <<
528  BuffToShort(&buffer[i + 27]) << std::endl; */
529 
530  // Convert scaled sensor data to SI units
531  gyro_rate_x_si = gyro_rate_x * deg_to_rad;
532  gyro_rate_y_si = gyro_rate_y * deg_to_rad;
533  // gyro_rate_z_si = gyro_rate_z * deg_to_rad;
534  accel_x_si = accel_x * grav;
535  accel_y_si = accel_y * grav;
536  accel_z_si = accel_z * grav;
537  // Store timestamp for next iteration
538  previous_timestamp = buffer[i];
539  // Calculate alpha for use with the complementary filter
540  m_alpha = m_tau / (m_tau + m_dt);
541  // Calculate complementary filter
542  if (m_first_run) {
543  accelAngleX = atan2f(
544  -accel_x_si,
545  sqrtf((accel_y_si * accel_y_si) + (-accel_z_si * -accel_z_si)));
546  accelAngleY =
547  atan2f(accel_y_si, sqrtf((-accel_x_si * -accel_x_si) +
548  (-accel_z_si * -accel_z_si)));
549  compAngleX = accelAngleX;
550  compAngleY = accelAngleY;
551  } else {
552  accelAngleX = atan2f(
553  -accel_x_si,
554  sqrtf((accel_y_si * accel_y_si) + (-accel_z_si * -accel_z_si)));
555  accelAngleY =
556  atan2f(accel_y_si, sqrtf((-accel_x_si * -accel_x_si) +
557  (-accel_z_si * -accel_z_si)));
558  accelAngleX = FormatAccelRange(accelAngleX, -accel_z_si);
559  accelAngleY = FormatAccelRange(accelAngleY, -accel_z_si);
560  compAngleX =
561  CompFilterProcess(compAngleX, accelAngleX, -gyro_rate_y_si);
562  compAngleY =
563  CompFilterProcess(compAngleY, accelAngleY, -gyro_rate_x_si);
564  }
565 
566  // Update global variables and state
567  {
568  std::scoped_lock sync(m_mutex);
569  // Ignore first, integrated sample
570  if (m_first_run) {
571  m_integ_gyro_angle_x = 0.0;
572  m_integ_gyro_angle_y = 0.0;
573  m_integ_gyro_angle_z = 0.0;
574  } else {
575  // Accumulate gyro for offset calibration
576  // Add most recent sample data to buffer
577  bufferAvgIndex = m_accum_count % m_avg_size;
578  m_offset_buffer[bufferAvgIndex] =
579  OffsetData{gyro_rate_x, gyro_rate_y, gyro_rate_z};
580  // Increment counter
581  m_accum_count++;
582  }
583  // Don't post accumulated data to the global variables until an
584  // initial gyro offset has been calculated
585  if (!m_start_up_mode) {
586  m_gyro_rate_x = gyro_rate_x;
587  m_gyro_rate_y = gyro_rate_y;
588  m_gyro_rate_z = gyro_rate_z;
589  m_accel_x = accel_x;
590  m_accel_y = accel_y;
591  m_accel_z = accel_z;
592  m_mag_x = mag_x;
593  m_mag_y = mag_y;
594  m_mag_z = mag_z;
595  m_baro = baro;
596  m_temp = temp;
597  m_compAngleX = compAngleX * rad_to_deg;
598  m_compAngleY = compAngleY * rad_to_deg;
599  m_accelAngleX = accelAngleX * rad_to_deg;
600  m_accelAngleY = accelAngleY * rad_to_deg;
601  // Accumulate gyro for angle integration and publish to global
602  // variables
603  m_integ_gyro_angle_x +=
604  (gyro_rate_x - m_gyro_rate_offset_x) * m_dt;
605  m_integ_gyro_angle_y +=
606  (gyro_rate_y - m_gyro_rate_offset_y) * m_dt;
607  m_integ_gyro_angle_z +=
608  (gyro_rate_z - m_gyro_rate_offset_z) * m_dt;

690 double ADIS16448_IMU::CompFilterProcess(double compAngle, double accelAngle,
691  double omega) {
692  compAngle = FormatFastConverge(compAngle, accelAngle);
693  compAngle =
694  m_alpha * (compAngle + omega * m_dt) + (1.0 - m_alpha) * accelAngle;
695  compAngle = FormatRange0to2PI(compAngle);
696  if (compAngle > wpi::numbers::pi) {
697  compAngle = compAngle - 2.0 * wpi::numbers::pi;
698  }
699  return compAngle;

702 int ADIS16448_IMU::ConfigDecRate(uint16_t DecimationSetting) {
703  uint16_t writeValue = DecimationSetting;
704  uint16_t readbackValue;
705  if (!SwitchToStandardSPI()) {
706  REPORT_ERROR("Failed to configure/reconfigure standard SPI.");
707  return 2;
708  }
709 
710  /* Check max */
711  if (DecimationSetting > 9) {
712  REPORT_ERROR("Attemted to write an invalid decimation value. Capping at 9");
713  DecimationSetting = 9;
714  }
715 
716  /* Shift decimation setting to correct position and select internal sync */
717  writeValue = (DecimationSetting << 8) | 0x1;
718 
719  /* Apply to IMU */
720  WriteRegister(SMPL_PRD, writeValue);
721 
722  /* Perform read back to verify write */
723  readbackValue = ReadRegister(SMPL_PRD);
724 
725  /* Throw error for invalid write */
726  if (readbackValue != writeValue) {
727  REPORT_ERROR("ADIS16448 SMPL_PRD write failed.");
728  }
729 
730  if (!SwitchToAutoSPI()) {
731  REPORT_ERROR("Failed to configure/reconfigure auto SPI.");
732  return 2;
733  }
734  return 0;
735 }
