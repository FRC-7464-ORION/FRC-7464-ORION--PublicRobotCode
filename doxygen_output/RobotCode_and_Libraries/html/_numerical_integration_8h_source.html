24 template <typename F, typename T>
25 T RK4(F&& f, T x, units::second_t dt) {
26  const auto h = dt.value();
27 
28  T k1 = f(x);
29  T k2 = f(x + h * 0.5 * k1);
30  T k3 = f(x + h * 0.5 * k2);
31  T k4 = f(x + h * k3);
32 
33  return x + h / 6.0 * (k1 + 2.0 * k2 + 2.0 * k3 + k4);
34 }

45 T RK4(F&& f, T x, U u, units::second_t dt) {
46  const auto h = dt.value();
47 
48  T k1 = f(x, u);
49  T k2 = f(x + h * 0.5 * k1, u);
50  T k3 = f(x + h * 0.5 * k2, u);
51  T k4 = f(x + h * k3, u);
52 
53  return x + h / 6.0 * (k1 + 2.0 * k2 + 2.0 * k3 + k4);
54 }

68 template <typename F, typename T, typename U>
69 T RKF45(F&& f, T x, U u, units::second_t dt, double maxError = 1e-6) {
70  // See
71  // https://en.wikipedia.org/wiki/Runge%E2%80%93Kutta%E2%80%93Fehlberg_method
72  // for the Butcher tableau the following arrays came from.
73  constexpr int kDim = 6;
74 
75  // clang-format off
76  constexpr double A[kDim - 1][kDim - 1]{
77  { 1.0 / 4.0},
78  { 3.0 / 32.0, 9.0 / 32.0},
79  {1932.0 / 2197.0, -7200.0 / 2197.0, 7296.0 / 2197.0},
80  { 439.0 / 216.0, -8.0, 3680.0 / 513.0, -845.0 / 4104.0},
81  { -8.0 / 27.0, 2.0, -3544.0 / 2565.0, 1859.0 / 4104.0, -11.0 / 40.0}};
82  // clang-format on
83 
84  constexpr std::array<double, kDim> b1{16.0 / 135.0, 0.0,
85  6656.0 / 12825.0, 28561.0 / 56430.0,
86  -9.0 / 50.0, 2.0 / 55.0};
87  constexpr std::array<double, kDim> b2{
88  25.0 / 216.0, 0.0, 1408.0 / 2565.0, 2197.0 / 4104.0, -1.0 / 5.0, 0.0};
89 
90  T newX;
91  double truncationError;
92 
93  double dtElapsed = 0.0;
94  double h = dt.value();
95 
96  // Loop until we've gotten to our desired dt
97  while (dtElapsed < dt.value()) {
98  do {
99  // Only allow us to advance up to the dt remaining
100  h = std::min(h, dt.value() - dtElapsed);
101 
102  // Notice how the derivative in the Wikipedia notation is dy/dx.
103  // That means their y is our x and their x is our t
104  // clang-format off
105  T k1 = f(x, u);
106  T k2 = f(x + h * (A[0][0] * k1), u);
107  T k3 = f(x + h * (A[1][0] * k1 + A[1][1] * k2), u);
108  T k4 = f(x + h * (A[2][0] * k1 + A[2][1] * k2 + A[2][2] * k3), u);
109  T k5 = f(x + h * (A[3][0] * k1 + A[3][1] * k2 + A[3][2] * k3 + A[3][3] * k4), u);
110  T k6 = f(x + h * (A[4][0] * k1 + A[4][1] * k2 + A[4][2] * k3 + A[4][3] * k4 + A[4][4] * k5), u);
111  // clang-format on
112 
113  newX = x + h * (b1[0] * k1 + b1[1] * k2 + b1[2] * k3 + b1[3] * k4 +
114  b1[4] * k5 + b1[5] * k6);
115  truncationError = (h * ((b1[0] - b2[0]) * k1 + (b1[1] - b2[1]) * k2 +
116  (b1[2] - b2[2]) * k3 + (b1[3] - b2[3]) * k4 +
117  (b1[4] - b2[4]) * k5 + (b1[5] - b2[5]) * k6))
118  .norm();
119 
120  h *= 0.9 * std::pow(maxError / truncationError, 1.0 / 5.0);
121  } while (truncationError > maxError);
122 
123  dtElapsed += h;
124  x = newX;
125  }
126 
127  return x;
128 }

141 template <typename F, typename T, typename U>
142 T RKDP(F&& f, T x, U u, units::second_t dt, double maxError = 1e-6) {
143  // See https://en.wikipedia.org/wiki/Dormand%E2%80%93Prince_method for the
144  // Butcher tableau the following arrays came from.
145 
146  constexpr int kDim = 7;
147 
148  // clang-format off
149  constexpr double A[kDim - 1][kDim - 1]{
150  { 1.0 / 5.0},
151  { 3.0 / 40.0, 9.0 / 40.0},
152  { 44.0 / 45.0, -56.0 / 15.0, 32.0 / 9.0},
153  {19372.0 / 6561.0, -25360.0 / 2187.0, 64448.0 / 6561.0, -212.0 / 729.0},
154  { 9017.0 / 3168.0, -355.0 / 33.0, 46732.0 / 5247.0, 49.0 / 176.0, -5103.0 / 18656.0},
155  { 35.0 / 384.0, 0.0, 500.0 / 1113.0, 125.0 / 192.0, -2187.0 / 6784.0, 11.0 / 84.0}};
156  // clang-format on
157 
158  constexpr std::array<double, kDim> b1{
159  35.0 / 384.0, 0.0, 500.0 / 1113.0, 125.0 / 192.0, -2187.0 / 6784.0,
160  11.0 / 84.0, 0.0};
161  constexpr std::array<double, kDim> b2{5179.0 / 57600.0, 0.0,
162  7571.0 / 16695.0, 393.0 / 640.0,
163  -92097.0 / 339200.0, 187.0 / 2100.0,
164  1.0 / 40.0};
165 
166  T newX;
167  double truncationError;
168 
169  double dtElapsed = 0.0;
170  double h = dt.value();
171 
172  // Loop until we've gotten to our desired dt
173  while (dtElapsed < dt.value()) {
174  do {
175  // Only allow us to advance up to the dt remaining
176  h = std::min(h, dt.value() - dtElapsed);
177 
178  // clang-format off
179  T k1 = f(x, u);
180  T k2 = f(x + h * (A[0][0] * k1), u);
181  T k3 = f(x + h * (A[1][0] * k1 + A[1][1] * k2), u);
182  T k4 = f(x + h * (A[2][0] * k1 + A[2][1] * k2 + A[2][2] * k3), u);
183  T k5 = f(x + h * (A[3][0] * k1 + A[3][1] * k2 + A[3][2] * k3 + A[3][3] * k4), u);
184  T k6 = f(x + h * (A[4][0] * k1 + A[4][1] * k2 + A[4][2] * k3 + A[4][3] * k4 + A[4][4] * k5), u);
185  // clang-format on
186 
187  // Since the final row of A and the array b1 have the same coefficients
188  // and k7 has no effect on newX, we can reuse the calculation.
189  newX = x + h * (A[5][0] * k1 + A[5][1] * k2 + A[5][2] * k3 +
190  A[5][3] * k4 + A[5][4] * k5 + A[5][5] * k6);
191  T k7 = f(newX, u);
192 
193  truncationError = (h * ((b1[0] - b2[0]) * k1 + (b1[1] - b2[1]) * k2 +
194  (b1[2] - b2[2]) * k3 + (b1[3] - b2[3]) * k4 +
195  (b1[4] - b2[4]) * k5 + (b1[5] - b2[5]) * k6 +
196  (b1[6] - b2[6]) * k7))
197  .norm();
198 
199  h *= 0.9 * std::pow(maxError / truncationError, 1.0 / 5.0);
200  } while (truncationError > maxError);
201 
202  dtElapsed += h;
203  x = newX;
204  }
205 
206  return x;
207 }
