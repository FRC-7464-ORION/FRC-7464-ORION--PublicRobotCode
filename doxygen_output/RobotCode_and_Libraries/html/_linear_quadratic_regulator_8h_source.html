84  LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
85  const Eigen::Matrix<double, States, Inputs>& B,
86  const Eigen::Matrix<double, States, States>& Q,
87  const Eigen::Matrix<double, Inputs, Inputs>& R,
88  units::second_t dt) {
89  Eigen::Matrix<double, States, States> discA;
90  Eigen::Matrix<double, States, Inputs> discB;
91  DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
92 
93  if (!IsStabilizable<States, Inputs>(discA, discB)) {
94  std::string msg = fmt::format(
95  "The system passed to the LQR is uncontrollable!\n\nA =\n{}\nB "
96  "=\n{}\n",
97  discA, discB);
98 
99  wpi::math::MathSharedStore::ReportError(msg);
100  throw std::invalid_argument(msg);
101  }
102 
103  Eigen::Matrix<double, States, States> S =
104  drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R);
105 
106  // K = (BᵀSB + R)⁻¹BᵀSA
107  m_K = (discB.transpose() * S * discB + R)
108  .llt()
109  .solve(discB.transpose() * S * discA);
110 
111  Reset();
112  }

124  LinearQuadraticRegulatorImpl(const Eigen::Matrix<double, States, States>& A,
125  const Eigen::Matrix<double, States, Inputs>& B,
126  const Eigen::Matrix<double, States, States>& Q,
127  const Eigen::Matrix<double, Inputs, Inputs>& R,
128  const Eigen::Matrix<double, States, Inputs>& N,
129  units::second_t dt) {
130  Eigen::Matrix<double, States, States> discA;
131  Eigen::Matrix<double, States, Inputs> discB;
132  DiscretizeAB<States, Inputs>(A, B, dt, &discA, &discB);
133 
134  Eigen::Matrix<double, States, States> S =
135  drake::math::DiscreteAlgebraicRiccatiEquation(discA, discB, Q, R, N);
136 
137  // K = (BᵀSB + R)⁻¹(BᵀSA + Nᵀ)
138  m_K = (discB.transpose() * S * discB + R)
139  .llt()
140  .solve(discB.transpose() * S * discA + N.transpose());
141 
142  Reset();
143  }

242  template <int Outputs>
243  void LatencyCompensate(const LinearSystem<States, Inputs, Outputs>& plant,
244  units::second_t dt, units::second_t inputDelay) {
245  Eigen::Matrix<double, States, States> discA;
246  Eigen::Matrix<double, States, Inputs> discB;
247  DiscretizeAB<States, Inputs>(plant.A(), plant.B(), dt, &discA, &discB);
248 
249  m_K = m_K * (discA - discB * m_K).pow(inputDelay / dt);
250  }
