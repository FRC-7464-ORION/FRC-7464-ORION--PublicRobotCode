11 Transform2d::Transform2d(Pose2d initial, Pose2d final) {
12  // We are rotating the difference between the translations
13  // using a clockwise rotation matrix. This transforms the global
14  // delta into a local delta (relative to the initial pose).
15  m_translation = (final.Translation() - initial.Translation())
16  .RotateBy(-initial.Rotation());
17 
18  m_rotation = final.Rotation() - initial.Rotation();
19 }
