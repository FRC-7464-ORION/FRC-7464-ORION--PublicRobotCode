55 int json_pointer::array_index(std::string_view s)
56 {

70 json& json_pointer::get_and_create(json& j) const
71 {
72  using size_type = typename json::size_type;
73  auto result = &j;
74 
75  // in case no reference tokens exist, return a reference to the JSON value
76  // j which will be overwritten by a primitive value
77  for (const auto& reference_token : reference_tokens)
78  {
79  switch (result->m_type)
80  {
81  case detail::value_t::null:
82  {
83  if (reference_token == "0")
84  {
85  // start a new array if reference token is 0
86  result = &result->operator[](0);
87  }
88  else
89  {
90  // start a new object otherwise
91  result = &result->operator[](reference_token);
92  }
93  break;
94  }
95 
96  case detail::value_t::object:
97  {
98  // create an entry in the object
99  result = &result->operator[](reference_token);
100  break;
101  }
102 
103  case detail::value_t::array:
104  {
105  // create an entry in the array
106  JSON_TRY
107  {
108  result = &result->operator[](static_cast<size_type>(array_index(reference_token)));
109  }
110  JSON_CATCH(std::invalid_argument&)
111  {
112  JSON_THROW(detail::parse_error::create(109, 0, fmt::format("array index '{}' is not a number", reference_token)));
113  }
114  break;
115  }
116 
117  /*
118  The following code is only reached if there exists a reference
119  token _and_ the current value is primitive. In this case, we have
120  an error situation, because primitive values may only occur as
121  single value; that is, with an empty list of reference tokens.
122  */
123  default:
124  JSON_THROW(detail::type_error::create(313, "invalid value to unflatten"));
125  }
126  }
127 
128  return *result;
129 }

131 json& json_pointer::get_unchecked(json* ptr) const
132 {
133  using size_type = typename json::size_type;
134  for (const auto& reference_token : reference_tokens)
135  {
136  // convert null values to arrays or objects before continuing
137  if (ptr->m_type == detail::value_t::null)
138  {
139  // check if reference token is a number
140  const bool nums =
141  std::all_of(reference_token.begin(), reference_token.end(),
142  [](const char x)
143  {
144  return (x >= '0' and x <= '9');
145  });
146 
147  // change value to array for numbers or "-" or to object otherwise
148  *ptr = (nums or reference_token == "-")
149  ? detail::value_t::array
150  : detail::value_t::object;
151  }
152 
153  switch (ptr->m_type)
154  {
155  case detail::value_t::object:
156  {
157  // use unchecked object access
158  ptr = &ptr->operator[](reference_token);
159  break;
160  }
161 
162  case detail::value_t::array:
163  {
164  // error condition (cf. RFC 6901, Sect. 4)
165  if (JSON_UNLIKELY(reference_token.size() > 1 and reference_token[0] == '0'))
166  {
167  JSON_THROW(detail::parse_error::create(106, 0,
168  fmt::format("array index '{}' must not begin with '0'", reference_token)));
169  }
170 
171  if (reference_token == "-")
172  {
173  // explicitly treat "-" as index beyond the end
174  ptr = &ptr->operator[](ptr->m_value.array->size());
175  }
176  else
177  {
178  // convert array index to number; unchecked access
179  JSON_TRY
180  {
181  ptr = &ptr->operator[](
182  static_cast<size_type>(array_index(reference_token)));
183  }
184  JSON_CATCH(std::invalid_argument&)
185  {
186  JSON_THROW(detail::parse_error::create(109, 0,
187  fmt::format("array index '{}' is not a number", reference_token)));
188  }
189  }
190  break;
191  }
192 
193  default:
194  JSON_THROW(detail::out_of_range::create(404,
195  fmt::format("unresolved reference token '{}'", reference_token)));
196  }
197  }
198 
199  return *ptr;
200 }

359 std::vector<std::string> json_pointer::split(std::string_view ref_str)
360 {
361  std::vector<std::string> result;
362 
363  // special case: empty reference string -> no reference tokens
364  if (ref_str.empty())
365  {
366  return result;
367  }
368 
369  // check if nonempty reference string begins with slash
370  if (JSON_UNLIKELY(ref_str[0] != '/'))
371  {
372  JSON_THROW(detail::parse_error::create(107, 1,
373  fmt::format("JSON pointer must be empty or begin with '/' - was: '{}'", ref_str)));
374  }
375 
376  // extract the reference tokens:
377  // - slash: position of the last read slash (or end of string)
378  // - start: position after the previous slash
379  for (
380  // search for the first slash after the first character
381  std::size_t slash = ref_str.find_first_of('/', 1),
382  // set the beginning of the first reference token
383  start = 1;
384  // we can stop if start == string::npos+1 = 0
385  start != 0;
386  // set the beginning of the next reference token
387  // (will eventually be 0 if slash == std::string::npos)
388  start = slash + 1,
389  // find next slash
390  slash = ref_str.find_first_of('/', start))
391  {
392  // use the text between the beginning of the reference token
393  // (start) and the last slash (slash).
394  auto reference_token = slice(ref_str, start, slash);
395 
396  // check reference tokens are properly escaped
397  for (std::size_t pos = reference_token.find_first_of('~');
398  pos != std::string_view::npos;
399  pos = reference_token.find_first_of('~', pos + 1))
400  {
401  assert(reference_token[pos] == '~');
402 
403  // ~ must be followed by 0 or 1
404  if (JSON_UNLIKELY(pos == reference_token.size() - 1 or
405  (reference_token[pos + 1] != '0' and
406  reference_token[pos + 1] != '1')))
407  {
408  JSON_THROW(detail::parse_error::create(108, 0, "escape character '~' must be followed with '0' or '1'"));
409  }
410  }
411 
412  // finally, store the reference token
413  std::string ref_tok{reference_token};
414  unescape(ref_tok);
415  result.emplace_back(std::move(ref_tok));
416  }
417 
418  return result;
419 }

446 void json_pointer::flatten(std::string_view reference_string,
447  const json& value,
448  json& result)
449 {
450  switch (value.m_type)
451  {
452  case detail::value_t::array:
453  {
454  if (value.m_value.array->empty())
455  {
456  // flatten empty array as null
457  result[reference_string] = nullptr;
458  }
459  else
460  {
461  // iterate array and use index as reference string
462  for (std::size_t i = 0; i < value.m_value.array->size(); ++i)
463  {
464  flatten(fmt::format("{}/{}", reference_string, i),
465  value.m_value.array->operator[](i), result);
466  }
467  }
468  break;
469  }
470 
471  case detail::value_t::object:
472  {
473  if (value.m_value.object->empty())
474  {
475  // flatten empty object as null
476  result[reference_string] = nullptr;
477  }
478  else
479  {
480  // iterate object and use keys as reference string
481  for (const auto& element : *value.m_value.object)
482  {
483  flatten(fmt::format("{}/{}", reference_string, escape(std::string{element.first()})), element.second, result);
484  }
485  }
486  break;
487  }
488 
489  default:
490  {
491  // add primitive value with its reference string
492  result[reference_string] = value;
493  break;
494  }
495  }
496 }

499 json_pointer::unflatten(const json& value)
500 {
501  if (JSON_UNLIKELY(not value.is_object()))
502  {
503  JSON_THROW(detail::type_error::create(314, "only objects can be unflattened"));
504  }
505 
506  // we need to iterate over the object values in sorted key order
507  SmallVector<StringMapConstIterator<json>, 64> sorted;
508  for (auto i = value.m_value.object->begin(),
509  end = value.m_value.object->end(); i != end; ++i)
510  {
511  if (!i->second.is_primitive())
512  {
513  JSON_THROW(detail::type_error::create(315, "values in object must be primitive"));
514  }
515  sorted.push_back(i);
516  }
517  std::sort(sorted.begin(), sorted.end(),
518  [](const StringMapConstIterator<json>& a,
519  const StringMapConstIterator<json>& b) {
520  return a->getKey() < b->getKey();
521  });
522 
523  json result;
524 
525  // iterate the sorted JSON object values
526  for (const auto& element : sorted)
527  {
528 
529  // assign value to reference pointed to by JSON pointer; Note
530  // that if the JSON pointer is "" (i.e., points to the whole
531  // value), function get_and_create returns a reference to
532  // result itself. An assignment will then create a primitive
533  // value.
534  json_pointer(element->first()).get_and_create(result) = element->second;
535  }
536 
537  return result;
538 }
