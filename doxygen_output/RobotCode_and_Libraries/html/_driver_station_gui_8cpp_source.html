312 JoystickModel::JoystickModel(int index) : m_index{index} {
313  HAL_JoystickAxes halAxes;
314  HALSIM_GetJoystickAxes(index, &halAxes);
315  axisCount = halAxes.count;
316  for (int i = 0; i < axisCount; ++i) {
317  axes[i] = std::make_unique<glass::DataSource>(
318  fmt::format("Joystick[{}] Axis[{}]", index, i));
319  }
320 
321  HAL_JoystickButtons halButtons;
322  HALSIM_GetJoystickButtons(index, &halButtons);
323  buttonCount = halButtons.count;
324  for (int i = 0; i < buttonCount; ++i) {
325  buttons[i] = new glass::DataSource(
326  fmt::format("Joystick[{}] Button[{}]", index, i + 1));
327  buttons[i]->SetDigital(true);
328  }
329  for (int i = buttonCount; i < 32; ++i) {
330  buttons[i] = nullptr;
331  }
332 
333  HAL_JoystickPOVs halPOVs;
334  HALSIM_GetJoystickPOVs(index, &halPOVs);
335  povCount = halPOVs.count;
336  for (int i = 0; i < povCount; ++i) {
337  povs[i] = std::make_unique<glass::DataSource>(
338  fmt::format("Joystick[{}] POV [{}]", index, i));
339  }
340 
341  m_callback =
342  HALSIM_RegisterDriverStationNewDataCallback(CallbackFunc, this, true);
343 }

345 void JoystickModel::CallbackFunc(const char*, void* param, const HAL_Value*) {
346  auto self = static_cast<JoystickModel*>(param);
347 
348  HAL_JoystickAxes halAxes;
349  HALSIM_GetJoystickAxes(self->m_index, &halAxes);
350  for (int i = 0; i < halAxes.count; ++i) {
351  if (auto axis = self->axes[i].get()) {
352  axis->SetValue(halAxes.axes[i]);
353  }
354  }
355 
356  HAL_JoystickButtons halButtons;
357  HALSIM_GetJoystickButtons(self->m_index, &halButtons);
358  for (int i = 0; i < halButtons.count; ++i) {
359  if (auto button = self->buttons[i]) {
360  button->SetValue((halButtons.buttons & (1u << i)) != 0 ? 1 : 0);
361  }
362  }
363 
364  HAL_JoystickPOVs halPOVs;
365  HALSIM_GetJoystickPOVs(self->m_index, &halPOVs);
366  for (int i = 0; i < halPOVs.count; ++i) {
367  if (auto pov = self->povs[i].get()) {
368  pov->SetValue(halPOVs.povs[i]);
369  }
370  }
371 }

373 void GlfwSystemJoystick::Update() {
374  bool wasPresent = m_present;
375  m_present = glfwJoystickPresent(m_index);
376 
377  if (!m_present) {
378  return;
379  }
380  m_axes = glfwGetJoystickAxes(m_index, &m_axisCount);
381  m_buttons = glfwGetJoystickButtons(m_index, &m_buttonCount);
382  m_hats = glfwGetJoystickHats(m_index, &m_hatCount);
383  m_isGamepad = glfwGetGamepadState(m_index, &m_gamepadState);
384 
385  m_anyButtonPressed = false;
386  for (int j = 0; j < m_buttonCount; ++j) {
387  if (m_buttons[j]) {
388  m_anyButtonPressed = true;
389  break;
390  }
391  }
392  for (int j = 0; j < m_hatCount; ++j) {
393  if (m_hats[j] != GLFW_HAT_CENTERED) {
394  m_anyButtonPressed = true;
395  break;
396  }
397  }
398 
399  if (!m_present || wasPresent) {
400  return;
401  }
402  m_name = glfwGetJoystickName(m_index);
403 
404  // try to find matching GUID
405  if (const char* guid = glfwGetJoystickGUID(m_index)) {
406  for (auto&& joy : gRobotJoysticks) {
407  if (guid == joy.guid) {
408  joy.sys = this;
409  break;
410  }
411  }
412  }
413 }

415 static int HatToAngle(unsigned char hat) {
416  switch (hat) {
417  case GLFW_HAT_UP:
418  return 0;
419  case GLFW_HAT_RIGHT:
420  return 90;
421  case GLFW_HAT_DOWN:
422  return 180;
423  case GLFW_HAT_LEFT:
424  return 270;
425  case GLFW_HAT_RIGHT_UP:
426  return 45;
427  case GLFW_HAT_RIGHT_DOWN:
428  return 135;
429  case GLFW_HAT_LEFT_UP:
430  return 315;
431  case GLFW_HAT_LEFT_DOWN:
432  return 225;
433  default:
434  return -1;
435  }
436 }

438 void GlfwSystemJoystick::GetData(HALJoystickData* data, bool mapGamepad) const {
439  if (!m_present) {
440  return;
441  }
442 
443  // use gamepad mappings if present and enabled
444  const float* sysAxes;
445  const unsigned char* sysButtons;
446  if (m_isGamepad && mapGamepad) {
447  sysAxes = m_gamepadState.axes;
448  // don't remap on windows
449 #ifdef _WIN32
450  sysButtons = m_buttons;
451 #else
452  sysButtons = m_gamepadState.buttons;
453 #endif
454  } else {
455  sysAxes = m_axes;
456  sysButtons = m_buttons;
457  }
458 
459  // copy into HAL structures
460  data->desc.isXbox = m_isGamepad ? 1 : 0;
461  data->desc.type = m_isGamepad ? 21 : 20;
462  std::strncpy(data->desc.name, m_name, sizeof(data->desc.name) - 1);
463  data->desc.name[sizeof(data->desc.name) - 1] = '\0';
464  data->desc.axisCount = (std::min)(m_axisCount, HAL_kMaxJoystickAxes);
465  // desc.axisTypes ???
466  data->desc.buttonCount = (std::min)(m_buttonCount, 32);
467  data->desc.povCount = (std::min)(m_hatCount, HAL_kMaxJoystickPOVs);
468 
469  data->buttons.count = data->desc.buttonCount;
470  for (int j = 0; j < data->buttons.count; ++j) {
471  data->buttons.buttons |= (sysButtons[j] ? 1u : 0u) << j;
472  }
473 
474  data->axes.count = data->desc.axisCount;
475  if (m_isGamepad && mapGamepad) {
476  // the FRC DriverStation maps gamepad (XInput) trigger values to 0-1 range
477  // on axis 2 and 3.
478  data->axes.axes[0] = sysAxes[0];
479  data->axes.axes[1] = sysAxes[1];
480  data->axes.axes[2] = 0.5 + sysAxes[4] / 2.0;
481  data->axes.axes[3] = 0.5 + sysAxes[5] / 2.0;
482  data->axes.axes[4] = sysAxes[2];
483  data->axes.axes[5] = sysAxes[3];
484 
485  // the start button for gamepads is not mapped on the FRC DriverStation
486  // platforms, so remove it if present
487  if (data->buttons.count == 11) {
488  --data->desc.buttonCount;
489  --data->buttons.count;
490  data->buttons.buttons = (data->buttons.buttons & 0xff) |
491  ((data->buttons.buttons >> 1) & 0x300);
492  }
493  } else {
494  std::memcpy(data->axes.axes, sysAxes,
495  data->axes.count * sizeof(&data->axes.axes[0]));
496  }
497 
498  data->povs.count = data->desc.povCount;
499  for (int j = 0; j < data->povs.count; ++j) {
500  data->povs.povs[j] = HatToAngle(m_hats[j]);
501  }
502 }

608 void KeyboardJoystick::SettingsDisplay() {
609  if (s_keyEdit) {
610  ImGuiIO& io = ImGui::GetIO();
611  for (int i = 0; i < IM_ARRAYSIZE(io.KeysDown); ++i) {
612  if (io.KeysDown[i]) {
613  // remove all other uses
614  for (auto&& joy : gKeyboardJoysticks) {
615  joy->ClearKey(i);
616  }
617  *s_keyEdit = i;
618  s_keyEdit = nullptr;
619  break;
620  }
621  }
622  }
623 
624  char label[64];
625  ImGui::PushItemWidth(ImGui::GetFontSize() * 6);
626  // axes
627  if (ImGui::CollapsingHeader("Axes", ImGuiTreeNodeFlags_DefaultOpen)) {
628  ImGui::PushID("Axes");
629  if (ImGui::InputInt("Count", &m_axisCount)) {
630  if (m_axisCount < 0) {
631  m_axisCount = 0;
632  } else if (m_axisCount > HAL_kMaxJoystickAxes) {
633  m_axisCount = HAL_kMaxJoystickAxes;
634  }
635  }
636  while (m_axisCount > static_cast<int>(m_axisConfig.size())) {
637  m_axisStorage.emplace_back(std::make_unique<glass::Storage>());
638  m_axisConfig.emplace_back(*m_axisStorage.back());
639  }

719 void KeyboardJoystick::Update() {
720  ImGuiIO& io = ImGui::GetIO();
721 
722  if (m_axisCount < 0) {
723  m_axisCount = 0;
724  }
725  if (m_buttonCount < 0) {
726  m_buttonCount = 0;
727  }
728  if (m_povCount < 0) {
729  m_povCount = 0;
730  }
731 
732  m_data.axes.count =
733  (std::min)(m_axisCount, static_cast<int>(m_axisConfig.size()));
734  m_data.buttons.count =
735  (std::min)(m_buttonCount, static_cast<int>(m_buttonKey.size()));
736  m_data.povs.count =
737  (std::min)(m_povCount, static_cast<int>(m_povConfig.size()));
738 
739  if (m_data.axes.count > 0 || m_data.buttons.count > 0 ||
740  m_data.povs.count > 0) {
741  m_present = true;
742  }
743 
744  // axes
745  for (int i = 0; i < m_data.axes.count; ++i) {
746  auto& config = m_axisConfig[i];
747  float& axisValue = m_data.axes.axes[i];
748  // increase/decrease while key held down (to saturation); decay back to 0
749  if (IsKeyDown(io, config.incKey)) {
750  axisValue += config.keyRate;
751  if (axisValue > config.maxAbsValue) {
752  axisValue = config.maxAbsValue;
753  }
754  } else if (axisValue > 0) {
755  if (axisValue < config.decayRate) {
756  axisValue = 0;
757  } else {
758  axisValue -= config.decayRate;
759  }
760  }
761 
762  if (IsKeyDown(io, config.decKey)) {
763  axisValue -= config.keyRate;
764  if (axisValue < -config.maxAbsValue) {
765  axisValue = -config.maxAbsValue;
766  }
767  } else if (axisValue < 0) {
768  if (axisValue > -config.decayRate) {
769  axisValue = 0;
770  } else {
771  axisValue += config.decayRate;
772  }
773  }
774  }
775 
776  // buttons
777  m_data.buttons.buttons = 0;
778  m_anyButtonPressed = false;
779  for (int i = 0; i < m_data.buttons.count; ++i) {
780  if (IsKeyDown(io, m_buttonKey[i])) {
781  m_data.buttons.buttons |= 1u << i;
782  m_anyButtonPressed = true;
783  }
784  }
785 
786  // povs
787  for (int i = 0; i < m_data.povs.count; ++i) {
788  auto& config = m_povConfig[i];
789  auto& povValue = m_data.povs.povs[i];
790  povValue = -1;
791  if (IsKeyDown(io, config.key0)) {
792  povValue = 0;
793  } else if (IsKeyDown(io, config.key45)) {
794  povValue = 45;
795  } else if (IsKeyDown(io, config.key90)) {
796  povValue = 90;
797  } else if (IsKeyDown(io, config.key135)) {
798  povValue = 135;
799  } else if (IsKeyDown(io, config.key180)) {
800  povValue = 180;
801  } else if (IsKeyDown(io, config.key225)) {
802  povValue = 225;
803  } else if (IsKeyDown(io, config.key270)) {
804  povValue = 270;
805  } else if (IsKeyDown(io, config.key315)) {
806  povValue = 315;
807  }
808  }
809 
810  // try to find matching GUID
811  for (auto&& joy : gRobotJoysticks) {
812  if (m_guid == joy.guid) {
813  joy.sys = this;
814  break;
815  }
816  }
817 
818  // update desc
819  m_data.desc.axisCount = m_data.axes.count;
820  m_data.desc.buttonCount = m_data.buttons.count;
821  m_data.desc.povCount = m_data.povs.count;
822 }

1017 static void DriverStationExecute() {
1018  // update sources
1019  for (int i = 0; i < HAL_kMaxJoysticks; ++i) {
1020  auto& source = gJoystickSources[i];
1021  int32_t axisCount, buttonCount, povCount;
1022  HALSIM_GetJoystickCounts(i, &axisCount, &buttonCount, &povCount);
1023  if (axisCount != 0 || buttonCount != 0 || povCount != 0) {
1024  if (!source || source->axisCount != axisCount ||
1025  source->buttonCount != buttonCount || source->povCount != povCount) {
1026  source.reset();
1027  source = std::make_unique<JoystickModel>(i);
1028  }
1029  } else {
1030  source.reset();
1031  }
1032  }
1033 
1034  static bool prevDisableDS = false;
1035 
1036  bool disableDS = IsDSDisabled();
1037  if (disableDS && !prevDisableDS) {
1038  if (auto win = HALSimGui::manager->GetWindow("System Joysticks")) {
1039  win->SetVisibility(glass::Window::kDisabled);
1040  }
1041  } else if (!disableDS && prevDisableDS) {
1042  if (auto win = HALSimGui::manager->GetWindow("System Joysticks")) {
1043  win->SetVisibility(glass::Window::kShow);
1044  }
1045  }
1046  prevDisableDS = disableDS;
1047  if (disableDS) {
1048  return;
1049  }
1050 
1051  double curTime = glfwGetTime();
1052 
1053  // update system joysticks
1054  gNumGlfwJoysticks = 0;
1055  for (int i = 0; i <= GLFW_JOYSTICK_LAST; ++i) {
1056  gGlfwJoysticks[i]->Update();
1057  if (gGlfwJoysticks[i]->IsPresent()) {
1058  gNumGlfwJoysticks = i + 1;
1059  }
1060  }
1061  for (auto&& joy : gKeyboardJoysticks) {
1062  joy->Update();
1063  }
1064 
1065  // update robot joysticks
1066  for (auto&& joy : gRobotJoysticks) {
1067  joy.Update();
1068  }
1069 
1070  bool isEnabled = HALSIM_GetDriverStationEnabled();
1071  bool isAuto = HALSIM_GetDriverStationAutonomous();
1072  bool isTest = HALSIM_GetDriverStationTest();
1073 
1074  // Robot state
1075  {
1076  // DS hotkeys
1077  bool enableHotkey = false;
1078  bool disableHotkey = false;
1079  if (gpUseEnableDisableHotkeys != nullptr && *gpUseEnableDisableHotkeys) {
1080  ImGuiIO& io = ImGui::GetIO();
1081  if (io.KeysDown[GLFW_KEY_ENTER] || io.KeysDown[GLFW_KEY_KP_ENTER]) {
1082  disableHotkey = true;
1083  } else if (io.KeysDown[GLFW_KEY_LEFT_BRACKET] &&
1084  io.KeysDown[GLFW_KEY_RIGHT_BRACKET] &&
1085  io.KeysDown[GLFW_KEY_BACKSLASH]) {
1086  enableHotkey = true;
1087  }
1088  }
1089  if (gpUseEstopHotkey != nullptr && *gpUseEstopHotkey) {
1090  ImGuiIO& io = ImGui::GetIO();
1091  if (io.KeysDown[GLFW_KEY_SPACE]) {
1092  HALSIM_SetDriverStationEnabled(false);
1093  }
1094  }
1095 
1096  ImGui::SetNextWindowPos(ImVec2{5, 20}, ImGuiCond_FirstUseEver);
1097  ImGui::Begin("Robot State", nullptr, ImGuiWindowFlags_AlwaysAutoResize);
1098  if (ImGui::Selectable("Disabled", !isEnabled) || disableHotkey) {
1099  HALSIM_SetDriverStationEnabled(false);
1100  }
1101  if (ImGui::Selectable("Autonomous", isEnabled && isAuto && !isTest)) {
1102  HALSIM_SetDriverStationAutonomous(true);
1103  HALSIM_SetDriverStationTest(false);
1104  HALSIM_SetDriverStationEnabled(true);
1105  }
1106  if (ImGui::Selectable("Teleoperated", isEnabled && !isAuto && !isTest) ||
1107  enableHotkey) {
1108  HALSIM_SetDriverStationAutonomous(false);
1109  HALSIM_SetDriverStationTest(false);
1110  HALSIM_SetDriverStationEnabled(true);
1111  }
1112  if (ImGui::Selectable("Test", isEnabled && isTest)) {
1113  HALSIM_SetDriverStationAutonomous(false);
1114  HALSIM_SetDriverStationTest(true);
1115  HALSIM_SetDriverStationEnabled(true);
1116  }
1117  ImGui::End();
1118  }
1119 
1120  // Update HAL
1121  for (int i = 0, end = gRobotJoysticks.size();
1122  i < end && i < HAL_kMaxJoysticks; ++i) {
1123  gRobotJoysticks[i].SetHAL(i);
1124  }
1125 
1126  // Send new data every 20 ms (may be slower depending on GUI refresh rate)
1127  static double lastNewDataTime = 0.0;
1128  if ((curTime - lastNewDataTime) > 0.02 && !HALSIM_IsTimingPaused()) {
1129  lastNewDataTime = curTime;
1130  HALSIM_NotifyDriverStationNewData();
1131  }
1132 }
