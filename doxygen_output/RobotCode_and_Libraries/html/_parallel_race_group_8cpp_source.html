45 void ParallelRaceGroup::AddCommands(
46  std::vector<std::unique_ptr<Command>>&& commands) {
47  if (!RequireUngrouped(commands)) {
48  return;
49  }
50 
51  if (isRunning) {
52  wpi_setWPIErrorWithContext(CommandIllegalUse,
53  "Commands cannot be added to a CommandGroup "
54  "while the group is running");
55  }
56 
57  for (auto&& command : commands) {
58  if (RequirementsDisjoint(this, command.get())) {
59  command->SetGrouped(true);
60  AddRequirements(command->GetRequirements());
61  m_runWhenDisabled &= command->RunsWhenDisabled();
62  m_commands.emplace_back(std::move(command));
63  } else {
64  wpi_setWPIErrorWithContext(CommandIllegalUse,
65  "Multiple commands in a parallel group cannot "
66  "require the same subsystems");
67  return;
68  }
69  }
70 }
