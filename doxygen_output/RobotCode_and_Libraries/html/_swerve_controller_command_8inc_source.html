62 template <size_t NumModules>
63 void SwerveControllerCommand<NumModules>::Execute() {
64  auto curTime = units::second_t(m_timer.Get());
65 
66  auto m_desiredState = m_trajectory.Sample(curTime);
67  auto m_desiredPose = m_desiredState.pose;
68 
69  auto m_poseError = m_desiredPose.RelativeTo(m_pose());
70 
71  auto targetXVel = units::meters_per_second_t(m_xController->Calculate(
72  (m_pose().Translation().X().template to<double>()),
73  (m_desiredPose.Translation().X().template to<double>())));
74  auto targetYVel = units::meters_per_second_t(m_yController->Calculate(
75  (m_pose().Translation().Y().template to<double>()),
76  (m_desiredPose.Translation().Y().template to<double>())));
77 
78  // Profiled PID Controller only takes meters as setpoint and measurement
79  // The robot will go to the desired rotation of the final pose in the
80  // trajectory, not following the poses at individual states.
81  auto targetAngularVel =
82  units::radians_per_second_t(m_thetaController->Calculate(
83  m_pose().Rotation().Radians(), m_finalPose.Rotation().Radians()));
84 
85  auto vRef = m_desiredState.velocity;
86 
87  targetXVel += vRef * m_poseError.Rotation().Cos();
88  targetYVel += vRef * m_poseError.Rotation().Sin();
89 
90  auto targetChassisSpeeds =
91  frc::ChassisSpeeds{targetXVel, targetYVel, targetAngularVel};
92 
93  auto targetModuleStates =
94  m_kinematics.ToSwerveModuleStates(targetChassisSpeeds);
95 
96  m_outputStates(targetModuleStates);
97 }
