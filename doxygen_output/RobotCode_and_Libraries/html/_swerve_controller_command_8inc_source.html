82 template <size_t NumModules>
83 void SwerveControllerCommand<NumModules>::Initialize() {
84  if (m_desiredRotation == nullptr) {
85  m_desiredRotation = [&] {
86  return m_trajectory.States().back().pose.Rotation();
87  };
88  }
89  m_timer.Reset();
90  m_timer.Start();
91 }

93 template <size_t NumModules>
94 void SwerveControllerCommand<NumModules>::Execute() {
95  auto curTime = units::second_t(m_timer.Get());
96  auto m_desiredState = m_trajectory.Sample(curTime);
97 
98  auto targetChassisSpeeds =
99  m_controller.Calculate(m_pose(), m_desiredState, m_desiredRotation());
100  auto targetModuleStates =
101  m_kinematics.ToSwerveModuleStates(targetChassisSpeeds);
102 
103  m_outputStates(targetModuleStates);
104 }
