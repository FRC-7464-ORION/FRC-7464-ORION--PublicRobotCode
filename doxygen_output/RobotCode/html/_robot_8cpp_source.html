494 // Periodic code for all modes should go here.
495 void Robot::RobotPeriodic() {
496 
497 #if TELEMETRY_DEBUG
498  // The time we started the telemetry output, in microseconds (usec)
499  uint64_t FPGATimeStart;
500  // The time we ended the telemetry output, in microseconds (usec)
501  uint64_t FPGATimeEnd;
502  // The elapsed time of the telemetry output, in microseconds (usec)
503  uint64_t FPGATimeElapsed;
504  // The elapsed time of the telemetry output, in milliseconds (msec, ms)
505  double TelemetryOutputTimeMS;
506  // A boolean to tell us when to output the telemetry time
507  bool output_telemetry_time = false;
508 
509  // If we are on a one second increment...
510  if(m_RobotTick % k_ExpectedRobotTicksPerSecond == 0)
511  {
512  // Get the starting FPGA time before we start telemetry
513  FPGATimeStart = frc::RobotController::GetFPGATime();
514 
515  // Indicate we can output telemetry time
516  output_telemetry_time = true;
517  }
518 #endif // #if TELEMETRY_DEBUG
519 
520  // If the telemetry output pointer has been initialized...
521  if(m_TelemetryOutputter != nullptr) {
522 
523  // Output the telemetry to the dashboard
524  m_TelemetryOutputter->OutputTelemetryToSmartDashboard();
525 
526 #if TELEMETRY_DEBUG
527 
528  if(output_telemetry_time) {
529 
530  // Get the ending time of the telemetry output
531  FPGATimeEnd = frc::RobotController::GetFPGATime();
532 
533 // Ignore the warning that FPGATimeStart is uninitialized
534 // It is initialized in the previous #if TELEMETRY
535 // See https://gcc.gnu.org/onlinedocs/gcc/Diagnostic-Pragmas.html
536 // and https://www.chiefdelphi.com/t/pragma-or-gradle-flag-for-ignoring-deprecated-warnings/375815
537 #pragma GCC diagnostic push
538 #pragma GCC diagnostic ignored "-Wmaybe-uninitialized"
539  // Calculate the elapsed time for telemetry output, in usec
540  FPGATimeElapsed = FPGATimeEnd - FPGATimeStart;
541 #pragma GCC diagnostic pop
542 
543  // Convert the elapsed time to milliseconds
544  TelemetryOutputTimeMS = FPGATimeElapsed / k_MicrosecondsPerMillisecond;
545 
546  // Output the telemetry output time to the SmartDashboard
547  frc::SmartDashboard::PutNumber("Telem. Out Time(ms)", TelemetryOutputTimeMS);
548 
549  // Create the telemetry message
550  m_telemetry_debug_msg = "Telemetry Output Time = " +
551  std::to_string(TelemetryOutputTimeMS) +
552  "ms.\n";
553 
554  // and output to the DriveStation as a warning
555  frc::DriverStation::ReportWarning(m_telemetry_debug_msg);
556 
557  } // end if(m_RobotTick % k_ExpectedRobotTicksPerSecond == 0)
558 
559 #endif // #if TELEMETRY_DEBUG
560 
561  } // if(m_TelemetryOutputter != nullptr)
562 
563 #if ORION_DEBUG
564 
565  else {
566 
567  // Output that we have not initialized the telemetry outputter
568  frc::DriverStation::ReportError("m_TelemetryOutputter NOT initialized!");
569 
570  } // end else (if(m_TelemetryOutputter != nullptr))
571 
572 #endif // #if ORION_DEBUG
573 
574  // Increment our robot tick counter
575  m_RobotTick++;
576 
577 } // end Robot::RobotPeriodic()
