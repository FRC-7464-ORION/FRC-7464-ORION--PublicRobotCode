99 
100  // Configure your button bindings here
101 
102  // NOTE: Using `new` here will leak these commands if they are ever no
103  // longer needed. This is usually a non-issue as button-bindings tend
104  // to be permanent - however, if you wish to avoid this, the commands
105  // should be stack-allocated and declared as members of RobotContainer.
106 
107  // DRIVE TRAIN COMMAND BUTTON BINDINGS
108 
109  // When the Start button of the primary joystick is pressed,
110  // run the command to toggle the drive direction
111  frc2::JoystickButton(&m_PrimaryJoystick, k_F310_StartButton)
112  .WhenPressed(m_cmdToggleDriveDirection);
113 
114  // When the B button of the primary joystick is pressed,
115  // run the command to toggle the smoothing mode
116  frc2::JoystickButton(&m_PrimaryJoystick, k_F310_B_Button)
117  .WhenPressed(m_cmdToggleSmoothingMode);
118 
119  // When the A button of the primary joystick is pressed (and held),
120  // run the command to enable turbo mode
121  frc2::JoystickButton(&m_PrimaryJoystick, k_F310_A_Button)
122  .WhenPressed(&m_cmdEnableTurboMode);
123

125  // run the command to disable turbo mode

127  .WhenReleased(&m_cmdDisableTurboMode);
128

132  // run the command CmdTurnWoFCCW using the PAT Turner subsystem

134  .WhenPressed(&m_cmdTurnWofCCW);
135 
136  // When the left rear button of the secondary joystick is pressed,
137  // run the command CmdTurnWoFCW using the PAT Turner subsystem

139  .WhenPressed(&m_cmdTurnWofCW);
140 
141  // HANS/FRANZ SYSTEM COMMAND BUTTON BINDINGS

148 
149  // HANS/FRANZ MUSCLES COMMAND BUTTON BINDINGS
150 
151  // When the back button of the secondary joystic is pressed,

154  .WhenPressed(&m_cmdDisableHansFranzMuscles);
155 
156  // When the North (top) POV button of the secondary joystick is pressed,

164  .WhenPressed(&m_cmdRetractHansFranzMuscles);

166  // PID PSSH COMMAND BUTTON BINDINGS
167 
168  // When the Y button on the secondary joystick is pressed,
169  // run the command to put Pssh mechanism in dump mode

172 
173  // When the Y button on the secondary joystick is released,

229 void RobotContainer::StartTelemetryOutputTimeMeasurement() {

231  // If we are on a one second increment after the 1st 5 seconds...
232  if(GetRobotTick() % k_ExpectedRobotTicksPerSecond == 0

242

247 // A method to finish measuring the telemetry output time
248 void RobotContainer::FinishTelemetryOutputTimeMeasurement() {
249 
250  // If we need to output the telemetry output time...
251  if(m_output_telemetry_time) {
252 
253  // Get the ending time of the telemetry output
254  m_FPGATimeEnd = frc::RobotController::GetFPGATime();
255 
256  // Calculate the elapsed time for telemetry output, in usec
257  m_FPGATimeElapsed = m_FPGATimeEnd - m_FPGATimeStart;
258 
259  // Convert the elapsed time to milliseconds
260  m_TelemetryOutputTimeMS =
261  m_FPGATimeElapsed / k_MicrosecondsPerMillisecond;
262 
263  // If we have a new maximum output time...
264  if(m_TelemetryOutputTimeMS > m_MaxTelemetryOutputTimeMS) {
265 
266  // Save the new max output time
267  m_MaxTelemetryOutputTimeMS = m_TelemetryOutputTimeMS;
268 
269  } // end if(m_TelemetryOutputTimeMS > m_MaxTelemetryOutputTimeMS)
270 
271  // If we have a new minimum output time...
272  if(m_TelemetryOutputTimeMS < m_MinTelemetryOutputTimeMS) {
273 
274  // Save the new min output time
275  m_MinTelemetryOutputTimeMS = m_TelemetryOutputTimeMS;
276 
277  } // end if(m_TelemetryOutputTimeMS > m_MinTelemetryOutputTimeMS)
278 
279  // Output the telemetry output time to the SmartDashboard
280  frc::SmartDashboard::PutNumber(k_TelemetryOutputTimeMS_Key,
281  m_TelemetryOutputTimeMS);
282 
283  // Output the maximum telemetry output time to the SmartDashboard
284  frc::SmartDashboard::PutNumber(k_MaxTelemetryOutputTimeMS_Key,
285  m_MaxTelemetryOutputTimeMS);
286 
287  // Output the minimum telemetry output time to the SmartDashboard
288  frc::SmartDashboard::PutNumber(k_MinTelemetryOutputTimeMS_Key,
289  m_MinTelemetryOutputTimeMS);
290 
291  // Don't output till the next second
292  m_output_telemetry_time = false;
293 
294  } // end if(m_output_telemetry_time)
295 
296 } // end RobotContainer::FinishTelemetryOutputTimeMeasurement()
