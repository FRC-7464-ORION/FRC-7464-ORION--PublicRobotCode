135 // Called repeatedly when this Command is scheduled to run
136 void CmdAutoTurnAngle::Execute() {
137 
138  // Calculate the speed that we have to rotate at for the controller
139  // to reach our target, clamping our speed to the maximum provided
140  m_RotateToAngleSpeed =
141  std::clamp(
142  m_turnController->Calculate(m_AHRS->GetAngle(), m_target_angle),
143  -1.0 * m_MaxRotateToAngleSpeed,
144  m_MaxRotateToAngleSpeed);
145 
146  // Set the left and right motors' speeds
147  // NOTE: For tank drive, to turn, left and right motors must
148  // go in opposite directions, hence the negative
149  // multiplier on the right motors' speed
150  m_left_motor_speed = m_RotateToAngleSpeed;
151  m_right_motor_speed = -1.0 * m_RotateToAngleSpeed;
152 
153  // Turn the robot
154  m_subSysDriveTrain->DriveTankStyle(m_left_motor_speed,
155  m_right_motor_speed);
156 
157  // Indicate our turn is completed when we are at the setpoint (target)
158  m_turn_completed = m_turnController->AtSetpoint();
