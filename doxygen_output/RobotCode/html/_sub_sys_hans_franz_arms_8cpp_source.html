138 // A little explanation is warranted.
139 //
140 // * We have the arms going up (extending) and going down (retracting).
141 //
142 // * We have two limit switches, one which closes when fully retracted,
143 // and one which closes when fully extended
144 //
145 // * These switches are connected to DIO 0 and 1 (see RobotMap.h)
146 //
147 // * DIO 0 and 1 have internal pull-up resistors between Vcc (+3.3V or
148 // +5V) and the signal. This makes the signal have a logic 1 (Vcc) when
149 // nothing else is connected.
150 //
151 // * The common (C) of a switch is connected to the signal of a DIO, and
152 // the the normally open (NO) of the switch will connect to ground.
153 //
154 // * When the switch closes, the NO contact makes contact with the common
155 // contact, and makes the signal go to a logic 0 (approx. 0V).
156 //
157 // * However, we use positive logic to see when the switch is set (closed).
158 // This means that a true indicates switch closed, and false for open.
159 //
160 // * Below is a table of inputs (motor direction, retracted switch,
161 // and extended switch), along with what state the arms would be in based
162 // on these inputs, along with the motor direction allowed.
163 //
164 // Dir Retracted Switch Extended Switch State Dir Allowed
165 // - OPEN OPEN Retracting 0,-,+
166 // - OPEN CLOSED StartRetracting 0,-
167 // - CLOSED OPEN Retracted 0, +
168 // - CLOSED CLOSED Fault 0
169 // 0 OPEN OPEN Unknown 0,-,+
170 // 0 OPEN CLOSED Extended 0,-
171 // 0 CLOSED OPEN Retracted 0, +
172 // 0 CLOSED CLOSED Fault 0
173 // + OPEN OPEN Extending 0,-,+
174 // + OPEN CLOSED Extended 0,-
175 // + CLOSED OPEN StartExtending 0,+
176 // + CLOSED CLOSED Fault 0
177 //
178 // If we reorder the above table by the direction allowed (a spreadsheet
179 // is very helpful to do this), we get:
180 //
181 // Dir Retracted Switch Extended Switch State Dir Allowed
182 // - CLOSED CLOSED Fault 0
183 // 0 CLOSED CLOSED Fault 0
184 // + CLOSED CLOSED Fault 0
185 // 0 OPEN CLOSED Extended 0,-
186 // + OPEN CLOSED Extended 0,-
187 // - OPEN CLOSED StartRetracting 0,-
188 // + OPEN OPEN Extending 0,-,+
189 // - OPEN OPEN Retracting 0,-,+
190 // 0 OPEN OPEN Unknown 0,-,+
191 // - CLOSED OPEN Retracted 0, +
192 // 0 CLOSED OPEN Retracted 0, +
193 // + CLOSED OPEN StartExtending 0, +
194 //
195 // We use the above table in the code below. Enumerations are in the
196 // header file.

199 void SubSysHansFranzArms::MoveHansFranzArms(double speed) {
200 
201  // If Hans and Franz arms are enabled...
202  if(m_HansFranzArmsEnabled)
203  {
204 
205  // Get the current state of Hans and Franz arms
206  m_HansFranzArmsState = GetCurrentHansFranzArmsState(speed);
207 
208  // Now set the allowed speed directions
209  SetAllowedSpeeds(m_HansFranzArmsState);
210 
211  // If the motor speed is positive AND positive is allowed,
212  // OR
213  // If the motor speed is negative AND negative is allowed...
214  if( ((GetMotorSpeedDir(speed) == POSITIVE) && (m_PositiveSpeedUpAllowed))
215  ||
216  ((GetMotorSpeedDir(speed) == NEGATIVE) && (m_NegativeSpeedDownAllowed)) )
217  {
218 
219  // Set the Hans and Franz arms motor to the passed in speed
220  m_HansFranzArmsController->Set(speed);
221 
222  }
223  else // positive or negative speeds are not allowed
224  {
225 
226  // Set the Hans and Franz arms motor to be stopped
227  m_HansFranzArmsController->Set(k_MotorStopSpeed);
228 
229  } // end ((GetMotorSpeedDir(speed) == POSITIVE)...
230 
231  // Switch on the state of the Hans and Franz Arms...
232  switch(m_HansFranzArmsState)
233  {
234 
235  // If we are extended or starting to retract
236  case EXTENDED:
237  case START_RETRACTING:
238  // Set state for dashboard to extended
239  m_HansFranzArmsSmartDashboardState = k_HansFranzArmsStateExtended;
240  // Reset the counter for the extended limit switch
241  ResetHansFranzArmsFullyExtendedCounter();
242  break;
243 
244  // If we are retracted or starting to extend...
245  case RETRACTED:
246  case START_EXTENDING:
247  // Set state for dashboard to retracted
248  m_HansFranzArmsSmartDashboardState = k_HansFranzArmsStateRetracted;
249  // Reset the counter for the retracted limit switch
250  ResetHansFranzArmsFullyRetractedCounter();
251  break;
252 
253  // If we are extending...
254  case EXTENDING:
255  // Set state for dashboard to extending
256  m_HansFranzArmsSmartDashboardState = k_HansFranzArmsStateExtending;
257  break;
258 
259  // If we are retracting...
260  case RETRACTING:
261  // Set state for dashboard to extending
262  m_HansFranzArmsSmartDashboardState = k_HansFranzArmsStateRetracting;
263  break;
264 
265  // If we don't know what state we are in...
266  case UNKNOWN:
267  // Set state for dashboard to unknown
268  m_HansFranzArmsSmartDashboardState = k_HansFranzArmsStateUnknown;
269  break;
270 
271  // If we have a fault
272  case FAULT:
273  default:
274  // Set state for dashboard to fault
275  m_HansFranzArmsSmartDashboardState = k_HansFranzArmsStateFault;
276 
277  } // end switch(m_HansFranzArmsState)
278 
279  } // end if(m_HansFranzArmsEnabled)

384 // Method to determine the current state of the arms based on inputs.
385 SubSysHansFranzArms::HansFranzArmsStates
386  SubSysHansFranzArms::GetCurrentHansFranzArmsState(double speed)
387 {
388 
389  // A state variable for Hans and Franz arms, set to unknown initially
390  HansFranzArmsStates state = UNKNOWN;
391 
392  // If both switches are closed...
393  if(IsHansFranzArmsFullyRetracted() && IsHansFranzArmsFullyExtended())
394  // we have a fault
395  state = FAULT;
396 
397  // If the retracted switch is open and the extended switch is closed...
398  if(!IsHansFranzArmsFullyRetracted() && IsHansFranzArmsFullyExtended())
399  {
400 
401  // If the motor speed is negative...
402  if(GetMotorSpeedDir(speed) == NEGATIVE)
403  state = START_RETRACTING;
404  else // Speed is ZERO or POSITIVE
405  state = EXTENDED;
406 
407  } // end if(!IsHansFranzArmsFullyRetracted() && IsHansFranzArmsFullyExtended())
408 
409  // If the both switches are open...
410  if(!IsHansFranzArmsFullyRetracted() && !IsHansFranzArmsFullyExtended())
411  {
412 
413  // If the motor speed is positive...
414  if(GetMotorSpeedDir(speed) == POSITIVE)
415  // We are extending
416  state = EXTENDING;
417 
418  // If the motor speed is negative...
419  else if(GetMotorSpeedDir(speed) == NEGATIVE)
420  // We are retracting
421  state = RETRACTING;
422 
423  // Must be zero speed
424  else
425  // So we don't know what we are doing
426  state = UNKNOWN;
427 
428  } // end if(!IsHansFranzArmsFullyRetracted() && !IsHansFranzArmsFullyExtended())
429 
430  // If the retracted switch is closed and the extended switch is open...
431  if(IsHansFranzArmsFullyRetracted() && !IsHansFranzArmsFullyExtended())
432  {
433 
434  // If the motor speed is positive...
435  if(GetMotorSpeedDir(speed) == POSITIVE)
436  // We are starting to extend from the retracted position
437  state = START_EXTENDING;
438  else // speed is negative or zero
439  state = RETRACTED;
440 
441  } // end if(IsHansFranzArmsFullyRetracted() && !IsHansFranzArmsFullyExtended())
442 
443  // Return the determined state
444  return state;
445 
446 } // end SubSysHansFranzArms::GetCurrentHansFranzArmsState(...)
