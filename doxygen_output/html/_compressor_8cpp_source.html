20 Compressor::Compressor(int pcmID) : m_module(pcmID) {
21  int32_t status = 0;
22  m_compressorHandle = HAL_InitializeCompressor(m_module, &status);
23  if (status != 0) {
24  wpi_setErrorWithContextRange(status, 0, HAL_GetNumPCMModules(), pcmID,
25  HAL_GetErrorMessage(status));
26  return;
27  }
28  SetClosedLoopControl(true);
29 
30  HAL_Report(HALUsageReporting::kResourceType_Compressor, pcmID);
31  SetName("Compressor", pcmID);
32 }
