94 void SmartDashboard::PutData(wpi::StringRef key, Sendable* data) {
95  if (data == nullptr) {
96  wpi_setGlobalWPIErrorWithContext(NullParameter, "value");
97  return;
98  }
99  auto& inst = Singleton::GetInstance();
100  std::lock_guard<wpi::mutex> lock(inst.tablesToDataMutex);
101  auto& sddata = inst.tablesToData[key];
102  if (!sddata.sendable || sddata.sendable != data) {
103  sddata = SmartDashboardData(data);
104  auto dataTable = inst.table->GetSubTable(key);
105  sddata.builder.SetTable(dataTable);
106  data->InitSendable(sddata.builder);
107  sddata.builder.UpdateTable();
108  sddata.builder.StartListeners();
109  dataTable->GetEntry(".name").SetString(key);
110  }
111 }

121 Sendable* SmartDashboard::GetData(wpi::StringRef key) {
122  auto& inst = Singleton::GetInstance();
123  std::lock_guard<wpi::mutex> lock(inst.tablesToDataMutex);
124  auto data = inst.tablesToData.find(key);
125  if (data == inst.tablesToData.end()) {
126  wpi_setGlobalWPIErrorWithContext(SmartDashboardMissingKey, key);
127  return nullptr;
128  }
129  return data->getValue().sendable;
130 }
