138 // The method for driving arcade style
139 void SubSysDriveTrain::DriveArcadeStyle(double y, double x) {
140 
141  // The multiplier for putting turbo mode on
142  double const TurboMultiplier =
143  k_DriveTrainAbsValMaxSpeedTurboOn/k_DriveTrainAbsValMaxSpeedNoTurbo;
144 
145  // The y axis value after being nulled, desensitized, and speed limited
146  double y_Nulled_Desensed_Limited;
147  // The x axis value after being nulled, desensitized, and speed limited
148  double x_Nulled_Desensed_Limited;
149  // The y axis value after being nulled, desensitized, speed limited,
150  // and turboed (on or off)
151  double y_Turboed;
152  // The x axis value after being nulled, desensitized, speed limited,
153  // and turboed (on or off)
154  double x_Turboed;
155  // The y axis value after being nulled, desensitized, speed limited,
156  // turboed (on or off), and smoothed (on or off)
157  double y_Smoothed;
158  // The x axis value after being nulled, desensitized, speed limited,
159  // turboed (on or off), and smoothed (on or off)
160  double x_Smoothed;
161 
162  // We were passed in a nulled/desensed/limited value for y,
163  // so we'll use that
164  y_Nulled_Desensed_Limited = y;
165  // We were passed in a nulled/desensed/limited value for x,
166  // so we'll use that
167  x_Nulled_Desensed_Limited = x;
168 
169  // If turbo mode is not enabled...
170  if(!m_TurboModeEnabled) {
171 
172  // We'll just used the passed in values...
173  y_Turboed = y_Nulled_Desensed_Limited;
174  x_Turboed = x_Nulled_Desensed_Limited;
175 
176  } // end if(!m_TurboModeEnabled)
177 
178  else { // Turbo mode is enabled...
179 
180  // We'll use the passed in values, multiplied by our
181  // turbo factor
182  y_Turboed = y_Nulled_Desensed_Limited*TurboMultiplier;
183  x_Turboed = x_Nulled_Desensed_Limited*TurboMultiplier;
184 
185  } // end else (if(!m_TurboModeEnabled))
186 
187  // If smoothing mode is not enabled...
188  if(!m_DriveTrain_Smoothing_Enabled) {
189 
190  // We'll just use the "turboed" y value
191  y_Smoothed = y_Turboed;
192  // We'll just use the "turboed" x value
193  x_Smoothed = x_Turboed;
194  }
195 
196  // Smoothing is enabled...
197  else {
198 
199  // The smoothed y value is calculated using the formula for
200  // an infinite impulse repsponse digital low pass filter,
201  // which is:
202  // y_curr_filt = d*y_curr_unfilt + (1-d)*y_past_filt
203  y_Smoothed =
204  (k_DriveTrain_IIR_DLPF_d * y_Turboed) +
205  ((1.0 - k_DriveTrain_IIR_DLPF_d) * m_speed1_past_filtered_output);
206 
207  // Save the current filtered y output for next time
208  m_speed1_past_filtered_output = y_Smoothed;
209 
210  // The smoothed x value is calculated using the formula for
211  // an infinite impulse repsponse digital low pass filter,
212  // which is:
213  // x_curr_filt = d*x_curr_unfilt + (1-d)*x_past_filt
214  x_Smoothed =
215  (k_DriveTrain_IIR_DLPF_d * x_Turboed) +
216  ((1.0 - k_DriveTrain_IIR_DLPF_d) * m_speed2_past_filtered_output);
217 
218  // Save the current filtered x output for next time
219  m_speed2_past_filtered_output = x_Smoothed;
220 
221  } // end else (if(!m_DriveTrain_Smoothing_Enabled))
222 
223  // Use arcade drive using the passed in values after:
224  // * Nulling, desensitizing, and speed limiting
225  // * Being Turboed (on or off)
226  // * Being Smoothed (on or off)
227  // * not squaring inputs
228  m_differentialDrive->ArcadeDrive(
229  y_Smoothed, x_Smoothed, k_DoNOTUseBuiltInSquaring);
230 
231 } // end SubSysDriveTrain::DriveArcadeStyle()
