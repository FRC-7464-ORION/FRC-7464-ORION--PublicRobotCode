75 void SendableBuilderImpl::AddBooleanProperty(const wpi::Twine& key,
76  std::function<bool()> getter,
77  std::function<void(bool)> setter) {
78  m_properties.emplace_back(*m_table, key);
79  if (getter) {
80  m_properties.back().update = [=](nt::NetworkTableEntry entry,
81  uint64_t time) {
82  entry.SetValue(nt::Value::MakeBoolean(getter(), time));
83  };
84  }
85  if (setter) {
86  m_properties.back().createListener =
87  [=](nt::NetworkTableEntry entry) -> NT_EntryListener {
88  return entry.AddListener(
89  [=](const nt::EntryNotification& event) {
90  if (!event.value->IsBoolean()) return;
91  setter(event.value->GetBoolean());
92  },
93  NT_NOTIFY_IMMEDIATE | NT_NOTIFY_NEW | NT_NOTIFY_UPDATE);
94  };
95  }
96 }
