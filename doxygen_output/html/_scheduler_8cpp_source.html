74 void Scheduler::Run() {
75  // Get button input (going backwards preserves button priority)
76  {
77  if (!m_impl->enabled) return;
78 
79  std::lock_guard<wpi::mutex> lock(m_impl->buttonsMutex);
80  for (auto& button : m_impl->buttons) {
81  button->Execute();
82  }
83  }
84 
85  // Call every subsystem's periodic method
86  for (auto& subsystem : m_impl->subsystems) {
87  subsystem->Periodic();
88  }
89 
90  m_impl->runningCommandsChanged = false;
91 
92  // Loop through the commands
93  for (auto cmdIter = m_impl->commands.begin();
94  cmdIter != m_impl->commands.end();) {
95  Command* command = *cmdIter;
96  // Increment before potentially removing to keep the iterator valid
97  ++cmdIter;
98  if (!command->Run()) {
99  Remove(command);
100  m_impl->runningCommandsChanged = true;
101  }
102  }
103 
104  // Add the new things
105  {
106  std::lock_guard<wpi::mutex> lock(m_impl->additionsMutex);
107  for (auto& addition : m_impl->additions) {
108  // Check to make sure no adding during adding
109  if (m_impl->adding) {
110  wpi_setWPIErrorWithContext(IncompatibleState,
111  "Can not start command from cancel method");
112  } else {
113  m_impl->ProcessCommandAddition(addition);
114  }
115  }
116  m_impl->additions.clear();
117  }
118 
119  // Add in the defaults
120  for (auto& subsystem : m_impl->subsystems) {
121  if (subsystem->GetCurrentCommand() == nullptr) {
122  if (m_impl->adding) {
123  wpi_setWPIErrorWithContext(IncompatibleState,
124  "Can not start command from cancel method");
125  } else {
126  m_impl->ProcessCommandAddition(subsystem->GetDefaultCommand());
127  }
128  }
129  subsystem->ConfirmCommand();
130  }
131 }

158 void Scheduler::InitSendable(SendableBuilder& builder) {
159  builder.SetSmartDashboardType("Scheduler");
160  auto namesEntry = builder.GetEntry("Names");
161  auto idsEntry = builder.GetEntry("Ids");
162  auto cancelEntry = builder.GetEntry("Cancel");
163  builder.SetUpdateTable([=]() {
164  // Get the list of possible commands to cancel
165  auto new_toCancel = cancelEntry.GetValue();
166  wpi::ArrayRef<double> toCancel;
167  if (new_toCancel) toCancel = new_toCancel->GetDoubleArray();
168 
169  // Cancel commands whose cancel buttons were pressed on the SmartDashboard
170  if (!toCancel.empty()) {
171  for (auto& command : m_impl->commands) {
172  for (const auto& cancelled : toCancel) {
173  if (command->GetID() == cancelled) {
174  command->Cancel();
175  }
176  }
177  }
178  nt::NetworkTableEntry(cancelEntry).SetDoubleArray({});
179  }
180 
181  // Set the running commands
182  if (m_impl->runningCommandsChanged) {
183  m_impl->commandsBuf.resize(0);
184  m_impl->idsBuf.resize(0);
185  for (const auto& command : m_impl->commands) {
186  m_impl->commandsBuf.emplace_back(command->GetName());
187  m_impl->idsBuf.emplace_back(command->GetID());
188  }
189  nt::NetworkTableEntry(namesEntry).SetStringArray(m_impl->commandsBuf);
190  nt::NetworkTableEntry(idsEntry).SetDoubleArray(m_impl->idsBuf);
191  }
192  });
193 }

213 void Scheduler::Impl::ProcessCommandAddition(Command* command) {
214  if (command == nullptr) return;
215 
216  // Only add if not already in
217  auto found = commands.find(command);
218  if (found == commands.end()) {
219  // Check that the requirements can be had
220  const auto& requirements = command->GetRequirements();
221  for (const auto& requirement : requirements) {
222  if (requirement->GetCurrentCommand() != nullptr &&
223  !requirement->GetCurrentCommand()->IsInterruptible())
224  return;
225  }
226 
227  // Give it the requirements
228  adding = true;
229  for (auto&& requirement : requirements) {
230  if (requirement->GetCurrentCommand() != nullptr) {
231  requirement->GetCurrentCommand()->Cancel();
232  Remove(requirement->GetCurrentCommand());
233  }
234  requirement->SetCurrentCommand(command);
235  }
236  adding = false;
237 
238  commands.insert(command);
239 
240  command->StartRunning();
241  runningCommandsChanged = true;
242  }
