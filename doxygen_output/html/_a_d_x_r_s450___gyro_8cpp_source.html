55 static bool CalcParity(int v) {
56  bool parity = false;
57  while (v != 0) {
58  parity = !parity;
59  v = v & (v - 1);
60  }
61  return parity;
62 }

72 uint16_t ADXRS450_Gyro::ReadRegister(int reg) {
73  int cmd = 0x80000000 | static_cast<int>(reg) << 17;
74  if (!CalcParity(cmd)) cmd |= 1u;
75 
76  // big endian
77  uint8_t buf[4] = {static_cast<uint8_t>((cmd >> 24) & 0xff),
78  static_cast<uint8_t>((cmd >> 16) & 0xff),
79  static_cast<uint8_t>((cmd >> 8) & 0xff),
80  static_cast<uint8_t>(cmd & 0xff)};
81 
82  m_spi.Write(buf, 4);
83  m_spi.Read(false, buf, 4);
84  if ((buf[0] & 0xe0) == 0) return 0; // error, return 0
85  return static_cast<uint16_t>((BytesToIntBE(buf) >> 5) & 0xffff);
86 }
