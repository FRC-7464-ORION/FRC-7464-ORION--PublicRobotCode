34 Command::Command(const wpi::Twine& name, double timeout) : SendableBase(false) {
35  // We use -1.0 to indicate no timeout.
36  if (timeout < 0.0 && timeout != -1.0)
37  wpi_setWPIErrorWithContext(ParameterOutOfRange, "timeout < 0.0");
38 
39  m_timeout = timeout;
40 
41  // If name contains an empty string
42  if (name.isTriviallyEmpty() ||
43  (name.isSingleStringRef() && name.getSingleStringRef().empty())) {
44  SetName("Command_" + wpi::Twine(typeid(*this).name()));
45  } else {
46  SetName(name);
47  }
48 }

91 bool Command::Run() {
92  if (!m_runWhenDisabled && m_parent == nullptr && RobotState::IsDisabled())
93  Cancel();
94 
95  if (IsCanceled()) return false;
96 
97  if (!m_initialized) {
98  m_initialized = true;
99  StartTiming();
100  _Initialize();
101  Initialize();
102  }
103  _Execute();
104  Execute();
105  return !IsFinished();
106 }
