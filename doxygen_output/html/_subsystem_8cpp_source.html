23 void Subsystem::SetDefaultCommand(Command* command) {
24  if (command == nullptr) {
25  m_defaultCommand = nullptr;
26  } else {
27  const auto& reqs = command->GetRequirements();
28  if (std::find(reqs.begin(), reqs.end(), this) == reqs.end()) {
29  wpi_setWPIErrorWithContext(
30  CommandIllegalUse, "A default command must require the subsystem");
31  return;
32  }
33 
34  m_defaultCommand = command;
35  }
36 }
