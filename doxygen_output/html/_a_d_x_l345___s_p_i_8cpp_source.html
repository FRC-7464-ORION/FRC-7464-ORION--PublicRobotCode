53 double ADXL345_SPI::GetAcceleration(ADXL345_SPI::Axes axis) {
54  uint8_t buffer[3];
55  uint8_t command[3] = {0, 0, 0};
56  command[0] = (kAddress_Read | kAddress_MultiByte | kDataRegister) +
57  static_cast<uint8_t>(axis);
58  m_spi.Transaction(command, buffer, 3);
59 
60  // Sensor is little endian... swap bytes
61  int16_t rawAccel = buffer[2] << 8 | buffer[1];
62  return rawAccel * kGsPerLSB;
63 }

65 ADXL345_SPI::AllAxes ADXL345_SPI::GetAccelerations() {
66  AllAxes data = AllAxes();
67  uint8_t dataBuffer[7] = {0, 0, 0, 0, 0, 0, 0};
68  int16_t rawData[3];
69 
70  // Select the data address.
71  dataBuffer[0] = (kAddress_Read | kAddress_MultiByte | kDataRegister);
72  m_spi.Transaction(dataBuffer, dataBuffer, 7);
73 
74  for (int i = 0; i < 3; i++) {
75  // Sensor is little endian... swap bytes
76  rawData[i] = dataBuffer[i * 2 + 2] << 8 | dataBuffer[i * 2 + 1];
77  }
78 
79  data.XAxis = rawData[0] * kGsPerLSB;
80  data.YAxis = rawData[1] * kGsPerLSB;
81  data.ZAxis = rawData[2] * kGsPerLSB;
82 
83  return data;
84 }
