40 void MecanumDrive::DriveCartesian(double ySpeed, double xSpeed,
41  double zRotation, double gyroAngle) {
42  if (!reported) {
43  HAL_Report(HALUsageReporting::kResourceType_RobotDrive, 4,
44  HALUsageReporting::kRobotDrive2_MecanumCartesian);
45  reported = true;
46  }
47 
48  ySpeed = Limit(ySpeed);
49  ySpeed = ApplyDeadband(ySpeed, m_deadband);
50 
51  xSpeed = Limit(xSpeed);
52  xSpeed = ApplyDeadband(xSpeed, m_deadband);
53 
54  // Compensate for gyro angle.
55  Vector2d input{ySpeed, xSpeed};
56  input.Rotate(-gyroAngle);
57 
58  double wheelSpeeds[4];
59  wheelSpeeds[kFrontLeft] = input.x + input.y + zRotation;
60  wheelSpeeds[kFrontRight] = -input.x + input.y - zRotation;
61  wheelSpeeds[kRearLeft] = -input.x + input.y + zRotation;
62  wheelSpeeds[kRearRight] = input.x + input.y - zRotation;
63 
64  Normalize(wheelSpeeds);
65 
66  m_frontLeftMotor.Set(wheelSpeeds[kFrontLeft] * m_maxOutput);
67  m_frontRightMotor.Set(wheelSpeeds[kFrontRight] * m_maxOutput *
68  m_rightSideInvertMultiplier);
69  m_rearLeftMotor.Set(wheelSpeeds[kRearLeft] * m_maxOutput);
70  m_rearRightMotor.Set(wheelSpeeds[kRearRight] * m_maxOutput *
71  m_rightSideInvertMultiplier);
72 
73  Feed();
74 }
