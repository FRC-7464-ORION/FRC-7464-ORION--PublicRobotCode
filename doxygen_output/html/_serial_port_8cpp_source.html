21  SerialPort::Parity parity,

28  wpi_setErrorWithContext(status, HAL_GetErrorMessage(status));
29  // Don't continue if initialization failed

31  HAL_SetSerialBaudRate(static_cast<HAL_SerialPort>(port), baudRate, &status);
