30 void DifferentialDrive::ArcadeDrive(double xSpeed, double zRotation,
31  bool squareInputs) {
32  static bool reported = false;
33  if (!reported) {
34  HAL_Report(HALUsageReporting::kResourceType_RobotDrive, 2,
35  HALUsageReporting::kRobotDrive2_DifferentialArcade);
36  reported = true;
37  }
38 
39  xSpeed = Limit(xSpeed);
40  xSpeed = ApplyDeadband(xSpeed, m_deadband);
41 
42  zRotation = Limit(zRotation);
43  zRotation = ApplyDeadband(zRotation, m_deadband);
44 
45  // Square the inputs (while preserving the sign) to increase fine control
46  // while permitting full power.
47  if (squareInputs) {
48  xSpeed = std::copysign(xSpeed * xSpeed, xSpeed);
49  zRotation = std::copysign(zRotation * zRotation, zRotation);
50  }
51 
52  double leftMotorOutput;
53  double rightMotorOutput;
54 
55  double maxInput =
56  std::copysign(std::max(std::abs(xSpeed), std::abs(zRotation)), xSpeed);
57 
58  if (xSpeed >= 0.0) {
59  // First quadrant, else second quadrant
60  if (zRotation >= 0.0) {
61  leftMotorOutput = maxInput;
62  rightMotorOutput = xSpeed - zRotation;
63  } else {
64  leftMotorOutput = xSpeed + zRotation;
65  rightMotorOutput = maxInput;
66  }
67  } else {
68  // Third quadrant, else fourth quadrant
69  if (zRotation >= 0.0) {
70  leftMotorOutput = xSpeed + zRotation;
71  rightMotorOutput = maxInput;
72  } else {
73  leftMotorOutput = maxInput;
74  rightMotorOutput = xSpeed - zRotation;
75  }
76  }
77 
78  m_leftMotor.Set(Limit(leftMotorOutput) * m_maxOutput);
79  m_rightMotor.Set(Limit(rightMotorOutput) * m_maxOutput *
80  m_rightSideInvertMultiplier);
81 
82  Feed();
83 }

85 void DifferentialDrive::CurvatureDrive(double xSpeed, double zRotation,
86  bool isQuickTurn) {
87  static bool reported = false;
88  if (!reported) {
89  HAL_Report(HALUsageReporting::kResourceType_RobotDrive, 2,
90  HALUsageReporting::kRobotDrive2_DifferentialCurvature);
91  reported = true;
92  }
93 
94  xSpeed = Limit(xSpeed);
95  xSpeed = ApplyDeadband(xSpeed, m_deadband);
96 
97  zRotation = Limit(zRotation);
98  zRotation = ApplyDeadband(zRotation, m_deadband);
99 
100  double angularPower;
101  bool overPower;
102 
103  if (isQuickTurn) {
104  if (std::abs(xSpeed) < m_quickStopThreshold) {
105  m_quickStopAccumulator = (1 - m_quickStopAlpha) * m_quickStopAccumulator +
106  m_quickStopAlpha * Limit(zRotation) * 2;
107  }
108  overPower = true;
109  angularPower = zRotation;
110  } else {
111  overPower = false;
112  angularPower = std::abs(xSpeed) * zRotation - m_quickStopAccumulator;
113 
114  if (m_quickStopAccumulator > 1) {
115  m_quickStopAccumulator -= 1;
116  } else if (m_quickStopAccumulator < -1) {
117  m_quickStopAccumulator += 1;
118  } else {
119  m_quickStopAccumulator = 0.0;
120  }
121  }
122 
123  double leftMotorOutput = xSpeed + angularPower;
124  double rightMotorOutput = xSpeed - angularPower;
125 
126  // If rotation is overpowered, reduce both outputs to within acceptable range
127  if (overPower) {
128  if (leftMotorOutput > 1.0) {
129  rightMotorOutput -= leftMotorOutput - 1.0;
130  leftMotorOutput = 1.0;
131  } else if (rightMotorOutput > 1.0) {
132  leftMotorOutput -= rightMotorOutput - 1.0;
133  rightMotorOutput = 1.0;
134  } else if (leftMotorOutput < -1.0) {
135  rightMotorOutput -= leftMotorOutput + 1.0;
136  leftMotorOutput = -1.0;
137  } else if (rightMotorOutput < -1.0) {
138  leftMotorOutput -= rightMotorOutput + 1.0;
139  rightMotorOutput = -1.0;
140  }
141  }
142 
143  // Normalize the wheel speeds
144  double maxMagnitude =
145  std::max(std::abs(leftMotorOutput), std::abs(rightMotorOutput));
146  if (maxMagnitude > 1.0) {
147  leftMotorOutput /= maxMagnitude;
148  rightMotorOutput /= maxMagnitude;
149  }
150 
151  m_leftMotor.Set(leftMotorOutput * m_maxOutput);
152  m_rightMotor.Set(rightMotorOutput * m_maxOutput *
153  m_rightSideInvertMultiplier);
154 
155  Feed();
156 }
