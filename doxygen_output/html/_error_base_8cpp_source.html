37  wpi::StringRef filename, wpi::StringRef function,
38  int lineNumber) const {
39  wpi::SmallString<128> buf;
40  wpi::raw_svector_ostream err(buf);
41  int errNo = errno;
42  if (errNo == 0) {
43  err << "OK: ";
44  } else {
45  err << std::strerror(errNo) << " (" << wpi::format_hex(errNo, 10, true)
46  << "): ";
47  }
48 
49  // Set the current error information for this object.
50  m_error.Set(-1, err.str() + contextMessage, filename, function, lineNumber,
51  this);
52 
53  // Update the global error if there is not one already set.
54  std::lock_guard<wpi::mutex> mutex(globalErrorsMutex);
55  globalErrors.insert(m_error);
56 }
