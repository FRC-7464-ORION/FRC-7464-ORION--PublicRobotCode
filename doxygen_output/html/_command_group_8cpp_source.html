123 void CommandGroup::_Execute() {
124  CommandGroupEntry* entry;
125  Command* cmd = nullptr;
126  bool firstRun = false;
127 
128  if (m_currentCommandIndex == -1) {
129  firstRun = true;
130  m_currentCommandIndex = 0;
131  }
132 
133  // While there are still commands in this group to run
134  while (static_cast<size_t>(m_currentCommandIndex) < m_commands.size()) {
135  // If a command is prepared to run
136  if (cmd != nullptr) {
137  // If command timed out, cancel it so it's removed from the Scheduler
138  if (entry->IsTimedOut()) cmd->_Cancel();
139 
140  // If command finished or was cancelled, remove it from Scheduler
141  if (cmd->Run()) {
142  break;
143  } else {
144  cmd->Removed();
145 
146  // Advance to next command in group
147  m_currentCommandIndex++;
148  firstRun = true;
149  cmd = nullptr;
150  continue;
151  }
152  }
153 
154  entry = &m_commands[m_currentCommandIndex];
155  cmd = nullptr;
156 
157  switch (entry->m_state) {
158  case CommandGroupEntry::kSequence_InSequence:
159  cmd = entry->m_command;
160  if (firstRun) {
161  cmd->StartRunning();
162  CancelConflicts(cmd);
163  firstRun = false;
164  }
165  break;
166 
167  case CommandGroupEntry::kSequence_BranchPeer:
168  // Start executing a parallel command and advance to next entry in group
169  m_currentCommandIndex++;
170  entry->m_command->Start();
171  break;
172 
173  case CommandGroupEntry::kSequence_BranchChild:
174  m_currentCommandIndex++;
175 
176  /* Causes scheduler to skip children of current command which require
177  * the same subsystems as it
178  */
179  CancelConflicts(entry->m_command);
180  entry->m_command->StartRunning();
181 
182  // Add current command entry to list of children of this group
183  m_children.push_back(entry);
184  break;
185  }
186  }
187 
188  // Run Children
189  for (auto& entry : m_children) {
190  auto child = entry->m_command;
191  if (entry->IsTimedOut()) {
192  child->_Cancel();
193  }
194 
195  // If child finished or was cancelled, set it to nullptr. nullptr entries
196  // are removed later.
197  if (!child->Run()) {
198  child->Removed();
199  entry = nullptr;
200  }
201  }
202 
203  m_children.erase(std::remove(m_children.begin(), m_children.end(), nullptr),
204  m_children.end());
205 }

227 void CommandGroup::CancelConflicts(Command* command) {
228  for (auto childIter = m_children.begin(); childIter != m_children.end();) {
229  Command* child = (*childIter)->m_command;
230  bool erased = false;
231 
232  for (auto&& requirement : command->GetRequirements()) {
233  if (child->DoesRequire(requirement)) {
234  child->_Cancel();
235  child->Removed();
236  childIter = m_children.erase(childIter);
237  erased = true;
238  break;
239  }
240  }
241  if (!erased) childIter++;
242  }
