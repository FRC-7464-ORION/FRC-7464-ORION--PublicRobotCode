39 double ADXL345_I2C::GetAcceleration(ADXL345_I2C::Axes axis) {
40  int16_t rawAccel = 0;
41  m_i2c.Read(kDataRegister + static_cast<int>(axis), sizeof(rawAccel),
42  reinterpret_cast<uint8_t*>(&rawAccel));
43  return rawAccel * kGsPerLSB;
44 }

46 ADXL345_I2C::AllAxes ADXL345_I2C::GetAccelerations() {
47  AllAxes data = AllAxes();
48  int16_t rawData[3];
49  m_i2c.Read(kDataRegister, sizeof(rawData),
50  reinterpret_cast<uint8_t*>(rawData));
51 
52  data.XAxis = rawData[0] * kGsPerLSB;
53  data.YAxis = rawData[1] * kGsPerLSB;
54  data.ZAxis = rawData[2] * kGsPerLSB;
55  return data;
56 }
